package io.github.tonywasher.joceanus.prometheus.preference;

import io.github.tonywasher.joceanus.metis.preference.MetisPreferenceSet.MetisPreferenceId;

/**
 * Enum class for preference types.
 */
public enum PrometheusPreferenceType
        implements MetisPreferenceId {
    /**
     * ByteArray.
     */
    BYTEARRAY,

    /**
     * CharArray.
     */
    CHARARRAY;

    /**
     * The String name.
     */
    private String theName;

    @Override
    public String toString() {
        /* If we have not yet loaded the name */
        if (theName == null) {
            /* Load the name */
            theName = PrometheusPreferenceResource.getKeyForPrefType(this).getValue();
        }

        /* return the name */
        return theName;
    }
}
