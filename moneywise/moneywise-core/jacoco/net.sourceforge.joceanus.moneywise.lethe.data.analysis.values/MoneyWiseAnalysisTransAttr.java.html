package net.sourceforge.joceanus.moneywise.lethe.data.analysis.values;

import net.sourceforge.joceanus.metis.data.MetisDataType;
import net.sourceforge.joceanus.moneywise.lethe.data.analysis.base.MoneyWiseAnalysisAttribute;

/**
 * TransactionAttribute enumeration.
 */
public enum MoneyWiseAnalysisTransAttr implements MoneyWiseAnalysisAttribute {
    /**
     * Income.
     */
    INCOME,

    /**
     * Expense.
     */
    EXPENSE,

    /**
     * Profit (NetIncome).
     */
    PROFIT;

    /**
     * The String name.
     */
    private String theName;

    @Override
    public String toString() {
        /* If we have not yet loaded the name */
        if (theName == null) {
            /* Load the name */
            theName = MoneyWiseAnalysisValuesResource.getKeyForTransactionAttr(this).getValue();
        }

        /* return the name */
        return theName;
    }

    @Override
    public boolean isCounter() {
        return true;
    }

    @Override
    public MetisDataType getDataType() {
        return MetisDataType.MONEY;
    }
}
