package io.github.tonywasher.joceanus.moneywise.atlas.reports;

import java.io.InputStream;

import io.github.tonywasher.joceanus.tethys.api.control.TethysUIHTMLManager.TethysUIStyleSheetId;

/**
 * Report StyleSheets.
 */
public enum MoneyWiseXReportStyleSheet implements TethysUIStyleSheetId {
    /**
     * Report StyleSheet.
     */
    CSS_REPORT("MoneyWiseReports.css");

    /**
     * The Source.
     */
    private String theSource;

    /**
     * Constructor.
     *
     * @param pSource the source
     */
    MoneyWiseXReportStyleSheet(final String pSource) {
        theSource = pSource;
    }

    @Override
    public String getSourceName() {
        return theSource;
    }

    @Override
    public InputStream getInputStream() {
        return MoneyWiseXReportStyleSheet.class.getResourceAsStream(theSource);
    }
}
