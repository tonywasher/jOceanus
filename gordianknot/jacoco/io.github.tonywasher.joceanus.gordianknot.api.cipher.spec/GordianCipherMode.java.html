package io.github.tonywasher.joceanus.gordianknot.api.cipher.spec;

/**
 * Cipher Modes. Available algorithms.
 */
public enum GordianCipherMode {
    /**
     * ECB Mode.
     */
    ECB,

    /**
     * CBC Mode.
     */
    CBC,

    /**
     * SIC(CTR) Mode.
     */
    SIC,

    /**
     * CFB Mode.
     */
    CFB,

    /**
     * CFB8 Mode.
     */
    CFB8,

    /**
     * OFB Mode.
     */
    OFB,

    /**
     * OFB8 Mode.
     */
    OFB8,

    /**
     * EAX Mode.
     */
    EAX,

    /**
     * CCM Mode.
     */
    CCM,

    /**
     * GCM Mode.
     */
    GCM,

    /**
     * OCB Mode.
     */
    OCB,

    /**
     * GCFB Mode.
     */
    GCFB,

    /**
     * GOFB Mode.
     */
    GOFB,

    /**
     * KCTR Mode.
     */
    KCTR,

    /**
     * KCCM Mode.
     */
    KCCM,

    /**
     * KGCM Mode.
     */
    KGCM,

    /**
     * G3413CBC Mode.
     */
    G3413CBC,

    /**
     * G3413CFB Mode.
     */
    G3413CFB,

    /**
     * G3413OFB Mode.
     */
    G3413OFB,

    /**
     * G3413CTR Mode.
     */
    G3413CTR,

    /**
     * GCMSIV Mode.
     */
    GCMSIV;
}
