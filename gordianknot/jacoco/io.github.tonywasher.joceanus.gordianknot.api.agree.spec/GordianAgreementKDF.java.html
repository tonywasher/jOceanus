package io.github.tonywasher.joceanus.gordianknot.api.agree.spec;

/**
 * KDF types.
 */
public enum GordianAgreementKDF {
    /**
     * None.
     */
    NONE,

    /**
     * SHA256 KDF.
     */
    SHA256KDF,

    /**
     * SHA512 KDF.
     */
    SHA512KDF,

    /**
     * SHA256 CKDF.
     */
    SHA256CKDF,

    /**
     * SHA512 CKDF.
     */
    SHA512CKDF,

    /**
     * SHA256 HKDF.
     */
    SHA256HKDF,

    /**
     * SHA512 HKDF.
     */
    SHA512HKDF,

    /**
     * KMAC128.
     */
    KMAC128,

    /**
     * KMAC256.
     */
    KMAC256,

    /**
     * SHAKE256.
     */
    SHAKE256;
}
