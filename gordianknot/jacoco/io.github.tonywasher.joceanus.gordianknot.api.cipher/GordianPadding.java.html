package io.github.tonywasher.joceanus.gordianknot.api.cipher;

/**
 * Padding Modes. Available algorithms.
 */
public enum GordianPadding {
    /**
     * CTS Padding.
     */
    CTS,

    /**
     * ISO7816-4 Padding.
     */
    ISO7816D4,

    /**
     * X9.23 Padding.
     */
    X923,

    /**
     * PKCS7 Padding.
     */
    PKCS7,

    /**
     * TBC Padding.
     */
    TBC,

    /**
     * No Padding.
     */
    NONE;
}
