package io.github.tonywasher.joceanus.gordianknot.api.keyset.spec;

import io.github.tonywasher.joceanus.gordianknot.api.base.GordianLength;

/**
 * KeySet Specification.
 */
public interface GordianKeySetSpec {
    /**
     * Minimum Cipher Steps.
     */
    Integer MINIMUM_CIPHER_STEPS = 3;

    /**
     * Maximum Cipher Steps.
     */
    Integer MAXIMUM_CIPHER_STEPS = 6;

    /**
     * Default Cipher Steps.
     */
    Integer DEFAULT_CIPHER_STEPS = 4;

    /**
     * Obtain the keyLength.
     *
     * @return the keyLength
     */
    GordianLength getKeyLength();

    /**
     * Obtain the number of Cipher Steps.
     *
     * @return the # of cipher steps
     */
    int getCipherSteps();

    /**
     * Is the Spec valid?
     *
     * @return true/false
     */
    boolean isValid();
}
