package io.github.tonywasher.joceanus.themis.stats;

/**
 * Basis Stats element interface.
 */
public interface ThemisStatsElement {
    /**
     * Obtain the name.
     *
     * @return the name
     */
    String getName();

    /**
     * Obtain the statistics.
     *
     * @return the statistics
     */
    ThemisStats getStats();

    /**
     * Add a method to the stats.
     *
     * @param pMethod the method element
     */
    default void addMethod(final ThemisStatsElement pMethod) {
        throw new IllegalStateException();
    }

    /**
     * Add a class to the stats.
     *
     * @param pClazz the class element
     */
    default void addClass(final ThemisStatsElement pClazz) {
        throw new IllegalStateException();
    }
}
