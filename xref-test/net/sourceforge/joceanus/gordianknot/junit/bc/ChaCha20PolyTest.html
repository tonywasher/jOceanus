129     /**
130      * Test the Cipher against the results.
131      * @param pCipher the cipher to test.
132      * @param pTestCase the testCase
133      */
134     void testAADCipher(final GordianChaChaPoly1305 pCipher,
135                        final TestCase pTestCase) {
136         try {
137             /* Access the expected bytes */
138             final byte[] myExpected = Hex.decode(pTestCase.theExpected);
139 
140             /* Access plainText */
141             final byte[] myData = Hex.decode(pTestCase.thePlainText);
142 
143             /* Access AAD */
144             final byte[] myAAD = Hex.decode(pTestCase.theAAD);
145 
146             /* Access the key and the iv */
147             final KeyParameter myKey = new KeyParameter(Hex.decode(pTestCase.theKey));
148             final byte[] myIV = Hex.decode(pTestCase.theIV);
149             final ParametersWithIV myIVParms = new ParametersWithIV(myKey, myIV);
150             final AEADParameters myAEADParms = new AEADParameters(myKey, 0, myIV, myAAD);
151 
152             /* Initialise the cipher and encrypt the data */
153             pCipher.init(true, myAEADParms);
154             final byte[] myOutput = new byte[pCipher.getOutputSize(myData.length)];
155             int iProcessed = pCipher.processBytes(myData, 0, myData.length, myOutput, 0);
156             pCipher.doFinal(myOutput, iProcessed);
157 
158             /* Check the encryption */
159             isTrue("Encryption mismatch", Arrays.areEqual(myExpected, myOutput));
160 
161             /* Check that auto-reset worked */
162             iProcessed = pCipher.processBytes(myData, 0, myData.length, myOutput, 0);
163             pCipher.doFinal(myOutput, iProcessed);
164 
165             /* Check the encryption */
166             isTrue("Encryption mismatch after reset", Arrays.areEqual(myExpected, myOutput));
167 
168             /* Initialise the cipher and decrypt the data */
169             pCipher.init(false, myAEADParms);
170             final byte[] myResult = new byte[pCipher.getOutputSize(myExpected.length)];
171             iProcessed = pCipher.processBytes(myExpected, 0, myExpected.length, myResult, 0);
172             pCipher.doFinal(myResult, iProcessed);
173 
174             /* Check the decryption */
175             isTrue("Decryption mismatch", Arrays.areEqual(myData, myResult));
176 
177             /* Loop to process differing block sizes */
178             for (int blockSize = 1; blockSize <= 64; blockSize <<= 1) {
179                 /* Process the decryption one block at a time */
180                 iProcessed = 0;
181                 int iRemaining = myExpected.length;
182                 final byte[] myResult2 = new byte[pCipher.getOutputSize(myExpected.length)];
183                 for (int i = 0; iRemaining > 0; i += blockSize, iRemaining -= blockSize) {
184                     int myLen = Math.min(blockSize, iRemaining);
185                     iProcessed += pCipher.processBytes(myExpected, i, myLen, myResult2, iProcessed);
186                 }
187                 pCipher.doFinal(myResult2, iProcessed);
188 
189                 /* Check the decryption */
190                 isTrue("Block Decryption mismatch for size " + blockSize, Arrays.areEqual(myData, myResult));
191             }
192 
193             /* Initialise the cipher and encrypt the data pass AAD explicitly */
194             pCipher.init(true, myIVParms);
195             pCipher.processAADBytes(myAAD, 0, myAAD.length);
196             final byte[] myOutput2 = new byte[pCipher.getOutputSize(myData.length)];
197             iProcessed = pCipher.processBytes(myData, 0, myData.length, myOutput2, 0);
198             pCipher.doFinal(myOutput2, iProcessed);
199 
200             /* Check the encryption */
201             isTrue("Encryption mismatch", Arrays.areEqual(myExpected, myOutput2));
202 
203             /* Catch exceptions */
204         } catch (InvalidCipherTextException e) {
205             fail("Failed to resolve mac", e);
206         }
207     }
