135      * @param pExpected the expected results
136      */
137     static void checkDigestStrings(final Digest pDigest,
138                                    final String[] pExpected) {
139         /* Check the array */
140         Assertions.assertEquals(INPUTS.length, pExpected.length, "Expected results must have same dimensions as Inputs");
141 
142         /* Create the output buffer */
143         final byte[] myOutput = new byte[pDigest.getDigestSize()];
144 
145         /* Loop through the input strings */
146         for(int i=0; i < INPUTS.length; i++) {
147             /* Create the hash */
148             final String myInput = INPUTS[i];
149             pDigest.update(myInput.getBytes(), 0, myInput.length());
150             pDigest.doFinal(myOutput, 0);
151 
152             /* Check the hash */
153             final byte[] myExpected = Hex.decode(pExpected[i]);
154             Assertions.assertArrayEquals(myExpected, myOutput, "Result mismatch");
155             pDigest.update(myInput.getBytes(), 0, myInput.length());
156             pDigest.doFinal(myOutput, 0);
157 
158             /* Check the hash */
159             Assertions.assertArrayEquals(myExpected, myOutput, "Result mismatch");
160         }
161     }

351     /**
352      * Run the kangaroo tests.
353      * @param pMsgLen the messageLength
354      * @param pStdMsg is this a  standard message
355      * @param pPersLen the personalLength
356      * @param pOutLen the outputLength
357      * @param pResult the expected result
358      */
359     static void testKangaroo(final int pMsgLen,
360                              final boolean pStdMsg,
361                              final int pPersLen,
362                              final int pOutLen,
363                              final String pResult) {
364         /* Access the expected result */
365         final byte[] myExpected = Hex.decode(pResult);
366         final int myXofLen = pOutLen == 0 ? myExpected.length : pOutLen;
367 
368         /* Create the message */
369         final byte[] myMsg = new byte[pMsgLen];
370         if (pStdMsg) {
371             buildStdBuffer(myMsg);
372         } else {
373             Arrays.fill(myMsg, (byte) 0xFF);
374         }
375 
376         /* Create the personalisation */
377         final byte[] myPers = pPersLen > 0 ? new byte[pPersLen] : null;
378         if (pPersLen > 0) {
379             buildStdBuffer(myPers);
380         }
381 
382         /* Create the output buffer */
383         byte[] myOutput = new byte[myXofLen];
384 
385         /* Initialise the mac */
386         final GordianKangarooTwelve myDigest = new GordianKangarooTwelve();
387         final GordianKeccakParameters myParams = new GordianKeccakParametersBuilder()
388                 .setPersonalisation(myPers)
389                 .build();
390         myDigest.init(myParams);
391         myDigest.update(myMsg, 0, pMsgLen);
392         myDigest.doFinal(myOutput, 0, myXofLen);
393 
394         /* If we are only looking at the last bit of the output */
395         if (pOutLen != 0) {
396             myOutput = Arrays.copyOfRange(myOutput, pOutLen - myExpected.length, pOutLen);
397         }
398 
399         /* Check the result */
400         Assertions.assertArrayEquals(myExpected, myOutput, "Result mismatch");
401     }

414     /**
415      * Run the Blake2Tree tests.
416      * @param pNumLeaves the number of leaves
417      * @param pFanOut the fanOut
418      * @param pMaxDepth the max depth of the tree
419      */
420     static void testBlake2Tree(final int pNumLeaves,
421                                final int pFanOut,
422                                final int pMaxDepth) {
423         /* Create the tree */
424         final GordianBlake2Tree myTree = new GordianBlake2Tree(new GordianBlake2bDigest(512));
425         final int myLeafLen = 4096;
426 
427         /* Build the parameters */
428         final GordianBlake2ParametersBuilder myBuilder = new GordianBlake2ParametersBuilder();
429         myBuilder.setKey(Arrays.copyOf(BLAKE2DATA, 32));
430         myBuilder.setTreeConfig(pFanOut, pMaxDepth, myLeafLen);
431         myTree.init(myBuilder.build());
432 
433         /* Build the leaf data */
434         final byte[] myLeaf = new byte[myLeafLen];
435 
436         /* Loop through the leaves */
437         for (int i = 0; i < pNumLeaves; i++) {
438             Arrays.fill(myLeaf, (byte) i);
439             myTree.update(myLeaf, 0, myLeafLen);
440         }
441 
442         /* Build the result */
443         final byte[] myResult = new byte[myTree.getDigestSize()];
444         myTree.doFinal(myResult, 0);
445 
446         /* Replace each leaf with 1-filled buffer */
447         Arrays.fill(myLeaf, (byte) -1);
448         for (int i = 0; i < pNumLeaves; i++ ) {
449             myTree.updateLeaf(i, myLeaf, 0);
450         }
451 
452         /* Obtain the updated result */
453         final byte[] myLeafResult = new byte[myTree.getDigestSize()];
454         myTree.obtainResult(myLeafResult, 0);
455 
456         /* Recalculate the entire tree */
457         final GordianBlake2Tree myAltTree = new GordianBlake2Tree(new GordianBlake2bDigest(512));
458         myAltTree.init(myBuilder.build());
459         for (int i = 0; i < pNumLeaves; i++) {
460             myAltTree.update(myLeaf, 0, myLeafLen);
461         }
462 
463         /* Build the result */
464         myAltTree.doFinal(myResult, 0);
465         Assertions.assertArrayEquals(myResult, myLeafResult, "Result mismatch");
466     }
