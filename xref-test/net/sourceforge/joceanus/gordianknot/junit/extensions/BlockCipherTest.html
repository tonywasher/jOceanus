113     /**
114      * Test the Cipher against the results.
115      * @param pCipher the cipher to test.
116      * @param pKey the key to test
117      * @param pData the data to test
118      * @param pExpected the expected results
119      */
120     static void testCipher(final BlockCipher pCipher,
121                            final String pKey,
122                            final String pData,
123                            final String pExpected) {
124         /* Create the output buffer */
125         final byte[] myOutput = new byte[pCipher.getBlockSize()];
126         final byte[] myFinal = new byte[pCipher.getBlockSize()];
127 
128         /* Access the key and the data */
129         final KeyParameter myKey = new KeyParameter(Hex.decode(pKey));
130         final byte[] myData = Hex.decode(pData);
131 
132         /* Initialise the cipher */
133         pCipher.init(true, myKey);
134         pCipher.processBlock(myData, 0, myOutput, 0);
135 
136         /* Check the encryption */
137         final byte[] myExpected = Hex.decode(pExpected);
138         Assertions.assertArrayEquals(myExpected, myOutput, "Encryption mismatch");
139 
140         /* Initialise the cipher */
141         pCipher.init(false, myKey);
142         pCipher.processBlock(myOutput, 0, myFinal, 0);
143         Assertions.assertArrayEquals(myData, myFinal, "Decryption mismatch");
144     }

782     /**
783      * Test the GCMSIVCipher against the results.
784      * @param pCipher the cipher to test.
785      * @param pKey the key to test
786      * @param pNonce the nonce
787      * @param pAEAD the AEAD
788      * @param pData the data to test
789      * @param pExpected the expected results
790      */
791     static void testSIVCipher(final GordianGCMSIVBlockCipher pCipher,
792                               final String pKey,
793                               final String pNonce,
794                               final String pAEAD,
795                               final String pData,
796                               final String pExpected) {
797         /* protect against exceptions */
798         try {
799             /* Access the key and the data */
800             final KeyParameter myKey = new KeyParameter(Hex.decode(pKey));
801             final byte[] myNonce = Hex.decode(pNonce);
802             final byte[] myAEAD = Hex.decode(pAEAD);
803             final byte[] myData = Hex.decode(pData);
804 
805             /* Initialise the cipher with AEAD */
806             AEADParameters myParamsA = new AEADParameters(myKey, 128, myNonce, myAEAD);
807             AEADParameters myParamsB = new AEADParameters(myKey, 128, myNonce);
808             pCipher.init(true, myParamsA);
809 
810             /* Create the output buffers */
811             final byte[] myOutput = new byte[pCipher.getOutputSize(myData.length)];
812             final byte[] myFinal = new byte[myData.length];
813 
814             /* Process the data */
815             pCipher.processBytes(myData, 0, myData.length, null, 0);
816             pCipher.doFinal(myOutput, 0);
817 
818             /* Check the encryption */
819             final byte[] myExpected = Hex.decode(pExpected);
820             Assertions.assertArrayEquals(myExpected, myOutput, "Encryption mismatch");
821 
822             /* Repeat processing one byte at a time */
823             for (byte b : myData) {
824                 pCipher.processByte(b, null, 0);
825             }
826             pCipher.doFinal(myOutput, 0);
827             Assertions.assertArrayEquals(myExpected, myOutput, "Encryption mismatch");
828 
829             /* Re-initialise the cipher with no AEAD */
830             pCipher.init(false, myParamsB);
831 
832             /* Process AEAD one byte at a time */
833             for (byte b : myAEAD) {
834                 pCipher.processAADByte(b);
835             }
836             pCipher.processBytes(myOutput, 0, myOutput.length, null, 0);
837             pCipher.doFinal(myFinal, 0);
838             Assertions.assertArrayEquals(myData, myFinal, "Decryption mismatch");
839 
840             /* Catch exceptions */
841         } catch (InvalidCipherTextException e) {
842             e.printStackTrace();
843         }
