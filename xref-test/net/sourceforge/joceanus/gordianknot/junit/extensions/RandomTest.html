204     /**
205      * Entropy Source Provider.
206      */
207     public static class GordianEntropySource
208             implements EntropySource {
209         /**
210          * The entropy data.
211          */
212         private final byte[] data;
213 
214         /**
215          * The bitsRequired each time.
216          */
217         private final int bitsRequired;
218 
219         /**
220          * Is the source prediction resistant?
221          */
222         private final boolean predictionResistant;
223 
224         /**
225          * The current index.
226          */
227         private int theIndex;
228 
229         /**
230          * Constructor.
231          * @param pData the entropy data
232          * @param pRequired the number of bits required each time
233          * @param pResistant is the entropy prediction resistant?
234          */
235         GordianEntropySource(final byte[] pData,
236                              final int pRequired,
237                              final boolean pResistant) {
238             data = pData;
239             bitsRequired = pRequired;
240             predictionResistant = pResistant;
241         }
242 
243         @Override
244         public boolean isPredictionResistant() {
245             return predictionResistant;
246         }
247 
248         @Override
249         public byte[] getEntropy() {
250             /* Create the required entropy */
251             byte[] rv = new byte[bitsRequired / 8];
252             System.arraycopy(data, theIndex, rv, 0, rv.length);
253 
254             /* Adjust index and return */
255             theIndex += bitsRequired / 8;
256             return rv;
257         }
258 
259         @Override
260         public int entropySize() {
261            return bitsRequired;
262         }
263     }
