378     /**
379      * check multi-output.
380      * @param pDigest the digest
381      * @param pBuffer the buffer
382      */
383     private static void checkMultiOutput(final GordianBlake3Digest pDigest,
384                                          final byte[] pBuffer) {
385         /* Take a copy of the state */
386         final GordianBlake3Digest mySave = pDigest.copy();
387 
388         /* Create alternative buffer */
389         final int myLen = pBuffer.length;
390         final byte[] myAlt = new byte[myLen];
391 
392         /* Perform standard output */
393         pDigest.doOutput(pBuffer, 0, myLen);
394         pDigest.reset(mySave);
395 
396         /* Loop creating output 5 bytes at a time */
397         for (int i = 0; i < myLen; i += PARTOUT) {
398             /* Output bytes */
399             int myRead = Math.min(PARTOUT, myLen - i);
400             pDigest.doOutput(myAlt, i, myRead);
401 
402             /* Around halfway through */
403             if (i == PARTBREAK) {
404                 /* Check save and reset in the middle of output */
405                 final GordianBlake3Digest myTemp  = pDigest.copy();
406                 pDigest.reset();
407                 pDigest.reset(myTemp);
408             }
409         }
410         pDigest.reset(mySave);
411         Assertions.assertArrayEquals(pBuffer, myAlt, "derivedDigest mismatch");
412     }
