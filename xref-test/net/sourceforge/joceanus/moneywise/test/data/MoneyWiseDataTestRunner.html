52  /**
53   * Transaction Test provider.
54   */
55  public class MoneyWiseDataTestRunner {
56      /**
57       * The view.
58       */
59      private final MoneyWiseView theView;
60  
61      /**
62       * The dataSet.
63       */
64      private final MoneyWiseDataSet theDataSet;
65  
66      /**
67       * Accounts Builder.
68       */
69      private final MoneyWiseDataTestAccounts theAccountBuilder;
70  
71      /**
72       * Analysis Builder.
73       */
74      private final MoneyWiseXAnalysisBuilder theAnalysisBuilder;
75  
76      /**
77       * The testCases.
78       */
79      private final List<MoneyWiseDataTestCase> theTestCases;
80  
81      /**
82       * Constructor.
83       *
84       * @param pView the view
85       * @throws OceanusException on error
86       */
87      MoneyWiseDataTestRunner(final MoneyWiseView pView) throws OceanusException {
88          /* Create the analyser */
89          theView = pView;
90          theAnalysisBuilder = new MoneyWiseXAnalysisBuilder(theView);
91  
92          /* Create the dataSet */
93          theDataSet = theView.getNewData();
94          theDataSet.doNewValidityChecks();
95  
96          /* Create the account builder */
97          theAccountBuilder = new MoneyWiseDataTestAccounts(theDataSet);
98  
99          theTestCases = createTestCases();
100     }
101 
102     /**
103      * Create the test suite.
104      *
105      * @param pView the view
106      * @return the test stream
107      * @throws OceanusException on error
108      */
109     public static Stream<DynamicNode> createTests(final MoneyWiseView pView) throws OceanusException {
110         /* Create the testRunner */
111         final MoneyWiseDataTestRunner myRunner = new MoneyWiseDataTestRunner(pView);
112 
113         /* Create the transaction testCases */
114         Stream<DynamicNode> myTranTests = Stream.empty();
115         for (MoneyWiseDataTestCase myTestCase : myRunner.getTestCases()) {
116             myTranTests = Stream.concat(myTranTests, Stream.of(DynamicTest.dynamicTest(myTestCase.getName(),
117                     () -> myRunner.runTestCase(myTestCase))));
118         }
119 
120         /* Create container for tests */
121         myTranTests = Stream.of(DynamicContainer.dynamicContainer("transactions", myTranTests));
122         return Stream.of(DynamicContainer.dynamicContainer("localData", Stream.concat(
123                 myTranTests, myRunner.createStorageTests())));
124     }
125 
126     /**
127      * Obtain the testCase list.
128      *
129      * @return the testCase list
130      */
131     private List<MoneyWiseDataTestCase> getTestCases() {
132         return theTestCases;
133     }
134 
135     /**
136      * Create the testCases.
137      *
138      * @return the testCases
139      */
140     private List<MoneyWiseDataTestCase> createTestCases() {
141         final List<MoneyWiseDataTestCase> myList = new ArrayList<>();
142         myList.add(new MoneyWiseDataTestTransfers(theAccountBuilder));
143         myList.add(new MoneyWiseDataTestExpense(theAccountBuilder));
144         myList.add(new MoneyWiseDataTestCash(theAccountBuilder));
145         myList.add(new MoneyWiseDataTestPayeeIncome(theAccountBuilder));
146         myList.add(new MoneyWiseDataTestDepositIncome(theAccountBuilder));
147         myList.add(new MoneyWiseDataTestCreditCard(theAccountBuilder));
148         myList.add(new MoneyWiseDataTestMortgage(theAccountBuilder));
149         myList.add(new MoneyWiseDataTestPrivateLoan(theAccountBuilder));
150         myList.add(new MoneyWiseDataTestShareBuySell(theAccountBuilder));
151         myList.add(new MoneyWiseDataTestShareDividend(theAccountBuilder));
152         return myList;
153     }
154 
155     /**
156      * Run the testCase.
157      *
158      * @param pTest the testCase
159      * @throws OceanusException on error
160      */
161     public void runTestCase(final MoneyWiseDataTestCase pTest) throws OceanusException {
162         /* Run the test */
163         final OceanusProfile myTask = theView.getNewProfile("Transaction TestCases");
164         prepareDataForTest(pTest);
165         pTest.checkErrors();
166         final MoneyWiseXAnalysis myAnalysis = theAnalysisBuilder.analyseNewData(theDataSet);
167         pTest.setAnalysis(myAnalysis);
168         pTest.checkAnalysis();
169 
170         reportTestCase(pTest);
171         myTask.end();
172     }
173 
174     /**
175      * Run the testCase.
176      *
177      * @param pTest the testCase
178      * @throws OceanusException on error
179      */
180     public void reportTestCase(final MoneyWiseDataTestCase pTest) throws OceanusException {
181         /* Run the test */
182         new MoneyWiseDataXDocBuilder(pTest);
183     }
184 
185     /**
186      * Prepare dataSet.
187      *
188      * @param pTest the testCase
189      * @throws OceanusException on error
190      */
191     private void prepareDataForTest(final MoneyWiseDataTestCase pTest) throws OceanusException {
192         /* Reset all data */
193         theAccountBuilder.resetData();
194 
195         /* Create data */
196         pTest.setUpAccounts();
197         pTest.defineRates();
198         pTest.definePrices();
199         pTest.defineTransactions();
200 
201         /* Resolve the transactions */
202         theDataSet.getTransactions().resolveDataSetLinks();
203     }
204 
205     /**
206      * Run the storage tests.
207      * @return the test stream
208      *
209      * @throws OceanusException on error
210      */
211     public Stream<DynamicNode> createStorageTests() throws OceanusException {
212         createOutputDirectory();
213         Stream<DynamicNode> myStream = Stream.of(DynamicTest.dynamicTest("initData", this::prepareFullData));
214         myStream = Stream.concat(myStream, MoneyWiseDataTest.storageTests(theDataSet, theView));
215         myStream = Stream.concat(myStream, Stream.of(DynamicTest.dynamicTest("editSet",
216                 () -> MoneyWiseDataTest.checkEditSet(theDataSet, theView))));
217         return Stream.of(DynamicContainer.dynamicContainer("allTrans", myStream));
218     }
219 
220     /**
221      * Prepare full data for test.
222      * @throws OceanusException on error
223      */
224     private void prepareFullData() throws OceanusException {
225         /* Reset all data */
226         theAccountBuilder.resetData();
227 
228         /* Loop through the tests */
229         for (MoneyWiseDataTestCase myTest : theTestCases) {
230             /* Create data */
231             myTest.setUpAccounts();
232             myTest.defineRates();
233             myTest.definePrices();
234             myTest.defineTransactions();
235         }
236 
237         /* Resolve the transactions */
238         theDataSet.getTransactions().resolveDataSetLinks();
239 
240         /* Initialise the security */
241         new MoneyWiseDataTestSecurity(theDataSet).initSecurity(theView);
242     }
243 
244     /**
245      * Ensure output directory.
246      * @throws OceanusException on error
247      */
248     private static void createOutputDirectory() throws OceanusException {
249         try {
250             final Path myDir = Paths.get(MoneyWiseDataXDocBuilder.OUTPUT_DIR);
251             if (!myDir.toFile().exists()) {
252                 Files.createDirectory(myDir);
253             }
254         } catch (IOException e) {
255             throw new MoneyWiseIOException("Failed to create directory", e);
256         }
257     }
