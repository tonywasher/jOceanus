19  import net.sourceforge.joceanus.moneywise.data.statics.MoneyWiseCurrencyClass;
20  import org.junit.jupiter.api.Assertions;
21  import org.junit.jupiter.api.DynamicNode;
22  import org.junit.jupiter.api.DynamicTest;
23  import org.junit.jupiter.api.TestFactory;
24  
25  import java.util.ArrayList;
26  import java.util.Arrays;
27  import java.util.Currency;
28  import java.util.List;
29  import java.util.stream.Stream;
30  
31  /**
32   * CurrencyCode checker.
33   */
34  final class MoneyWiseCurrencyTest {
35      /**
36       * List of irrelevant currency codes.
37       */
38      private static final String[] IRRELEVANT =
39      { "XBB", "ZWR", "XBA", "YUM", "AZM", "XFO", "COU", "SDD", "MTL", "XBC", "XDR", "ROL", "USS", "AFA", "XXX",
40              "MZM", "PTE", "SIT", "ATS", "CYP", "ZWN", "SKK", "ADP", "FIM", "XTS", "FRF", "USN", "ZMK", "CHE",
41              "GWP", "VEB", "DEM", "XBD", "UYI", "BYB", "XPT", "EEK", "ZWD", "GHC", "BEF", "XFU", "XAU", "XSU",
42              "BGL", "MGF", "IEP", "TPE", "XUA", "CLF", "LUF", "XPD", "SRG", "RUR", "ESP", "TMM", "GRD", "CHW",
43              "TRL", "XAG", "BOV", "ITL", "CSD", "MXV", "NLG", "AYM", "BYR" };
44  
45      /**
46       * Create the analysis test suite.
47       * @return the test stream
48       */
49      @TestFactory
50      Stream<DynamicNode> checkCurrencies() {
51          return Stream.of(
52                  DynamicTest.dynamicTest("checkCurrency", MoneyWiseCurrencyTest::checkCurrency)
53          );
54      }
55  
56      /**
57       * Main test.
58       */
59      private static void checkCurrency() {
60          /* Create list of irrelevant codes */
61          final List<String> myIrrelevantCurrencies = new ArrayList<>(Arrays.asList(IRRELEVANT));
62  
63          /* Create collections */
64          final List<MoneyWiseCurrencyClass> mySupportedCurrencies = new ArrayList<>();
65          final List<MoneyWiseCurrencyClass> myInvalidCurrencies = new ArrayList<>();
66          final List<Currency> myMissingCurrencies = new ArrayList<>();
67  
68          /* Check Available Currencies */
69          for (Currency myCurr : Currency.getAvailableCurrencies()) {
70              /* Determine whether the currency is supported */
71              final String myCode = myCurr.getCurrencyCode();
72              try {
73                  final MoneyWiseCurrencyClass myClass = MoneyWiseCurrencyClass.valueOf(myCode);
74                  mySupportedCurrencies.add(myClass);
75              } catch (IllegalArgumentException e) {
76                  if (!myIrrelevantCurrencies.contains(myCode)) {
77                      myMissingCurrencies.add(myCurr);
78                  }
79              }
80          }
81  
82          /* Loop through all the declared currencies */
83          for (final MoneyWiseCurrencyClass myClass : MoneyWiseCurrencyClass.values()) {
84              /* Check for invalid currencies */
85              if (!mySupportedCurrencies.contains(myClass)) {
86                  myInvalidCurrencies.add(myClass);
87              }
88          }
89  
90          /* Report failures */
91          Assertions.assertTrue(myInvalidCurrencies.isEmpty(), "Invalid currencies:\n" + myInvalidCurrencies);
92          Assertions.assertTrue(myMissingCurrencies.isEmpty(), "Missing currencies:\n" + myMissingCurrencies);
93      }
94  }
