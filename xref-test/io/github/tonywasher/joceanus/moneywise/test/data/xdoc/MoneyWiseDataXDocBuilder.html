18  
19  import io.github.tonywasher.joceanus.oceanus.base.OceanusException;
20  import io.github.tonywasher.joceanus.moneywise.exc.MoneyWiseIOException;
21  import io.github.tonywasher.joceanus.moneywise.test.data.trans.MoneyWiseDataTestCase;

23  import java.io.File;
24  import java.io.PrintWriter;
25  
26  /**
27   * XDoc Report Builder.
28   */
29  public class MoneyWiseDataXDocBuilder {
30      /**
31       * The output directory.
32       */
33      public static final String OUTPUT_DIR = "target/html";
34  
35      /**
36       * Analysis Group.
37       */
38      static final String GRP_ANALYSIS = "analysis";
39  
40      /**
41       * Holdings Group.
42       */
43      static final String GRP_HOLDINGS = "holdings";
44  
45      /**
46       * Accounts Group.
47       */
48      static final String GRP_ACCOUNTS = "accounts";
49  
50      /**
51       * Data Group.
52       */
53      static final String GRP_DATA = "data";
54  
55      /**
56       * Date Table Header.
57       */
58      static final String HDR_DATE = "Date";
59  
60      /**
61       * Name Table Header.
62       */
63      static final String HDR_NAME = "Name";
64  
65      /**
66       * Category Table Header.
67       */
68      static final String HDR_CATEGORY = "Category";
69  
70      /**
71       * Parent Table Header.
72       */
73      static final String HDR_PARENT = "Parent";
74  
75      /**
76       * Currency Table Header.
77       */
78      static final String HDR_CURRENCY = "Currency";
79  
80      /**
81       * Opening Balance Table Header.
82       */
83      static final String HDR_OPENING = "Starting Balance";
84  
85      /**
86       * Profit Table Header.
87       */
88      static final String HDR_PROFIT = "Profit";
89  
90      /**
91       * Constructor.
92       *
93       * @param pTest the test case
94       * @throws OceanusException on error
95       */
96      public MoneyWiseDataXDocBuilder(final MoneyWiseDataTestCase pTest) throws OceanusException {
97          /* Initialise the document */
98          final MoneyWiseDataXDocReport myReport = new MoneyWiseDataXDocReport();
99          myReport.startReport(pTest);
100 
101         /* Create definitions */
102         myReport.newSection("Definitions");
103         myReport.addParagraph("We start with the following accounts and transaction categories");
104         final MoneyWiseDataXDocAssets myAssets = new MoneyWiseDataXDocAssets(myReport, pTest);
105         myAssets.createAssetDefinitions();
106         final MoneyWiseDataXDocPayee myPayee = new MoneyWiseDataXDocPayee(myReport, pTest);
107         myPayee.declareParentPayees(myAssets.getParentPayees());
108         myPayee.createPayeeDefinitions();
109         final MoneyWiseDataXDocTransCat myTransCat = new MoneyWiseDataXDocTransCat(myReport, pTest);
110         myTransCat.createTransDefinitions();
111 
112         /* Define transaction data */
113         myReport.newSection("Transactions");
114         myReport.addParagraph("We have the following events");
115         final MoneyWiseDataXDocTrans myTrans = new MoneyWiseDataXDocTrans(myReport, pTest);
116         myTrans.createTransactions();
117         final MoneyWiseDataXDocRate myRates = new MoneyWiseDataXDocRate(myReport, pTest);
118         myRates.createExchangeRates();
119         final MoneyWiseDataXDocPrice myPrices = new MoneyWiseDataXDocPrice(myReport, pTest);
120         myPrices.createSecurityPrices();
121 
122         /* Define analysis */
123         myReport.newSection("Analysis");
124         myReport.addParagraph("The analysis of these transactions is as follows");
125         myAssets.createAssetAnalysis();
126         myPayee.createPayeeAnalysis();
127         myTransCat.createTransAnalysis();
128         final MoneyWiseDataXDocTax myTax = new MoneyWiseDataXDocTax(myReport, pTest);
129         myTax.createTaxAnalysis();
130 
131         /* Output the file */
132         final File myFile = new File(OUTPUT_DIR, pTest.getName() + ".xml");
133         try (PrintWriter out = new PrintWriter(myFile)) {
134             out.println(myReport.formatXML());
135         } catch (Exception e) {
136             throw new MoneyWiseIOException("Failed to write file", e);
137         }
138     }
139 }
