101     /**
102      * Analyse the data.
103      *
104      * @param pData the dataSet to analyse
105      * @param pView the view
106      * @throws OceanusException on error
107      */
108     public void analyseData(final MoneyWiseDataSet pData,
109                             final MoneyWiseView pView) throws OceanusException {
110         /* Update the maps */
111         pData.updateMaps();
112 
113         /* Create the analysis */
114         final OceanusProfile myTask = pView.getNewProfile("Dummy");
115         pView.setData(pData);
116         final PrometheusEditSet myEditSet = new PrometheusEditSet(pView);
117         final MoneyWiseAnalysisTransAnalyser myAnalyser = new MoneyWiseAnalysisTransAnalyser(myTask, myEditSet, pView.getPreferenceManager());
118 
119         /* Post process the analysis */
120         myAnalyser.postProcessAnalysis();
121 
122         /* Access the analysis */
123         final MoneyWiseAnalysis myAnalysis = myAnalyser.getAnalysis();
124         final MoneyWiseAnalysisManager myAnalysisMgr = new MoneyWiseAnalysisManager(myAnalysis);
125         myAnalysisMgr.analyseBase();
126 
127         /* Obtain deposit totals */
128         final MoneyWiseAnalysisDepositCategoryBucket myDepCat = myAnalysis.getDepositCategories().getTotals();
129         OceanusMoney myDepTotal = myDepCat.getValues().getMoneyValue(MoneyWiseAnalysisAccountAttr.VALUEDELTA);
130         myDepTotal = myDepTotal == null ? new OceanusMoney() : new OceanusMoney(myDepTotal);
131 
132         /* Add in cash totals */
133         final MoneyWiseAnalysisCashCategoryBucket myCashCat = myAnalysis.getCashCategories().getTotals();
134         final OceanusMoney myCashTotal = myCashCat.getValues().getMoneyValue(MoneyWiseAnalysisAccountAttr.VALUEDELTA);
135         if (myCashTotal != null) {
136             myDepTotal.addAmount(myCashTotal);
137         }
138 
139         /* Add in loan totals */
140         final MoneyWiseAnalysisLoanCategoryBucket myLoanCat = myAnalysis.getLoanCategories().getTotals();
141         final OceanusMoney myLoanTotal = myLoanCat.getValues().getMoneyValue(MoneyWiseAnalysisAccountAttr.VALUEDELTA);
142         if (myLoanTotal != null) {
143             myDepTotal.addAmount(myLoanTotal);
144         }
145 
146         /* Add in portfolio totals */
147         final MoneyWiseAnalysisPortfolioBucket myPort = myAnalysis.getPortfolios().getTotals();
148         final OceanusMoney myPortTotal = myPort.getValues().getMoneyValue(MoneyWiseAnalysisSecurityAttr.VALUEDELTA);
149         if (myPortTotal != null) {
150             myDepTotal.addAmount(myPortTotal);
151         }
152 
153         /* Validate Payee totals */
154         final MoneyWiseAnalysisPayeeBucket myPayeeTotals = myAnalysis.getPayees().getTotals();
155         OceanusMoney myPayTotal = myPayeeTotals.getValues().getMoneyValue(MoneyWiseAnalysisPayeeAttr.PROFIT);
156         if (myPayTotal == null) {
157             myPayTotal = new OceanusMoney();
158         }
159         Assertions.assertEquals(myDepTotal, myPayTotal, "Payee total mismatch");
160 
161         /* Validate transaction totals */
162         final MoneyWiseAnalysisTransCategoryBucket myTransTotals = myAnalysis.getTransCategories().getTotals();
163         OceanusMoney myEvtTotal = myTransTotals.getValues().getMoneyValue(MoneyWiseAnalysisTransAttr.PROFIT);
164         if (myEvtTotal == null) {
165             myEvtTotal = new OceanusMoney();
166         }
167         Assertions.assertEquals(myDepTotal, myEvtTotal, "Transaction total mismatch");
168 
169         /* Validate tax totals */
170         final MoneyWiseAnalysisTaxBasisBucket myTaxTotals = myAnalysis.getTaxBasis().getTotals();
171         final OceanusMoney myTaxTotal = myTaxTotals.getValues().getMoneyValue(MoneyWiseAnalysisTaxBasisAttr.GROSS);
172         Assertions.assertEquals(myDepTotal, myTaxTotal, "TaxBasis total mismatch");
173     }

175     /**
176      * Create the archiveData tests.
177      *
178      * @param pView the view
179      * @return the testStream
180      */
181     private Stream<DynamicNode> archiveDataTests(final MoneyWiseView pView) {
182         /* Create the stream */
183         final MoneyWiseDataSet myData = pView.getNewData();
184         final TethysUIThreadManager myThreadMgr = new MoneyWiseNullThreadMgr();
185         Stream<DynamicNode> myStream = Stream.of(DynamicTest.dynamicTest("initData",
186                 () -> new MoneyWiseDataTestArchiveFile(myThreadMgr).performTest(myData, pView)));
187         myStream = Stream.concat(myStream, storageTests(myData, pView));
188         myStream = Stream.concat(myStream, Stream.of(DynamicTest.dynamicTest("analyseData", () -> analyseData(myData, pView))));
189         myStream = Stream.concat(myStream, Stream.of(DynamicTest.dynamicTest("editSet", () -> checkEditSet(myData, pView))));
190 
191         /* Return the stream */
192         return Stream.of(DynamicContainer.dynamicContainer("archiveData", myStream));
193     }

195     /**

198      * @param pData the dataSet to populate
199      * @param pView the view
200      * @return the testStream
201      */
202     static Stream<DynamicNode> storageTests(final MoneyWiseDataSet pData,
203                                             final MoneyWiseView pView) {
204         /* Return the stream */
205         final TethysUIThreadManager myThreadMgr = new MoneyWiseNullThreadMgr();
206         return Stream.of(DynamicContainer.dynamicContainer("Storage Tests", Stream.of(
207                 DynamicTest.dynamicTest("XML File", () -> new MoneyWiseDataTestXMLFile(myThreadMgr).performTest(pData, pView)),
208                 DynamicTest.dynamicTest("ODS File", () -> new MoneyWiseDataTestODSFile(myThreadMgr).performTest(pData, pView)),
209                 DynamicTest.dynamicTest("dataBase", () -> new MoneyWiseDataTestDatabase(myThreadMgr).performTest(pData, pView)))));
210     }
