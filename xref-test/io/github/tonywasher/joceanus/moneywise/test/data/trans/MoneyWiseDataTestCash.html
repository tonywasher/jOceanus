24  /**
25   * Test AutoCash.
26   */
27  public class MoneyWiseDataTestCash
28          extends MoneyWiseDataTestCase {
29      /**
30       * TransactionBuilder.
31       */
32      private final MoneyWiseTransactionBuilder theTransBuilder;
33  
34      /**
35       * Constructor.
36       *
37       * @param pBuilder the builder
38       */
39      public MoneyWiseDataTestCash(final MoneyWiseDataTestAccounts pBuilder) {
40          /* Store parameters */
41          super(pBuilder);
42          theTransBuilder = getTransBuilder();
43      }
44  
45      @Override
46      public String getName() {
47          return "Cash";
48      }
49  
50      @Override
51      public String getTitle() {
52          return "Cash and AutoCash Transactions";
53      }
54  
55      @Override
56      public String getDesc() {
57          return "Cash accounts can be used as normal accounts or as autoCash";
58      }
59  
60      @Override
61      public void setUpAccounts() throws OceanusException {
62          createPayees(MoneyWiseDataTestAccounts.IDPY_ASDA);
63          createDeposits(MoneyWiseDataTestAccounts.IDDP_BARCLAYS_CURRENT,
64                  MoneyWiseDataTestAccounts.IDDP_STARLING_EURO);
65          createCash(MoneyWiseDataTestAccounts.IDCS_CASH,
66                  MoneyWiseDataTestAccounts.IDCS_EUROS_CASH,
67                  MoneyWiseDataTestAccounts.IDCS_CASH_WALLET,
68                  MoneyWiseDataTestAccounts.IDCS_EUROS_WALLET);
69      }
70  
71      @Override
72      public void defineRates() throws OceanusException {
73          createXchgRate(MoneyWiseCurrencyClass.USD, "06-Apr-1980", "0.8");
74          createXchgRate(MoneyWiseCurrencyClass.EUR, "06-Apr-1980", "0.9");
75          createXchgRate(MoneyWiseCurrencyClass.USD, "01-Jan-2025", "0.85");
76          createXchgRate(MoneyWiseCurrencyClass.EUR, "01-Jan-2025", "0.95");
77      }
78  
79      /**
80       * Create simple transfers.
81       *
82       * @throws OceanusException on error
83       */
84      @Override
85      public void defineTransactions() throws OceanusException {
86          /* A simple payment from one account to autoCash */
87          theTransBuilder.date("01-Jun-1987").category(MoneyWiseDataTestCategories.IDTC_TRANSFER)
88                  .account(MoneyWiseDataTestAccounts.IDDP_BARCLAYS_CURRENT).amount("50")
89                  .to().partner(MoneyWiseDataTestAccounts.IDCS_CASH)
90                  .build();
91  
92          /* A simple refund to one account from autoCash */
93          theTransBuilder.date("02-Jun-1987").category(MoneyWiseDataTestCategories.IDTC_TRANSFER)
94                  .account(MoneyWiseDataTestAccounts.IDCS_CASH).amount("20")
95                  .to().partner(MoneyWiseDataTestAccounts.IDDP_BARCLAYS_CURRENT)
96                  .build();
97  
98          /* A simple payment from autoCash */
99          theTransBuilder.date("03-Jun-1987").category(MoneyWiseDataTestCategories.IDTC_SHOP_FOOD)
100                 .account(MoneyWiseDataTestAccounts.IDCS_CASH).amount("12")
101                 .to().partner(MoneyWiseDataTestAccounts.IDPY_ASDA)
102                 .build();
103 
104         /* A simple refund to autoCash */
105         theTransBuilder.date("04-Jun-1987").category(MoneyWiseDataTestCategories.IDTC_SHOP_FOOD)
106                 .account(MoneyWiseDataTestAccounts.IDCS_CASH).amount("6")
107                 .from().partner(MoneyWiseDataTestAccounts.IDPY_ASDA)
108                 .build();
109 
110         /* A simple transfer from one account to cash */
111         theTransBuilder.date("05-Jun-1987").category(MoneyWiseDataTestCategories.IDTC_TRANSFER)
112                 .account(MoneyWiseDataTestAccounts.IDDP_BARCLAYS_CURRENT).amount("22")
113                 .to().partner(MoneyWiseDataTestAccounts.IDCS_CASH_WALLET)
114                 .build();
115 
116         /* A simple refund to one account from cash */
117         theTransBuilder.date("06-Jun-1987").category(MoneyWiseDataTestCategories.IDTC_TRANSFER)
118                 .account(MoneyWiseDataTestAccounts.IDCS_CASH_WALLET).amount("7")
119                 .to().partner(MoneyWiseDataTestAccounts.IDDP_BARCLAYS_CURRENT)
120                 .build();
121 
122         /* A simple payment from one account to autoCash in foreign currency */
123         theTransBuilder.date("07-Jun-1987").category(MoneyWiseDataTestCategories.IDTC_TRANSFER)
124                 .account(MoneyWiseDataTestAccounts.IDDP_STARLING_EURO).amount("65")
125                 .to().partner(MoneyWiseDataTestAccounts.IDCS_EUROS_CASH)
126                 .build();
127 
128         /* A simple refund to one account from autoCash in foreign currency */
129         theTransBuilder.date("08-Jun-1987").category(MoneyWiseDataTestCategories.IDTC_TRANSFER)
130                 .account(MoneyWiseDataTestAccounts.IDCS_EUROS_CASH).amount("14")
131                 .to().partner(MoneyWiseDataTestAccounts.IDDP_STARLING_EURO)
132                 .build();
133 
134         /* A simple payment from autoCash in foreign currency */
135         theTransBuilder.date("09-Jun-1987").category(MoneyWiseDataTestCategories.IDTC_SHOP_FOOD)
136                 .account(MoneyWiseDataTestAccounts.IDCS_EUROS_CASH).amount("17")
137                 .to().partner(MoneyWiseDataTestAccounts.IDPY_ASDA)
138                 .build();
139 
140         /* A simple refund to autoCash in foreign currency */
141         theTransBuilder.date("10-Jun-1987").category(MoneyWiseDataTestCategories.IDTC_SHOP_FOOD)
142                 .account(MoneyWiseDataTestAccounts.IDCS_EUROS_CASH).amount("8")
143                 .from().partner(MoneyWiseDataTestAccounts.IDPY_ASDA)
144                 .build();
145 
146         /* A simple transfer from one account to cash in foreign currency */
147         theTransBuilder.date("11-Jun-1987").category(MoneyWiseDataTestCategories.IDTC_TRANSFER)
148                 .account(MoneyWiseDataTestAccounts.IDDP_STARLING_EURO).amount("13")
149                 .to().partner(MoneyWiseDataTestAccounts.IDCS_EUROS_WALLET)
150                 .build();
151 
152         /* A simple refund to one account from cash in foreign currency */
153         theTransBuilder.date("12-Jun-1987").category(MoneyWiseDataTestCategories.IDTC_TRANSFER)
154                 .account(MoneyWiseDataTestAccounts.IDCS_EUROS_WALLET).amount("8.5")
155                 .to().partner(MoneyWiseDataTestAccounts.IDDP_STARLING_EURO)
156                 .build();
157     }
