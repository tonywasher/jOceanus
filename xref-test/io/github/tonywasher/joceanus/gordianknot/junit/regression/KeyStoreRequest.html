182     /**
183      * test signed keyPairRequest.
184      *
185      * @param pKeyPairSpec the keyPairSpec
186      * @throws GordianException on error
187      */
188     private void signedKeyPairRequest(final GordianKeyPairSpec pKeyPairSpec) throws GordianException {
189         /* Access details */
190         final GordianCoreKeyStoreManager myMgr = theState.getManager();
191         final GordianCoreKeyStore myStore = theState.getStore();
192         final GordianKeyStorePair myIntermediate = theState.getIntermediate();
193 
194         /* Handle a disabled keyPairSpec */
195         if (!myMgr.getKeyStore().getFactory().getAsyncFactory().getKeyPairFactory().supportedKeyPairSpecs().test(pKeyPairSpec)) {
196             return;
197         }
198 
199         /* Create and configure gateway */
200         final GordianKeyStoreGateway myGateway = myStore.getFactory().getAsyncFactory().getKeyStoreFactory().createKeyStoreGateway(myMgr);
201         myGateway.setPasswordResolver(theState::passwordResolver);
202         myGateway.setCertifier(KeyStoreAlias.CERTIFIER.getName());
203         myGateway.setMACSecretResolver(n -> DEF_MACSECRET);
204 
205         /* Create a signature keyPair */
206         final X500Name mySignName = KeyStoreUtils.buildX500Name(KeyStoreAlias.SIGNER);
207         final GordianKeyPairUsage myUsage = new GordianKeyPairUsage(GordianKeyPairUse.SIGNATURE);
208         myMgr.createKeyPair(pKeyPairSpec, mySignName, myUsage, myIntermediate, KeyStoreAlias.SIGNER.getName(), KeyStoreUtils.DEF_PASSWORD);
209 
210         /* Build the CertificateRequest */
211         final ByteArrayOutputStream myOutStream = new ByteArrayOutputStream();
212         myGateway.createCertificateRequest(KeyStoreAlias.SIGNER.getName(), myOutStream);
213 
214         /* Process the certificateRequest */
215         ByteArrayInputStream myInputStream = new ByteArrayInputStream(myOutStream.toByteArray());
216         myOutStream.reset();
217         myGateway.processCertificateRequest(myInputStream, myOutStream);
218 
219         /* Process the certificateResponse */
220         myInputStream = new ByteArrayInputStream(myOutStream.toByteArray());
221         myOutStream.reset();
222         final Integer myRespId = myGateway.processCertificateResponse(myInputStream, myOutStream);
223 
224         /* Cleanup */
225         myStore.deleteEntry(KeyStoreAlias.SIGNER.getName());
226         myStore.deleteEntry(((GordianCoreKeyStoreGateway) myGateway).getCertificateAlias(myRespId));
227     }

229     /**
230      * test encrypted keyPairRequest.
231      *
232      * @param pKeyPairSpec the keyPairSpec
233      * @param pUsage       the keyUsage
234      * @throws GordianException on error
235      */
236     private void encryptedKeyPairRequest(final GordianKeyPairSpec pKeyPairSpec,
237                                          final GordianKeyPairUsage pUsage) throws GordianException {
238         /* Access details */
239         final GordianCoreKeyStoreManager myMgr = theState.getManager();
240         final GordianCoreKeyStore myStore = theState.getStore();
241         final GordianKeyStorePair myIntermediate = theState.getIntermediate();
242 
243         /* Create the keyPair */
244         final KeyStoreAlias myAlias = pUsage.hasUse(GordianKeyPairUse.KEYENCRYPT)
245                 ? KeyStoreAlias.ENCRYPT
246                 : KeyStoreAlias.AGREE;
247         final X500Name myCertName = KeyStoreUtils.buildX500Name(myAlias);
248         myMgr.createKeyPair(pKeyPairSpec, myCertName, pUsage, myIntermediate, myAlias.getName(), KeyStoreUtils.DEF_PASSWORD);
249 
250         /* Create and configure gateway */
251         final GordianKeyStoreGateway myGateway = myStore.getFactory().getAsyncFactory().getKeyStoreFactory().createKeyStoreGateway(myMgr);
252         myGateway.setPasswordResolver(theState::passwordResolver);
253         myGateway.setCertifier(KeyStoreAlias.CERTIFIER.getName());
254         myGateway.setMACSecretResolver(n -> DEF_MACSECRET);
255 
256         /* Build the CertificateRequest */
257         final ByteArrayOutputStream myOutStream = new ByteArrayOutputStream();
258         myGateway.createCertificateRequest(myAlias.getName(), myOutStream);
259 
260         /* Process the certificateRequest */
261         ByteArrayInputStream myInputStream = new ByteArrayInputStream(myOutStream.toByteArray());
262         myOutStream.reset();
263         myGateway.processCertificateRequest(myInputStream, myOutStream);
264 
265         /* Process the certificateResponse */
266         myInputStream = new ByteArrayInputStream(myOutStream.toByteArray());
267         myOutStream.reset();
268         final Integer myRespId = myGateway.processCertificateResponse(myInputStream, myOutStream);
269 
270         /* Process the certificateAck */
271         myInputStream = new ByteArrayInputStream(myOutStream.toByteArray());
272         myOutStream.reset();
273         myGateway.processCertificateAck(myInputStream);
274 
275         /* Cleanup */
276         myStore.deleteEntry(myAlias.getName());
277         myStore.deleteEntry(((GordianCoreKeyStoreGateway) myGateway).getCertificateAlias(myRespId));
278     }

280     /**
281      * KeyPairCertReqState.
282      */
283     private static class KeyPairCertReqState {
284         /**
285          * The keyStoreManager.
286          */
287         private final GordianCoreKeyStoreManager theManager;
288 
289         /**
290          * The intermediate signer.
291          */
292         private GordianKeyStorePair theIntermediate;
293 
294         /**
295          * The certifier signer.
296          */
297         private GordianKeyStorePair theCertifier;
298 
299         /**
300          * Constructor.
301          *
302          * @param pManager the keyStore Manager
303          */
304         KeyPairCertReqState(final GordianKeyStoreManager pManager) {
305             /* Store the manager */
306             theManager = (GordianCoreKeyStoreManager) pManager;
307         }
308 
309         /**
310          * Obtain the keyStoreManager.
311          *
312          * @return the keyStoreManager
313          */
314         GordianCoreKeyStoreManager getManager() {
315             return theManager;
316         }
317 
318         /**
319          * Obtain the keyStore.
320          *
321          * @return the keyStore
322          */
323         GordianCoreKeyStore getStore() {
324             return theManager.getKeyStore();
325         }
326 
327         /**
328          * Obtain the intermediate.
329          *
330          * @return the intermediate keyPair
331          */
332         GordianKeyStorePair getIntermediate() {
333             return theIntermediate;
334         }
335 
336         /**
337          * Obtain the certifier.
338          *
339          * @return the certifier keyPair
340          */
341         GordianKeyStorePair getCertifier() {
342             return theCertifier;
343         }
344 
345         /**
346          * Initialise.
347          *
348          * @throws GordianException on error
349          */
350         void initialise() throws GordianException {
351             /* Reset the keyStore */
352             theManager.getKeyStore().reset();
353 
354             /* Create specs */
355             final GordianKeyPairSpec myRSASpec = GordianKeyPairSpecBuilder.rsa(GordianRSAModulus.MOD2048);
356             final GordianKeyPairSpec myECSpec = GordianKeyPairSpecBuilder.ec(GordianDSAElliptic.SECT571K1);
357 
358             /* Create root certificate */
359             final X500Name myRootName = KeyStoreUtils.buildX500Name(KeyStoreAlias.ROOT);
360             final GordianKeyStorePair myRoot = theManager.createRootKeyPair(myRSASpec, myRootName, KeyStoreAlias.ROOT.getName(), DEF_PASSWORD);
361 
362             /* Create intermediate */
363             GordianKeyPairUsage myUsage = new GordianKeyPairUsage(GordianKeyPairUse.CERTIFICATE);
364             final X500Name myInterName = KeyStoreUtils.buildX500Name(KeyStoreAlias.INTER);
365             theIntermediate = theManager.createKeyPair(myECSpec, myInterName, myUsage, myRoot, KeyStoreAlias.INTER.getName(), DEF_PASSWORD);
366 
367             /* Create certifier */
368             final X500Name myCertifierName = KeyStoreUtils.buildX500Name(KeyStoreAlias.CERTIFIER);
369             theCertifier = theManager.createKeyPair(myRSASpec, myCertifierName, myUsage, myRoot, KeyStoreAlias.CERTIFIER.getName(), DEF_PASSWORD);
370         }
371 
372         /**
373          * Cleanup.
374          */
375         void cleanUp() {
376             /* Access the store */
377             final GordianCoreKeyStore myStore = getStore();
378 
379             /* delete the entries */
380             myStore.deleteEntry(KeyStoreAlias.ROOT.getName());
381             myStore.deleteEntry(KeyStoreAlias.INTER.getName());
382             myStore.deleteEntry(KeyStoreAlias.CERTIFIER.getName());
383 
384             /* Check that we have deleted all values */
385             int mySize = myStore.size();
386             Assertions.assertEquals(0, mySize);
387         }
388 
389         /**
390          * Password resolver.
391          *
392          * @param pAlias the alias
393          * @return the password
394          */
395         public char[] passwordResolver(final String pAlias) {
396             return DEF_PASSWORD.clone();
397         }
398     }
