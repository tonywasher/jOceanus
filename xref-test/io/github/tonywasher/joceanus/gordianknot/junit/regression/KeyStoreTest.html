17  package io.github.tonywasher.joceanus.gordianknot.junit.regression;
18  
19  import io.github.tonywasher.joceanus.gordianknot.api.base.GordianException;
20  import io.github.tonywasher.joceanus.gordianknot.api.factory.GordianFactory;
21  import io.github.tonywasher.joceanus.gordianknot.api.factory.GordianFactoryType;
22  import io.github.tonywasher.joceanus.gordianknot.api.keystore.GordianKeyStore;
23  import io.github.tonywasher.joceanus.gordianknot.api.keystore.GordianKeyStoreFactory;
24  import io.github.tonywasher.joceanus.gordianknot.api.keystore.GordianKeyStoreManager;
25  import io.github.tonywasher.joceanus.gordianknot.api.lock.GordianPasswordLockSpec;
26  import io.github.tonywasher.joceanus.gordianknot.util.GordianGenerator;
27  import org.junit.jupiter.api.DynamicNode;
28  import org.junit.jupiter.api.TestFactory;
29  
30  import java.util.stream.Stream;
31  
32  /**
33   * KeyStore Tests.
34   */
35  class KeyStoreTest {
36      /**
37       * The KeySetHashSpec.
38       */
39      private static final GordianPasswordLockSpec KEYSETLOCKSPEC = new GordianPasswordLockSpec(KeyStoreSymmetric.KEYSETSPEC);
40  
41      /**
42       * Create the keyStore test suite.
43       *
44       * @return the test stream
45       * @throws GordianException on error
46       */
47      @TestFactory
48      Stream<DynamicNode> bouncyCastle() throws GordianException {
49          return keyStoreTests(GordianFactoryType.BC);
50      }
51  
52      /**
53       * Create the jca keyStore test suite.
54       *
55       * @return the test stream
56       * @throws GordianException on error
57       */
58      @TestFactory
59      Stream<DynamicNode> jca() throws GordianException {
60          return keyStoreTests(GordianFactoryType.JCA);
61      }
62  
63      /**
64       * Create the keySet test suite for a factoryType.
65       *
66       * @param pFactoryType the factoryType
67       * @return the test stream
68       * @throws GordianException on error
69       */
70      private Stream<DynamicNode> keyStoreTests(final GordianFactoryType pFactoryType) throws GordianException {
71          /* Create the factory */
72          final GordianFactory myFactory = GordianGenerator.createRandomFactory(pFactoryType);
73  
74          /* Access keyStoreFactory and create a keyStore */
75          final GordianKeyStoreFactory myKSFactory = myFactory.getAsyncFactory().getKeyStoreFactory();
76          final GordianKeyStore myStore = myKSFactory.createKeyStore(KEYSETLOCKSPEC);
77          final GordianKeyStoreManager myMgr = myKSFactory.createKeyStoreManager(myStore);
78  
79          /* Return the stream */
80          return Stream.of(
81                  new KeyStoreSymmetric(myMgr).symmetricTest(),
82                  new KeyStorePairs(myMgr).keyPairsTest(),
83                  new KeyStoreRequest(myMgr).keyPairRequestTest()
84          );
85      }
86  }
