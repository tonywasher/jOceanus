55      /**
56       * Create the mac test suite for a factory.
57       *
58       * @param pFactory the factory
59       * @param pPartner the partner
60       * @return the test stream
61       */
62      static Stream<DynamicNode> macTests(final GordianFactory pFactory,
63                                          final GordianFactory pPartner) {
64          /* Create the default stream */
65          Stream<DynamicNode> myTests = Stream.empty();
66  
67          /* Loop through the keyLengths */
68          Iterator<GordianLength> myIterator = GordianKeyLengths.iterator();
69          while (myIterator.hasNext()) {
70              final GordianLength myKeyLen = myIterator.next();
71  
72              /* Build tests for this keyLength */
73              final Stream<DynamicNode> myTest = macTests(pFactory, pPartner, myKeyLen);
74              if (myTest != null) {
75                  myTests = Stream.concat(myTests, myTest);
76              }
77          }
78  
79          /* Return the tests */
80          return Stream.of(DynamicContainer.dynamicContainer("Macs", myTests));
81      }

83      /**
84       * Create the mac test suite for a factory.
85       *
86       * @param pFactory the factory
87       * @param pPartner the partner
88       * @param pKeyLen  the keyLength
89       * @return the test stream or null
90       */
91      private static Stream<DynamicNode> macTests(final GordianFactory pFactory,
92                                                  final GordianFactory pPartner,
93                                                  final GordianLength pKeyLen) {
94          /* Add mac Tests */
95          List<FactoryMacSpec> myMacs = SymmetricStore.macProvider(pFactory, pPartner, pKeyLen);
96          if (!myMacs.isEmpty()) {
97              Stream<DynamicNode> myTests = myMacs.stream().map(x -> DynamicContainer.dynamicContainer(x.toString(), macTests(x)));
98              return Stream.of(DynamicContainer.dynamicContainer(pKeyLen.toString(), myTests));
99          }
100 
101         /* No mac Tests */
102         return null;
103     }

105     /**
106      * Create the mac test suite for a macSpec.
107      *
108      * @param pMacSpec the macSpec
109      * @return the test stream
110      */
111     private static Stream<DynamicNode> macTests(final FactoryMacSpec pMacSpec) {
112         /* Add profile test */
113         Stream<DynamicNode> myTests = Stream.of(DynamicTest.dynamicTest("profile", () -> profileMac(pMacSpec)));
114 
115         /* Add Multi test */
116         myTests = Stream.concat(myTests, Stream.of(DynamicTest.dynamicTest("multi", () -> multiMac(pMacSpec))));
117 
118         /* Add Xof test if this is a Xof */
119         if (pMacSpec.getSpec().isXof()
120                 && GordianFactoryType.BC.equals(pMacSpec.getFactory().getFactoryType())) {
121             myTests = Stream.concat(myTests, Stream.of(DynamicTest.dynamicTest("xof", () -> checkXof(pMacSpec))));
122         }
123 
124         /* Add algorithmId test */
125         myTests = Stream.concat(myTests, Stream.of(DynamicTest.dynamicTest("checkAlgId", () -> checkMacAlgId(pMacSpec))));
126 
127         /* Add externalId test */
128         myTests = Stream.concat(myTests, Stream.of(DynamicTest.dynamicTest("externalId", () -> SymmetricTest.checkExternalId(pMacSpec))));
129 
130         /* Add partner test if  the partner supports this macSpec */
131         if (pMacSpec.getPartner() != null) {
132             myTests = Stream.concat(myTests, Stream.of(DynamicTest.dynamicTest("Partner", () -> checkPartnerMac(pMacSpec))));
133         }
134 
135         /* Return the tests */
136         return myTests;
137     }

139     /**
140      * Profile mac.
141      *
142      * @param pMacSpec the mac to profile
143      * @throws GordianException on error
144      */
145     private static void profileMac(final FactoryMacSpec pMacSpec) throws GordianException {
146         final GordianFactory myFactory = pMacSpec.getFactory();
147         final GordianMacSpec mySpec = pMacSpec.getSpec();
148         final GordianMacFactory myMacFactory = myFactory.getMacFactory();
149         final GordianMac myMac1 = myMacFactory.createMac(pMacSpec.getSpec());
150         final GordianKey<GordianMacSpec> myKey = pMacSpec.getKey();
151 
152         /* Check that the macLength is correct */
153         Assertions.assertEquals(mySpec.getMacLength().getByteLength(), myMac1.getMacSize(), "MacLength incorrect");
154 
155         /* Define the input */
156         final byte[] myBytes = "MacInput".getBytes();
157         boolean isInconsistent = false;
158 
159         /* Access the two macs */
160         final GordianMacType myType = mySpec.getMacType();
161         final boolean twoMacs = GordianMacType.GMAC.equals(myType);
162         final boolean needsReInit = myType.needsReInitialisation();
163         final GordianMac myMac2 = twoMacs
164                 ? myMacFactory.createMac(mySpec)
165                 : myMac1;
166 
167         /* Start loop */
168         final long myStart = System.nanoTime();
169         for (int i = 0; i < SymmetricTest.profileRepeat; i++) {
170             /* Use first mac */
171             myMac1.init(GordianMacParameters.keyWithRandomNonce(myKey));
172             myMac1.update(myBytes);
173             final byte[] myFirst = myMac1.finish();
174 
175             /* If we need to reInitialise */
176             if (needsReInit) {
177                 myMac2.init(GordianMacParameters.keyAndNonce(myKey, myMac1.getInitVector()));
178             }
179 
180             /* Use second mac */
181             myMac2.update(myBytes);
182             final byte[] mySecond = myMac2.finish();
183             if (!Arrays.areEqual(myFirst, mySecond)) {
184                 isInconsistent = true;
185             }
186         }
187 
188         /* Record elapsed */
189         long myElapsed = System.nanoTime() - myStart;
190         myElapsed /= 2 * SymmetricTest.MILLINANOS * (long) SymmetricTest.profileRepeat;
191         if (SymmetricTest.fullProfiles) {
192             System.out.println(pMacSpec + ":" + myElapsed);
193         }
194         Assertions.assertFalse(isInconsistent, pMacSpec + " inconsistent");
195     }

197     /**
198      * Multi-call mac.
199      *
200      * @param pMacSpec the mac to profile
201      */
202     private static void multiMac(final FactoryMacSpec pMacSpec) throws GordianException {
203         /* Create the mac */
204         final GordianFactory myFactory = pMacSpec.getFactory();
205         final GordianMacSpec mySpec = pMacSpec.getSpec();
206         final GordianMacFactory myMacFactory = myFactory.getMacFactory();
207         final GordianMac myMac1 = myMacFactory.createMac(mySpec);
208         final GordianMac myMac2 = myMacFactory.createMac(mySpec);
209         final GordianKey<GordianMacSpec> myKey = pMacSpec.getKey();
210 
211         /* Check that the macLength is correct */
212         Assertions.assertEquals(mySpec.getMacLength().getByteLength(), myMac1.getMacSize(), "MacLength incorrect");
213 
214         /* Create the mac as a single block */
215         final byte[] myBytes = SymmetricTest.getTestData();
216         myMac1.init(GordianMacParameters.keyWithRandomNonce(myKey));
217         myMac1.update(myBytes);
218         final byte[] mySingle = myMac1.finish();
219 
220         /* Create the mac as partial blocks */
221         myMac2.init(GordianMacParameters.keyAndNonce(myKey, myMac1.getInitVector()));
222         for (int myPos = 0; myPos < SymmetricTest.DATALEN; myPos += SymmetricTest.PARTIALLEN) {
223             final int myLen = Math.min(SymmetricTest.PARTIALLEN, SymmetricTest.DATALEN - myPos);
224             myMac2.update(myBytes, myPos, myLen);
225         }
226         final byte[] myMulti = myMac2.finish();
227 
228         /* Check that the results are identical */
229         Assertions.assertArrayEquals(mySingle, myMulti, "Multi-Block and Single-Block results differ");
230     }

232     /**
233      * Check xof.
234      *
235      * @param pMacSpec the macSpec
236      * @throws GordianException on error
237      */
238     private static void checkXof(final FactoryMacSpec pMacSpec) throws GordianException {
239         /* Create the digest */
240         final GordianFactory myFactory = pMacSpec.getFactory();
241         final GordianMacSpec mySpec = pMacSpec.getSpec();
242         final GordianMacFactory myMacFactory = myFactory.getMacFactory();
243         final GordianMac myMac = myMacFactory.createMac(mySpec);
244         final GordianXof myXof = (GordianXof) myMac;
245         final GordianKey<GordianMacSpec> myKey = pMacSpec.getKey();
246 
247         /* Create the data */
248         final byte[] myData = SymmetricTest.getTestData();
249         myMac.init(GordianMacParameters.keyWithRandomNonce(myKey));
250 
251         /* Update the Xofs with the data */
252         myXof.update(myData, 0, SymmetricTest.DATALEN);
253 
254         /* Extract Xofs as single block */
255         final byte[] myFull = new byte[SymmetricTest.DATALEN];
256         myXof.finish(myFull, 0, SymmetricTest.DATALEN);
257 
258         /* Update the Xofs with the data */
259         myXof.update(myData, 0, SymmetricTest.DATALEN);
260         final byte[] myPart = new byte[SymmetricTest.DATALEN];
261 
262         /* Create the xof as partial blocks */
263         for (int myPos = 0; myPos < SymmetricTest.DATALEN; ) {
264             final int myLen = Math.min(SymmetricTest.PARTIALLEN, SymmetricTest.DATALEN - myPos);
265             myPos += myXof.output(myPart, myPos, myLen);
266         }
267         myXof.finish(myPart, 0, 0);
268 
269         /* Check that they are identical */
270         Assertions.assertArrayEquals(myPart, myFull, "Mismatch on partial vs full xof");
271     }

273     /**
274      * Check partner mac.
275      *
276      * @param pMacSpec the mac to check
277      */
278     private static void checkPartnerMac(final FactoryMacSpec pMacSpec) throws GordianException {
279         /* Create the macs */
280         final GordianFactory myFactory = pMacSpec.getFactory();
281         final GordianFactory myPartner = pMacSpec.getPartner();
282         final GordianMacSpec mySpec = pMacSpec.getSpec();
283         final GordianMacFactory myMacFactory = myFactory.getMacFactory();
284         final GordianMac myMac = myMacFactory.createMac(mySpec);
285         final GordianMacFactory myPartnerFactory = myPartner.getMacFactory();
286         final GordianMac myPartnerMac = myPartnerFactory.createMac(mySpec);
287         final GordianKey<GordianMacSpec> myKey = pMacSpec.getKey();
288         final GordianKey<GordianMacSpec> myPartnerKey = pMacSpec.getPartnerKey();
289 
290         /* Calculate macs */
291         final byte[] myBytes = "MacInput".getBytes();
292         myMac.init(GordianMacParameters.keyWithRandomNonce(myKey));
293         final byte[] myIV = myMac.getInitVector();
294         myMac.update(myBytes);
295         final byte[] myFirst = myMac.finish();
296         if (myIV == null) {
297             myPartnerMac.init(GordianMacParameters.key(myPartnerKey));
298         } else {
299             myPartnerMac.init(GordianMacParameters.keyAndNonce(myPartnerKey, myIV));
300         }
301         myPartnerMac.update(myBytes);
302         final byte[] mySecond = myPartnerMac.finish();
303 
304         /* Check that the macs match */
305         Assertions.assertArrayEquals(myFirst, mySecond, "Mac misMatch");
306     }

308     /**
309      * Check macAlgId.
310      *
311      * @param pSpec the Spec to check
312      */
313     private static void checkMacAlgId(final FactoryMacSpec pSpec) {
314         /* Access the factory */
315         final GordianBaseFactory myFactory = (GordianBaseFactory) pSpec.getFactory();
316 
317         /* Check that we have an id */
318         final AlgorithmIdentifier myId = myFactory.getIdentifierForSpec(pSpec.getSpec());
319         Assertions.assertNotNull(myId, "Unknown AlgorithmId for " + pSpec.getSpec());
320 
321         /* Check unique mapping */
322         final GordianKeySpec mySpec = myFactory.getKeySpecForIdentifier(myId);
323         Assertions.assertEquals(pSpec.getSpec(), mySpec, "Invalid mapping for  " + pSpec.getSpec());
324     }
