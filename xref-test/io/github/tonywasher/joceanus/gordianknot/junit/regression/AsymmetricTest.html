105     /**
106      * Create the asymmetric test suite for a factory.
107      *
108      * @param pFactory the factory
109      * @param pPartner the partner factory
110      * @return the test stream
111      */
112     Stream<DynamicNode> asymmetricTests(final GordianFactory pFactory,
113                                         final GordianFactory pPartner) {
114         /* Create an empty stream */
115         Stream<DynamicNode> myStream = Stream.empty();
116 
117         /* Loop through the possible keySpecs */
118         for (final FactoryKeySpec myKeySpec : AsymmetricStore.keySpecProvider(pFactory, pPartner)) {
119             /* Create a stream */
120             Stream<DynamicNode> myKeyStream = Stream.of(DynamicTest.dynamicTest("generate", () -> generateKeyPairs(myKeySpec)));
121             myKeyStream = Stream.concat(myKeyStream, Stream.of(DynamicTest.dynamicTest("keySpec", () -> checkKeyPair(myKeySpec))));
122             myKeyStream = Stream.concat(myKeyStream, Stream.of(DynamicTest.dynamicTest("keyWrap", () -> checkKeyWrap(myKeySpec))));
123 
124             /* Add signature Tests */
125             AsymmetricStore.signatureProvider(myKeySpec);
126             if (!myKeySpec.getSignatures().isEmpty()) {
127                 Stream<DynamicNode> myTests = myKeySpec.getSignatures().stream()
128                         .map(x -> DynamicContainer.dynamicContainer(x.toString(), AsymmetricSignScripts.signatureTests(x)));
129                 myTests = Stream.of(DynamicContainer.dynamicContainer("Signatures", myTests));
130                 myKeyStream = Stream.concat(myKeyStream, myTests);
131             }
132 
133             /* Add agreement Tests */
134             AsymmetricStore.agreementProvider(myKeySpec);
135             if (!myKeySpec.getAgreements().isEmpty()) {
136                 Stream<DynamicNode> myTests = myKeySpec.getAgreements().stream()
137                         .map(x -> DynamicContainer.dynamicContainer(x.toString(), AsymmetricXAgreeScripts.xAgreementTests(x)));
138                 myTests = Stream.of(DynamicContainer.dynamicContainer("XAgreements", myTests));
139                 myKeyStream = Stream.concat(myKeyStream, myTests);
140             }
141 
142             /* Add encryptor Tests */
143             AsymmetricStore.encryptorProvider(myKeySpec);
144             if (!myKeySpec.getEncryptors().isEmpty()) {
145                 Stream<DynamicNode> myTests = myKeySpec.getEncryptors().stream()
146                         .map(x -> DynamicContainer.dynamicContainer(x.toString(), AsymmetricEncryptScripts.encryptorTests(x)));
147                 myTests = Stream.of(DynamicContainer.dynamicContainer("Encryptors", myTests));
148                 myKeyStream = Stream.concat(myKeyStream, myTests);
149             }
150 
151             /* Add the stream */
152             myStream = Stream.concat(myStream, Stream.of(DynamicContainer.dynamicContainer(myKeySpec.getKeySpec().toString(), myKeyStream)));
153         }
154 
155         /* Return the stream */
156         return myStream;
157     }

159     /**
160      * Generate KeyPairs.
161      *
162      * @param pKeySpec the keySpec
163      * @throws GordianException on error
164      */
165     private void generateKeyPairs(final FactoryKeySpec pKeySpec) throws GordianException {
166         /* Access the keyPairs */
167         final FactoryKeyPairs myPairs = pKeySpec.getKeyPairs();
168 
169         /* Force creation of the pairs */
170         myPairs.getKeyPair();
171         myPairs.getMirrorKeyPair();
172         myPairs.getPartnerSelfKeyPair();
173         myPairs.getPartnerTargetKeyPair();
174         myPairs.getTargetKeyPair();
175     }
176 
177     /**
178      * Check KeyPair.
179      *
180      * @param pKeySpec the keySpec
181      * @throws GordianException on error
182      */
183     private void checkKeyPair(final FactoryKeySpec pKeySpec) throws GordianException {
184         /* Access the keyPairs */
185         final FactoryKeyPairs myPairs = pKeySpec.getKeyPairs();
186         final GordianKeyPairFactory myFactory = pKeySpec.getFactory().getKeyPairFactory();
187         final GordianKeyPairSpec mySpec = pKeySpec.getKeySpec();
188 
189         /* Check X509Encodings */
190         final X509EncodedKeySpec myPublic = myPairs.getX509Encoding();
191         Assertions.assertEquals(mySpec, myFactory.determineKeyPairSpec(myPublic), "X509 has wrong keySpec");
192 
193         /* Check PKCS8Encodings */
194         final PKCS8EncodedKeySpec myPrivate = myPairs.getPKCS8Encoding();
195         Assertions.assertEquals(mySpec, myFactory.determineKeyPairSpec(myPrivate), "PKCS8 has wrong keySpec");
196 
197         /* Derive identical keyPair */
198         final GordianKeyPair myPair = myPairs.getKeyPair();
199         final GordianKeyPair myMirror = myPairs.getMirrorKeyPair();
200         Assertions.assertEquals(myPair, myMirror, "Derived pair has wrong keySpec");
201 
202         /* If the keyPair is stateAware */
203         if (mySpec.isStateAware()) {
204             /* Check for StateAware */
205             Assertions.assertInstanceOf(GordianStateAwareKeyPair.class, myPair, "Pair");
206             Assertions.assertInstanceOf(GordianStateAwareKeyPair.class, myMirror, "Mirror");
207             Assertions.assertInstanceOf(GordianStateAwareKeyPair.class, myPairs.getTargetKeyPair(), "Target");
208             Assertions.assertInstanceOf(GordianStateAwareKeyPair.class, myPairs.getPartnerSelfKeyPair(), "PartnerSelf");
209             Assertions.assertInstanceOf(GordianStateAwareKeyPair.class, myPairs.getPartnerTargetKeyPair(), "PartnerTarget");
210         }
211     }

213     /**
214      * Check KeyPair.
215      *
216      * @param pKeySpec the keySpec
217      * @throws GordianException on error
218      */
219     private void checkKeyWrap(final FactoryKeySpec pKeySpec) throws GordianException {
220         /* Access the keyPairs */
221         final FactoryKeyPairs myPairs = pKeySpec.getKeyPairs();
222         final GordianCoreAsyncFactory myFactory = (GordianCoreAsyncFactory) pKeySpec.getFactory();
223         final GordianKeyPair myPair = myPairs.getKeyPair();
224         final X509EncodedKeySpec myPublic = myPairs.getX509Encoding();
225 
226         /* Create a keySet */
227         final GordianKeySetFactory myKeySetFactory = myFactory.getFactory().getKeySetFactory();
228         final GordianKeySet myKeySet = myKeySetFactory.generateKeySet(AsymmetricXAgreeScripts.KEYSETSPEC);
229         final byte[] mySecured = myKeySet.securePrivateKey(myPair);
230         final GordianKeyPair myDerived = myKeySet.deriveKeyPair(myPublic, mySecured);
231         Assertions.assertEquals(myPair, myDerived, "Incorrect derived pair");
232         Assertions.assertEquals(myKeySet.getPrivateKeyWrapLength(myPair), mySecured.length, "Incorrect wrapped length");
233     }
