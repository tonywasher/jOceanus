60      /**
61       * Create the streamKey test suite for a factory.
62       *
63       * @param pFactory the factory
64       * @param pPartner the partner
65       * @return the test stream or null
66       */
67      static Stream<DynamicNode> streamKeyTests(final GordianFactory pFactory,
68                                                final GordianFactory pPartner) {
69          /* Create the default stream */
70          Stream<DynamicNode> myTests = Stream.empty();
71  
72          /* Loop through the keyLengths */
73          Iterator<GordianLength> myIterator = GordianKeyLengths.iterator();
74          while (myIterator.hasNext()) {
75              final GordianLength myKeyLen = myIterator.next();
76  
77              /* Build tests for this keyLength */
78              final Stream<DynamicNode> myTest = streamKeyTests(pFactory, pPartner, myKeyLen);
79              if (myTest != null) {
80                  myTests = Stream.concat(myTests, myTest);
81              }
82          }
83  
84          /* Return the tests */
85          return Stream.of(DynamicContainer.dynamicContainer("streamKeys", myTests));
86      }

88      /**
89       * Create the streamKey test suite for a factory.
90       *
91       * @param pFactory the factory
92       * @param pPartner the partner
93       * @param pKeyLen  the keyLength
94       * @return the test stream or null
95       */
96      private static Stream<DynamicNode> streamKeyTests(final GordianFactory pFactory,
97                                                        final GordianFactory pPartner,
98                                                        final GordianLength pKeyLen) {
99          /* Add streamKey Tests */
100         List<FactoryStreamKeySpec> myKeys = SymmetricStore.streamKeyProvider(pFactory, pPartner, pKeyLen);
101         if (!myKeys.isEmpty()) {
102             Stream<DynamicNode> myTests = myKeys.stream().map(x -> DynamicContainer.dynamicContainer(x.toString(), streamKeyTests(x)));
103             return Stream.of(DynamicContainer.dynamicContainer(pKeyLen.toString(), myTests));
104         }
105 
106         /* No stream Tests */
107         return null;
108     }

110     /**
111      * Create the streamKey test suite for a streamKeySpec.
112      *
113      * @param pKeySpec the keySpec
114      * @return the test stream
115      */
116     private static Stream<DynamicNode> streamKeyTests(final FactoryStreamKeySpec pKeySpec) {
117         /* Add profile test */
118         Stream<DynamicNode> myTests = Stream.of(DynamicTest.dynamicTest("profile", () -> profileStreamKey(pKeySpec)));
119 
120         /* Add cipher test */
121         final FactoryStreamCipherSpec myCipherSpec = new FactoryStreamCipherSpec(pKeySpec, GordianStreamCipherSpecBuilder.stream(pKeySpec.getSpec()));
122         myTests = Stream.concat(myTests, Stream.of(DynamicTest.dynamicTest("checkCipher", () -> checkStreamCipher(myCipherSpec))));
123 
124         /* Add multi test */
125         myTests = Stream.concat(myTests, Stream.of(DynamicTest.dynamicTest("multi", () -> multiStreamCipher(myCipherSpec))));
126 
127         /* Add externalId tests */
128         myTests = Stream.concat(myTests, Stream.of(DynamicTest.dynamicTest("externalKeyId", () -> SymmetricTest.checkExternalId(pKeySpec))));
129         myTests = Stream.concat(myTests, Stream.of(DynamicTest.dynamicTest("externalCipherId", () -> SymmetricTest.checkExternalId(myCipherSpec))));
130 
131         /* Add algorithmId tests */
132         myTests = Stream.concat(myTests, Stream.of(DynamicTest.dynamicTest("algorithmKeyId", () -> checkStreamKeyAlgId(pKeySpec))));
133         myTests = Stream.concat(myTests, Stream.of(DynamicTest.dynamicTest("algorithmCipherId", () -> checkStreamCipherAlgId(myCipherSpec))));
134 
135         /* Add partner test if  the partner supports this streamKeySpec */
136         if (pKeySpec.getPartner() != null) {
137             myTests = Stream.concat(myTests, Stream.of(DynamicTest.dynamicTest("Partner", () -> checkPartnerStreamKey(pKeySpec))));
138         }
139 
140         /* Add PBE tests */
141         myTests = Stream.concat(myTests, Stream.of(DynamicContainer.dynamicContainer("PBE", streamPBECipherTests(myCipherSpec))));
142 
143         /* Add AAD cipher tests if required */
144         if (pKeySpec.hasAADVersion()) {
145             final FactoryStreamCipherSpec myAADSpec = new FactoryStreamCipherSpec(pKeySpec, GordianStreamCipherSpecBuilder.stream(pKeySpec.getSpec(), true));
146             myTests = Stream.concat(myTests, Stream.of(DynamicContainer.dynamicContainer("AAD", streamAADCipherTests(myAADSpec))));
147         }
148 
149         /* Return the tests */
150         return myTests;
151     }

189     /**
190      * Check stream cipher.
191      *
192      * @param pCipherSpec the keySpec
193      * @throws GordianException on error
194      */
195     private static void checkStreamCipher(final FactoryStreamCipherSpec pCipherSpec) throws GordianException {
196         /* If this cipher is in fact an AEAD cipher */
197         /* Elephant, ISAP, PhotonBeetle and Xoodyak do not currently support AEADParameters so treat as non-AEAD for the time being */
198         if (pCipherSpec.getSpec().isAEAD()) {
199             /* Test as AEAD cipher */
200             checkStreamAADCipher(pCipherSpec);
201             return;
202         }
203 
204         /* Access details */
205         final FactoryStreamKeySpec myOwner = pCipherSpec.getOwner();
206         final GordianFactory myFactory = myOwner.getFactory();
207         final GordianStreamKeySpec myKeySpec = myOwner.getSpec();
208         final GordianCipherFactory myCipherFactory = myFactory.getCipherFactory();
209         final GordianKey<GordianStreamKeySpec> myStreamKey = myOwner.getKey();
210 
211         /* Access Data */
212         final byte[] myTestData = SymmetricTest.getTestData();
213 
214         /* Create the Cipher */
215         final GordianStreamCipherSpec myCipherSpec = GordianStreamCipherSpecBuilder.stream(myKeySpec);
216         final GordianStreamCipher myCipher = myCipherFactory.createStreamKeyCipher(myCipherSpec);
217         GordianCipherParameters myParms = GordianCipherParameters.keyWithRandomNonce(myStreamKey);
218         myCipher.initForEncrypt(myParms);
219         final byte[] myIV = myCipher.getInitVector();
220         myParms = GordianCipherParameters.keyAndNonce(myStreamKey, myIV);
221         final byte[] myEncrypted = myCipher.finish(myTestData);
222         if (myCipherSpec.getKeyType().getStreamKeyType().needsReInit()) {
223             myCipher.initForEncrypt(myParms);
224         }
225         final byte[] myEncrypted2 = myCipher.finish(myTestData);
226         myCipher.initForDecrypt(myParms);
227         final byte[] myResult = myCipher.finish(myEncrypted);
228         myCipher.initForDecrypt(myParms);
229         final byte[] myResult2 = myCipher.finish(myEncrypted2);
230         Assertions.assertArrayEquals(myTestData, myResult, "Failed to encrypt/decrypt");
231         Assertions.assertArrayEquals(myResult, myResult2, "Failed to reset properly");
232     }

234     /**
235      * Multi-call symCipher.
236      *
237      * @param pCipherSpec the cipher to profile
238      */
239     private static void multiStreamCipher(final FactoryStreamCipherSpec pCipherSpec) throws GordianException {
240         /* Create the cipher */
241         final GordianFactory myFactory = pCipherSpec.getFactory();
242         final GordianStreamCipherSpec mySpec = pCipherSpec.getSpec();
243         final GordianCipherFactory myCipherFactory = myFactory.getCipherFactory();
244         final GordianStreamCipher myCipher1 = myCipherFactory.createStreamKeyCipher(mySpec);
245         final GordianStreamCipher myCipher2 = myCipherFactory.createStreamKeyCipher(mySpec);
246         final GordianKey<GordianStreamKeySpec> myKey = pCipherSpec.getKey();
247 
248         /* Access the test data */
249         final byte[] myBytes = SymmetricTest.getTestData();
250 
251         /* Encrypt the data as a single block */
252         myCipher1.initForEncrypt(GordianCipherParameters.keyWithRandomNonce(myKey));
253         byte[] myEncrypt = new byte[myCipher1.getOutputLength(myBytes.length)];
254         int myOut = myCipher1.update(myBytes, 0, myBytes.length, myEncrypt, 0);
255         int myXpected = myOut + myCipher1.getOutputLength(0);
256         myOut += myCipher1.finish(myEncrypt, myOut);
257         Assertions.assertEquals(myOut, myXpected, "Output length fails on encryption");
258         myEncrypt = Arrays.copyOf(myEncrypt, myOut);
259 
260         /* Decrypt the data as partial blocks */
261         myCipher2.initForDecrypt(GordianCipherParameters.keyAndNonce(myKey, myCipher1.getInitVector()));
262         byte[] myMulti = new byte[myCipher2.getOutputLength(myEncrypt.length)];
263         myOut = 0;
264         for (int myPos = 0; myPos < myEncrypt.length; myPos += SymmetricTest.PARTIALLEN) {
265             final int myLen = Math.min(SymmetricTest.PARTIALLEN, myEncrypt.length - myPos);
266             myOut += myCipher2.update(myEncrypt, myPos, myLen, myMulti, myOut);
267         }
268         myOut += myCipher2.finish(myMulti, myOut);
269         myMulti = Arrays.copyOf(myMulti, myOut);
270 
271         /* Check that the results are identical */
272         Assertions.assertEquals(myBytes.length, myOut, "Multi-Block decrypt length failed");
273         Assertions.assertArrayEquals(myBytes, myMulti, "Multi-Block decrypt failed");
274 
275         /* Encrypt the data as partial blocks */
276         myCipher2.initForEncrypt(GordianCipherParameters.keyWithRandomNonce(myKey));
277         myOut = 0;
278         for (int myPos = 0; myPos < myBytes.length; myPos += SymmetricTest.PARTIALLEN) {
279             final int myLen = Math.min(SymmetricTest.PARTIALLEN, myBytes.length - myPos);
280             myOut += myCipher2.update(myBytes, myPos, myLen, myEncrypt, myOut);
281         }
282         myOut += myCipher2.finish(myEncrypt, myOut);
283         myEncrypt = Arrays.copyOf(myEncrypt, myOut);
284 
285         /* Decrypt the data as single block */
286         myCipher1.initForDecrypt(GordianCipherParameters.keyAndNonce(myKey, myCipher2.getInitVector()));
287         byte[] mySingle = new byte[myCipher1.getOutputLength(myEncrypt.length)];
288         myOut = myCipher1.update(myEncrypt, 0, myEncrypt.length, mySingle, 0);
289         myXpected = myOut + myCipher1.getOutputLength(0);
290         myOut += myCipher1.finish(mySingle, myOut);
291         Assertions.assertEquals(myOut, myXpected, "Output length fails on decryption");
292         mySingle = Arrays.copyOf(mySingle, myOut);
293 
294         /* Check that the results are identical */
295         Assertions.assertEquals(myBytes.length, myOut, "Multi-Block encrypt length failed");
296         Assertions.assertArrayEquals(myBytes, mySingle, "Multi-Block encrypt failed");
297     }

299     /**
300      * Check streamKey PBE CipherMode.
301      *
302      * @param pCipherSpec the cipherSpec
303      * @throws GordianException on error
304      */
305     private static void checkStreamPBECipher(final FactoryStreamPBECipherSpec pCipherSpec) throws GordianException {
306         /* Access details */
307         final GordianFactory myFactory = pCipherSpec.getFactory();
308         final FactoryStreamCipherSpec myOwner = pCipherSpec.getOwner();
309         final GordianStreamCipherSpec myCipherSpec = myOwner.getSpec();
310         final GordianPBESpec myPBESpec = pCipherSpec.getSpec();
311         final GordianCipherFactory myCipherFactory = myFactory.getCipherFactory();
312 
313         /* Access Data */
314         final byte[] myTestData = SymmetricTest.getTestData();
315         final char[] myPassword = "HelloThere".toCharArray();
316 
317         /* Create the Spec */
318         final GordianStreamCipher myCipher = myCipherFactory.createStreamKeyCipher(myCipherSpec);
319         GordianCipherParameters myParms = GordianCipherParameters.pbe(myPBESpec, myPassword);
320         myCipher.initForEncrypt(myParms);
321 
322         /* Check encryption */
323         final byte[] mySalt = myCipher.getPBESalt();
324         myParms = GordianCipherParameters.pbeAndNonce(myPBESpec, myPassword, mySalt);
325         final byte[] myEncrypted = myCipher.finish(myTestData);
326         myCipher.initForDecrypt(myParms);
327         final byte[] myResult = myCipher.finish(myEncrypted);
328         Assertions.assertArrayEquals(myTestData, myResult, "Failed to encrypt/decrypt");
329     }

337     private static void checkStreamAADCipher(final FactoryStreamCipherSpec pCipherSpec) throws GordianException {

340         final GordianStreamCipherSpec mySpec = pCipherSpec.getSpec();

354         myCipher.initForDecrypt(myParms);
355         final byte[] myResult = myCipher.finish(myEncrypted);
356         myCipher.initForDecrypt(myParms);
357         final byte[] myResult2 = myCipher.finish(myEncrypted);
358         Assertions.assertArrayEquals(myResult, myResult2, "Failed to reset properly");
359         Assertions.assertArrayEquals(myTestData, myResult, "Failed to encrypt/decrypt");
360     }

395     /**
396      * Profile streamKey.
397      *
398      * @param pStreamKeySpec the keySpec
399      * @throws GordianException on error
400      */
401     private static void profileStreamKey(final FactoryStreamKeySpec pStreamKeySpec) throws GordianException {
402         final GordianFactory myFactory = pStreamKeySpec.getFactory();
403         final GordianStreamKeySpec myKeySpec = pStreamKeySpec.getSpec();
404         final GordianCipherFactory myCipherFactory = myFactory.getCipherFactory();
405         final GordianKey<GordianStreamKeySpec> myStreamKey = pStreamKeySpec.getKey();
406         byte[] myBytes = SymmetricTest.getTestData();
407         final GordianStreamCipherSpec myCipherSpec = GordianStreamCipherSpecBuilder.stream(myKeySpec);
408         final GordianStreamCipher myCipher = myCipherFactory.createStreamKeyCipher(myCipherSpec);
409         GordianCipherParameters myParms = GordianCipherParameters.keyWithRandomNonce(myStreamKey);
410         final long myStart = System.nanoTime();
411         for (int i = 0; i < SymmetricTest.profileRepeat; i++) {
412             myCipher.initForEncrypt(myParms);
413             myBytes = myCipher.finish(myBytes);
414         }
415         long myElapsed = System.nanoTime() - myStart;
416         myElapsed /= SymmetricTest.MILLINANOS * (long) SymmetricTest.profileRepeat;
417         if (SymmetricTest.fullProfiles) {
418             System.out.println(myKeySpec.toString() + ":" + myElapsed);
419         }
420     }
