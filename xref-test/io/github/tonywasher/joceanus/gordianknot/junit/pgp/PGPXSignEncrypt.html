61  /**
62   * PGP File sign and encrypt.
63   */
64  public class PGPXSignEncrypt {
65      /* Secure Random */
66      private static final SecureRandom RANDOM = new SecureRandom();
67  
68      /* Target Style */
69      private static final PGPXKeyRingStyle STYLE = PGPXKeyRingStyle.DUO;
70  
71      /**
72       * Main program.
73       *
74       * @param pArgs arguments
75       */
76      public static void main(final String[] pArgs) {
77          /* Protect against exceptions */
78          try {
79              /* Access the encryption keys */
80              final List<BcPGPPublicKeyRing> myRings = loadRings();
81  
82              /* Access the target file */
83              try (OutputStream myOutput = new FileOutputStream(PGPXKeyRingUtil.FILEDIR + "PGPTest.new.asc");
84                   BufferedOutputStream myBufferedOut = new BufferedOutputStream(myOutput);
85                   ArmoredOutputStream myArmoredOut = new ArmoredOutputStream(myBufferedOut)) {
86                  /* Encrypt the file */
87                  encryptFile(myArmoredOut, myRings);
88              }
89  
90          } catch (IOException | PGPException e) {
91              e.printStackTrace();
92          }
93      }
94  
95      /**
96       * Load the list of target rings
97       *
98       * @return the ring list
99       */
100     private static List<BcPGPPublicKeyRing> loadRings() throws PGPException, IOException {
101         final List<BcPGPPublicKeyRing> myRings = new ArrayList<>();
102         for (final PGPXKeyRing myRing : PGPXKeyRing.values()) {
103             if (myRing.getKeyPairStyle() == STYLE) {
104                 myRings.add(PGPXKeyRingUtil.loadPublicKeyRing(myRing));
105             }
106         }
107         return myRings;
108     }

110     /**
111      * Encrypt a file.
112      *
113      * @param pOutput   the output stream
114      * @param pKeyRings the keyRings to encrypt to
115      */
116     public static void encryptFile(final OutputStream pOutput,
117                                    final List<BcPGPPublicKeyRing> pKeyRings) throws PGPException, IOException {
118         /* Determine algorithms */
119         final PGPXFeatures myAlgs = PGPXFeatures.determinePreferences(pKeyRings);
120 
121         /* Init encrypted data generator */
122         final PGPDataEncryptorBuilder myEncBuilder = new BcPGPDataEncryptorBuilder(myAlgs.getSymAlgorithm())
123                 .setSecureRandom(RANDOM).setWithIntegrityPacket(myAlgs.withIntegrity());
124         final PGPEncryptedDataGenerator encryptedDataGenerator = new PGPEncryptedDataGenerator(myEncBuilder);
125         for (BcPGPPublicKeyRing myRing : pKeyRings) {
126             final PGPPublicKey myEncKey = obtainEncryptionKey(myRing);
127             encryptedDataGenerator.addMethod(new BcPublicKeyKeyEncryptionMethodGenerator(myEncKey));
128         }
129         final OutputStream encryptedOut = encryptedDataGenerator.open(pOutput, new byte[PGPXKeyRingUtil.BUFFER_SIZE]);
130 
131         /* start compression */
132         final PGPCompressedDataGenerator compressedDataGenerator = new PGPCompressedDataGenerator(myAlgs.getCompAlgorithm());
133         final OutputStream compressedOut = compressedDataGenerator.open(encryptedOut, new byte[PGPXKeyRingUtil.BUFFER_SIZE]);
134 
135         /* Create the signature builders */
136         final List<PGPSignatureGenerator> mySigners = createSigners(compressedOut, myAlgs.getHashAlgorithm());
137 
138         /* Create the Literal Data generator output stream */
139         final PGPLiteralDataGenerator literalDataGenerator = new PGPLiteralDataGenerator();
140         final File actualFile = new File(PGPXKeyRingUtil.FILEDIR + "PGPTest.docx");
141         final OutputStream literalOut = literalDataGenerator.open(compressedOut,
142                 PGPLiteralData.BINARY, "PGPTest.docx",
143                 new Date(actualFile.lastModified()), new byte[PGPXKeyRingUtil.BUFFER_SIZE]);
144 
145         /* read input file and write to target file using a buffer */
146         final InputStream myInput = new FileInputStream(PGPXKeyRingUtil.FILEDIR + "PGPTest.docx");
147         final BufferedInputStream myBufferedIn = new BufferedInputStream(myInput);
148         final byte[] buf = new byte[PGPXKeyRingUtil.BUFFER_SIZE];
149         int len;
150         while ((len = myBufferedIn.read(buf, 0, buf.length)) > 0) {
151             literalOut.write(buf, 0, len);
152             for (final PGPSignatureGenerator mySigner : mySigners) {
153                 mySigner.update(buf, 0, len);
154             }
155         }
156 
157         /* Close input/output files */
158         myBufferedIn.close();
159         literalOut.close();
160         literalDataGenerator.close();
161 
162         /* Generate signatures */
163         for (final PGPSignatureGenerator mySigner : mySigners) {
164             mySigner.generate().encode(compressedOut);
165         }
166 
167         /* Close compressed output */
168         compressedOut.close();
169         compressedDataGenerator.close();
170 
171         /* Close encrypted output */
172         encryptedOut.close();
173         encryptedDataGenerator.close();
174     }

176     /**
177      * Try to find an encryption key in the keyRing
178      *
179      * @param pRing the keyRing
180      * @return first encryption key
181      */
182     private static PGPPublicKey obtainEncryptionKey(final PGPPublicKeyRing pRing) {
183         /* Access the master key */
184         final PGPPublicKey myMaster = pRing.getPublicKey();
185 
186         /* Loop through keyRing */
187         final Iterator<PGPPublicKey> kIt = pRing.getPublicKeys();
188         while (kIt.hasNext()) {
189             final PGPPublicKey k = kIt.next();
190 
191             /* Access the binding signature */
192             final PGPSignature mySig = PGPXKeyRingUtil.obtainKeyIdSignature(k, myMaster.getKeyID());
193 
194             /* Make sure that we can encrypt storage */
195             final PGPSignatureSubpacketVector v = mySig.getHashedSubPackets();
196             if (PGPXKeyRingUtil.checkKeyValidity(k, mySig)
197                     && (v.getKeyFlags() & PGPKeyFlags.CAN_ENCRYPT_STORAGE) != 0) {
198                 return k;
199             }
200         }
201 
202         /* No valid encryption key */
203         throw new IllegalArgumentException("Can't find encryption key in key ring.");
204     }

206     /**
207      * Create signers for secret keys
208      *
209      * @param pCompressed the compressed stream
210      * @param pHashAlgId  the hashAlgId
211      * @return the signer list
212      */
213     private static List<PGPSignatureGenerator> createSigners(final OutputStream pCompressed,
214                                                              final int pHashAlgId) throws PGPException, IOException {
215         final List<PGPXKeyRing> myRings = new ArrayList<>();
216         final List<PGPSignatureGenerator> mySigners = new ArrayList<>();
217         for (final PGPXKeyRing myRing : PGPXKeyRing.values()) {
218             if (myRing.getKeyPairStyle() == STYLE) {
219                 myRings.add(myRing);
220             }
221         }
222         final int myNumRings = myRings.size();
223         for (int i = 0; i < myNumRings; i++) {
224             final PGPXKeyRing myRing = myRings.get(i);
225             final PGPSignatureGenerator mySigner = createSigner(myRing, pHashAlgId);
226             mySigner.generateOnePassVersion(i < myNumRings - 1).encode(pCompressed);
227             mySigners.add(0, mySigner);
228         }
229         return mySigners;
230     }

232     /**
233      * Create signer for secret key
234      *
235      * @param pSecret    the keyRing type
236      * @param pHashAlgId the hashAlgId
237      * @return the signer
238      */
239     private static PGPSignatureGenerator createSigner(final PGPXKeyRing pSecret,
240                                                       final int pHashAlgId) throws PGPException, IOException {
241         /* Access the signing key */
242         final BcPGPSecretKeyRing pgpSec = PGPXKeyRingUtil.loadSecretKeyRing(pSecret);
243         final PGPSecretKey mySignSecret = obtainSigningKey(pgpSec);
244         final PGPPrivateKey mySignKey = obtainPrivateKey(pSecret, mySignSecret);
245 
246         /* Create the signature builder */
247         final BcPGPContentSignerBuilder mySignerBuilder = new BcPGPContentSignerBuilder(mySignSecret.getPublicKey().getAlgorithm(),
248                 pHashAlgId);
249         final int myAlg = mySignSecret.getPublicKey().getAlgorithm();
250         if (myAlg != PGPPublicKey.Ed25519 && myAlg != PGPPublicKey.Ed448) {
251             mySignerBuilder.setSecureRandom(RANDOM);
252         }
253 
254         /* Create the signer */
255         final PGPSignatureGenerator mySigner = new PGPSignatureGenerator(mySignerBuilder, pgpSec.getPublicKey());
256         mySigner.init(PGPSignature.BINARY_DOCUMENT, mySignKey);
257 
258         /* Build signature attributes */
259         final PGPSignatureSubpacketGenerator spGen = new PGPSignatureSubpacketGenerator();
260         spGen.setIssuerFingerprint(false, mySignSecret);
261         final Iterator<String> myUserids = mySignSecret.getUserIDs();
262         if (myUserids.hasNext()) {
263             final String userId = myUserids.next();
264             spGen.addSignerUserID(false, userId.getBytes(StandardCharsets.UTF_8));
265         }
266         mySigner.setHashedSubpackets(spGen.generate());
267 
268         /* Return the signer */
269         return mySigner;
270     }

272     /**
273      * Try to find a signing key in the KeyRing
274      *
275      * @param pRing the keyRing collection
276      * @return first signing key
277      */
278     private static PGPSecretKey obtainSigningKey(final PGPSecretKeyRing pRing) {
279         /* Access the master key */
280         final PGPPublicKey myMaster = pRing.getPublicKey();
281 
282         /* Loop through keyRing */
283         final Iterator<PGPSecretKey> kIt = pRing.getSecretKeys();
284         while (kIt.hasNext()) {
285             final PGPSecretKey k = kIt.next();
286             final PGPSignature mySig = PGPXKeyRingUtil.obtainKeyIdSignature(k.getPublicKey(), myMaster.getKeyID());
287 
288             /* Make sure that we can sign data */
289             final PGPSignatureSubpacketVector v = mySig.getHashedSubPackets();
290             if (PGPXKeyRingUtil.checkKeyValidity(k.getPublicKey(), mySig)
291                     && (v.getKeyFlags() & PGPKeyFlags.CAN_SIGN) != 0) {
292                 return k;
293             }
294         }
295 
296         /* No valid signing key */
297         throw new IllegalArgumentException("Can't find signing key in key ring.");
298     }

300     /**
301      * Obtain private key for secret key
302      *
303      * @param pSecret the keyRing type
304      * @param pKey    the secretKey
305      * @return password
306      */
307     private static PGPPrivateKey obtainPrivateKey(final PGPXKeyRing pSecret,
308                                                   final PGPSecretKey pKey) throws PGPException {
309         final String myPass = pSecret.obtainPassword4Secret();
310         final PBESecretKeyDecryptor myDecryptor = new BcPBESecretKeyDecryptorBuilder(new BcPGPDigestCalculatorProvider()).build(myPass.toCharArray());
311         return pKey.extractPrivateKey(myDecryptor);
312     }
