211     /**
212      * Test the Cipher against the results.
213      *
214      * @param pCipher   the cipher to test.
215      * @param pTestCase the testCase
216      */
217     static void testCipher(final StreamCipher pCipher,
218                            final TestCase pTestCase) {
219         /* Access the expected bytes */
220         final byte[] myExpected = Hex.decode(pTestCase.theExpected);
221 
222         /* Create the output buffer */
223         final byte[] myOutput = new byte[myExpected.length];
224 
225         /* Access plainText or nulls */
226         final byte[] myData = pTestCase.thePlainText != null
227                 ? Hex.decode(pTestCase.thePlainText)
228                 : new byte[myExpected.length];
229 
230         /* Access the key and the iv */
231         final KeyParameter myKey = new KeyParameter(Hex.decode(pTestCase.theKey));
232         final byte[] myIV = Hex.decode(pTestCase.theIV);
233         final ParametersWithIV myParms = new ParametersWithIV(myKey, myIV);
234 
235         /* Initialise the cipher and create the keyStream */
236         pCipher.init(true, myParms);
237         pCipher.processBytes(myData, 0, myData.length, myOutput, 0);
238 
239         /* Check the encryption */
240         if (!Arrays.equals(myExpected, myOutput)) {
241             System.out.println(Hex.toHexString(myOutput, 0, myOutput.length));
242         }
243         Assertions.assertArrayEquals(myExpected, myOutput, "Encryption mismatch");
244     }

246     /**
247      * Test the Mac against the results.
248      *
249      * @param pMac      the mac to test.
250      * @param pOnes     use all ones as data?
251      * @param pTestCase the testCase
252      */
253     static void testMac(final Mac pMac,
254                         final boolean pOnes,
255                         final TestCase pTestCase) {
256         /* Access the expected bytes */
257         final byte[] myExpected = Hex.decode(pTestCase.theExpected);
258 
259         /* Create the output buffer and the data */
260         final byte[] myOutput = new byte[pMac.getMacSize()];
261         final byte[] myData = new byte[(pOnes ? 4000 : 400) / 8];
262         Arrays.fill(myData, (byte) (pOnes ? 0x11 : 0));
263 
264         /* Access the key and the iv */
265         final KeyParameter myKey = new KeyParameter(Hex.decode(pTestCase.theKey));
266         final byte[] myIV = Hex.decode(pTestCase.theIV);
267         final ParametersWithIV myParms = new ParametersWithIV(myKey, myIV);
268 
269         /* Initialise the cipher and create the keyStream */
270         pMac.init(myParms);
271         pMac.update(myData, 0, myData.length);
272         pMac.doFinal(myOutput, 0);
273 
274         /* Check the encryption */
275         Assertions.assertArrayEquals(myExpected, myOutput, "Mac mismatch");
276     }

278     /**
279      * Test the Stream Cipher against the limit.
280      *
281      * @param pCipher   the cipher to test.
282      * @param pTestCase the testCase
283      * @param pLimit    the limit in bits.
284      */
285     static void testStreamLimit(final StreamCipher pCipher,
286                                 final TestCase pTestCase,
287                                 final int pLimit) {
288         /* Check the limit is a whole number of integers */
289         Assertions.assertTrue((pLimit % Integer.SIZE == 0), "Invalid limit");
290         final int myNumBytes = pLimit / Byte.SIZE;
291 
292         /* Create the maximum # of bytes */
293         final byte[] myData = new byte[myNumBytes];
294         final byte[] myOutput = new byte[myNumBytes];
295 
296         /* Access the key and the iv */
297         final KeyParameter myKey = new KeyParameter(Hex.decode(pTestCase.theKey));
298         final byte[] myIV = Hex.decode(pTestCase.theIV);
299         final ParametersWithIV myParms = new ParametersWithIV(myKey, myIV);
300 
301         /* Initialise the cipher and create the keyStream */
302         pCipher.init(true, myParms);
303         pCipher.processBytes(myData, 0, myData.length, myOutput, 0);
304 
305         /* Check that next encryption throws exception */
306         Assertions.assertThrows(IllegalStateException.class, () -> pCipher.processBytes(myData, 0, 1, myOutput, 0), "Limit failure");
307     }

314      * @param pLimit    the limit in bits.
315      */
316     static void testMacLimit(final Mac pMac,
317                              final TestCase pTestCase,
318                              final int pLimit) {
319         /* Check the limit is a whole numbet of integers */
320         Assertions.assertTrue((pLimit % Integer.SIZE == 0), "Invalid limit");
321         final int myNumBytes = pLimit / Byte.SIZE;
322 
323         /* Create the maximum # of bytes */
324         final byte[] myData = new byte[myNumBytes];
325         final byte[] myOutput = new byte[myNumBytes];
326 
327         /* Access the key and the iv */
328         final KeyParameter myKey = new KeyParameter(Hex.decode(pTestCase.theKey));
329         final byte[] myIV = Hex.decode(pTestCase.theIV);
330         final ParametersWithIV myParms = new ParametersWithIV(myKey, myIV);
331 
332         /* Initialise the mac and create the result */
333         pMac.init(myParms);
334         pMac.update(myData, 0, myData.length);
335         pMac.doFinal(myOutput, 0);
336 
337         /* Initialise the mac and process as much data as possible */
338         pMac.init(myParms);

341         /* We expect a failure on processing a further byte */
342         Assertions.assertThrows(IllegalStateException.class, () -> {
343             pMac.update(myData, 0, 1);
344             pMac.doFinal(myOutput, 0);
345         }, "Limit failure");
346     }

348     /**
349      * Test the Cipher against the results.
350      *
351      * @param pCipher   the cipher to test.
352      * @param pTestCase the testCase
353      * @throws GordianException on error
354      */
355     static void testAADCipher(final GordianChaChaPoly1305 pCipher,
356                               final TestCase pTestCase) throws GordianException {
357         try {
358             /* Access the expected bytes */
359             final byte[] myExpected = Hex.decode(pTestCase.theExpected);
360 
361             /* Access plainText */
362             final byte[] myData = Hex.decode(pTestCase.thePlainText);
363 
364             /* Access AAD */
365             final byte[] myAAD = Hex.decode(pTestCase.theAAD);
366 
367             /* Access the key and the iv */
368             final KeyParameter myKey = new KeyParameter(Hex.decode(pTestCase.theKey));
369             final byte[] myIV = Hex.decode(pTestCase.theIV);
370             final ParametersWithIV myIVParms = new ParametersWithIV(myKey, myIV);
371             final AEADParameters myAEADParms = new AEADParameters(myKey, 0, myIV, myAAD);
372 
373             /* Initialise the cipher and encrypt the data */
374             pCipher.init(true, myAEADParms);
375             final byte[] myOutput = new byte[pCipher.getOutputSize(myData.length)];
376             int iProcessed = pCipher.processBytes(myData, 0, myData.length, myOutput, 0);
377             pCipher.doFinal(myOutput, iProcessed);
378 
379             /* Check the encryption */
380             Assertions.assertArrayEquals(myExpected, myOutput, "Encryption mismatch");
381 
382             /* Check that auto-reset worked */
383             iProcessed = pCipher.processBytes(myData, 0, myData.length, myOutput, 0);
384             pCipher.doFinal(myOutput, iProcessed);
385 
386             /* Check the encryption */
387             Assertions.assertArrayEquals(myExpected, myOutput, "Encryption mismatch after reset");
388 
389             /* Initialise the cipher and decrypt the data */
390             pCipher.init(false, myAEADParms);
391             final byte[] myResult = new byte[pCipher.getOutputSize(myExpected.length)];
392             iProcessed = pCipher.processBytes(myExpected, 0, myExpected.length, myResult, 0);
393             pCipher.doFinal(myResult, iProcessed);
394 
395             /* Check the decryption */
396             Assertions.assertArrayEquals(myData, myResult, "Decryption mismatch");
397 
398             /* Loop to process differing block sizes */
399             for (int blockSize = 1; blockSize <= 64; blockSize <<= 1) {
400                 /* Process the decryption one block at a time */
401                 iProcessed = 0;
402                 int iRemaining = myExpected.length;
403                 final byte[] myResult2 = new byte[pCipher.getOutputSize(myExpected.length)];
404                 for (int i = 0; iRemaining > 0; i += blockSize, iRemaining -= blockSize) {
405                     int myLen = Math.min(blockSize, iRemaining);
406                     iProcessed += pCipher.processBytes(myExpected, i, myLen, myResult2, iProcessed);
407                 }
408                 pCipher.doFinal(myResult2, iProcessed);
409 
410                 /* Check the decryption */
411                 Assertions.assertArrayEquals(myData, myResult, "Block Decryption mismatch for size " + blockSize);
412             }
413 
414             /* Initialise the cipher and encrypt the data pass AAD explicitly */
415             pCipher.init(true, myIVParms);
416             pCipher.processAADBytes(myAAD, 0, myAAD.length);
417             final byte[] myOutput2 = new byte[pCipher.getOutputSize(myData.length)];
418             iProcessed = pCipher.processBytes(myData, 0, myData.length, myOutput2, 0);
419             pCipher.doFinal(myOutput2, iProcessed);
420 
421             /* Check the encryption */
422             Assertions.assertArrayEquals(myExpected, myOutput2, "Encryption mismatch");
423 
424             /* Catch exceptions */
425         } catch (InvalidCipherTextException e) {
426             throw new GordianCryptoException("Failed to resolve mac", e);
427         }
428     }
