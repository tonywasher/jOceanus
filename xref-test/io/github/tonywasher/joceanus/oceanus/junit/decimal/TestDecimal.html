100     /**
101      * Check addition.
102      *
103      * @param pFirst  the first value
104      * @param pSecond the second value
105      */
106     private static void checkAddition(final String pFirst,
107                                       final String pSecond) {
108         /* Obtain the two decimals */
109         final OceanusDecimal myD1 = new OceanusDecimal(pFirst);
110         final OceanusDecimal myD2 = new OceanusDecimal(pSecond);
111         final int myScale = Math.max(myD1.scale(), myD2.scale());
112 
113         /* Obtain the two BigDecimals */
114         final BigDecimal myB1 = new BigDecimal(pFirst);
115         final BigDecimal myB2 = new BigDecimal(pSecond);
116 
117         /* Add the two values */
118         final BigDecimal myB3 = myB1.add(myB2);
119 
120         /* Check addition with rounding */
121         final OceanusDecimal myD4 = new OceanusDecimal(myD1);
122         myD4.addValue(myD2);
123 
124         /* Assert equality */
125         Assertions.assertEquals(myB3.setScale(myD4.scale(), RoundingMode.HALF_UP), myD4.toBigDecimal(),
126                 "Failed Rounded Addition: " + pFirst + " + " + pSecond);
127 
128         /* Check addition with rounding */
129         final OceanusDecimal myD5 = new OceanusDecimal(myD2);
130         myD5.addValue(myD1);
131 
132         /* Assert equality */
133         Assertions.assertEquals(myB3.setScale(myD5.scale(), RoundingMode.HALF_UP), myD5.toBigDecimal(),
134                 "Failed Rounded Addition: " + pFirst + " + " + pSecond);
135 
136         /* Assert equality */
137         final OceanusDecimal myD3 = myD1.scale() > myD2.scale() ? myD4 : myD5;
138         Assertions.assertEquals(myB3, myD3.toBigDecimal(), "Failed Addition: " + pFirst + " + " + pSecond);
139     }

141     /**
142      * Check subtraction.
143      *
144      * @param pFirst  the first value
145      * @param pSecond the second value
146      */
147     private static void checkSubtraction(final String pFirst,
148                                          final String pSecond) {
149         /* Obtain the two decimals */
150         final OceanusDecimal myD1 = new OceanusDecimal(pFirst);
151         final OceanusDecimal myD2 = new OceanusDecimal(pSecond);
152 
153         /* Obtain the two BigDecimals */
154         final BigDecimal myB1 = new BigDecimal(pFirst);
155         final BigDecimal myB2 = new BigDecimal(pSecond);
156 
157         /* Subtract the two values */
158         final BigDecimal myB3 = myB1.subtract(myB2);
159 
160         /* Check subtraction with rounding */
161         final OceanusDecimal myD4 = new OceanusDecimal(myD1);
162         myD4.subtractValue(myD2);
163 
164         /* Assert equality */
165         Assertions.assertEquals(myB3.setScale(myD4.scale(), RoundingMode.HALF_UP), myD4.toBigDecimal(),
166                 "Failed Rounded Subtraction: " + pFirst + " - " + pSecond);
167 
168         /* Check subtraction with rounding */
169         final OceanusDecimal myD5 = new OceanusDecimal(myD2);
170         myD5.subtractValue(myD1);
171         myD5.negate();
172 
173         /* Assert equality */
174         Assertions.assertEquals(myB3.setScale(myD5.scale(), RoundingMode.HALF_UP), myD5.toBigDecimal(),
175                 "Failed Rounded Subtraction: -" + pSecond + " + " + pFirst);
176 
177         /* Assert equality */
178         final OceanusDecimal myD3 = myD1.scale() > myD2.scale() ? myD4 : myD5;
179         Assertions.assertEquals(myB3, myD3.toBigDecimal(), "Failed Subtraction: " + pFirst + " - " + pSecond);
180     }

182     /**
183      * Check multiplication.
184      *
185      * @param pFirst  the first value
186      * @param pSecond the second value
187      */
188     private static void checkMultiplication(final String pFirst,
189                                             final String pSecond) {
190         /* Obtain the two decimals */
191         final OceanusDecimal myD1 = new OceanusDecimal(pFirst);
192         final OceanusDecimal myD2 = new OceanusDecimal(pSecond);
193 
194         /* Obtain the two BigDecimals */
195         final BigDecimal myB1 = new BigDecimal(pFirst);
196         final BigDecimal myB2 = new BigDecimal(pSecond);
197 
198         /* Multiply the two values */
199         final OceanusDecimal myD3 = new OceanusDecimal(myD1).multiply(myD2);
200         final BigDecimal myB3 = myB1.multiply(myB2);
201 
202         /* Assert equality */
203         Assertions.assertEquals(myB3, myD3.toBigDecimal(), "Failed Multiplication: " + pFirst + " * " + pSecond);
204     }

206     /**
207      * Check division.
208      *
209      * @param pFirst  the first value
210      * @param pSecond the second value
211      */
212     private static void checkDivision(final String pFirst,
213                                       final String pSecond) {
214         /* Obtain the two decimals */
215         final OceanusDecimal myD1 = new OceanusDecimal(pFirst);
216         final OceanusDecimal myD2 = new OceanusDecimal(pSecond);
217 
218         /* Obtain the two BigDecimals */
219         final BigDecimal myB1 = new BigDecimal(pFirst);
220         final BigDecimal myB2 = new BigDecimal(pSecond);
221 
222         /* Divide the two values */
223         final OceanusDecimal myD3 = new OceanusDecimal(myD1).divide(myD2);
224         final BigDecimal myB3 = myB1.divide(myB2, RoundingMode.HALF_UP);
225 
226         /* Assert equality */
227         Assertions.assertEquals(myB3, myD3.toBigDecimal(), "Failed Division: " + pFirst + " / " + pSecond);
228     }

233     private static void checkBytes() {
234         /* Obtain the two values */
235         final OceanusRate myRate = new OceanusRate("0.25");
236         final OceanusUnits myUnits = new OceanusUnits("25.678");
237         final OceanusRatio myRatio = new OceanusRatio("5425.68");
238         final OceanusMoney myMoney = new OceanusMoney("76.90");
239         final OceanusPrice myPrice = new OceanusPrice("0.9856");
240         final OceanusDate myDate = new OceanusDate();
241 
242         /* Check Rate */
243         final OceanusRate myRate2 = new OceanusRate(myRate.toBytes());
244         Assertions.assertEquals(myRate, myRate2, "Failed Rate");
245 
246         /* Check Units */
247         final OceanusUnits myUnits2 = new OceanusUnits(myUnits.toBytes());
248         Assertions.assertEquals(myUnits, myUnits2, "Failed Units");
249 
250         /* Check Ratio */
251         final OceanusRatio myRatio2 = new OceanusRatio(myRatio.toBytes());
252         Assertions.assertEquals(myRatio, myRatio2, "Failed Ratio");
253 
254         /* Check Money */
255         final OceanusMoney myMoney2 = new OceanusMoney(myMoney.toBytes());
256         Assertions.assertEquals(myMoney, myMoney2, "Failed Money");
257 
258         /* Check Price */
259         final OceanusPrice myPrice2 = new OceanusPrice(myPrice.toBytes());
260         Assertions.assertEquals(myPrice, myPrice2, "Failed Price");
261 
262         /* Check Date */
263         final OceanusDateFormatter myFormatter = new OceanusDateFormatter();
264         final OceanusDate myDate2 = myFormatter.fromBytes(myFormatter.toBytes(myDate));
265         Assertions.assertEquals(myDate, myDate2, "Failed Date");
266     }
