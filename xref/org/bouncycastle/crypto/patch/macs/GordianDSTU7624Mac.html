1   package org.bouncycastle.crypto.patch.macs;
2   
3   import org.bouncycastle.crypto.CipherParameters;
4   import org.bouncycastle.crypto.DataLengthException;
5   import org.bouncycastle.crypto.Mac;
6   import org.bouncycastle.crypto.OutputLengthException;
7   import org.bouncycastle.crypto.engines.DSTU7624Engine;
8   import org.bouncycastle.crypto.params.KeyParameter;
9   import org.bouncycastle.util.Arrays;
10  
11  /**
12   * Implementation of DSTU7624 MAC mode
13   */
14  public class GordianDSTU7624Mac
15          implements Mac {
16      private final static int BITS_IN_BYTE = 8;
17  
18      private byte[] buf;
19      private int bufOff;
20  
21      private int macSize;
22      private int blockSize;
23      private DSTU7624Engine engine;
24  
25      private byte[] c, cTemp, kDelta;
26  
27      public GordianDSTU7624Mac(int blockBitLength,
28                                int q) {
29          this.engine = new DSTU7624Engine(blockBitLength);
30          this.blockSize = blockBitLength / BITS_IN_BYTE;
31          this.macSize = q / BITS_IN_BYTE;
32          this.c = new byte[blockSize];
33          this.kDelta = new byte[blockSize];
34          this.cTemp = new byte[blockSize];
35          this.buf = new byte[blockSize];
36      }
37  
38      public void init(CipherParameters params)
39              throws IllegalArgumentException {
40          if (params instanceof KeyParameter) {
41              engine.init(true, params);
42              reset();
43          } else {
44              throw new IllegalArgumentException("Invalid parameter passed to DSTU7624Mac");
45          }
46      }
47  
48      public String getAlgorithmName() {
49          return "DSTU7624Mac";
50      }
51  
52      public int getMacSize() {
53          return macSize;
54      }
55  
56      public void update(byte in) {
57          if (bufOff == buf.length) {
58              processBlock(buf, 0);
59              bufOff = 0;
60          }
61  
62          buf[bufOff++] = in;
63      }
64  
65      public void update(byte[] in, int inOff, int len) {
66          if (len < 0) {
67              throw new IllegalArgumentException(
68                      "can't have a negative input length!");
69          }
70  
71          int blockSize = engine.getBlockSize();
72          int gapLen = blockSize - bufOff;
73  
74          if (len > gapLen) {
75              System.arraycopy(in, inOff, buf, bufOff, gapLen);
76  
77              processBlock(buf, 0);
78  
79              bufOff = 0;
80              len -= gapLen;
81              inOff += gapLen;
82  
83              while (len > blockSize) {
84                  processBlock(in, inOff);
85  
86                  len -= blockSize;
87                  inOff += blockSize;
88              }
89          }
90  
91          System.arraycopy(in, inOff, buf, bufOff, len);
92  
93          bufOff += len;
94      }
95  
96      private void processBlock(byte[] in, int inOff) {
97          xor(c, 0, in, inOff, cTemp);
98  
99          engine.processBlock(cTemp, 0, c, 0);
100     }
101 
102     public int doFinal(byte[] out, int outOff)
103             throws DataLengthException, IllegalStateException {
104         if (bufOff % buf.length != 0) {
105             /* Pad using Poly1305 style padding */
106             buf[bufOff] = 1;
107             for (int i = bufOff + 1; i < buf.length; i++) {
108                 buf[i] = 0;
109             }
110         }
111 
112         // Last block
113         xor(c, 0, buf, 0, cTemp);
114         xor(cTemp, 0, kDelta, 0, c);
115         engine.processBlock(c, 0, c, 0);
116 
117         if (macSize + outOff > out.length) {
118             throw new OutputLengthException("output buffer too short");
119         }
120 
121         System.arraycopy(c, 0, out, outOff, macSize);
122         reset();
123 
124         return macSize;
125     }
126 
127     public void reset() {
128         Arrays.fill(c, (byte) 0x00);
129         Arrays.fill(cTemp, (byte) 0x00);
130         Arrays.fill(kDelta, (byte) 0x00);
131         Arrays.fill(buf, (byte) 0x00);
132         engine.reset();
133         engine.processBlock(kDelta, 0, kDelta, 0);
134         bufOff = 0;
135     }
136 
137     private void xor(byte[] x, int xOff, byte[] y, int yOff, byte[] x_xor_y) {
138 
139         if (x.length - xOff < blockSize || y.length - yOff < blockSize || x_xor_y.length < blockSize) {
140             throw new IllegalArgumentException("some of input buffers too short");
141         }
142         for (int byteIndex = 0; byteIndex < blockSize; byteIndex++) {
143             x_xor_y[byteIndex] = (byte) (x[byteIndex + xOff] ^ y[byteIndex + yOff]);
144         }
145     }
146 }
