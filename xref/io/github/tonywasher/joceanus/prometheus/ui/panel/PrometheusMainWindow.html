157     /**
158      * Build the main window.
159      *
160      * @param pView    the Data view
161      * @param pToolkit the toolkit
162      * @throws OceanusException on error
163      */
164     public void buildMainWindow(final PrometheusDataControl pView,
165                                 final PrometheusToolkit pToolkit) throws OceanusException {
166         /* Store the view */
167         theView = pView;
168         theToolkit = pToolkit.getToolkit();
169         theGuiFactory = theToolkit.getGuiFactory();
170         theThreadMgr = theToolkit.getThreadManager();
171 
172         /* Obtain the active profile */
173         final OceanusProfile myTask = theView.getActiveTask();
174         myTask.startTask("buildGUI");
175 
176         /* Create the panel */
177         thePanel = theGuiFactory.paneFactory().newBorderPane();
178 
179         /* Build the Main Panel */
180         final TethysUIComponent myMainPanel = buildMainPanel();
181 
182         /* Access the status bar and set to invisible */
183         theStatusBar = theThreadMgr.getStatusManager();
184         theStatusBar.setVisible(false);
185         theThreadMgr.getEventRegistrar().addEventListener(TethysUIThreadEvent.THREADEND, e -> {
186             setVisibility();
187             theStatusBar.setVisible(false);
188         });
189 
190         /* Create the panel */
191         thePanel.setNorth(theStatusBar);
192         thePanel.setCentre(myMainPanel);
193 
194         /* Build the Main Menu */
195         buildMainMenu();
196 
197         /* Create the data window */
198         theDataWdw = theToolkit.newViewerWindow();
199         theDataWdw.getEventRegistrar().addEventListener(TethysUIEvent.WINDOWCLOSED, e -> theMenuBar.setEnabled(PrometheusMenuId.DATAVIEWER, true));
200 
201         /* Complete task */
202         myTask.end();
203     }

303     /**
304      * Set visibility.
305      */
306     public void setVisibility() {
307         /* Determine whether we have any updates */
308         final boolean hasUpdates = hasUpdates();
309         final boolean hasChanges = hasChanges();
310         final boolean hasControl = getView().getData().getControl() != null;
311 
312         /* Note whether we have a worker thread */
313         final boolean hasWorker = hasWorker();
314 
315         /* Disable menus if we have a worker thread */
316         theMenuBar.setEnabled(PrometheusMenuId.DATA, !hasWorker);
317         theMenuBar.setEnabled(PrometheusMenuId.BACKUP, !hasWorker);
318         theMenuBar.setEnabled(PrometheusMenuId.SECURITY, !hasWorker && hasControl);
319 
320         /* If we have changes but no updates enable the undo/reset options */
321         if (hasWorker || !hasControl) {
322             theMenuBar.setEnabled(PrometheusMenuId.EDIT, false);
323         } else {
324             theMenuBar.setEnabled(PrometheusMenuId.EDIT, !hasUpdates && hasChanges);
325         }
326 
327         /* If we have changes disable the create backup options */
328         final boolean allowBackups = !hasChanges && !hasUpdates && hasControl;
329         theMenuBar.setEnabled(PrometheusThreadId.CREATEBACKUP, allowBackups);
330         theMenuBar.setEnabled(PrometheusThreadId.CREATEXML, allowBackups);
331         theMenuBar.setEnabled(PrometheusThreadId.CREATEXTRACT, allowBackups);
332 
333         /* If we have changes disable the security options */
334         final boolean allowSecurity = !hasChanges && !hasUpdates;
335         theMenuBar.setEnabled(PrometheusThreadId.CHANGEPASS, allowSecurity);
336         theMenuBar.setEnabled(PrometheusThreadId.RENEWSECURITY, allowSecurity);
337 
338         /* If we have updates disable the load backup/database option */
339         theMenuBar.setEnabled(PrometheusThreadId.RESTOREBACKUP, !hasUpdates);
340         theMenuBar.setEnabled(PrometheusThreadId.RESTOREXML, !hasUpdates);
341         theMenuBar.setEnabled(PrometheusThreadId.LOADDB, !hasUpdates);
342 
343         /* If we have updates or no changes disable the save database */
344         theMenuBar.setEnabled(PrometheusThreadId.STOREDB, !hasUpdates && hasChanges);
345     }

539     /**
540      * Display Help.
541      */
542     private void displayHelp() {
543         try {
544             /* Create the help window */
545             final MetisHelpWindow myHelpWdw = theToolkit.newHelpWindow();
546             myHelpWdw.setModule(getHelpModule());
547 
548             /* Listen for its closure */
549             myHelpWdw.getEventRegistrar().addEventListener(e -> {
550                 theMenuBar.setEnabled(PrometheusMenuId.SHOWHELP, true);
551                 myHelpWdw.hideDialog();
552             });
553 
554             /* Disable the menu item */
555             theMenuBar.setEnabled(PrometheusMenuId.SHOWHELP, false);
556 
557             /* Display it */
558             myHelpWdw.showDialog();
559 
560         } catch (OceanusException e) {
561             LOGGER.error("Failed to start Help Window", e);
562         }
563     }

570     @Override
571     public boolean handleAppClose() {
572         /* If we have updates or changes */
573         if (hasUpdates() || hasChanges()) {
574             /* Ask whether to continue */
575             final TethysUIAlert myAlert = theGuiFactory.dialogFactory().newAlert();
576             myAlert.setMessage(PROMPT_DISCARD);
577             myAlert.setTitle(TITLE_CLOSE);
578             final boolean myResult = myAlert.confirmYesNo();
579 
580             /* Ignore if not confirmed */
581             if (!myResult) {
582                 return false;
583             }
584         }
585 
586         /* terminate the executor */
587         theThreadMgr.shutdown();
588         return true;
589     }
