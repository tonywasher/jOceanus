20  
21  /**
22   * Style Item.
23   */
24  public enum PrometheusOdfStyleItem
25          implements PrometheusOdfItem {
26      /**
27       * Style.
28       */
29      STYLE("style"),
30  
31      /**
32       * Name.
33       */
34      NAME("name"),
35  
36      /**
37       * Family.
38       */
39      FAMILY("family"),
40  
41      /**
42       * ParentStyle.
43       */
44      PARENTSTYLE("parent-style-name"),
45  
46      /**
47       * DataStyle.
48       */
49      DATASTYLE("data-style-name"),
50  
51      /**
52       * ColumnProps.
53       */
54      TABLEPROPS("table-properties"),
55  
56      /**
57       * Display.
58       */
59      DISPLAY("display"),
60  
61      /**
62       * BreakBefore.
63       */
64      BREAKBEFORE("break-before"),
65  
66      /**
67       * TextAlign.
68       */
69      TEXTALIGN("text-align"),
70  
71      /**
72       * ColumnProps.
73       */
74      COLUMNPROPS("table-column-properties"),
75  
76      /**
77       * ColumnWidth.
78       */
79      COLUMNWIDTH("column-width"),
80  
81      /**
82       * RowProps.
83       */
84      ROWPROPS("table-row-properties"),
85  
86      /**
87       * RowHeight.
88       */
89      ROWHEIGHT("row-height"),
90  
91      /**
92       * CellProps.
93       */
94      CELLPROPS("table-cell-properties"),
95  
96      /**
97       * Protect.
98       */
99      PROTECT("cell-protect"),
100 
101     /**
102      * ParaProps.
103      */
104     PARAGRAPHPROPS("paragraph-properties"),
105 
106     /**
107      * TextProps.
108      */
109     TEXTPROPS("text-properties"),
110 
111     /**
112      * Map.
113      */
114     MAP("map"),
115 
116     /**
117      * ApplyStyle.
118      */
119     APPLYSTYLE("apply-style-name"),
120 
121     /**
122      * Condition.
123      */
124     CONDITION("condition"),
125 
126     /**
127      * Colour.
128      */
129     COLOR("color"),
130 
131     /**
132      * FontName.
133      */
134     FONTNAME("font-name"),
135 
136     /**
137      * FontSize.
138      */
139     FONTSIZE("font-size"),
140 
141     /**
142      * FontSize.
143      */
144     FONTWEIGHT("font-weight");
145 
146     /**
147      * Name.
148      */
149     private final String theName;
150 
151     /**
152      * Qualified Name.
153      */
154     private String theQualifiedName;
155 
156     /**
157      * Constructor.
158      *
159      * @param pName the name
160      */
161     PrometheusOdfStyleItem(final String pName) {
162         theName = pName;
163     }
164 
165     @Override
166     public String getName() {
167         return theName;
168     }
169 
170     @Override
171     public String getQualifiedName() {
172         if (theQualifiedName == null) {
173             theQualifiedName = PrometheusOdfNameSpace.buildQualifiedName(this);
174         }
175         return theQualifiedName;
176     }
177 
178     @Override
179     public PrometheusOdfNameSpace getNameSpace() {
180         switch (this) {
181             case BREAKBEFORE:
182             case TEXTALIGN:
183             case COLOR:
184                 return PrometheusOdfNameSpace.FORMAT;
185             default:
186                 return PrometheusOdfNameSpace.STYLE;
187         }
188     }
189 }
