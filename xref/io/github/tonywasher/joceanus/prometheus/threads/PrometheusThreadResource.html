17  package io.github.tonywasher.joceanus.prometheus.threads;
18  
19  import io.github.tonywasher.joceanus.oceanus.resource.OceanusBundleId;
20  import io.github.tonywasher.joceanus.oceanus.resource.OceanusBundleLoader;
21  
22  import java.util.ResourceBundle;
23  
24  /**
25   * Resource IDs for Prometheus Threads.
26   */
27  public enum PrometheusThreadResource
28          implements OceanusBundleId {
29      /**
30       * LoadDB.
31       */
32      THREAD_LOADDB("LoadDB"),
33  
34      /**
35       * StoreDB.
36       */
37      THREAD_STOREDB("StoreDB"),
38  
39      /**
40       * CreateDB.
41       */
42      THREAD_CREATEDB("CreateDB"),
43  
44      /**
45       * CreateDBTables.
46       */
47      THREAD_CREATEDBTABLES("CreateDBTables"),
48  
49      /**
50       * PurgeDB.
51       */
52      THREAD_PURGEDB("PurgeDB"),
53  
54      /**
55       * Create Backup.
56       */
57      THREAD_BACKUPCREATE("CreateBackup"),
58  
59      /**
60       * Restore Backup.
61       */
62      THREAD_BACKUPRESTORE("RestoreBackup"),
63  
64      /**
65       * Create XML Backup.
66       */
67      THREAD_XMLCREATE("CreateXml"),
68  
69      /**
70       * Restore XMLBackup MenuItem.
71       */
72      THREAD_XMLLOAD("RestoreXml"),
73  
74      /**
75       * Create XmlExtract.
76       */
77      THREAD_XTRACTCREATE("CreateXtract"),
78  
79      /**
80       * RenewSecurity.
81       */
82      THREAD_SECURERENEW("RenewSecurity"),
83  
84      /**
85       * Change Password.
86       */
87      THREAD_CHANGEPASS("ChangePass"),
88  
89      /**
90       * Seloect Backup File.
91       */
92      TASK_SELECT_BACKUP("task.SelectBackup");
93  
94      /**
95       * The Resource Loader.
96       */
97      private static final OceanusBundleLoader LOADER = OceanusBundleLoader.getLoader(PrometheusThreadResource.class.getCanonicalName(),
98              ResourceBundle::getBundle);
99  
100     /**
101      * The Id.
102      */
103     private final String theKeyName;
104 
105     /**
106      * The Value.
107      */
108     private String theValue;
109 
110     /**
111      * Constructor.
112      *
113      * @param pKeyName the key name
114      */
115     PrometheusThreadResource(final String pKeyName) {
116         theKeyName = pKeyName;
117     }
118 
119     @Override
120     public String getKeyName() {
121         return theKeyName;
122     }
123 
124     @Override
125     public String getNameSpace() {
126         return "Prometheus.thread";
127     }
128 
129     @Override
130     public String getValue() {
131         /* If we have not initialised the value */
132         if (theValue == null) {
133             /* Derive the value */
134             theValue = LOADER.getValue(this);
135         }
136 
137         /* return the value */
138         return theValue;
139     }
140 }
