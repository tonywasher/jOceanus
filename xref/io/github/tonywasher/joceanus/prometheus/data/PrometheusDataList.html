438     /**
439      * Populate a list extract.
440      *
441      * @param pList the list to populate
442      * @throws OceanusException on error
443      */
444     protected void populateList(final PrometheusDataList<?> pList) throws OceanusException {
445         /* Determine special styles */
446         final PrometheusListStyle myStyle = pList.getStyle();
447         final boolean isUpdate = myStyle == PrometheusListStyle.UPDATE;
448         final boolean isClone = myStyle == PrometheusListStyle.CLONE;
449 
450         /* Create an iterator for all items in the list */
451         final Iterator<? extends PrometheusDataItem> myIterator = iterator();
452 
453         /* Loop through the list */
454         while (myIterator.hasNext()) {
455             /* Access the item and its state */
456             final PrometheusDataItem myCurr = myIterator.next();
457             final MetisDataState myState = myCurr.getState();
458 
459             /* If this is an UPDATE list, ignore clean elements */
460             if (isUpdate && myState == MetisDataState.CLEAN) {
461                 continue;
462             }
463 
464             /* Copy the item */
465             pList.addCopyItem(myCurr);
466         }
467 
468         /* If this is a Clone list */
469         if (isClone) {
470             /* Adjust the links */
471             pList.resolveDataSetLinks();
472         }
473     }

492     /**
493      * Construct a difference extract between two DataLists. The difference extract will only have
494      * items that differ between the two lists. Items that are in the new list, but not in the old
495      * list will be viewed as inserted. Items that are in the old list but not in the new list will
496      * be viewed as deleted. Items that are in both list but differ will be viewed as changed
497      *
498      * @param pDataSet the difference DataSet
499      * @param pOld     The old list to compare to
500      * @return the difference list
501      */
502     public PrometheusDataList<T> deriveDifferences(final PrometheusDataSet pDataSet,
503                                                    final PrometheusDataList<?> pOld) {
504         /* Obtain an empty list of the correct style */
505         final PrometheusDataList<T> myList = getEmptyList(PrometheusListStyle.DIFFER);
506         myList.theDataSet = pDataSet;
507 
508         /* Access an Id Map of the old list */
509         final Map<Integer, ?> myOld = pOld.copyIdMap();
510 
511         /* Loop through the new list */
512         final Iterator<T> myIterator = iterator();
513         while (myIterator.hasNext()) {
514             /* Locate the item in the old list */
515             final PrometheusDataItem myCurr = myIterator.next();
516             PrometheusDataItem myItem = (PrometheusDataItem) myOld.get(myCurr.getIndexedId());
517 
518             /* If the item does not exist in the old list */
519             if (myItem == null) {
520                 /* Insert a new item */
521                 myItem = myList.addCopyItem(myCurr);
522                 myItem.setNewVersion();
523 
524                 /* else the item exists in the old list */
525             } else {
526                 /* If the item has changed */
527                 if (!myCurr.equals(myItem)) {
528                     /* Copy the item */
529                     final PrometheusDataItem myNew = myList.addCopyItem(myCurr);
530                     myNew.setBase(myItem);
531 
532                     /* Ensure that we record the correct history */
533                     myNew.setHistory(myItem);
534                 }
535 
536                 /* Remove the item from the map */
537                 myOld.remove(myItem.getIndexedId());
538             }
539         }
540 
541         /* Loop through the remaining items in the old list */
542         final Iterator<?> myOldIterator = myOld.values().iterator();
543         while (myOldIterator.hasNext()) {
544             /* Insert a new item */
545             final PrometheusDataItem myCurr = (PrometheusDataItem) myOldIterator.next();
546             final PrometheusDataItem myItem = myList.addCopyItem(myCurr);
547             myItem.setBase(null);
548             myItem.setDeleted(true);
549         }
550 
551         /* Return the difference list */
552         return myList;
553     }

555     /**
556      * Re-base the list against a database image. This method is used to re-synchronise between two
557      * sources. Items that are in this list, but not in the base list will be viewed as inserted.
558      * Items that are in the base list but not in this list list will be viewed as deleted. Items
559      * that are in both list but differ will be viewed as changed
560      *
561      * @param pBase The base list to re-base on
562      * @return are there any changes
563      */
564     public boolean reBase(final PrometheusDataList<?> pBase) {
565         /* Access an Id Map of the old list */
566         final Map<Integer, ?> myBase = pBase.copyIdMap();
567         boolean bChanges = false;
568 
569         /* Loop through this list */
570         final Iterator<T> myIterator = iterator();
571         while (myIterator.hasNext()) {
572             /* Locate the item in the base list */
573             final T myCurr = myIterator.next();
574             final PrometheusDataItem myItem = (PrometheusDataItem) myBase.get(myCurr.getIndexedId());
575 
576             /* If the underlying item does not exist */
577             if (myItem == null) {
578                 /* Mark this as a new item */
579                 myCurr.getValues().setVersion(getVersion() + 1);
580                 myCurr.setBase(null);
581                 bChanges = true;
582 
583                 /* else the item exists in the old list */
584             } else {
585                 /* if it has changed */
586                 if (!myCurr.equals(myItem)) {
587                     /* Set correct history */
588                     myCurr.setHistory(myItem);
589                     myCurr.setBase(null);
590                     bChanges = true;
591 
592                     /* else it is identical */
593                 } else {
594                     /* Mark this as a clean item */
595                     myCurr.clearHistory();
596                     myCurr.setBase(null);
597                 }
598 
599                 /* Remove the old item */
600                 myBase.remove(myItem.getIndexedId());
601             }
602         }
603 
604         /* Loop through the remaining items in the base list */
605         final Iterator<?> myBaseIterator = myBase.values().iterator();
606         while (myBaseIterator.hasNext()) {
607             /* Insert a new item */
608             final PrometheusDataItem myCurr = (PrometheusDataItem) myBaseIterator.next();
609             final T myItem = addCopyItem(myCurr);
610             myItem.setBase(null);
611             myItem.setHistory(myCurr);
612             myItem.getValues().setDeletion(true);
613             bChanges = true;
614         }
615 
616         /* Return flag */
617         return bChanges;
618     }

620     /**
621      * Is the Id unique in this list.
622      *
623      * @param uId the Id to check
624      * @return Whether the id is unique <code>true/false</code>
625      */
626     public boolean isIdUnique(final Integer uId) {
627         /* Its unique if its unassigned or greater than the max id */
628         if (uId == null
629                 || uId == 0
630                 || uId > theList.getNextId()) {
631             return true;
632         }
633 
634         /* Check in list */
635         return !theList.containsId(uId);
636     }

694     /**
695      * Validate the data items.
696      *
697      * @return the error list (or null if no errors)
698      */
699     public PrometheusDataErrorList validate() {
700         /* Allocate error list */
701         PrometheusDataErrorList myErrors = null;
702         MetisDataEditState myState = MetisDataEditState.CLEAN;
703 
704         /* Loop through the items */
705         final Iterator<T> myIterator = iterator();
706         while (myIterator.hasNext()) {
707             final T myCurr = myIterator.next();
708 
709             /* Clear errors for the item */
710             myCurr.clearErrors();
711 
712             /* Skip deleted items */
713             if (myCurr.isDeleted()) {
714                 myCurr.setValidEdit();
715                 myState = myState.combineState(MetisDataEditState.VALID);
716                 continue;
717             }
718 
719             /* Validate the item and build up the state */
720             myCurr.validate();
721             myState = myState.combineState(myCurr.getEditState());
722 
723             /* If the item is in error */
724             if (myCurr.hasErrors()) {
725                 /* If this is the first error */
726                 if (myErrors == null) {
727                     /* Allocate error list */
728                     myErrors = new PrometheusDataErrorList();
729                 }
730 
731                 /* Add to the error list */
732                 myErrors.add(myCurr);
733             }
734         }
735 
736         /* Store the edit state */
737         theEdit = myState;
738 
739         /* Return the errors */
740         return myErrors;
741     }

784     /**
785      * Build map of the data.
786      */
787     public void mapData() {
788         /* Ensure the map */
789         ensureMap();
790 
791         /* Loop through the items */
792         final Iterator<T> myIterator = iterator();
793         while (myIterator.hasNext()) {
794             final T myItem = myIterator.next();
795 
796             /* If the item is not deleted */
797             if (!myItem.isDeleted()) {

842     /**
843      * postProcessOnUpdate.
844      */
845     public void postProcessOnUpdate() {
846         /* Note whether this is a DataInfoList */
847         final boolean isDataInfo = this instanceof PrometheusDataInfoList;
848 
849         /* Reset the map */
850         if (theDataMap != null) {
851             theDataMap.resetMap();
852         }
853 
854         /* Loop through items clearing active flag */
855         final Iterator<T> myIterator = iterator();
856         MetisDataEditState myState = MetisDataEditState.CLEAN;
857         while (myIterator.hasNext()) {
858             final T myCurr = myIterator.next();
859 
860             /* Clear errors for the item */
861             myCurr.clearErrors();
862 
863             /* Skip deleted items */
864             if (myCurr.isDeleted()) {
865                 myCurr.setValidEdit();
866                 myState = myState.combineState(MetisDataEditState.VALID);
867                 continue;
868             }
869 
870             /* If this is not a DataInfo */
871             if (!isDataInfo) {
872                 /* Adjust touches and update map */
873                 myCurr.touchOnUpdate();
874                 myCurr.adjustMapForItem();
875             }
876 
877             /* Validate the item and build up the state */
878             myCurr.validate();
879             myState = myState.combineState(myCurr.getEditState());
880         }
881 
882         /* Store the edit state */
883         theEdit = myState;
884     }

920     /**
921      * Rewind items to the required version.
922      *
923      * @param pVersion the version to rewind to
924      */
925     public void rewindToVersion(final int pVersion) {
926         /* Loop through the elements */
927         final Iterator<T> myIterator = theList.iterator();
928         while (myIterator.hasNext()) {
929             final T myCurr = myIterator.next();
930 
931             /* If the version is before required version */
932             if (myCurr.getValues().getVersion() <= pVersion) {
933                 /* Ignore */
934                 continue;
935             }
936 
937             /* If the item was created after the required version */
938             if (myCurr.getOriginalValues().getVersion() > pVersion) {
939                 /* Remove from list */
940                 myIterator.remove();
941                 myCurr.deRegister();
942 
943                 /* Re-Loop */
944                 continue;
945             }
946 
947             /* Adjust values */
948             myCurr.rewindToVersion(pVersion);
949         }
950 
951         /* Adjust list value */
952         setVersion(pVersion);
953 
954         /* ReSort the list unless we are an edit list */
955         if (theStyle != PrometheusListStyle.EDIT) {
956             theList.sortList();
957         }
958     }

960     /**
961      * Condense history.
962      *
963      * @param pNewVersion the new maximum version
964      */
965     public void condenseHistory(final int pNewVersion) {
966         /* Loop through the elements */
967         final Iterator<T> myIterator = theList.iterator();
968         while (myIterator.hasNext()) {
969             final T myCurr = myIterator.next();
970 
971             /* If the version is before required version */
972             if (myCurr.getValues().getVersion() < pNewVersion) {
973                 /* Ignore */
974                 continue;
975             }
976 
977             /* If the item is in DELNEW state */
978             if (myCurr.isDeleted()
979                     && (myCurr.getOriginalValues().getVersion() >= pNewVersion)) {
980                 /* Remove from list */
981                 myIterator.remove();
982                 myCurr.deRegister();
983 
984                 /* Re-Loop */
985                 continue;
986             }
987 
988             /* Condense the history */
989             myCurr.condenseHistory(pNewVersion);
990         }
991 
992         /* Adjust list value */
993         setVersion(pNewVersion);
994     }
