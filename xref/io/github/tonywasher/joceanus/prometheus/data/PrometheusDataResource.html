490     /**
491      * The Resource Loader.
492      */
493     private static final OceanusBundleLoader LOADER = OceanusBundleLoader.getLoader(PrometheusDataResource.class.getCanonicalName(),
494             ResourceBundle::getBundle);
495 
496     /**
497      * The Id.
498      */
499     private final String theKeyName;
500 
501     /**
502      * The Value.
503      */
504     private String theValue;
505 
506     /**
507      * Constructor.
508      *
509      * @param pKeyName the key name
510      */
511     PrometheusDataResource(final String pKeyName) {
512         theKeyName = pKeyName;
513     }
514 
515     @Override
516     public String getKeyName() {
517         return theKeyName;
518     }
519 
520     @Override
521     public String getNameSpace() {
522         return "Prometheus.data";
523     }
524 
525     @Override
526     public String getValue() {
527         /* If we have not initialised the value */
528         if (theValue == null) {
529             /* Derive the value */
530             theValue = LOADER.getValue(this);
531         }
532 
533         /* return the value */
534         return theValue;
535     }
536 
537     @Override
538     public String getId() {
539         return getValue();
540     }
541 
542     @Override
543     public String toString() {
544         return getValue();
545     }
