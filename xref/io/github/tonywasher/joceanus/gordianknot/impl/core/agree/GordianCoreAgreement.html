181     /**
182      * Check status.
183      *
184      * @param pStatus the required status
185      * @throws GordianException on error
186      */
187     protected void checkStatus(final GordianAgreementStatus pStatus) throws GordianException {
188         /* If we are in the wrong state */
189         final GordianAgreementStatus myStatus = theState.getStatus();
190         if (myStatus != pStatus) {
191             throw new GordianLogicException("Invalid State: " + myStatus);
192         }
193     }

209     /**
210      * Set the client certificate.
211      *
212      * @param pClient the client certificate
213      * @throws GordianException on error
214      */
215     void setClientCertificate(final GordianCertificate pClient) throws GordianException {
216         /* Handle null client certificate */
217         if (pClient == null) {
218             final GordianAgreementType myType = theSpec.getAgreementType();
219             if (!myType.isAnonymous() && !myType.isSigned()) {
220                 throw new GordianDataException("Client Certificate must be provided");
221             }
222             return;
223         }
224 
225         /* Store the certificate */
226         theBuilder.setClientCertificate(pClient);
227     }

261     @Override
262     public void updateParams(final GordianAgreementParams pParams) throws GordianException {
263         /* Must be looking for serverPrivate */
264         checkStatus(GordianAgreementStatus.AWAITING_SERVERPRIVATE);
265 
266         /* Ensure that we are updating from correct parameters */
267         if (!Objects.equals(theParams.getId(), ((GordianCoreAgreementParams) pParams).getId())) {
268             throw new GordianDataException("Invalid parameters provided");
269         }
270 
271         /* Determine agreement type */
272         final GordianAgreementSpec mySpec = theState.getSpec();
273         final boolean isSigned = mySpec.getAgreementType().isSigned();
274 
275         /* If this is a signed agreement */
276         if (isSigned) {
277             /* Handle no signer certificate */
278             final GordianCertificate mySignerCert = pParams.getSignerCertificate();
279             if (mySignerCert == null) {
280                 throw new GordianLogicException("No signer declared for Signed agreement");
281             }
282 
283             /* Declare the signer */
284             setSignerCertificate(pParams.getSignatureSpec(), mySignerCert);
285 
286         } else {
287             /* Ensure that the server has a private key */
288             final GordianCertificate myServerCert = pParams.getServerCertificate();
289             if (myServerCert.getKeyPair().isPublicOnly()) {
290                 throw new GordianDataException("Server Certificate is Public Only");
291             }
292 
293             /* Update the server certificate */
294             setServerCertificate(myServerCert);
295         }
296 
297         /* Store additional data */
298         theState.setAdditionalData(pParams.getAdditionalData());
299 
300         /* Update the parameters */
301         theParams = new GordianCoreAgreementParams((GordianCoreAgreementParams) pParams);
302 
303         /* Process the augmented clientHello and return the agreement */
304         processClientHello();
305     }

307     @Override
308     public void setError(final String pError) throws GordianException {
309         /* Only allowed while awaiting ServerPrivate */
310         checkStatus(GordianAgreementStatus.AWAITING_SERVERPRIVATE);
311         theBuilder.setError(pError);
312 
313         /* If we are not anonymous */
314         if (!theSpec.getAgreementType().isAnonymous()) {
315             /* Create the rejection serverHello */
316             setNextMessage(theBuilder.newServerHello());
317         }
318 
319         /* Set result available */
320         theBuilder.setStatus(GordianAgreementStatus.RESULT_AVAILABLE);
321     }

352     /**
353      * Build the clientHello.
354      *
355      * @throws GordianException on error
356      */
357     void buildClientHello() throws GordianException {
358         /* Take a snapshot of the parameters */
359         theParams = new GordianCoreAgreementParams(theBuilder);
360 
361         /* Create ClientId and InitVector */
362         if (!theSpec.getAgreementType().isAnonymous()) {
363             theBuilder.newClientId();
364         }
365         theBuilder.newClientIV();
366 
367         /* Create clientEphemeral if needed */
368         if (needClientEphemeral()) {
369             theBuilder.newClientEphemeral();
370         }
371 
372         /* Create the new ClientHello */
373         theEngine.buildClientHello();
374         final GordianCoreAgreementMessageASN1 myMsg = theBuilder.newClientHello();
375 
376         /* Set next message and status */
377         setNextMessage(myMsg);
378         theBuilder.setStatus(theSpec.getAgreementType().isAnonymous()
379                 ? GordianAgreementStatus.RESULT_AVAILABLE
380                 : GordianAgreementStatus.AWAITING_SERVERHELLO);
381 
382         /* Store into cache if required */
383         if (!theSpec.getAgreementType().isAnonymous()) {
384             theSupplier.storeAgreement(myMsg.getClientId(), this);
385         }
386     }

388     /**
389      * Process the clientHello.
390      *
391      * @param pClientHello the clientHello
392      * @throws GordianException on error
393      */
394     void parseClientHello(final GordianCoreAgreementMessageASN1 pClientHello) throws GordianException {
395         /* Parse the clientHello */
396         theBuilder.parseClientHello(pClientHello);
397         theParams = new GordianCoreAgreementParams(theBuilder);
398         theBuilder.setStatus(GordianAgreementStatus.AWAITING_SERVERPRIVATE);
399     }

401     /**
402      * Process the clientHello.
403      *
404      * @throws GordianException on error
405      */
406     void processClientHello() throws GordianException {
407         /* Create ServerId and InitVector */
408         if (!theSpec.getAgreementType().isAnonymous()) {
409             theBuilder.newServerId();
410             theBuilder.newServerIV();
411         }
412 
413         /* Create serverEphemeral if needed */
414         if (needServerEphemeral()) {
415             theBuilder.newServerEphemeral();
416         }
417 
418         /* Copy ephemerals to keyPairs for signed */
419         if (theSpec.getAgreementType().isSigned()) {
420             theBuilder.copyEphemerals();
421         }
422 
423         /* Process the clientHello */
424         theEngine.processClientHello();
425 
426         /* If we are anonymous */
427         if (theSpec.getAgreementType().isAnonymous()) {
428             /* Set that the result is available */
429             theBuilder.setStatus(GordianAgreementStatus.RESULT_AVAILABLE);
430 
431             /* Else we need to build a serverHello */
432         } else {
433             /* Build the new serverHello */
434             final GordianCoreAgreementMessageASN1 myMsg = theBuilder.newServerHello();
435             setNextMessage(myMsg);
436             theBuilder.setStatus(Boolean.TRUE.equals(theSpec.withConfirm())
437                     ? GordianAgreementStatus.AWAITING_CLIENTCONFIRM
438                     : GordianAgreementStatus.RESULT_AVAILABLE);
439 
440             /* Store into cache if required */
441             if (Boolean.TRUE.equals(theSpec.withConfirm())) {
442                 theSupplier.storeAgreement(myMsg.getServerId(), this);
443             }
444         }
445     }

447     /**
448      * Process the serverHello.
449      *
450      * @param pServerHello the serverHello
451      * @throws GordianException on error
452      */
453     public void processServerHello(final GordianCoreAgreementMessageASN1 pServerHello) throws GordianException {
454         /* Check that we are expecting a serverHello */
455         checkStatus(GordianAgreementStatus.AWAITING_SERVERHELLO);
456 
457         /* Parse the serverHello */
458         final boolean bSuccess = theBuilder.parseServerHello(pServerHello);
459         if (bSuccess) {
460             /* Copy ephemerals to keyPairs for signed */
461             if (theSpec.getAgreementType().isSigned()) {
462                 theBuilder.copyEphemerals();
463             }
464 
465             /* Process the serverHello */
466             theEngine.processServerHello();
467         }
468 
469         /* If we need to send confirm */
470         if (bSuccess && Boolean.TRUE.equals(theSpec.withConfirm())) {
471             /* Build the new clientConfirm */
472             setNextMessage(theBuilder.newClientConfirm());
473         } else {
474             setNextMessage(null);
475         }
476         theBuilder.setStatus(GordianAgreementStatus.RESULT_AVAILABLE);
477 
478         /* remove from cache */
479         theSupplier.removeAgreement(pServerHello.getClientId());
480     }

482     /**
483      * Process the clientConfirm.
484      *
485      * @param pClientConfirm the clientConfirm
486      * @throws GordianException on error
487      */
488     public void processClientConfirm(final GordianCoreAgreementMessageASN1 pClientConfirm) throws GordianException {
489         /* Check that we are expecting a confirmation */
490         checkStatus(GordianAgreementStatus.AWAITING_CLIENTCONFIRM);
491 
492         /* Parse the clientConfirm */
493         if (theBuilder.parseClientConfirm(pClientConfirm)) {
494             /* Process if we have no error */
495             theEngine.processClientConfirm();
496         }
497 
498         /* Update status */
499         setNextMessage(null);
500         theBuilder.setStatus(GordianAgreementStatus.RESULT_AVAILABLE);
501 
502         /* remove from cache */
503         theSupplier.removeAgreement(pClientConfirm.getServerId());
504     }

506     /**
507      * Do we need a client ephemeral?
508      *
509      * @return true/false
510      */
511     private boolean needClientEphemeral() {
512         switch (theSpec.getAgreementType()) {
513             case ANON:
514             case SIGNED:
515             case SM2:
516             case MQV:
517             case UNIFIED:
518                 return true;
519             case KEM:
520             case BASIC:
521             default:
522                 return false;
523         }
524     }
525

534             case SM2:
535             case MQV:
536             case UNIFIED:
537                 return true;
538             case ANON:
539             case KEM:
540             case BASIC:
541             default:
542                 return false;
543         }
