52       * CipherOID branch.
53       */
54      private static final ASN1ObjectIdentifier CIPHEROID = GordianASN1Util.SYMOID.branch("4");

56      /**
57       * Map of cipherSpec to Identifier.
58       */
59      private final Map<GordianCipherSpec<?>, AlgorithmIdentifier> theSpecMap;
60  
61      /**
62       * Map of Identifier to cipherSpec.
63       */
64      private final Map<AlgorithmIdentifier, GordianCipherSpec<?>> theIdentifierMap;
65  
66      /**
67       * Constructor.
68       *
69       * @param pFactory the factory
70       */
71      GordianCipherAlgId(final GordianBaseFactory pFactory) {
72          /* Create the maps */
73          theSpecMap = new HashMap<>();
74          theIdentifierMap = new HashMap<>();
75  
76          /* Access the cipherFactory  */
77          final GordianCipherFactory myFactory = pFactory.getCipherFactory();
78  
79          /* Populate with the public standards */
80          addWellKnownCiphers128();
81          addWellKnownCiphers192();
82          addWellKnownCiphers256();
83  
84          /* For each KeyLength */
85          final Iterator<GordianLength> myIterator = GordianKeyLengths.iterator();
86          while (myIterator.hasNext()) {
87              final GordianLength myKeyLen = myIterator.next();
88  
89              /* Loop through the possible SymKeySpecs */
90              for (GordianSymKeySpec mySymKey : myFactory.listAllSupportedSymKeySpecs(myKeyLen)) {
91                  /* Loop through the possible CipherSpecs */
92                  for (GordianSymCipherSpec mySpec : myFactory.listAllSupportedSymCipherSpecs(mySymKey)) {
93                      /* Add any non-standard symCiphers */
94                      ensureSymCipher(mySpec);
95                  }
96              }
97  
98              /* Loop through the possible StreamCipherSpecs */
99              for (GordianStreamCipherSpec mySpec : myFactory.listAllSupportedStreamCipherSpecs(myKeyLen)) {
100                 /* Add any non-standard streamCiphers */
101                 ensureStreamCipher(mySpec);
102             }
103         }
104     }

106     /**
107      * Obtain Identifier for cipherSpec.
108      *
109      * @param pSpec the cipherSpec.
110      * @return the Identifier
111      */
112     public AlgorithmIdentifier getIdentifierForSpec(final GordianCipherSpec<?> pSpec) {
113         return theSpecMap.get(pSpec);
114     }

116     /**
117      * Obtain symCipherSpec for Identifier.
118      *
119      * @param pIdentifier the identifier.
120      * @return the cipherSpec (or null if not found)
121      */
122     GordianCipherSpec<?> getCipherSpecForIdentifier(final AlgorithmIdentifier pIdentifier) {
123         return theIdentifierMap.get(pIdentifier);

265     /**
266      * Add symCipherSpec to map if supported and not already present.
267      *
268      * @param pSpec the symCipherSpec
269      */
270     private void ensureSymCipher(final GordianSymCipherSpec pSpec) {
271         /* If the cipher is not already known */
272         if (!theSpecMap.containsKey(pSpec)) {
273             addSymCipher(pSpec);
274         }
275     }
276 
277     /**
278      * Create Identifier for a symCipherSpec.
279      *
280      * @param pSpec the cipherSpec
281      */
282     private void addSymCipher(final GordianSymCipherSpec pSpec) {
283         /* Build SymKey id */
284         ASN1ObjectIdentifier myId = GordianKeyAlgId.appendSymKeyOID(CIPHEROID.branch("1"), true, pSpec.getKeyType());
285 
286         /* Add mode */
287         final GordianCipherMode myMode = pSpec.getCipherMode();
288         myId = myId.branch(Integer.toString(myMode.ordinal() + 1));
289 
290         /* Add padding if necessary */
291         if (myMode.hasPadding()) {
292             myId = myId.branch(Integer.toString(pSpec.getPadding().ordinal() + 1));
293         }
294 
295         /* Add the spec to the maps */
296         addToMaps(pSpec, new AlgorithmIdentifier(myId, DERNull.INSTANCE));
297     }

299     /**
300      * Add streamCipherSpec to map if supported and not already present.
301      *
302      * @param pSpec the streamCipherSpec
303      */
304     private void ensureStreamCipher(final GordianStreamCipherSpec pSpec) {
305         /* If the cipher is not already known */
306         if (!theSpecMap.containsKey(pSpec)) {
307             addStreamCipher(pSpec);
308         }
309     }

311     /**
312      * Create Identifier for a streamCipherSpec.
313      *
314      * @param pSpec the cipherSpec
315      */
316     private void addStreamCipher(final GordianStreamCipherSpec pSpec) {
317         /* Create a branch for cipher based on the KeyType */
318         final GordianStreamKeySpec mySpec = pSpec.getKeyType();
319         ASN1ObjectIdentifier myId = GordianKeyAlgId.appendStreamKeyOID(CIPHEROID.branch("2"), mySpec);
320         if (mySpec.supportsAEAD()) {
321             myId = myId.branch(Integer.toString(pSpec.isAEAD() ? 2 : 1));
322         }
323 
324         /* Add the spec to the maps */
325         addToMaps(pSpec, new AlgorithmIdentifier(myId, DERNull.INSTANCE));
326     }
327 
328     /**
329      * Add symCipherSpec to maps.
330      *
331      * @param pSpec       the cipherSpec
332      * @param pIdentifier the identifier
333      */
334     private void addToMaps(final GordianCipherSpec<?> pSpec,
335                            final AlgorithmIdentifier pIdentifier) {
336         theSpecMap.put(pSpec, pIdentifier);
337         theIdentifierMap.put(pIdentifier, pSpec);
338     }
