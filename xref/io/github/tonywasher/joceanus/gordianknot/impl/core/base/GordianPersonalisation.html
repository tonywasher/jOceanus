101     /**
102      * Obtain an array of digests for personalisation.
103      *
104      * @param pFactory the factory
105      * @return the digests
106      * @throws GordianException on error
107      */
108     private static GordianDigest[] determineDigests(final GordianFactory pFactory) throws GordianException {
109         /* Access digest factory */
110         final GordianDigestFactory myFactory = pFactory.getDigestFactory();
111 
112         /* Initialise variables */
113         final GordianDigestType[] myTypes = GordianDigestType.values();
114         final GordianDigest[] myDigests = new GordianDigest[myTypes.length];
115         int myLen = 0;
116 
117         /* Loop through the digestTypes */
118         for (final GordianDigestType myType : GordianDigestType.values()) {
119             /* Add the digest if it is available as 256-bit and supports largeData */
120             final GordianDigestSpec mySpec = new GordianDigestSpec(myType, HASH_LEN);
121             if (myType.supportsLargeData()
122                     && myFactory.supportedDigestSpecs().test(mySpec)) {
123                 myDigests[myLen++] = myFactory.createDigest(mySpec);
124             }
125         }
126 
127         /* Return the array */
128         return Arrays.copyOf(myDigests, myLen);
129     }

141      * determine hostName.
142      *

145     private static String getHostName() {
146         /* Protect against exceptions */
147         try {
148             final InetAddress myAddr = InetAddress.getLocalHost();
149             return myAddr.getHostName();
150 
151         } catch (UnknownHostException e) {
152             return "localhost";
153         }

156     /**
157      * Create an array of hashes from personalisation.
158      *
159      * @param pFactory the factory
160      * @return the hashes
161      * @throws GordianException on error
162      */
163     private static byte[][] personalise(final GordianBaseSupplier pFactory) throws GordianException {
164         /* Determine the digests */
165         final GordianDigest[] myDigests = determineDigests(pFactory);
166 
167         /* Allocate buffers */
168         final byte[][] myHashes = new byte[myDigests.length][];
169         final byte[] myConfig = new byte[HASH_LEN.getByteLength()];
170         byte[] myExpanded = null;
171         GordianHKDFParams myParams = null;
172         final byte[] myKeySetVec = pFactory.getKeySetSeed();
173 
174         /* Obtain configuration */
175         byte[] myPhraseBytes = pFactory.getSecuritySeed();
176         if (myPhraseBytes == null) {
177             myPhraseBytes = GordianDataConverter.stringToByteArray(getHostName());
178         }
179 
180         /* Protect against exceptions */
181         try {
182             /* Initialise hashes */
183             for (int i = 0; i < myDigests.length; i++) {
184                 /* Initialise the digests */
185                 final GordianDigest myDigest = myDigests[i];
186                 myDigest.update(BASE_PERSONAL);
187                 myDigest.update(myPhraseBytes);
188 
189                 /* Finish the update and store the buffer */
190                 final byte[] myResult = myDigest.finish();
191                 buildHashResult(myConfig, myResult);
192                 myHashes[i] = myResult;
193             }
194 
195             /* Determine the number of iterations */
196             final int myIterations = pFactory.isInternal()
197                     ? NUM_ITERATIONS
198                     : NUM_ITERATIONS << PHRASE_SHIFT;
199 
200             /* Loop the required amount of times to cross-fertilise */
201             for (int i = 0; i < myIterations; i++) {
202                 iterateHashes(myDigests, myHashes, myConfig);
203             }
204 
205             /* Determine the number of results that we want from HKDF */
206             final int numResults = myKeySetVec != null ? NUM_RESULTS - 1 : NUM_RESULTS;
207 
208             /* Use HKDF to expand to the required length */
209             final GordianDigestSpec mySpec = determineHKDFDigestSpec(pFactory, myConfig);
210             final GordianHKDFEngine myEngine = new GordianHKDFEngine(pFactory, mySpec);
211             myParams = GordianHKDFParams.expandOnly(myConfig, RESULT_LEN * numResults).withInfo(BASE_PERSONAL);
212             myExpanded = myEngine.deriveBytes(myParams);
213 
214             /* Extract the results */
215             final byte[][] myResults = new byte[NUM_RESULTS][];
216             myResults[0] = new byte[RESULT_LEN];
217             System.arraycopy(myExpanded, 0, myResults[0], 0, RESULT_LEN);
218             myResults[1] = new byte[RESULT_LEN];
219             System.arraycopy(myExpanded, RESULT_LEN, myResults[1], 0, RESULT_LEN);
220 
221             /* Handle the keySetVector */
222             if (myKeySetVec != null) {
223                 myResults[2] = myKeySetVec;
224             } else {
225                 myResults[2] = new byte[RESULT_LEN];
226                 System.arraycopy(myExpanded, RESULT_LEN << 1, myResults[2], 0, RESULT_LEN);
227             }
228 
229             /* Return the array */
230             return myResults;
231 
232             /* Clear intermediate arrays */
233         } finally {
234             /* Clear intermediate hashes */
235             for (int i = 0; i < myDigests.length; i++) {
236                 if (myHashes[i] != null) {
237                     Arrays.fill(myHashes[i], (byte) 0);
238                 }
239             }
240 
241             /* Clear intermediate result */
242             Arrays.fill(myConfig, (byte) 0);
243 
244             /* Clear temporary output buffer */
245             if (myExpanded != null) {
246                 Arrays.fill(myExpanded, (byte) 0);
247             }
248 
249             /* Clear HKDF parameters */
250             if (myParams != null) {
251                 myParams.clearParameters();
252             }
253         }
254     }

256     /**
257      * Iterate the hashes.
258      *
259      * @param pDigests the digest array
260      * @param pHashes  the hashes array
261      * @param pResult  the result array
262      * @throws GordianException on error
263      */
264     private static void iterateHashes(final GordianDigest[] pDigests,
265                                       final byte[][] pHashes,
266                                       final byte[] pResult) throws GordianException {
267         /* Update all the digests */
268         for (final GordianDigest myDigest : pDigests) {
269             /* Update with the results */
270             for (int k = 0; k < pDigests.length; k++) {
271                 myDigest.update(pHashes[k]);
272             }
273         }
274 
275         /* Finish all the digests */
276         for (int j = 0; j < pDigests.length; j++) {
277             /* Update with the results */
278             final GordianDigest myDigest = pDigests[j];
279             final byte[] myResult = pHashes[j];
280             myDigest.finish(myResult, 0);
281             buildHashResult(pResult, myResult);
282         }
283     }

336     /**
337      * Obtain mask from personalisation.
338      *
339      * @param pOffSet the offset within the array
340      * @return the result
341      */
342     private int getPersonalisedMask(final int pOffSet) {
343         /* Loop to obtain the personalised byte */
344         int myVal = 0;
345         for (int i = 0, myOffSet = pOffSet; i < Integer.BYTES; i++, myOffSet++) {
346             myVal <<= Byte.SIZE;
347             myVal |= thePersonalisation[myOffSet] & GordianDataConverter.BYTE_MASK;
348         }
349 
350         /* Return the value */
351         return myVal;
352     }

354     /**
355      * Obtain seeded random.
356      *
357      * @param pPrefixId the prefixId
358      * @param pBaseSeed the seed.
359      * @return the random
360      */
361     public Random getSeededRandom(final GordianPersonalId pPrefixId,
362                                   final byte[] pBaseSeed) {
363         /* Build the 48-bit seed and return the seeded random */
364         final long myPrefix = ((long) getPersonalisedInteger(pPrefixId)) << Short.SIZE;
365         final long myBaseSeed = Integer.toUnsignedLong(GordianDataConverter.byteArrayToInteger(pBaseSeed));
366         final long mySeed = myPrefix ^ myBaseSeed;
367         return new Random(mySeed);
368     }

370     /**
371      * Detremine the HKDFDigestSpec.
372      *
373      * @param pFactory  the factory
374      * @param pBaseSeed the seed.
375      * @return the digestSpec
376      */
377     private static GordianDigestSpec determineHKDFDigestSpec(final GordianBaseSupplier pFactory,
378                                                              final byte[] pBaseSeed) {
379         /* Build the 64-bit seed and create the seeded random */
380         final long mySeed = GordianDataConverter.byteArrayToLong(pBaseSeed);
381         final Random myRandom = new Random(mySeed);
382 
383         /* Access the list to select from */
384         final GordianValidator myValidator = pFactory.getValidator();
385         final List<GordianDigestType> myTypes = myValidator.listAllExternalDigestTypes();
386 
387         /* Select from the list */
388         final int myIndex = myRandom.nextInt(myTypes.size());
389         final GordianDigestType myType = myTypes.get(myIndex);
390 
391         /* return the selected digestSpec */
392         return new GordianDigestSpec(myType, GordianLength.LEN_512);
393     }

395     /**
396      * Simple function to combine hashes. Hashes are simply XORed together.
397      *
398      * @param pFirst  the first Hash
399      * @param pSecond the second Hash
400      * @return the combined hash
401      */
402     public static byte[] combineHashes(final byte[] pFirst,
403                                        final byte[] pSecond) {
404         /* Handle nulls */
405         if (pFirst == null) {
406             return pSecond;
407         }
408         if (pSecond == null) {
409             return pFirst;
410         }
411 
412         /* If the target is smaller than the source */
413         byte[] myTarget = pSecond;
414         byte[] mySource = pFirst;
415         if (myTarget.length < mySource.length) {
416             /* Reverse the order to make use of all bits */
417             myTarget = pFirst;
418             mySource = pSecond;
419         }
420 
421         /* Allocate the target as a copy of the input */
422         myTarget = Arrays.copyOf(myTarget, myTarget.length);
423 
424         /* Determine length of operation */
425         final int myLen = mySource.length;
426 
427         /* Loop through the array bytes */
428         for (int i = 0; i < myTarget.length; i++) {
429             /* Combine the bytes */
430             myTarget[i] ^= mySource[i
431                     % myLen];
432         }
433 
434         /* return the array */
435         return myTarget;
436     }

438     /**
439      * Simple function to build a hash result.
440      *
441      * @param pResult the result Hash
442      * @param pHash   the calculated Hash
443      * @throws GordianException on error
444      */
445     public static void buildHashResult(final byte[] pResult,
446                                        final byte[] pHash) throws GordianException {
447         /* If the target is smaller than the source */
448         final int myLen = pResult.length;
449         if (myLen != pHash.length) {
450             throw new GordianDataException("Hashes are different lengths");
451         }
452         /* Loop through the array bytes */
453         for (int i = 0; i < myLen; i++) {
454             /* Combine the bytes */
455             pResult[i] ^= pHash[i];
456         }
457     }
