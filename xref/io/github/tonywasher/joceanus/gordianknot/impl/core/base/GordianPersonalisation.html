104     /**
105      * Obtain an array of digests for personalisation.
106      *
107      * @param pFactory the factory
108      * @return the digests
109      * @throws GordianException on error
110      */
111     private static GordianDigest[] determineDigests(final GordianFactory pFactory) throws GordianException {
112         /* Access digest factory */
113         final GordianDigestFactory myFactory = pFactory.getDigestFactory();
114 
115         /* Initialise variables */
116         final GordianCoreDigestType[] myTypes = GordianCoreDigestType.values();
117         final GordianDigest[] myDigests = new GordianDigest[myTypes.length];
118         int myLen = 0;
119 
120         /* Loop through the digestTypes */
121         final GordianDigestSpecBuilder myBuilder = GordianCoreDigestSpecBuilder.newInstance();
122         for (final GordianCoreDigestType myType : myTypes) {
123             /* Add the digest if it is available as 256-bit and supports largeData */
124             final GordianDigestSpec mySpec = myBuilder.digest(myType.getType(), HASH_LEN);
125             if (myType.supportsLargeData()
126                     && myFactory.supportedDigestSpecs().test(mySpec)) {
127                 myDigests[myLen++] = myFactory.createDigest(mySpec);
128             }
129         }
130 
131         /* Return the array */
132         return Arrays.copyOf(myDigests, myLen);
133     }

145      * determine hostName.
146      *
147      * @return the hostName
148      */
149     private static String getHostName() {
150         /* Protect against exceptions */
151         try {
152             final InetAddress myAddr = InetAddress.getLocalHost();
153             return myAddr.getHostName();
154 
155         } catch (UnknownHostException e) {
156             return "localhost";
157         }

160     /**
161      * Create an array of hashes from personalisation.
162      *
163      * @param pFactory the factory
164      * @return the hashes
165      * @throws GordianException on error
166      */
167     private static byte[][] personalise(final GordianBaseSupplier pFactory) throws GordianException {
168         /* Determine the digests */
169         final GordianDigest[] myDigests = determineDigests(pFactory);
170 
171         /* Allocate buffers */
172         final byte[][] myHashes = new byte[myDigests.length][];
173         final byte[] myConfig = new byte[HASH_LEN.getByteLength()];
174         byte[] myExpanded = null;
175         GordianHKDFParams myParams = null;
176         final byte[] myKeySetVec = pFactory.getKeySetSeed();
177 
178         /* Obtain configuration */
179         byte[] myPhraseBytes = pFactory.getSecuritySeed();
180         if (myPhraseBytes == null) {
181             myPhraseBytes = GordianDataConverter.stringToByteArray(getHostName());
182         }
183 
184         /* Protect against exceptions */
185         try {
186             /* Initialise hashes */
187             for (int i = 0; i < myDigests.length; i++) {
188                 /* Initialise the digests */
189                 final GordianDigest myDigest = myDigests[i];
190                 myDigest.update(BASE_PERSONAL);
191                 myDigest.update(myPhraseBytes);
192 
193                 /* Finish the update and store the buffer */
194                 final byte[] myResult = myDigest.finish();
195                 buildHashResult(myConfig, myResult);
196                 myHashes[i] = myResult;
197             }
198 
199             /* Determine the number of iterations */
200             final int myIterations = pFactory.isInternal()
201                     ? NUM_ITERATIONS
202                     : NUM_ITERATIONS << PHRASE_SHIFT;
203 
204             /* Loop the required amount of times to cross-fertilise */
205             for (int i = 0; i < myIterations; i++) {
206                 iterateHashes(myDigests, myHashes, myConfig);
207             }
208 
209             /* Determine the number of results that we want from HKDF */
210             final int numResults = myKeySetVec != null ? NUM_RESULTS - 1 : NUM_RESULTS;
211 
212             /* Use HKDF to expand to the required length */
213             final GordianDigestSpec mySpec = determineHKDFDigestSpec(pFactory, myConfig);
214             final GordianHKDFEngine myEngine = new GordianHKDFEngine(pFactory, mySpec);
215             myParams = GordianHKDFParams.expandOnly(myConfig, RESULT_LEN * numResults).withInfo(BASE_PERSONAL);
216             myExpanded = myEngine.deriveBytes(myParams);
217 
218             /* Extract the results */
219             final byte[][] myResults = new byte[NUM_RESULTS][];
220             myResults[0] = new byte[RESULT_LEN];
221             System.arraycopy(myExpanded, 0, myResults[0], 0, RESULT_LEN);
222             myResults[1] = new byte[RESULT_LEN];
223             System.arraycopy(myExpanded, RESULT_LEN, myResults[1], 0, RESULT_LEN);
224 
225             /* Handle the keySetVector */
226             if (myKeySetVec != null) {
227                 myResults[2] = myKeySetVec;
228             } else {
229                 myResults[2] = new byte[RESULT_LEN];
230                 System.arraycopy(myExpanded, RESULT_LEN << 1, myResults[2], 0, RESULT_LEN);
231             }
232 
233             /* Return the array */
234             return myResults;
235 
236             /* Clear intermediate arrays */
237         } finally {
238             /* Clear intermediate hashes */
239             for (int i = 0; i < myDigests.length; i++) {
240                 if (myHashes[i] != null) {
241                     Arrays.fill(myHashes[i], (byte) 0);
242                 }
243             }
244 
245             /* Clear intermediate result */
246             Arrays.fill(myConfig, (byte) 0);
247 
248             /* Clear temporary output buffer */
249             if (myExpanded != null) {
250                 Arrays.fill(myExpanded, (byte) 0);
251             }
252 
253             /* Clear HKDF parameters */
254             if (myParams != null) {
255                 myParams.clearParameters();
256             }
257         }
258     }

260     /**
261      * Iterate the hashes.
262      *
263      * @param pDigests the digest array
264      * @param pHashes  the hashes array
265      * @param pResult  the result array
266      * @throws GordianException on error
267      */
268     private static void iterateHashes(final GordianDigest[] pDigests,
269                                       final byte[][] pHashes,
270                                       final byte[] pResult) throws GordianException {
271         /* Update all the digests */
272         for (final GordianDigest myDigest : pDigests) {
273             /* Update with the results */
274             for (int k = 0; k < pDigests.length; k++) {
275                 myDigest.update(pHashes[k]);
276             }
277         }
278 
279         /* Finish all the digests */
280         for (int j = 0; j < pDigests.length; j++) {
281             /* Update with the results */
282             final GordianDigest myDigest = pDigests[j];
283             final byte[] myResult = pHashes[j];
284             myDigest.finish(myResult, 0);
285             buildHashResult(pResult, myResult);
286         }
287     }

340     /**
341      * Obtain mask from personalisation.
342      *
343      * @param pOffSet the offset within the array
344      * @return the result
345      */
346     private int getPersonalisedMask(final int pOffSet) {
347         /* Loop to obtain the personalised byte */
348         int myVal = 0;
349         for (int i = 0, myOffSet = pOffSet; i < Integer.BYTES; i++, myOffSet++) {
350             myVal <<= Byte.SIZE;
351             myVal |= thePersonalisation[myOffSet] & GordianDataConverter.BYTE_MASK;
352         }
353 
354         /* Return the value */
355         return myVal;
356     }

358     /**
359      * Obtain seeded random.
360      *
361      * @param pPrefixId the prefixId
362      * @param pBaseSeed the seed.
363      * @return the random
364      */
365     public Random getSeededRandom(final GordianPersonalId pPrefixId,
366                                   final byte[] pBaseSeed) {
367         /* Build the 48-bit seed and return the seeded random */
368         final long myPrefix = ((long) getPersonalisedInteger(pPrefixId)) << Short.SIZE;
369         final long myBaseSeed = Integer.toUnsignedLong(GordianDataConverter.byteArrayToInteger(pBaseSeed));
370         final long mySeed = myPrefix ^ myBaseSeed;
371         return new Random(mySeed);
372     }

374     /**
375      * Detremine the HKDFDigestSpec.
376      *
377      * @param pFactory  the factory
378      * @param pBaseSeed the seed.
379      * @return the digestSpec
380      */
381     private static GordianDigestSpec determineHKDFDigestSpec(final GordianBaseSupplier pFactory,
382                                                              final byte[] pBaseSeed) {
383         /* Build the 64-bit seed and create the seeded random */
384         final long mySeed = GordianDataConverter.byteArrayToLong(pBaseSeed);
385         final Random myRandom = new Random(mySeed);
386 
387         /* Access the list to select from */
388         final GordianValidator myValidator = pFactory.getValidator();
389         final List<GordianDigestType> myTypes = myValidator.listAllExternalDigestTypes();
390 
391         /* Select from the list */
392         final int myIndex = myRandom.nextInt(myTypes.size());
393         final GordianDigestType myType = myTypes.get(myIndex);
394 
395         /* return the selected digestSpec */
396         final GordianDigestSpecBuilder myBuilder = GordianCoreDigestSpecBuilder.newInstance();
397         return myBuilder.digest(myType, GordianLength.LEN_512);
398     }

400     /**
401      * Simple function to combine hashes. Hashes are simply XORed together.
402      *
403      * @param pFirst  the first Hash
404      * @param pSecond the second Hash
405      * @return the combined hash
406      */
407     public static byte[] combineHashes(final byte[] pFirst,
408                                        final byte[] pSecond) {
409         /* Handle nulls */
410         if (pFirst == null) {
411             return pSecond;
412         }
413         if (pSecond == null) {
414             return pFirst;
415         }
416 
417         /* If the target is smaller than the source */
418         byte[] myTarget = pSecond;
419         byte[] mySource = pFirst;
420         if (myTarget.length < mySource.length) {
421             /* Reverse the order to make use of all bits */
422             myTarget = pFirst;
423             mySource = pSecond;
424         }
425 
426         /* Allocate the target as a copy of the input */
427         myTarget = Arrays.copyOf(myTarget, myTarget.length);
428 
429         /* Determine length of operation */
430         final int myLen = mySource.length;
431 
432         /* Loop through the array bytes */
433         for (int i = 0; i < myTarget.length; i++) {
434             /* Combine the bytes */
435             myTarget[i] ^= mySource[i
436                     % myLen];
437         }
438 
439         /* return the array */
440         return myTarget;
441     }

443     /**
444      * Simple function to build a hash result.
445      *
446      * @param pResult the result Hash
447      * @param pHash   the calculated Hash
448      * @throws GordianException on error
449      */
450     public static void buildHashResult(final byte[] pResult,
451                                        final byte[] pHash) throws GordianException {
452         /* If the target is smaller than the source */
453         final int myLen = pResult.length;
454         if (myLen != pHash.length) {
455             throw new GordianDataException("Hashes are different lengths");
456         }
457         /* Loop through the array bytes */
458         for (int i = 0; i < myLen; i++) {
459             /* Combine the bytes */
460             pResult[i] ^= pHash[i];
461         }
462     }
