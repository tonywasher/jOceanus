99      /**
100      * Prepare for encryption.
101      *
102      * @param pCertificate the target certificate
103      * @return the CRM result
104      * @throws GordianException on error
105      */
106     GordianCRMResult prepareForEncryption(final GordianCoreCertificate pCertificate) throws GordianException {
107         /* Try to send an encrypted proof */
108         final GordianKeyPair myKeyPair = pCertificate.getKeyPair();
109         final GordianKeyPairSpec mySpec = myKeyPair.getKeyPairSpec();
110         final GordianEncryptorSpec myEncSpec = theFactory.getAsyncFactory().getEncryptorFactory().defaultForKeyPair(mySpec);
111         if (myEncSpec != null) {
112             return prepareForEncryption(myEncSpec, pCertificate);
113         }
114 
115         /* Try to send an agreed proof */
116         final GordianAgreementSpec myAgreeSpec = theFactory.getAsyncFactory().getAgreementFactory().defaultForKeyPair(mySpec);
117         if (myAgreeSpec != null) {
118             return prepareAgreedEncryption(myAgreeSpec, pCertificate);
119         }
120 
121         /* Reject the request */
122         throw new GordianDataException("Unable to prepare for encryption");
123     }
124 
125     /**
126      * Prepare an agreed encryption.
127      *
128      * @param pAgreeSpec   the agreementSpec
129      * @param pCertificate the target certificate
130      * @return the CRM result
131      * @throws GordianException on error
132      */
133     private GordianCRMResult prepareAgreedEncryption(final GordianAgreementSpec pAgreeSpec,
134                                                      final GordianCoreCertificate pCertificate) throws GordianException {
135         /* Create the agreement */
136         final GordianAsyncFactory myFactory = theFactory.getAsyncFactory();
137         final GordianCoreAgreementFactory myAgreeFactory = (GordianCoreAgreementFactory) myFactory.getAgreementFactory();
138         final GordianCertificate myCert = myAgreeFactory.newMiniCertificate(SERVER, pCertificate.getKeyPair(),
139                 new GordianKeyPairUsage(GordianKeyPairUse.AGREEMENT));
140         final GordianAgreementParams myParams = myAgreeFactory.newAgreementParams(pAgreeSpec, new GordianKeySetSpec())
141                 .setServerCertificate(myCert);
142         final GordianAgreement myAgree = myAgreeFactory.createAgreement(myParams);
143         final byte[] myHello = myAgree.nextMessage();
144         final GordianKeySet myKeySet = myAgree.getKeySetResult();
145 
146         /* Create the recipient info */
147         final AlgorithmIdentifier myAlgId = myAgreeFactory.getIdentifierForSpec(pAgreeSpec);
148         final IssuerAndSerialNumber myId = new IssuerAndSerialNumber(pCertificate.getIssuer().getName(), pCertificate.getSerialNo());
149         final KeyTransRecipientInfo myKTInfo = new KeyTransRecipientInfo(new RecipientIdentifier(myId), myAlgId, new BEROctetString(myHello));
150         final RecipientInfo myInfo = new RecipientInfo(myKTInfo);
151 
152         /* Return the result */
153         return new GordianCRMResult(myInfo, myKeySet);
154     }

156     /**
157      * Prepare for encryption.
158      *
159      * @param pEncryptSpec the encryptionSpec
160      * @param pCertificate the target certificate
161      * @return the CRM result
162      * @throws GordianException on error
163      */
164     private GordianCRMResult prepareForEncryption(final GordianEncryptorSpec pEncryptSpec,
165                                                   final GordianCoreCertificate pCertificate) throws GordianException {
166         /* Create the random key */
167         final byte[] myKey = createKeyForKeySet();
168 
169         /* Derive the keySet from the key */
170         final GordianKeySet myKeySet = deriveKeySetFromKey(myKey.clone());
171 
172         /* Create recipientInfo */
173         final RecipientInfo myInfo = createRecipientInfo(myKey, pCertificate, pEncryptSpec);
174 
175         /* Return the result */
176         return new GordianCRMResult(myInfo, myKeySet);
177     }

179     /**
180      * Create a random key for KeySet.
181      *
182      * @return the new key
183      */
184     private byte[] createKeyForKeySet() {
185         final byte[] myKey = new byte[GordianLength.LEN_1024.getByteLength()];
186         theFactory.getRandomSource().getRandom().nextBytes(myKey);
187         return myKey;
188     }

190     /**
191      * Derive a keySet from a key.
192      *
193      * @param pKey the key
194      * @return the keySet
195      * @throws GordianException on error
196      */
197     private GordianKeySet deriveKeySetFromKey(final byte[] pKey) throws GordianException {
198         /* Create a new Factory using the key */
199         final GordianParameters myParams = new GordianParameters(pKey);
200         final GordianFactory myFactory = theFactory.newFactory(myParams);
201         return myFactory.getEmbeddedKeySet();
202     }

204     /**
205      * Encrypt the key with a keyPair.
206      *
207      * @param pKey         the key to encrypt
208      * @param pCertificate the target certificate
209      * @param pSpec        the encryptorSpec
210      * @return the encrypted key
211      * @throws GordianException on error
212      */
213     private RecipientInfo createRecipientInfo(final byte[] pKey,
214                                               final GordianCoreCertificate pCertificate,
215                                               final GordianEncryptorSpec pSpec) throws GordianException {
216         /* Create the encrypted key */
217         final GordianAsyncFactory myFactory = theFactory.getAsyncFactory();
218         final GordianCoreEncryptorFactory myEncFactory = (GordianCoreEncryptorFactory) myFactory.getEncryptorFactory();
219         final GordianEncryptor myEncryptor = myEncFactory.createEncryptor(pSpec);
220 
221         /* Create the encrypted key */
222         myEncryptor.initForEncrypt(pCertificate.getKeyPair());
223         final byte[] myEncryptedKey = myEncryptor.encrypt(pKey);
224         Arrays.fill(pKey, (byte) 0);
225 
226         /* Build recipientInfo */
227         final AlgorithmIdentifier myAlgId = myEncFactory.getIdentifierForSpec(pSpec);
228         final IssuerAndSerialNumber myId = new IssuerAndSerialNumber(pCertificate.getIssuer().getName(), pCertificate.getSerialNo());
229         final KeyTransRecipientInfo myKTInfo = new KeyTransRecipientInfo(new RecipientIdentifier(myId), myAlgId, new BEROctetString(myEncryptedKey));
230         return new RecipientInfo(myKTInfo);
231     }

233     /**
234      * Build the encryptedContentInfo for a PrivateKey.
235      *
236      * @param pKeySet        the keySet to encrypt with
237      * @param pPKCS8Encoding the PKCS8Encoded privateKey
238      * @param pCertificate   the local certificate
239      * @return the encryptedContentInfo
240      * @throws GordianException on error
241      */
242     public static EncryptedContentInfo buildEncryptedContentInfo(final GordianKeySet pKeySet,
243                                                                  final PKCS8EncodedKeySpec pPKCS8Encoding,
244                                                                  final GordianCertificate pCertificate) throws GordianException {
245         /* Protect against exceptions */
246         try {
247             /* Obtain the PrivateKeyInfo */
248             final PrivateKeyInfo myInfo = PrivateKeyInfo.getInstance(pPKCS8Encoding.getEncoded());
249             final EncKeyWithID myKeyWithId = new EncKeyWithID(myInfo, new GeneralName(pCertificate.getSubject().getName()));
250             final byte[] myData = pKeySet.encryptBytes(myKeyWithId.getEncoded());
251             final GordianKeySetSpecASN1 myASN1 = new GordianKeySetSpecASN1(pKeySet.getKeySetSpec());
252             final AlgorithmIdentifier myAlgId = myASN1.getAlgorithmId();
253             return new EncryptedContentInfo(CRMFObjectIdentifiers.id_ct_encKeyWithID, myAlgId, new BEROctetString(myData));
254 
255         } catch (IOException e) {
256             throw new GordianIOException("Failed to create EncryptedContentInfo", e);
257         }
258     }

260     /**
261      * Build the encryptedContentInfo.
262      *
263      * @param pKeySet      the keySet to encrypt with
264      * @param pCertificate the certificate to encrypt
265      * @return the encryptedContentInfo
266      * @throws GordianException on error
267      */
268     public static EncryptedContentInfo buildEncryptedContentInfo(final GordianKeySet pKeySet,
269                                                                  final GordianCertificate pCertificate) throws GordianException {
270         /* Obtain the encrypted certificate */
271         final byte[] myData = pKeySet.encryptBytes(pCertificate.getEncoded());
272         final GordianKeySetSpecASN1 myASN1 = new GordianKeySetSpecASN1(pKeySet.getKeySetSpec());
273         final AlgorithmIdentifier myAlgId = myASN1.getAlgorithmId();
274         return new EncryptedContentInfo(PKCSObjectIdentifiers.x509Certificate, myAlgId, new BEROctetString(myData));
275     }

277     /**
278      * Derive the keySet via a keyPairSet issuer.
279      *
280      * @param pRecInfo     the recipient info
281      * @param pCertificate the receiving certificate
282      * @param pKeyPair     the keyPair
283      * @return the keySet
284      * @throws GordianException on error
285      */
286     public GordianKeySet deriveKeySetFromRecInfo(final KeyTransRecipientInfo pRecInfo,
287                                                  final GordianCertificate pCertificate,
288                                                  final GordianKeyPair pKeyPair) throws GordianException {
289         /* Extract details */
290         final AlgorithmIdentifier myAlgId = pRecInfo.getKeyEncryptionAlgorithm();
291         final byte[] myEncryptedKey = pRecInfo.getEncryptedKey().getOctets();
292 
293         /* Derive the keySet appropriately */
294         final GordianKeyPairUsage myUsage = pCertificate.getUsage();
295         return myUsage.hasUse(GordianKeyPairUse.KEYENCRYPT)
296                 ? deriveEncryptedKeySet(pKeyPair, myAlgId, myEncryptedKey)
297                 : deriveAgreedKeySet(pKeyPair, myEncryptedKey);
298     }
299 
300     /**
301      * Derive an encrypted keySet.
302      *
303      * @param pKeyPair      the keyPair
304      * @param pAlgId        the algorithm Identifier
305      * @param pEncryptedKey the encrypted key
306      * @return the derived keySet
307      * @throws GordianException on error
308      */
309     private GordianKeySet deriveEncryptedKeySet(final GordianKeyPair pKeyPair,
310                                                 final AlgorithmIdentifier pAlgId,
311                                                 final byte[] pEncryptedKey) throws GordianException {
312         /* Handle decryption */
313         final GordianAsyncFactory myFactory = theFactory.getAsyncFactory();
314         final GordianCoreEncryptorFactory myEncFactory = (GordianCoreEncryptorFactory) myFactory.getEncryptorFactory();
315         final GordianEncryptorSpec myEncSpec = myEncFactory.getSpecForIdentifier(pAlgId);
316         final GordianEncryptor myEncryptor = myEncFactory.createEncryptor(myEncSpec);
317         myEncryptor.initForDecrypt(pKeyPair);
318         final byte[] myKey = myEncryptor.decrypt(pEncryptedKey);
319         final GordianKeySet myKeySet = deriveKeySetFromKey(myKey);
320         Arrays.fill(myKey, (byte) 0);
321         return myKeySet;
322     }
323 
324     /**
325      * Derive an agreed keySet.
326      *
327      * @param pKeyPair the keyPair
328      * @param pHello   the clientHello
329      * @return the derived keySet
330      * @throws GordianException on error
331      */
332     private GordianKeySet deriveAgreedKeySet(final GordianKeyPair pKeyPair,
333                                              final byte[] pHello) throws GordianException {
334         /* Handle agreement */
335         final GordianAsyncFactory myFactory = theFactory.getAsyncFactory();
336         final GordianAgreementFactory myAgreeFactory = myFactory.getAgreementFactory();
337         final GordianCertificate myCert = myAgreeFactory.newMiniCertificate(SERVER, pKeyPair, new GordianKeyPairUsage(GordianKeyPairUse.AGREEMENT));
338         final GordianAgreement myAgree = myAgreeFactory.parseAgreementMessage(pHello);
339         final GordianAgreementParams myParams = myAgree.getAgreementParams().setServerCertificate(myCert);
340         myAgree.updateParams(myParams);
341         return myAgree.getKeySetResult();
342     }
