112     /**
113      * Create a Certificate request.
114      *
115      * @param pKeyPair   the keyStore entry
116      * @param pRequestId the reqId
117      * @return the certificate request message
118      * @throws GordianException on error
119      */
120     public CertReqMsg createCertificateRequest(final GordianKeyStorePair pKeyPair,
121                                                final int pRequestId) throws GordianException {
122         /* Access the certificate */
123         final GordianCoreCertificate myCert = (GordianCoreCertificate) pKeyPair.getCertificateChain().get(0);
124 
125         /* Create the Certificate request */
126         final CertRequest myCertReq = createCertRequest(myCert, pRequestId);
127 
128         /* Create the ProofOfPossession */
129         final ProofOfPossession myProof = createKeyPairProofOfPossession(pKeyPair, myCert, myCertReq);
130 
131         /* Create control if necessary */
132         AttributeTypeAndValue[] myAttrs = null;
133         final X500Name myName = myCert.getSubjectName();
134         final byte[] myMACSecret = theGateway.getMACSecret(myName);
135         if (myMACSecret != null) {
136             /* Create the PKMACValue Control */
137             final PKMACValue myMACValue = createPKMACValue(myMACSecret, myCertReq.getCertTemplate().getPublicKey());
138             myAttrs = new AttributeTypeAndValue[]{new AttributeTypeAndValue(MACVALUEATTROID, myMACValue)};
139         }
140 
141         /* Create a CRMF */
142         return new CertReqMsg(myCertReq, myProof, myAttrs);
143     }

145     /**
146      * Create a Certificate request.
147      *
148      * @param pCertificate the local certificate
149      * @param pRequestId   the reqId
150      * @return the certificate request
151      * @throws GordianException on error
152      */
153     private static CertRequest createCertRequest(final GordianCoreCertificate pCertificate,
154                                                  final int pRequestId) throws GordianException {
155         /* Protect against exceptions */
156         try {
157             /* Create the Certificate template Builder */
158             final CertTemplateBuilder myBuilder = new CertTemplateBuilder();
159 
160             /* Record certificate name */
161             myBuilder.setSubject(pCertificate.getSubject().getName());
162 
163             /* Record certificate publicKey */
164             final X509EncodedKeySpec myX509 = pCertificate.getX509KeySpec();
165             final SubjectPublicKeyInfo myPublicInfo = SubjectPublicKeyInfo.getInstance(myX509.getEncoded());
166             myBuilder.setPublicKey(myPublicInfo);
167 
168             /* record extensions */
169             final ExtensionsGenerator myGenerator = new ExtensionsGenerator();
170             myGenerator.addExtension(Extension.keyUsage, true, pCertificate.getUsage().getKeyUsage());
171             myGenerator.addExtension(Extension.basicConstraints, false, new BasicConstraints(false));
172             myBuilder.setExtensions(myGenerator.generate());
173 
174             /* Create the Certificate request */
175             return new CertRequest(pRequestId, myBuilder.build(), null);
176 
177         } catch (IOException e) {
178             throw new GordianIOException("Failed to create Certificate request", e);
179         }
180     }

182     /**
183      * Create KeyPair Proof of Possession.
184      *
185      * @param pKeyPair     the keyStore entry
186      * @param pCertificate the local certificate
187      * @param pCertRequest the certificate request
188      * @return the proof of possession
189      * @throws GordianException on error
190      */
191     private ProofOfPossession createKeyPairProofOfPossession(final GordianKeyStorePair pKeyPair,
192                                                              final GordianCoreCertificate pCertificate,
193                                                              final CertRequest pCertRequest) throws GordianException {
194         /* Try to send a signed proof */
195         final GordianKeyPair myKeyPair = pKeyPair.getKeyPair();
196         final GordianKeyPairSpec mySpec = myKeyPair.getKeyPairSpec();
197         final GordianSignatureSpec mySignSpec = theGateway.getFactory().getAsyncFactory().getSignatureFactory().defaultForKeyPair(mySpec);
198         if (mySignSpec != null) {
199             return createKeyPairSignedProof(myKeyPair, mySignSpec, pCertRequest);
200         }
201 
202         /* Send encrypted key via targeted encryption or request encrypted certificate */
203         final GordianCoreCertificate myTarget = theGateway.getTarget();
204         return myTarget != null
205                 ? createTargetedProofOfPossession(myKeyPair, pCertificate)
206                 : new ProofOfPossession(ProofOfPossession.TYPE_KEY_ENCIPHERMENT, new POPOPrivKey(SubsequentMessage.encrCert));
207     }
208 
209     /**
210      * Create Targeted Proof of Possession.
211      *
212      * @param pKeyPair     the keyPair
213      * @param pCertificate the local certificate
214      * @return the proof of possession
215      * @throws GordianException on error
216      */
217     private ProofOfPossession createTargetedProofOfPossession(final GordianKeyPair pKeyPair,
218                                                               final GordianCoreCertificate pCertificate) throws GordianException {
219         /* Obtain the PKCS8Encoding of the private key */
220         final GordianKeyPairFactory myFactory = theGateway.getFactory().getAsyncFactory().getKeyPairFactory();
221         final GordianKeyPairSpec mySpec = pKeyPair.getKeyPairSpec();
222         final GordianKeyPairGenerator myGenerator = myFactory.getKeyPairGenerator(mySpec);
223         final PKCS8EncodedKeySpec myPKCS8Encoding = myGenerator.getPKCS8Encoding(pKeyPair);
224 
225         /* Prepare for encryption */
226         final GordianCRMEncryptor myEncryptor = theGateway.getEncryptor();
227         final GordianCoreCertificate myTarget = theGateway.getTarget();
228         final GordianCRMResult myResult = myEncryptor.prepareForEncryption(myTarget);
229 
230         /* Derive the keySet from the key */
231         final GordianKeySet myKeySet = myResult.getKeySet();
232 
233         /* Create the encrypted data */
234         final EncryptedContentInfo myInfo = GordianCRMEncryptor.buildEncryptedContentInfo(myKeySet, myPKCS8Encoding, pCertificate);
235 
236         /* Create the Proof of possession */
237         final EnvelopedData myEnvData = new EnvelopedData(null, new BERSet(myResult.getRecipient()), myInfo, (BERSet) null);
238         return new ProofOfPossession(ProofOfPossession.TYPE_KEY_ENCIPHERMENT, new XPOPOPrivKey(myEnvData));
239     }

241     /**
242      * Create a Signed KeyPair Proof of Possession.
243      *
244      * @param pKeyPair     the keyPair
245      * @param pSignSpec    the signatureSpec
246      * @param pCertRequest the certificate request
247      * @return the proof of possession
248      * @throws GordianException on error
249      */
250     ProofOfPossession createKeyPairSignedProof(final GordianKeyPair pKeyPair,
251                                                final GordianSignatureSpec pSignSpec,
252                                                final CertRequest pCertRequest) throws GordianException {
253         /* Create the signer */
254         final GordianAsyncFactory myFactory = theGateway.getFactory().getAsyncFactory();
255         final GordianCoreSignatureFactory mySignFactory = (GordianCoreSignatureFactory) myFactory.getSignatureFactory();
256         final GordianSignature mySigner = mySignFactory.createSigner(pSignSpec);
257         final AlgorithmIdentifier myAlgId = mySignFactory.getIdentifierForSpecAndKeyPair(pSignSpec, pKeyPair);
258 
259         /* Build the signed proof */
260         return createSignedProof(pKeyPair, myAlgId, mySigner, pCertRequest);
261     }

263     /**
264      * Create a Signed Proof of Possession.
265      *
266      * @param pKeyPair     the keyPair
267      * @param pAlgId       the algorithmId
268      * @param pSigner      the signer
269      * @param pCertRequest the certificate request
270      * @return the proof of possession
271      * @throws GordianException on error
272      */
273     private static ProofOfPossession createSignedProof(final GordianKeyPair pKeyPair,
274                                                        final AlgorithmIdentifier pAlgId,
275                                                        final GordianSignature pSigner,
276                                                        final CertRequest pCertRequest) throws GordianException {
277         /* Protect against exceptions */
278         try {
279             /* Create the signature */
280             pSigner.initForSigning(GordianSignParams.keyPair(pKeyPair));
281             pSigner.update(pCertRequest.getEncoded());
282             final byte[] mySignature = pSigner.sign();
283 
284             /* Build the signing key */
285             final POPOSigningKey myKey = new POPOSigningKey(null, pAlgId, new DERBitString(mySignature));
286             return new ProofOfPossession(myKey);
287 
288         } catch (IOException e) {
289             throw new GordianIOException("Failed to create Signed Proof of Possession", e);
290         }
291     }

293     /**
294      * Extended POPOPrivKey to allow encryptedKey.
295      */
296     static class XPOPOPrivKey extends POPOPrivKey {
297         /**
298          * The encrypted key.
299          */
300         private final EnvelopedData theKey;
301 
302         /**
303          * Constructor.
304          *
305          * @param pEncryptedKey the encryptedKey.
306          */
307         XPOPOPrivKey(final EnvelopedData pEncryptedKey) {
308             super((PKMACValue) null);
309             theKey = pEncryptedKey;
310         }
311 
312         @Override
313         public ASN1Primitive toASN1Primitive() {
314             return new DERTaggedObject(false, encryptedKey, theKey);
315         }
316     }

319      * Create MACValue.
320      *
321      * @param pSecret the secret
322      * @param pData   the data to calculate over
323      * @return the MACValue
324      * @throws GordianException on error
325      */
326     public PKMACValue createPKMACValue(final byte[] pSecret,
327                                        final ASN1Object pData) throws GordianException {
328         final PBMParameter myParams = generatePBMParameters();
329         return calculatePKMacValue(theGateway.getFactory(), pSecret, pData, myParams);
330     }

332     /**
333      * Check PKMACValue.
334      *
335      * @param pSecret   the secret
336      * @param pData     the data to calculate over
337      * @param pMACValue the supplied MACValue
338      * @throws GordianException on error
339      */
340     public void checkPKMACValue(final byte[] pSecret,
341                                 final ASN1Object pData,
342                                 final PKMACValue pMACValue) throws GordianException {
343         final PBMParameter myParams = PBMParameter.getInstance(pMACValue.getAlgId().getParameters());
344         final PKMACValue myMACValue = calculatePKMacValue(theGateway.getFactory(), pSecret, pData, myParams);
345         if (!pMACValue.equals(myMACValue)) {
346             throw new GordianDataException("Invalid PKMacValue");
347         }
348     }

350     /**
351      * Create PBMParameters.
352      *
353      * @return the PBMParameters
354      */
355     private PBMParameter generatePBMParameters() {
356         /* Create the salt value */
357         final byte[] mySalt = new byte[GordianLength.LEN_256.getByteLength()];
358         final GordianRandomSource mySource = theGateway.getFactory().getRandomSource();
359         mySource.getRandom().nextBytes(mySalt);
360 
361         /* Access algorithm ids */
362         final AlgorithmIdentifier myHashId = new AlgorithmIdentifier(NISTObjectIdentifiers.id_sha256, DERNull.INSTANCE);
363         final AlgorithmIdentifier myMacId = new AlgorithmIdentifier(PKCSObjectIdentifiers.id_hmacWithSHA256, DERNull.INSTANCE);
364 
365         /* Create the PBMParameter */
366         return new PBMParameter(mySalt, myHashId, PBM_ITERATIONS, myMacId);
367     }

369     /**
370      * Calculate PKMacValue.
371      *
372      * @param pFactory the factory
373      * @param pSecret  the secret value
374      * @param pObject  the object
375      * @param pParams  the PBM Parameters
376      * @return the PKMacValue
377      * @throws GordianException on error
378      */
379     private static PKMACValue calculatePKMacValue(final GordianBaseFactory pFactory,
380                                                   final byte[] pSecret,
381                                                   final ASN1Object pObject,
382                                                   final PBMParameter pParams) throws GordianException {
383         /* Protect against exceptions */
384         try {
385             /* Create the digest */
386             final GordianCoreDigestFactory myDigests = (GordianCoreDigestFactory) pFactory.getDigestFactory();
387             final GordianDigestSpec myDigestSpec = myDigests.getDigestSpecForIdentifier(pParams.getOwf());
388             final GordianDigest myDigest = myDigests.createDigest(myDigestSpec);
389 
390             /* Run through the first iteration */
391             myDigest.update(pSecret);
392             myDigest.update(pParams.getSalt().getOctets());
393             byte[] myKey = new byte[myDigest.getDigestSize()];
394             myKey = myDigest.finish(myKey);
395 
396             /* Loop through the remaining iterations */
397             final int numIterations = pParams.getIterationCount().intValueExact() - 1;
398             for (int i = 0; i < numIterations; i++) {
399                 myKey = myDigest.finish(myKey);
400             }
401 
402             /* Create the mac */
403             final GordianMacFactory myMacs = pFactory.getMacFactory();
404             final GordianMacSpec myMacSpec = (GordianMacSpec) pFactory.getKeySpecForIdentifier(pParams.getMac());
405             final GordianMac myMac = myMacs.createMac(myMacSpec);
406             myMac.initKeyBytes(myKey);
407 
408             /* Create the result */
409             myMac.update(pObject.toASN1Primitive().getEncoded());
410             final byte[] myResult = myMac.finish();
411             return new PKMACValue(pParams, new DERBitString(myResult));
412 
413             /* Handle exceptions */
414         } catch (IOException e) {
415             throw new GordianIOException("Failed to calculate PKMACValue", e);
416         }
417     }

419     /**
420      * Calculate AckValue.
421      *
422      * @param pCertificate the certificate
423      * @return the AckValue
424      * @throws GordianException on error
425      */
426     public byte[] calculateAckValue(final GordianCoreCertificate pCertificate) throws GordianException {
427         /* Access the MACSecret */
428         final X500Name mySubject = pCertificate.getSubjectName();
429         final byte[] myMACSecret = theGateway.getMACSecret(mySubject);
430 
431         /* Create the digest */
432         final GordianBaseFactory myFactory = theGateway.getFactory();
433         final GordianDigestFactory myDigests = myFactory.getDigestFactory();
434         final GordianDigest myDigest = myDigests.createDigest(GordianDigestSpecBuilder.sha2(GordianLength.LEN_256));
435 
436         /* Calculate the digest */
437         if (myMACSecret != null) {
438             myDigest.update(myMACSecret);
439         }
440         myDigest.update(pCertificate.getEncoded());
441         return myDigest.finish();
442     }
