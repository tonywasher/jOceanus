52  public class GordianCoreAgreementCalculator {
53      /**
54       * The factory.
55       */
56      private final GordianBaseFactory theFactory;

58      /**
59       * The state.
60       */
61      private final GordianCoreAgreementState theState;

63      /**
64       * constructor.
65       *
66       * @param pFactory the factory
67       * @param pState   the state
68       */
69      GordianCoreAgreementCalculator(final GordianBaseFactory pFactory,
70                                     final GordianCoreAgreementState pState) {
71          theFactory = pFactory;
72          theState = pState;
73      }

75      /**
76       * Process the secret.
77       *
78       * @param pSecret     the secret
79       * @param pResultType the resultType
80       * @return the result
81       * @throws GordianException on error
82       */
83      Object processSecret(final byte[] pSecret,
84                           final Object pResultType) throws GordianException {
85          /* If the resultType is a FactoryType */
86          if (pResultType instanceof GordianFactoryType myType) {
87              /* derive the factory */
88              return deriveFactory(myType, pSecret);
89  
90              /* If the resultType is a KeySetSpec */
91          } else if (pResultType instanceof GordianKeySetSpec mySpec) {
92              /* Derive the keySet */
93              return deriveKeySet(mySpec, pSecret);
94  
95              /* If the resultType is a SymCipherSpec */
96          } else if (pResultType instanceof GordianSymCipherSpec mySpec) {
97              /* Derive the key */
98              return deriveCipher(mySpec, pSecret);
99  
100             /* If the resultType is a StreamCipherSpec */
101         } else if (pResultType instanceof GordianStreamCipherSpec mySpec) {
102             /* Derive the key */
103             return deriveCipher(mySpec, pSecret);
104 
105             /* If the resultType is an Integer */
106         } else if (pResultType instanceof Integer myLength) {
107             /* Derive the key */
108             return deriveBytes(pSecret, myLength);
109         }
110 
111         /* Return the raw secret */
112         return Arrays.clone(pSecret);
113     }

115     /**
116      * Derive factory from the secret.
117      *
118      * @param pFactoryType the factoryType
119      * @param pSecret      the secret
120      * @return the factory
121      * @throws GordianException on error
122      */
123     private GordianFactory deriveFactory(final GordianFactoryType pFactoryType,
124                                          final byte[] pSecret) throws GordianException {
125         /* Ensure that we clear out the seed */
126         byte[] mySeed = null;
127         try {
128             /* Calculate the seed */
129             mySeed = calculateDerivedSecret(pSecret, GordianDerivationId.FACTORY, GordianParameters.SEED_LEN.getByteLength());
130 
131             /* Create a new Factory using the phrase */
132             final GordianParameters myParams = new GordianParameters(pFactoryType, mySeed);
133             return theFactory.newFactory(myParams);
134 
135             /* Clear buffer */
136         } finally {
137             if (mySeed != null) {
138                 Arrays.fill(mySeed, (byte) 0);
139             }
140         }
141     }
142 
143     /**
144      * Derive a keySet from the secret.
145      *
146      * @param pSpec   the keySetSpec
147      * @param pSecret the secret
148      * @return the derived keySet
149      * @throws GordianException on error
150      */
151     private GordianKeySet deriveKeySet(final GordianKeySetSpec pSpec,
152                                        final byte[] pSecret) throws GordianException {
153         /* Derive a shared factory */
154         final GordianFactory myFactory = deriveFactory(GordianFactoryType.BC, pSecret);
155 
156         /* Ensure that we clear out the secret */
157         byte[] mySecret = null;
158         try {
159             /* Calculate the secret */
160             mySecret = calculateDerivedSecret(pSecret, GordianDerivationId.KEYSET, GordianParameters.SECRET_LEN.getByteLength());
161 
162             /* Derive the keySet */
163             final GordianCoreKeySetFactory myKeySets = (GordianCoreKeySetFactory) myFactory.getKeySetFactory();
164             final GordianCoreKeySet myKeySet = myKeySets.createKeySet(pSpec);
165             myKeySet.buildFromSecret(mySecret);
166             return myKeySet;
167 
168         } finally {
169             if (mySecret != null) {
170                 Arrays.fill(mySecret, (byte) 0);
171             }
172         }
173     }

175     /**
176      * Derive a symKeyCipher pair from the secret.
177      *
178      * @param pCipherSpec the cipherSpec
179      * @param pSecret     the secret
180      * @return the ciphers
181      * @throws GordianException on error
182      */
183     private GordianSymCipher[] deriveCipher(final GordianSymCipherSpec pCipherSpec,
184                                             final byte[] pSecret) throws GordianException {
185         /* Derive a shared factory */
186         final GordianFactory myFactory = deriveFactory(GordianFactoryType.BC, pSecret);
187 
188         /* Generate the key */
189         final GordianKey<GordianSymKeySpec> myKey = deriveKey(myFactory, pCipherSpec.getKeyType(), pSecret);
190 
191         /* Create the cipher and initialise with key */
192         final GordianCipherFactory myCiphers = myFactory.getCipherFactory();
193         final GordianSymCipher myOutCipher = myCiphers.createSymKeyCipher(pCipherSpec);
194         final GordianSymCipher myInCipher = myCiphers.createSymKeyCipher(pCipherSpec);
195 
196         /* If we need an IV */
197         if (pCipherSpec.needsIV()) {
198             /* Calculate the IV */
199             final byte[] myIV = deriveIV(pSecret, pCipherSpec.getIVLength());
200 
201             /* Initialise the ciphers */
202             final GordianCipherParameters myParms = GordianCipherParameters.keyAndNonce(myKey, myIV);
203             myOutCipher.initForEncrypt(myParms);
204             myInCipher.initForDecrypt(myParms);
205 
206             /* else no IV */
207         } else {
208             final GordianCipherParameters myParms = GordianCipherParameters.key(myKey);
209             myOutCipher.initForEncrypt(myParms);
210             myInCipher.initForDecrypt(myParms);
211         }
212         return new GordianSymCipher[]{myOutCipher, myInCipher};
213     }

256     /**
257      * Derive a key from the secret.
258      *
259      * @param <T>      the key type
260      * @param pFactory the factory
261      * @param pKeyType the keyType
262      * @param pSecret  the secret
263      * @return the key
264      * @throws GordianException on error
265      */
266     private <T extends GordianKeySpec> GordianKey<T> deriveKey(final GordianFactory pFactory,
267                                                                final T pKeyType,
268                                                                final byte[] pSecret) throws GordianException {
269         /* Ensure that we clear out the key */
270         byte[] myKey = null;
271         try {
272             /* Calculate the secret */
273             myKey = calculateDerivedSecret(pSecret, GordianDerivationId.KEY, pKeyType.getKeyLength().getByteLength());
274 
275             /* Derive the key */
276             final GordianCipherFactory myCiphers = pFactory.getCipherFactory();
277             final GordianCoreKeyGenerator<T> myGenerator = (GordianCoreKeyGenerator<T>) myCiphers.getKeyGenerator(pKeyType);
278             return myGenerator.buildKeyFromBytes(myKey);
279 
280             /* Clear buffers */
281         } finally {
282             if (myKey != null) {
283                 Arrays.fill(myKey, (byte) 0);
284             }
285         }
286     }

296     private byte[] deriveIV(final byte[] pSecret,

302     /**
303      * Derive bytes.
304      *
305      * @param pSecret the secret
306      * @param pLength the length of bytes
307      * @return the factory
308      * @throws GordianException on error
309      */
310     private byte[] deriveBytes(final byte[] pSecret,

312         /* Return the secret */
313         return calculateDerivedSecret(pSecret, GordianDerivationId.BYTES, pLength);
314     }

316     /**
317      * Calculate the derived secret.
318      *
319      * @param pSecret    the secret
320      * @param pId        the derivation Id
321      * @param pResultLen the length of the result
322      * @return the derived secret
323      * @throws GordianException on error
324      */
325     byte[] calculateDerivedSecret(final byte[] pSecret,
326                                   final GordianDerivationId pId,
327                                   final int pResultLen) throws GordianException {
328         /* Build the 64-bit seed and create the seeded random */
329         final long mySeed = GordianDataConverter.byteArrayToLong(pSecret);
330         final Random myRandom = new Random(mySeed);
331 
332         /* Protect against exceptions */
333         final GordianHKDFParams myParams = GordianHKDFParams.extractThenExpand(pResultLen);
334         try {
335             /* Customise the HKDF parameters */
336             final GordianDigestSpec myDigestSpec = new GordianDigestSpec(pId.getDigestType());
337             final byte[] myBytes = new byte[Long.BYTES];
338             myRandom.nextBytes(myBytes);
339             myParams.withIKM(pSecret).withIKM(pId.getId())
340                     .withSalt(theState.getClient().getInitVector())
341                     .withSalt(theState.getServer().getInitVector())
342                     .withInfo(myBytes);
343 
344             /* Derive the bytes */
345             final GordianHKDFEngine myEngine = new GordianHKDFEngine(theFactory, myDigestSpec);
346             return myEngine.deriveBytes(myParams);
347 
348         } finally {
349             myParams.clearParameters();
350         }
351     }

353     /**
354      * Derived secret id.
355      */
356     public enum GordianDerivationId {
357         /**
358          * Factory.
359          */
360         FACTORY("Factory"),
361 
362         /**
363          * KeySet.
364          */
365         KEYSET("KeySet"),
366 
367         /**
368          * Key.
369          */
370         KEY("Key"),
371 
372         /**
373          * IV.
374          */
375         IV("IV"),
376 
377         /**
378          * Bytes.
379          */
380         BYTES("Bytes"),
381 
382         /**
383          * Tags.
384          */
385         TAGS("Tags"),
386 
387         /**
388          * Composite.
389          */
390         COMPOSITE("Composite");
391 
392         /**
393          * The id.
394          */
395         private final byte[] theId;
396 
397         /**
398          * Constructor.
399          *
400          * @param pId the id
401          */
402         GordianDerivationId(final String pId) {
403             theId = pId.getBytes(StandardCharsets.UTF_8);
404         }
405 
406         /**
407          * Obtain the id.
408          *
409          * @return the id.
410          */
411         byte[] getId() {
412             return theId;
413         }
414 
415         /**
416          * Obtain digest type for id.
417          *
418          * @return the id
419          */
420         public GordianDigestType getDigestType() {
421             /*
422              * Assign a different digestType to each Id.
423              * Note that each type must be available as an HMAC in JCA.
424              */
425             switch (this) {
426                 case FACTORY:
427                     return GordianDigestType.SHA3;
428                 case KEYSET:
429                     return GordianDigestType.SHA2;
430                 case KEY:
431                     return GordianDigestType.SKEIN;
432                 case IV:
433                     return GordianDigestType.RIPEMD;
434                 case BYTES:
435                     return GordianDigestType.STREEBOG;
436                 case TAGS:
437                     return GordianDigestType.WHIRLPOOL;
438                 case COMPOSITE:
439                     return GordianDigestType.SM3;
440                 default:
441                     throw new IllegalArgumentException("Invalid ID");
442             }
443         }
444     }
