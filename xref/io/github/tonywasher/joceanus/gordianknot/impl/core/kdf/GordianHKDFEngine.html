17  package io.github.tonywasher.joceanus.gordianknot.impl.core.kdf;
18  
19  import io.github.tonywasher.joceanus.gordianknot.api.base.GordianException;
20  import io.github.tonywasher.joceanus.gordianknot.api.digest.GordianDigest;
21  import io.github.tonywasher.joceanus.gordianknot.api.digest.GordianDigestFactory;
22  import io.github.tonywasher.joceanus.gordianknot.api.digest.GordianDigestSpec;
23  import io.github.tonywasher.joceanus.gordianknot.api.factory.GordianFactory;
24  import io.github.tonywasher.joceanus.gordianknot.api.mac.GordianMac;
25  import io.github.tonywasher.joceanus.gordianknot.api.mac.GordianMacFactory;
26  import io.github.tonywasher.joceanus.gordianknot.api.mac.GordianMacSpec;
27  import io.github.tonywasher.joceanus.gordianknot.api.mac.GordianMacSpecBuilder;
28  
29  import java.util.Arrays;
30  import java.util.Iterator;
31  
32  /**
33   * HKDF functions.
34   */
35  public class GordianHKDFEngine {
36      /**
37       * The Digest.
38       */
39      private final GordianDigest theDigest;
40  
41      /**
42       * The HMac.
43       */
44      private final GordianMac theHMac;
45  
46      /**
47       * Is this a primary engine?
48       */
49      private boolean isPrimary;
50  
51      /**
52       * Constructor.
53       *
54       * @param pFactory    the security factory
55       * @param pDigestSpec the digestSpec
56       * @throws GordianException on error
57       */
58      public GordianHKDFEngine(final GordianFactory pFactory,
59                               final GordianDigestSpec pDigestSpec) throws GordianException {
60          /* Create the digest */
61          final GordianDigestFactory myDigestFactory = pFactory.getDigestFactory();
62          theDigest = myDigestFactory.createDigest(pDigestSpec);
63  
64          /* Create the hMac */
65          final GordianMacFactory myMacFactory = pFactory.getMacFactory();
66          final GordianMacSpec myMacSpec = GordianMacSpecBuilder.hMac(pDigestSpec);
67          theHMac = myMacFactory.createMac(myMacSpec);
68      }
69  
70      /**
71       * Derive bytes.
72       *
73       * @param pParams the parameters
74       * @return the derived bytes
75       * @throws GordianException on error
76       */
77      public byte[] deriveBytes(final GordianHKDFParams pParams) throws GordianException {
78          /* Check parameters */
79          if (pParams == null) {
80              throw new IllegalStateException("Null HKDF parameters");
81          }
82  
83          /* Determine the mode */
84          final GordianHKDFMode myMode = pParams.getMode();
85          byte[] myOutput = null;
86  
87          /* If we should extract the information */
88          if (myMode.doExtract()) {
89              myOutput = extractKeyingMaterial(pParams.saltIterator(), pParams.ikmIterator());
90          }
91  
92          /* If we should expand the information */
93          if (myMode.doExpand()) {
94              /* Save the intermediate value */
95              final byte[] myIntermediate = myOutput;
96  
97              /* Determine PRK and expand it */
98              final byte[] myPRK = myOutput == null ? pParams.getPRK() : myOutput;
99              myOutput = expandKeyingMaterial(pParams, myPRK);
100 
101             /* Clear intermediate result */
102             if (myIntermediate != null) {
103                 Arrays.fill(myIntermediate, (byte) 0);
104             }
105         }
106 
107         /* Return the result */
108         return myOutput;
109     }
110 
111     /**
112      * Extract keying material.
113      *
114      * @param saltIterator the iterator over the salts
115      * @param ikmIterator  the iterator over the initial keying material
116      * @return the extracted material
117      * @throws GordianException on error
118      */
119     private byte[] extractKeyingMaterial(final Iterator<byte[]> saltIterator,
120                                          final Iterator<byte[]> ikmIterator) throws GordianException {
121         /* Determine the key */
122         while (saltIterator.hasNext()) {
123             theDigest.update(saltIterator.next());
124         }
125         theHMac.initKeyBytes(theDigest.finish());
126 
127         /* Extract the keying material */
128         while (ikmIterator.hasNext()) {
129             theHMac.update(ikmIterator.next());
130         }
131         return theHMac.finish();
132     }
133 
134     /**
135      * Expand the pseudo-random key.
136      *
137      * @param pParams the parameters
138      * @param pPRK    the pseudo-random key
139      * @return the expanded material
140      * @throws GordianException on error
141      */
142     private byte[] expandKeyingMaterial(final GordianHKDFParams pParams,
143                                         final byte[] pPRK) throws GordianException {
144         /* Initialise the HMac */
145         theHMac.initKeyBytes(pPRK);
146 
147         /* Allocate the output buffer */
148         int myLenRemaining = pParams.getLength();
149         final byte[] myOutput = new byte[myLenRemaining];
150         final int myHashLen = theHMac.getMacSize();
151 
152         /* Initialise variables */
153         byte[] myInput = null;
154         int myOffset = 0;
155         byte myCounter = 0;
156 
157         /* Loop while we have more data to obtain */
158         while (myLenRemaining > 0) {
159             /* Update with the results of the last loop */
160             if (myInput != null) {
161                 theHMac.update(myInput);
162                 Arrays.fill(myInput, (byte) 0);
163             }
164 
165             /* Update with the info */
166             final Iterator<byte[]> myIterator = pParams.infoIterator();
167             while (myIterator.hasNext()) {
168                 theHMac.update(myIterator.next());
169             }
170 
171             /* Update with the counter */
172             theHMac.update(myCounter++);
173 
174             /* Calculate the hash */
175             myInput = theHMac.finish();
176 
177             /* Output the required bytes */
178             final int myLenToCopy = Math.min(myLenRemaining, myHashLen);
179             System.arraycopy(myInput, 0, myOutput, myOffset, myLenToCopy);
180             myOffset += myLenToCopy;
181             myLenRemaining -= myLenToCopy;
182         }
183 
184         /* Clear final intermediate results */
185         if (myInput != null) {
186             Arrays.fill(myInput, (byte) 0);
187         }
188 
189         /* Return the expanded key */
190         return myOutput;
191     }
192 }
