105     @Override
106     public GordianAgreement createAgreement(final GordianAgreementParams pParams) throws GordianException {
107         /* Check validity of Agreement */
108         final GordianAgreementSpec mySpec = pParams.getAgreementSpec();
109         checkAgreementSpec(mySpec);
110 
111         /* Create the agreement */
112         final GordianCoreAgreementEngine myEngine = createEngine(mySpec);
113         final GordianCoreAgreement myAgreement = new GordianCoreAgreement(myEngine);
114 
115         /* Set the details */
116         myAgreement.setClientCertificate(pParams.getClientCertificate());
117         myAgreement.setServerCertificate(pParams.getServerCertificate());
118         myAgreement.setResultType(pParams.getResultType());
119         myAgreement.setAdditionalData(pParams.getAdditionalData());
120 
121         /* Build the clientHello */
122         myAgreement.buildClientHello();
123 
124         /* Return the agreement */
125         return myAgreement;
126     }

128     @Override
129     public GordianAgreement parseAgreementMessage(final byte[] pMessage) throws GordianException {
130         /* Parse the message */
131         final GordianCoreAgreementMessageASN1 myASN1 = GordianCoreAgreementMessageASN1.getInstance(pMessage);
132 
133         /* Switch on the messageType */
134         switch (myASN1.getMessageType()) {
135             case CLIENTHELLO:
136                 return parseClientHello(myASN1);
137             case SERVERHELLO:
138                 return parseServerHello(myASN1);
139             case CLIENTCONFIRM:
140                 return parseClientConfirm(myASN1);
141             default:
142                 throw new GordianDataException("Unexpected MessageType: " + myASN1.getMessageType());
143         }
144     }

146     /**
147      * Parse a clientHello message.
148      *
149      * @param pClientHello the message
150      * @return the agreement
151      * @throws GordianException error
152      */
153     private GordianAgreement parseClientHello(final GordianCoreAgreementMessageASN1 pClientHello) throws GordianException {
154         /* Create a new agreement */
155         final GordianAgreementSpec mySpec = getSpecForIdentifier(pClientHello.getAgreementId());
156         final GordianCoreAgreementEngine myEngine = createEngine(mySpec);
157         final GordianCoreAgreement myAgreement = new GordianCoreAgreement(myEngine);
158 
159         /* Parse the clientHello */
160         myAgreement.parseClientHello(pClientHello);
161 
162         /* Return the agreement */
163         return myAgreement;
164     }

166     /**
167      * Parse a serverHello message.
168      *
169      * @param pServerHello the message
170      * @return the agreement
171      * @throws GordianException error
172      */
173     private GordianAgreement parseServerHello(final GordianCoreAgreementMessageASN1 pServerHello) throws GordianException {
174         /* Look up the agreement in the cache */
175         final GordianAgreementSpec mySpec = getSpecForIdentifier(pServerHello.getAgreementId());
176         final GordianCoreAgreement myAgreement = theCache.lookUpAgreement(pServerHello.getClientId(), mySpec);
177 
178         /* Process the serverHello */
179         myAgreement.processServerHello(pServerHello);
180 
181         /* Return the agreement */
182         return myAgreement;
183     }

204     /**
205      * Create engine.
206      *
207      * @param pSpec the agreement Spec
208      * @return the engine
209      * @throws GordianException on error
210      */
211     protected GordianCoreAgreementEngine createEngine(final GordianAgreementSpec pSpec) throws GordianException {
212         /* If this is a composite agreement */
213         if (pSpec.getKeyPairSpec().getKeyPairType() == GordianKeyPairType.COMPOSITE) {
214             /* Create an engine for each sub-agreement */
215             final List<GordianAgreementSpec> mySpecs = GordianCoreAgreementComposite.getSubAgreements(pSpec);
216             final List<GordianCoreAgreementEngine> myEngines = new ArrayList<>();
217             for (GordianAgreementSpec mySpec : mySpecs) {
218                 myEngines.add(createEngine(mySpec));
219             }
220             return new GordianCoreAgreementComposite(this, pSpec, myEngines);
221         }
222 
223         /* Unsupported spec */
224         throw new GordianDataException(GordianBaseData.getInvalidText(pSpec));
225     }

263     @Override
264     public void setSigner(final GordianCertificate pSigner,
265                           final GordianSignatureSpec pSignSpec) throws GordianException {
266         /* Check that certificate can sign data */
267         if (pSigner == null || !pSigner.getUsage().hasUse(GordianKeyPairUse.SIGNATURE)) {
268             throw new GordianDataException("Certificate must be capable of signing data");
269         }
270 
271         /* Check that certificate can sign data */
272         final GordianSignatureFactory mySignFactory = theFactory.getAsyncFactory().getSignatureFactory();
273         if (!mySignFactory.validSignatureSpecForKeyPair(pSigner.getKeyPair(), pSignSpec)) {
274             throw new GordianDataException(GordianBaseData.getInvalidText(pSignSpec));
275         }
276 
277         /* Store parameters */
278         theSignerCertificate = pSigner;
279         theSignSpec = pSignSpec;
280     }

282     /**
283      * Check AgreementSpec.
284      *
285      * @param pSpec the agreementSpec
286      * @return true/false
287      */
288     protected boolean validAgreementSpec(final GordianAgreementSpec pSpec) {
289         /* Reject invalid agreementSpec */
290         if (pSpec == null || !pSpec.isValid()) {
291             return false;
292         }
293 
294         /* Check that spec is supported */
295         return pSpec.isSupported();
296     }

298     @Override
299     public boolean validAgreementSpecForKeyPairSpec(final GordianKeyPairSpec pKeyPairSpec,
300                                                     final GordianAgreementSpec pAgreementSpec) {
301         /* Check that the agreementSpec is supported */
302         if (!validAgreementSpec(pAgreementSpec)) {
303             return false;
304         }
305 
306         /* Check agreement matches keySpec */
307         if (!pAgreementSpec.getKeyPairSpec().equals(pKeyPairSpec)) {
308             return false;
309         }
310 
311         /* For Edwards XDH, disallow 512KDF for 25519 and 256KDF for 448 */
312         if (pKeyPairSpec.getKeyPairType() == GordianKeyPairType.XDH) {
313             final GordianEdwardsElliptic myEdwards = pKeyPairSpec.getEdwardsElliptic();
314             switch (pAgreementSpec.getKDFType()) {
315                 case SHA256KDF:
316                 case SHA256CKDF:
317                 case SHA256HKDF:
318                     return myEdwards.is25519();
319                 case SHA512KDF:
320                 case SHA512CKDF:
321                 case SHA512HKDF:
322                     return !myEdwards.is25519();
323                 default:
324                     break;
325             }
326         }
327 
328         /* For Composite AgreementSpec */
329         if (pKeyPairSpec.getKeyPairType() == GordianKeyPairType.COMPOSITE) {
330             /* Access the subSpecs  */
331             final List<GordianAgreementSpec> mySubAgrees = GordianCoreAgreementComposite.getSubAgreements(pAgreementSpec);
332 
333             /* Loop through the subAgreements */
334             for (GordianAgreementSpec mySpec : mySubAgrees) {
335                 if (!validAgreementSpecForKeyPairSpec(mySpec.getKeyPairSpec(), mySpec)) {
336                     return false;
337                 }
338             }
339 
340             /* Check confirmation */
341             if (Boolean.TRUE.equals(pAgreementSpec.withConfirm())
342                     && !pAgreementSpec.getAgreementType().canConfirm()) {
343                 return false;
344             }
345 
346             /* Disallow SM2 with confirm */
347             return pAgreementSpec.getAgreementType() != GordianAgreementType.SM2
348                     || !pAgreementSpec.withConfirm();
349         }
350 
351         /* OK */
352         return true;
353     }

392     @Override
393     public List<GordianAgreementSpec> listAllSupportedAgreements(final GordianKeyPairSpec pKeyPairSpec) {
394         return listPossibleAgreements(pKeyPairSpec)
395                 .stream()
396                 .filter(supportedAgreements())
397                 .filter(s -> validAgreementSpecForKeyPairSpec(pKeyPairSpec, s))
398                 .toList();
399     }

401     /**
402      * Obtain a list of all possible agreements for the keyPairSpec.
403      *
404      * @param pKeyPairSpec the keyPairSpec
405      * @return the list
406      */
407     private List<GordianAgreementSpec> listPossibleAgreements(final GordianKeyPairSpec pKeyPairSpec) {
408         /* Create list */
409         final List<GordianAgreementSpec> myAgreements = new ArrayList<>();
410 
411         /* Switch on keyPairType */
412         switch (pKeyPairSpec.getKeyPairType()) {
413             case RSA:
414                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.KEM));
415                 break;
416             case NEWHOPE:
417             case CMCE:
418             case FRODO:
419             case SABER:
420             case MLKEM:
421             case HQC:
422             case BIKE:
423             case NTRU:
424             case NTRUPRIME:
425                 myAgreements.add(GordianAgreementSpecBuilder.kem(pKeyPairSpec, GordianAgreementKDF.NONE));
426                 break;
427             case EC:
428             case SM2:
429             case GOST2012:
430                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.ANON));
431                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.KEM));
432                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.BASIC));
433                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SIGNED));
434                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED));
435                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED, Boolean.TRUE));
436                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.MQV));
437                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.MQV, Boolean.TRUE));
438                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SM2));
439                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SM2, Boolean.TRUE));
440                 break;
441             case DH:
442             case DSTU4145:
443                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.ANON));
444                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.KEM));
445                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.BASIC));
446                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SIGNED));
447                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED));
448                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED, Boolean.TRUE));
449                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.MQV));
450                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.MQV, Boolean.TRUE));
451                 break;
452             case XDH:
453                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.ANON));
454                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.BASIC));
455                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SIGNED));
456                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED));
457                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED, Boolean.TRUE));
458                 break;
459             case COMPOSITE:
460                 /* Loop through the possible keySpecs for the first key */
461                 final Iterator<GordianKeyPairSpec> myIterator = pKeyPairSpec.keySpecIterator();
462                 for (GordianAgreementSpec mySpec : listPossibleAgreements(myIterator.next())) {
463                     final GordianAgreementSpec myTest = new GordianAgreementSpec(pKeyPairSpec, mySpec.getAgreementType(), mySpec.getKDFType(), mySpec.withConfirm());
464                     if (myTest.isValid()) {
465                         myAgreements.add(myTest);
466                     }
467                 }
468                 break;
469             default:
470                 break;
471         }
472 
473         /* Return the list */
474         return myAgreements;
475     }

489     @Override
490     public GordianAgreementSpec defaultForKeyPair(final GordianKeyPairSpec pKeySpec) {
491         final Iterator<GordianAgreementSpec> myIterator = listAllSupportedAgreements(pKeySpec).iterator();
492         return myIterator.hasNext() ? myIterator.next() : null;
493     }

495     /**
496      * Create list of KDF variants.
497      *
498      * @param pKeyPairSpec   the keyPairSpec
499      * @param pAgreementType the agreementType
500      * @param pConfirm       with key confirmation
501      * @return the list
502      */
503     public static List<GordianAgreementSpec> listAllKDFs(final GordianKeyPairSpec pKeyPairSpec,
504                                                          final GordianAgreementType pAgreementType,
505                                                          final Boolean pConfirm) {
506         /* Create list */
507         final List<GordianAgreementSpec> myAgreements = new ArrayList<>();
508 
509         /* Loop through the KDFs */
510         for (final GordianAgreementKDF myKDF : GordianAgreementKDF.values()) {
511             myAgreements.add(new GordianAgreementSpec(pKeyPairSpec, pAgreementType, myKDF, pConfirm));
512         }
513 
514         /* Return the list */
515         return myAgreements;
516     }
