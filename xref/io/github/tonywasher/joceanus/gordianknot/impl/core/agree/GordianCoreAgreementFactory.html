112     @Override
113     public GordianAgreement createAgreement(final GordianAgreementParams pParams) throws GordianException {
114         /* Check validity of Agreement */
115         final GordianAgreementSpec mySpec = pParams.getAgreementSpec();
116         checkAgreementSpec(mySpec);
117 
118         /* Create the agreement */
119         final GordianCoreAgreementEngine myEngine = createEngine(mySpec);
120         final GordianCoreAgreement myAgreement = new GordianCoreAgreement(myEngine);
121 
122         /* Set the details */
123         myAgreement.setClientCertificate(pParams.getClientCertificate());
124         myAgreement.setServerCertificate(pParams.getServerCertificate());
125         myAgreement.setResultType(pParams.getResultType());
126         myAgreement.setAdditionalData(pParams.getAdditionalData());
127 
128         /* Build the clientHello */
129         myAgreement.buildClientHello();
130 
131         /* Return the agreement */
132         return myAgreement;
133     }

135     @Override
136     public GordianAgreement parseAgreementMessage(final byte[] pMessage) throws GordianException {
137         /* Parse the message */
138         final GordianCoreAgreementMessageASN1 myASN1 = GordianCoreAgreementMessageASN1.getInstance(pMessage);
139 
140         /* Switch on the messageType */
141         return switch (myASN1.getMessageType()) {
142             case CLIENTHELLO -> parseClientHello(myASN1);
143             case SERVERHELLO -> parseServerHello(myASN1);
144             case CLIENTCONFIRM -> parseClientConfirm(myASN1);
145             default -> throw new GordianDataException("Unexpected MessageType: " + myASN1.getMessageType());
146         };
147     }

149     /**
150      * Parse a clientHello message.
151      *
152      * @param pClientHello the message
153      * @return the agreement
154      * @throws GordianException error
155      */
156     private GordianAgreement parseClientHello(final GordianCoreAgreementMessageASN1 pClientHello) throws GordianException {
157         /* Create a new agreement */
158         final GordianAgreementSpec mySpec = getSpecForIdentifier(pClientHello.getAgreementId());
159         final GordianCoreAgreementEngine myEngine = createEngine(mySpec);
160         final GordianCoreAgreement myAgreement = new GordianCoreAgreement(myEngine);
161 
162         /* Parse the clientHello */
163         myAgreement.parseClientHello(pClientHello);
164         myAgreement.setSignerCertificate(theSignSpec, theSignerCertificate);
165 
166         /* Return the agreement */
167         return myAgreement;
168     }

170     /**
171      * Parse a serverHello message.
172      *
173      * @param pServerHello the message
174      * @return the agreement
175      * @throws GordianException error
176      */
177     private GordianAgreement parseServerHello(final GordianCoreAgreementMessageASN1 pServerHello) throws GordianException {
178         /* Look up the agreement in the cache */
179         final GordianAgreementSpec mySpec = getSpecForIdentifier(pServerHello.getAgreementId());
180         final GordianCoreAgreement myAgreement = theCache.lookUpAgreement(pServerHello.getClientId(), mySpec);
181 
182         /* Process the serverHello */
183         myAgreement.processServerHello(pServerHello);
184 
185         /* Return the agreement */
186         return myAgreement;
187     }

208     /**
209      * Create engine.
210      *
211      * @param pSpec the agreement Spec
212      * @return the engine
213      * @throws GordianException on error
214      */
215     protected GordianCoreAgreementEngine createEngine(final GordianAgreementSpec pSpec) throws GordianException {
216         /* If this is a composite agreement */
217         if (pSpec.getKeyPairSpec().getKeyPairType() == GordianKeyPairType.COMPOSITE) {
218             /* Create an engine for each sub-agreement */
219             final List<GordianAgreementSpec> mySpecs = GordianCoreAgreementComposite.getSubAgreements(pSpec);
220             final List<GordianCoreAgreementEngine> myEngines = new ArrayList<>();
221             for (GordianAgreementSpec mySpec : mySpecs) {
222                 myEngines.add(createEngine(mySpec));
223             }
224             return new GordianCoreAgreementComposite(this, (GordianCoreAgreementSpec) pSpec, myEngines);
225         }
226 
227         /* Unsupported spec */
228         throw new GordianDataException(GordianBaseData.getInvalidText(pSpec));
229     }

260     @Override
261     public void setSigner(final GordianCertificate pSigner) throws GordianException {
262         final GordianSignatureFactory mySignFactory = theFactory.getAsyncFactory().getSignatureFactory();
263         final GordianSignatureSpec mySignSpec = pSigner == null ? null : mySignFactory.defaultForKeyPair(pSigner.getKeyPair().getKeyPairSpec());
264         setSigner(pSigner, mySignSpec);
265     }

267     @Override
268     public void setSigner(final GordianCertificate pSigner,
269                           final GordianSignatureSpec pSignSpec) throws GordianException {
270         /* Check that certificate can sign data */
271         if (pSigner == null || !pSigner.getUsage().hasUse(GordianKeyPairUse.SIGNATURE)) {
272             throw new GordianDataException("Certificate must be capable of signing data");
273         }
274 
275         /* Check that certificate can sign data */
276         final GordianSignatureFactory mySignFactory = theFactory.getAsyncFactory().getSignatureFactory();
277         if (!mySignFactory.validSignatureSpecForKeyPair(pSigner.getKeyPair(), pSignSpec)) {
278             throw new GordianDataException(GordianBaseData.getInvalidText(pSignSpec));
279         }
280 
281         /* Store parameters */
282         theSignerCertificate = pSigner;
283         theSignSpec = pSignSpec;
284     }

286     /**
287      * Check AgreementSpec.
288      *
289      * @param pSpec the agreementSpec
290      * @return true/false
291      */
292     protected boolean validAgreementSpec(final GordianAgreementSpec pSpec) {
293         /* Reject invalid agreementSpec */
294         if (pSpec == null || !pSpec.isValid()) {
295             return false;
296         }
297 
298         /* Check that spec is supported */
299         return pSpec.isSupported();
300     }
301 
302     @Override
303     public boolean validAgreementSpecForKeyPairSpec(final GordianKeyPairSpec pKeyPairSpec,
304                                                     final GordianAgreementSpec pAgreementSpec) {
305         /* Check that the agreementSpec is supported */
306         if (!validAgreementSpec(pAgreementSpec)) {
307             return false;
308         }
309 
310         /* Check agreement matches keySpec */
311         if (!pAgreementSpec.getKeyPairSpec().equals(pKeyPairSpec)) {
312             return false;
313         }
314 
315         /* For Edwards XDH, disallow 512KDF for 25519 and 256KDF for 448 */
316         final GordianCoreAgreementSpec myAgreeSpec = (GordianCoreAgreementSpec) pAgreementSpec;
317         final GordianCoreKeyPairSpec myKeySpec = (GordianCoreKeyPairSpec) pKeyPairSpec;
318         if (pKeyPairSpec.getKeyPairType() == GordianKeyPairType.XDH) {
319             final GordianCoreEdwardsSpec myEdwards = myKeySpec.getEdwardsSpec();
320             switch (pAgreementSpec.getKDFType()) {
321                 case SHA256KDF, SHA256CKDF, SHA256HKDF:
322                     return myEdwards.is25519();
323                 case SHA512KDF, SHA512CKDF, SHA512HKDF:
324                     return !myEdwards.is25519();
325                 default:
326                     break;
327             }
328         }
329 
330         /* For Composite AgreementSpec */
331         if (pKeyPairSpec.getKeyPairType() == GordianKeyPairType.COMPOSITE) {
332             /* Access the subSpecs  */
333             final List<GordianAgreementSpec> mySubAgrees = GordianCoreAgreementComposite.getSubAgreements(pAgreementSpec);
334 
335             /* Loop through the subAgreements */
336             for (GordianAgreementSpec mySpec : mySubAgrees) {
337                 if (!validAgreementSpecForKeyPairSpec(mySpec.getKeyPairSpec(), mySpec)) {
338                     return false;
339                 }
340             }
341 
342             /* Check confirmation */
343             if (pAgreementSpec.withConfirm()
344                     && !myAgreeSpec.getCoreAgreementType().canConfirm()) {
345                 return false;
346             }
347 
348             /* Disallow SM2 with confirm */
349             return pAgreementSpec.getAgreementType() != GordianAgreementType.SM2
350                     || !pAgreementSpec.withConfirm();
351         }
352 
353         /* OK */
354         return true;
355     }

394     @Override
395     public List<GordianAgreementSpec> listAllSupportedAgreements(final GordianKeyPairSpec pKeyPairSpec) {
396         return listPossibleAgreements(pKeyPairSpec)
397                 .stream()
398                 .filter(supportedAgreements())
399                 .filter(s -> validAgreementSpecForKeyPairSpec(pKeyPairSpec, s))
400                 .toList();
401     }

414     public GordianAgreementSpec defaultForKeyPair(final GordianKeyPairSpec pKeySpec) {
415         final Iterator<GordianAgreementSpec> myIterator = listAllSupportedAgreements(pKeySpec).iterator();
416         return myIterator.hasNext() ? myIterator.next() : null;
