58  /**
59   * Mappings from EncodedId to AgreementSpec.
60   */
61  public class GordianCoreAgreementAlgId {
62      /**
63       * AgreementOID branch.
64       */
65      private static final ASN1ObjectIdentifier AGREEOID = GordianASN1Util.ASYMOID.branch("2");
66  
67      /**
68       * AgreementSpec OID branch.
69       */
70      private static final ASN1ObjectIdentifier AGREESPECOID = AGREEOID.branch("1");
71  
72      /**
73       * AgreementKeyPairSpec OID branch.
74       */
75      private static final ASN1ObjectIdentifier KEYPAIRSPECOID = AGREEOID.branch("2");
76  
77      /**
78       * Byte Result OID branch.
79       */
80      static final ASN1ObjectIdentifier BYTERESULTOID = AGREEOID.branch("3");
81  
82      /**
83       * Error Result OID branch.
84       */
85      static final ASN1ObjectIdentifier ERRORRESULTOID = AGREEOID.branch("4");
86  
87      /**
88       * Null KeyPairSpec for Partial AgreementSpec.
89       */
90      private static final GordianKeyPairSpec NULLKEYPAIRSPEC = GordianKeyPairSpecBuilder.ed448();
91  
92      /**
93       * The factory.
94       */
95      private final GordianFactory theFactory;
96  
97      /**
98       * Map of AgreementSpec to ObjectIdentifier.
99       */
100     private final Map<GordianAgreementSpec, ASN1ObjectIdentifier> theAgree2IdMap;
101 
102     /**
103      * Map of Identifier to AgreementSpec.
104      */
105     private final Map<ASN1ObjectIdentifier, GordianAgreementSpec> theId2AgreeMap;
106 
107     /**
108      * Map of KeyPairSpec to ObjectIdentifier.
109      */
110     private final Map<GordianKeyPairSpec, ASN1ObjectIdentifier> theKeyPair2IdMap;
111 
112     /**
113      * Map of Identifier to AgreementSpec.
114      */
115     private final Map<ASN1ObjectIdentifier, GordianKeyPairSpec> theId2KeyPairMap;
116 
117     /**
118      * Constructor.
119      *
120      * @param pFactory the factory
121      */
122     GordianCoreAgreementAlgId(final GordianFactory pFactory) {
123         /* Store parameters */
124         theFactory = pFactory;
125 
126         /* Create the maps */
127         theId2AgreeMap = new HashMap<>();
128         theId2KeyPairMap = new HashMap<>();
129         theAgree2IdMap = new HashMap<>();
130         theKeyPair2IdMap = new HashMap<>();
131 
132         /* Add all agreement types and keyPairs */
133         addAllAgreements();
134         addAllKeyPairs();
135     }
136 
137     /**
138      * Add all agreements to the maps.
139      */
140     private void addAllAgreements() {
141         /* Loop through all the Agreement types */
142         for (GordianAgreementType myType : GordianAgreementType.values()) {
143             /* Loop through all the KDF types */
144             for (GordianAgreementKDF myKDF : GordianAgreementKDF.values()) {
145                 /* Add agreements */
146                 addAgreement(new GordianAgreementSpec(NULLKEYPAIRSPEC, myType, myKDF));
147                 addAgreement(new GordianAgreementSpec(NULLKEYPAIRSPEC, myType, myKDF, Boolean.TRUE));
148             }
149         }
150     }
151 
152     /**
153      * Add the agreement to the maps.
154      *
155      * @param pSpec the agreementSpec
156      */
157     private void addAgreement(final GordianAgreementSpec pSpec) {
158         /* Add branch for agreementType */
159         final GordianAgreementType myType = pSpec.getAgreementType();
160         ASN1ObjectIdentifier myId = AGREESPECOID.branch(Integer.toString(myType.ordinal() + 1));
161 
162         /* Add branch for kdfType */
163         final GordianAgreementKDF myKDFType = pSpec.getKDFType();
164         myId = myId.branch(Integer.toString(myKDFType.ordinal() + 1));
165 
166         /* Add branch for confirm (if present) */
167         if (Boolean.TRUE.equals(pSpec.withConfirm())) {
168             myId = myId.branch("1");
169         }
170 
171         /* Add the spec to the maps */
172         addToMaps(pSpec, myId);
173     }
174 
175     /**
176      * Add all keyPairSpecs to the maps.
177      */
178     private void addAllKeyPairs() {
179         /* Loop through all the Agreement types */
180         for (GordianKeyPairSpec mySpec : theFactory.getAsyncFactory().getKeyPairFactory().listPossibleKeySpecs()) {
181             /* Add agreements */
182             addKeyPair(mySpec);
183         }
184         addKeyPair(GordianKeyPairSpecBuilder.composite());
185     }
186 
187     /**
188      * Add the keyPairSpec to the maps.
189      *
190      * @param pSpec the keyPairSpec
191      */
192     private void addKeyPair(final GordianKeyPairSpec pSpec) {
193         /* Add branch for agreementType */
194         final GordianKeyPairType myType = pSpec.getKeyPairType();
195         ASN1ObjectIdentifier myId = KEYPAIRSPECOID.branch(Integer.toString(myType.ordinal() + 1));
196 
197         /* switch on type */
198         switch (myType) {
199             case RSA:
200                 myId = myId.branch(Integer.toString(pSpec.getRSAModulus().ordinal() + 1));
201                 break;
202             case DH:
203                 myId = myId.branch(Integer.toString(pSpec.getDHGroup().ordinal() + 1));
204                 break;
205             case XDH:
206                 myId = myId.branch(Integer.toString(pSpec.getEdwardsElliptic().ordinal() + 1));
207                 break;
208             case EC:
209                 myId = myId.branch(Integer.toString(((GordianDSAElliptic) pSpec.getElliptic()).ordinal() + 1));
210                 break;
211             case SM2:
212                 myId = myId.branch(Integer.toString(((GordianSM2Elliptic) pSpec.getElliptic()).ordinal() + 1));
213                 break;
214             case GOST2012:
215                 myId = myId.branch(Integer.toString(((GordianGOSTElliptic) pSpec.getElliptic()).ordinal() + 1));
216                 break;
217             case DSTU4145:
218                 myId = myId.branch(Integer.toString(((GordianDSTU4145Elliptic) pSpec.getElliptic()).ordinal() + 1));
219                 break;
220             case CMCE:
221                 myId = myId.branch(Integer.toString(pSpec.getCMCEKeySpec().ordinal() + 1));
222                 break;
223             case FRODO:
224                 myId = myId.branch(Integer.toString(pSpec.getFRODOKeySpec().ordinal() + 1));
225                 break;
226             case SABER:
227                 myId = myId.branch(Integer.toString(pSpec.getSABERKeySpec().ordinal() + 1));
228                 break;
229             case MLKEM:
230                 myId = myId.branch(Integer.toString(pSpec.getMLKEMKeySpec().ordinal() + 1));
231                 break;
232             case HQC:
233                 myId = myId.branch(Integer.toString(pSpec.getHQCKeySpec().ordinal() + 1));
234                 break;
235             case BIKE:
236                 myId = myId.branch(Integer.toString(pSpec.getBIKEKeySpec().ordinal() + 1));
237                 break;
238             case NTRU:
239                 myId = myId.branch(Integer.toString(pSpec.getNTRUKeySpec().ordinal() + 1));
240                 break;
241             case NTRUPRIME:
242                 final GordianNTRUPrimeSpec myNTRUPrime = pSpec.getNTRUPrimeKeySpec();
243                 myId = myId.branch(Integer.toString(myNTRUPrime.getType().ordinal() + 1));
244                 myId = myId.branch(Integer.toString(myNTRUPrime.getParams().ordinal() + 1));
245                 break;
246             case NEWHOPE:
247             case COMPOSITE:
248             default:
249                 break;
250         }
251 
252         /* Add the spec to the maps */
253         addToMaps(pSpec, myId);
254     }

256     /**
257      * Determine Identifier for an agreementSpec.
258      *
259      * @param pSpec the agreementSpec
260      * @return the identifier
261      */
262     public AlgorithmIdentifier determineIdentifier(final GordianAgreementSpec pSpec) {
263         /* Add branch for agreementType */
264         final GordianAgreementSpec myPartial = new GordianAgreementSpec(NULLKEYPAIRSPEC, pSpec.getAgreementType(),
265                 pSpec.getKDFType(), pSpec.withConfirm());
266         final ASN1ObjectIdentifier myAgreeId = theAgree2IdMap.get(myPartial);
267 
268         /* Handle composite keyPairSpec */
269         final GordianKeyPairSpec mySpec = pSpec.getKeyPairSpec();
270         if (mySpec.getKeyPairType() == GordianKeyPairType.COMPOSITE) {
271             final ASN1EncodableVector v = new ASN1EncodableVector();
272             final ASN1ObjectIdentifier myId = theKeyPair2IdMap.get(GordianKeyPairSpecBuilder.composite());
273             final Iterator<GordianKeyPairSpec> myIterator = mySpec.keySpecIterator();
274             while (myIterator.hasNext()) {
275                 final GordianKeyPairSpec myPair = myIterator.next();
276                 v.add(theKeyPair2IdMap.get(myPair));
277             }
278             final AlgorithmIdentifier myPairId = new AlgorithmIdentifier(myId, new DERSequence(v));
279             return new AlgorithmIdentifier(myAgreeId, myPairId);
280 
281         } else {
282             return new AlgorithmIdentifier(myAgreeId, new AlgorithmIdentifier(theKeyPair2IdMap.get(pSpec.getKeyPairSpec())));
283         }
284     }

286     /**
287      * DetermineIdentifier the agreementSpec for an AlgorithmIdentifier.
288      *
289      * @param pId the identifier
290      * @return the agreementSpec
291      */
292     public GordianAgreementSpec determineAgreementSpec(final AlgorithmIdentifier pId) {
293         /* Obtain the ObjectIdentifiers */
294         final ASN1ObjectIdentifier myId = pId.getAlgorithm();
295         final AlgorithmIdentifier myPairId = AlgorithmIdentifier.getInstance(pId.getParameters());
296 
297         /* Derive the specs */
298         final GordianAgreementSpec mySpec = theId2AgreeMap.get(myId);
299         final GordianKeyPairSpec myPairSpec = theId2KeyPairMap.get(myPairId.getAlgorithm());
300         if (myPairSpec.getKeyPairType() == GordianKeyPairType.COMPOSITE) {
301             final ASN1Sequence myPairs = ASN1Sequence.getInstance(myPairId.getParameters());
302             final List<GordianKeyPairSpec> mySpecs = new ArrayList<>();
303             final Enumeration<?> en = myPairs.getObjects();
304             while (en.hasMoreElements()) {
305                 mySpecs.add(theId2KeyPairMap.get(ASN1ObjectIdentifier.getInstance(en.nextElement())));
306             }
307             return new GordianAgreementSpec(GordianKeyPairSpecBuilder.composite(mySpecs), mySpec.getAgreementType(), mySpec.getKDFType(), mySpec.withConfirm());
308         }
309 
310         /* Return the AgreementSpec */
311         return new GordianAgreementSpec(myPairSpec, mySpec.getAgreementType(), mySpec.getKDFType(), mySpec.withConfirm());
312     }

338     /**
339      * Obtain identifier for result.
340      *
341      * @param pResultType the result type
342      * @return the identifier
343      * @throws GordianException on error
344      */
345     protected AlgorithmIdentifier getIdentifierForResult(final Object pResultType) throws GordianException {
346         if (pResultType instanceof GordianFactoryType) {
347             final ASN1ObjectIdentifier myOID = pResultType == GordianFactoryType.BC
348                     ? GordianBaseData.BCFACTORYOID
349                     : GordianBaseData.JCAFACTORYOID;
350             return new AlgorithmIdentifier(myOID, null);
351         }
352         if (pResultType instanceof GordianKeySetSpec mySpec) {
353             final GordianKeySetSpecASN1 myParms = new GordianKeySetSpecASN1(mySpec);
354             return myParms.getAlgorithmId();
355         }
356         if (pResultType instanceof GordianSymCipherSpec mySpec) {
357             final GordianCoreCipherFactory myCipherFactory = (GordianCoreCipherFactory) theFactory.getCipherFactory();
358             return myCipherFactory.getIdentifierForSpec(mySpec);
359         }
360         if (pResultType instanceof GordianStreamCipherSpec mySpec) {
361             final GordianCoreCipherFactory myCipherFactory = (GordianCoreCipherFactory) theFactory.getCipherFactory();
362             return myCipherFactory.getIdentifierForSpec(mySpec);
363         }
364         if (pResultType instanceof Integer myInt) {
365             return new AlgorithmIdentifier(BYTERESULTOID, new ASN1Integer(myInt));
366         }
367         if (pResultType instanceof String myError) {
368             return new AlgorithmIdentifier(ERRORRESULTOID, new DERGeneralString(myError));
369         }
370         throw new GordianDataException("Illegal resultType set");
371     }

373     /**
374      * process result algorithmId.
375      *
376      * @param pResId the result algorithmId.
377      * @return the resultType
378      * @throws GordianException on error
379      */
380     public Object processResultIdentifier(final AlgorithmIdentifier pResId) throws GordianException {
381         /* Look for a Factory */
382         final ASN1ObjectIdentifier myAlgId = pResId.getAlgorithm();
383         if (GordianBaseData.BCFACTORYOID.equals(myAlgId)) {
384             return GordianFactoryType.BC;
385         }
386         if (GordianBaseData.JCAFACTORYOID.equals(myAlgId)) {
387             return GordianFactoryType.JCA;
388         }
389 
390         /* Look for a keySet Spec */
391         if (GordianKeySetSpecASN1.KEYSETALGID.equals(myAlgId)) {
392             return GordianKeySetSpecASN1.getInstance(pResId.getParameters()).getSpec();
393         }
394 
395         /* Look for a cipher Spec */
396         final GordianCoreCipherFactory myCipherFactory = (GordianCoreCipherFactory) theFactory.getCipherFactory();
397         final GordianCipherSpec<?> mySpec = myCipherFactory.getCipherSpecForIdentifier(pResId);
398         if (mySpec != null) {
399             return mySpec;
400         }
401 
402         /* Look for a Byte Type */
403         if (BYTERESULTOID.equals(myAlgId)) {
404             return ASN1Integer.getInstance(pResId.getParameters()).intValueExact();
405         }
406 
407         /* Look for an error */
408         if (ERRORRESULTOID.equals(myAlgId)) {
409             return ASN1GeneralString.getInstance(pResId.getParameters()).getString();
410         }
411 
412         /* Unrecognised */
413         throw new GordianDataException("Unrecognised resultType");
414     }
