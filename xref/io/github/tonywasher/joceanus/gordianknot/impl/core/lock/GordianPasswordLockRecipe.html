48       * Hash margins.

50      private static final int HASH_MARGIN = 4;

87      /**
88       * Constructor for random choices.
89       *
90       * @param pFactory  the factory
91       * @param pLockSpec the passwordLockSpec
92       */
93      GordianPasswordLockRecipe(final GordianBaseFactory pFactory,
94                                final GordianPasswordLockSpec pLockSpec) {
95          /* Access the secureRandom */
96          final SecureRandom myRandom = pFactory.getRandomSource().getRandom();
97  
98          /* Create the Salt vector */
99          theSalt = new byte[GordianLockData.SALTLEN];
100         myRandom.nextBytes(theSalt);
101 
102         /* Calculate the initVector */
103         final GordianPersonalisation myPersonal = pFactory.getPersonalisation();
104         theInitVector = myPersonal.adjustIV(theSalt);
105 
106         /* Allocate new set of parameters */
107         theParams = new GordianPasswordLockParams(pFactory);
108         theRecipe = theParams.getRecipe();
109         theLockSpec = pLockSpec;
110         theHashBytes = null;
111         thePayload = null;
112     }

114     /**
115      * Constructor for external form parse.
116      *
117      * @param pFactory    the factory
118      * @param pPassLength the password length
119      * @param pLockASN1   the lockASN1
120      */
121     GordianPasswordLockRecipe(final GordianBaseFactory pFactory,
122                               final int pPassLength,
123                               final GordianPasswordLockASN1 pLockASN1) {
124         /* Parse the ASN1 external form */
125         final byte[] myHashBytes = pLockASN1.getHashBytes();
126         theLockSpec = pLockASN1.getLockSpec();
127         thePayload = pLockASN1.getPayload();
128 
129         /* Create the byte arrays */
130         theRecipe = new byte[GordianLockData.RECIPELEN];
131         theSalt = new byte[GordianLockData.SALTLEN];
132         theHashBytes = new byte[GordianLockData.HASHLEN];
133 
134         /* Determine offset position */
135         int myOffSet = Math.max(pPassLength, HASH_MARGIN);
136         myOffSet = Math.min(myOffSet, GordianLockData.HASHLEN
137                 - HASH_MARGIN);
138 
139         /* Copy Data into buffers */
140         System.arraycopy(myHashBytes, 0, theHashBytes, 0, myOffSet);
141         System.arraycopy(myHashBytes, myOffSet, theRecipe, 0, GordianLockData.RECIPELEN);
142         System.arraycopy(myHashBytes, myOffSet
143                 + GordianLockData.RECIPELEN, theSalt, 0, GordianLockData.SALTLEN);
144         System.arraycopy(myHashBytes, myOffSet
145                 + GordianLockData.RECIPELEN
146                 + GordianLockData.SALTLEN, theHashBytes, myOffSet, GordianLockData.HASHLEN
147                 - myOffSet);
148 
149         /* Calculate the initVector */
150         final GordianPersonalisation myPersonal = pFactory.getPersonalisation();
151         theInitVector = myPersonal.adjustIV(theSalt);
152 
153         /* Allocate new set of parameters */
154         theParams = new GordianPasswordLockParams(pFactory, theRecipe);
155     }

166     /**
167      * Build lockBytes for hash and password length.
168      *
169      * @param pPassLength the password length
170      * @param pPayload    the payload
171      * @return the lockBytes
172      */
173     GordianPasswordLockASN1 buildLockASN1(final int pPassLength,
174                                           final byte[] pPayload) {
175         /* Allocate the new buffer */
176         final int myHashLen = theHashBytes.length;
177         final int myLen = GordianLockData.RECIPELEN
178                 + GordianLockData.SALTLEN
179                 + myHashLen;
180         final byte[] myBuffer = new byte[myLen];
181 
182         /* Determine offset position */
183         int myOffSet = Math.max(pPassLength, HASH_MARGIN);
184         myOffSet = Math.min(myOffSet, myHashLen
185                 - HASH_MARGIN);
186 
187         /* Copy Data into buffer */
188         System.arraycopy(theHashBytes, 0, myBuffer, 0, myOffSet);
189         System.arraycopy(theRecipe, 0, myBuffer, myOffSet, GordianLockData.RECIPELEN);
190         System.arraycopy(theSalt, 0, myBuffer, myOffSet
191                 + GordianLockData.RECIPELEN, GordianLockData.SALTLEN);
192         System.arraycopy(theHashBytes, myOffSet, myBuffer, myOffSet
193                 + GordianLockData.RECIPELEN
194                 + GordianLockData.SALTLEN, myHashLen
195                 - myOffSet);
196 
197         /* Build the ASN1 form */
198         return new GordianPasswordLockASN1(theLockSpec, myBuffer, pPayload);
199     }

201     /**
202      * Process password.
203      *
204      * @param pFactory  the factory
205      * @param pPassword the password for the keys
206      * @return the locking KeySet
207      * @throws GordianException on error
208      */
209     GordianCoreKeySet processPassword(final GordianBaseFactory pFactory,
210                                       final byte[] pPassword) throws GordianException {
211         /* Obtain configuration details */
212         final GordianPersonalisation myPersonal = pFactory.getPersonalisation();
213         final int iIterations = theLockSpec.getNumIterations();
214         final int iFinal = theParams.getAdjustment()
215                 + iIterations;
216 
217         /* Create a byte array of the iterations */
218         final byte[] myLoops = GordianDataConverter.integerToByteArray(iFinal);
219 
220         /* Access factories */
221         final GordianDigestFactory myDigests = pFactory.getDigestFactory();
222         final GordianMacFactory myMacs = pFactory.getMacFactory();
223 
224         /* Create the primeMac */
225         GordianMacSpec myMacSpec = GordianMacSpecBuilder.hMac(theParams.getPrimeDigest());
226         final GordianMac myPrimeMac = myMacs.createMac(myMacSpec);
227         myPrimeMac.initKeyBytes(pPassword);
228 
229         /* Create the alternateMac */
230         myMacSpec = GordianMacSpecBuilder.hMac(theParams.getSecondaryDigest());
231         final GordianMac mySecondaryMac = myMacs.createMac(myMacSpec);
232         mySecondaryMac.initKeyBytes(pPassword);
233 
234         /* Create the alternateMac */
235         myMacSpec = GordianMacSpecBuilder.hMac(theParams.getTertiaryDigest());
236         final GordianMac myTertiaryMac = myMacs.createMac(myMacSpec);
237         myTertiaryMac.initKeyBytes(pPassword);
238 
239         /* Create the secretMac */
240         myMacSpec = GordianMacSpecBuilder.hMac(new GordianDigestSpec(theParams.getSecretDigest(), GordianLength.LEN_512));
241         final GordianMac mySecretMac = myMacs.createMac(myMacSpec);
242         mySecretMac.initKeyBytes(pPassword);
243 
244         /* Initialise hash bytes and counter */
245         final byte[] myPrimeBytes = new byte[myPrimeMac.getMacSize()];
246         final byte[] mySecondaryBytes = new byte[mySecondaryMac.getMacSize()];
247         final byte[] myTertiaryBytes = new byte[myTertiaryMac.getMacSize()];
248         final byte[] mySecretBytes = new byte[mySecretMac.getMacSize()];
249         final byte[] myPrimeHash = new byte[myPrimeMac.getMacSize()];
250         final byte[] mySecondaryHash = new byte[mySecondaryMac.getMacSize()];
251         final byte[] myTertiaryHash = new byte[myTertiaryMac.getMacSize()];
252         final byte[] mySecretHash = new byte[mySecretMac.getMacSize()];
253 
254         /* Access final digest */
255         final GordianDigestSpec myDigestSpec = new GordianDigestSpec(theParams.getExternalDigest(), GordianLength.LEN_512);
256         final GordianDigest myDigest = myDigests.createDigest(myDigestSpec);
257 
258         /* Initialise the hash input values as the salt bytes */
259         final byte[] mySaltBytes = theInitVector;
260         byte[] myPrimeInput = mySaltBytes;
261         byte[] mySecondaryInput = mySaltBytes;
262         byte[] myTertiaryInput = mySaltBytes;
263         byte[] mySecretInput = mySaltBytes;
264 
265         /* Protect from exceptions */
266         try {
267             /* Update each Hash with the personalisation */
268             myPersonal.updateMac(myPrimeMac);
269             myPersonal.updateMac(mySecondaryMac);
270             myPersonal.updateMac(myTertiaryMac);
271             myPersonal.updateMac(mySecretMac);
272 
273             /* Update each Hash with the loops */
274             myPrimeMac.update(myLoops);
275             mySecondaryMac.update(myLoops);
276             myTertiaryMac.update(myLoops);
277             mySecretMac.update(myLoops);
278 
279             /* Loop through the iterations */
280             for (int iPass = 0; iPass < iFinal; iPass++) {
281                 /* Update the prime Mac */
282                 myPrimeMac.update(mySecondaryInput);
283                 myPrimeMac.update(myTertiaryInput);
284 
285                 /* Update the secondary Mac */
286                 mySecondaryMac.update(myPrimeInput);
287                 mySecondaryMac.update(myTertiaryInput);
288 
289                 /* Update the tertiary Mac */
290                 myTertiaryMac.update(myPrimeInput);
291                 myTertiaryMac.update(mySecondaryInput);
292 
293                 /* Update the secret Mac */
294                 mySecretMac.update(mySecretInput);
295                 mySecretMac.update(myPrimeInput);
296                 mySecretMac.update(mySecondaryInput);
297                 mySecretMac.update(myTertiaryInput);
298 
299                 /* Update inputs */
300                 myPrimeInput = myPrimeHash;
301                 mySecondaryInput = mySecondaryHash;
302                 myTertiaryInput = myTertiaryHash;
303                 mySecretInput = mySecretHash;
304 
305                 /* Recalculate hashes and combine them */
306                 myPrimeMac.finish(myPrimeHash, 0);
307                 GordianPersonalisation.buildHashResult(myPrimeBytes, myPrimeHash);
308                 mySecondaryMac.finish(mySecondaryHash, 0);
309                 GordianPersonalisation.buildHashResult(mySecondaryBytes, mySecondaryHash);
310                 myTertiaryMac.finish(myTertiaryHash, 0);
311                 GordianPersonalisation.buildHashResult(myTertiaryBytes, myTertiaryHash);
312                 mySecretMac.finish(mySecretHash, 0);
313                 GordianPersonalisation.buildHashResult(mySecretBytes, mySecretHash);
314             }
315 
316             /* Combine the Primary, Secondary and Tertiary bytes to form the external hash */
317             myDigest.update(myPrimeBytes);
318             myDigest.update(mySecondaryBytes);
319             myDigest.update(myTertiaryBytes);
320             final byte[] myHashBytes = myDigest.finish();
321 
322             /* If we are resolving the lock, check the hash */
323             if (theHashBytes != null
324                     && !Arrays.equals(theHashBytes, myHashBytes)) {
325                 /* Fail the password attempt */
326                 throw new GordianBadCredentialsException("Invalid Password");
327             }
328             theHashBytes = myHashBytes;
329 
330             /* Create the Key Set */
331             final GordianCoreKeySet myKeySet = ((GordianCoreKeySetFactory) pFactory.getKeySetFactory()).createKeySet(theLockSpec.getKeySetSpec());
332             myKeySet.buildFromSecret(mySecretBytes);
333 
334             /* Return to caller */
335             return myKeySet;
336 
337             /* Clear intermediate arrays */
338         } finally {
339             Arrays.fill(myPrimeHash, (byte) 0);
340             Arrays.fill(myPrimeBytes, (byte) 0);
341             Arrays.fill(mySecondaryHash, (byte) 0);
342             Arrays.fill(mySecondaryBytes, (byte) 0);
343             Arrays.fill(myTertiaryHash, (byte) 0);
344             Arrays.fill(myTertiaryBytes, (byte) 0);
345             Arrays.fill(mySecretHash, (byte) 0);
346             Arrays.fill(mySecretBytes, (byte) 0);
347         }
348     }

350     /**
351      * The parameters class.
352      */
353     private static final class GordianPasswordLockParams {
354         /**
355          * The Recipe.
356          */
357         private final byte[] theRecipe;
358 
359         /**
360          * The secret hMac type.
361          */
362         private final GordianDigestType theSecretDigest;
363 
364         /**
365          * The hMac types.
366          */
367         private final GordianDigestType[] theDigests;
368 
369         /**
370          * The external Digest type.
371          */
372         private final GordianDigestType theExternalDigest;
373 
374         /**
375          * The Adjustment.
376          */
377         private final int theAdjust;
378 
379         /**
380          * Construct the parameters from random.
381          *
382          * @param pFactory the factory
383          */
384         GordianPasswordLockParams(final GordianBaseFactory pFactory) {
385             /* Obtain Id manager and random */
386             final GordianIdManager myManager = pFactory.getIdManager();
387             final GordianPersonalisation myPersonal = pFactory.getPersonalisation();
388             final SecureRandom myRandom = pFactory.getRandomSource().getRandom();
389 
390             /* Generate recipe and derive digestTypes */
391             final int mySeed = myRandom.nextInt();
392             theRecipe = GordianDataConverter.integerToByteArray(mySeed);
393             final Random mySeededRandom = myPersonal.getSeededRandom(GordianPersonalId.LOCKRANDOM, theRecipe);
394             theSecretDigest = myManager.deriveLockSecretTypeFromSeed(mySeededRandom);
395             theDigests = myManager.deriveLockDigestTypesFromSeed(mySeededRandom, GordianLockData.NUM_DIGESTS);
396             theExternalDigest = myManager.deriveExternalDigestTypeFromSeed(mySeededRandom);
397 
398             /* Derive random adjustment value */
399             theAdjust = mySeededRandom.nextInt(GordianDataConverter.NYBBLE_MASK + 1);
400         }
401 
402         /**
403          * Construct the parameters from recipe.
404          *
405          * @param pFactory the factory
406          * @param pRecipe  the recipe bytes
407          */
408         GordianPasswordLockParams(final GordianBaseFactory pFactory,
409                                   final byte[] pRecipe) {
410             /* Obtain Id manager */
411             final GordianIdManager myManager = pFactory.getIdManager();
412             final GordianPersonalisation myPersonal = pFactory.getPersonalisation();
413 
414             /* Store recipe and derive digestTypes */
415             theRecipe = pRecipe;
416             final Random mySeededRandom = myPersonal.getSeededRandom(GordianPersonalId.LOCKRANDOM, theRecipe);
417             theSecretDigest = myManager.deriveLockSecretTypeFromSeed(mySeededRandom);
418             theDigests = myManager.deriveLockDigestTypesFromSeed(mySeededRandom, GordianLockData.NUM_DIGESTS);
419             theExternalDigest = myManager.deriveExternalDigestTypeFromSeed(mySeededRandom);
420 
421             /* Derive random adjustment value */
422             theAdjust = mySeededRandom.nextInt(GordianDataConverter.NYBBLE_MASK + 1);
423         }
