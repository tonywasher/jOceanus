110     /**
111      * Does a certificate request require encryption?
112      *
113      * @param pCertReq the certificate request
114      * @return true/false
115      */
116     static boolean requiresEncryption(final CertReqMsg pCertReq) {
117         /* Only encipherment keys may be encrypted */
118         final ProofOfPossession myProof = pCertReq.getPop();
119         if (myProof.getType() != ProofOfPossession.TYPE_KEY_ENCIPHERMENT) {
120             return false;
121         }
122 
123         /* Check that we are responding with an encrypted certificate */
124         final POPOPrivKey myPOP = POPOPrivKey.getInstance(myProof.getObject());
125         final int myProofType = myPOP.getType();
126         return myProofType == POPOPrivKey.subsequentMessage
127                 && ASN1Integer.getInstance(myPOP.getValue()).intValueExact() == SubsequentMessage.encrCert.intValueExact();
128     }

130     /**
131      * process a certificate request.
132      *
133      * @param pRequest the request
134      * @return the signed certificate chain
135      * @throws GordianException on error
136      */
137     List<GordianCertificate> processCertificateRequest(final CertReqMsg pRequest) throws GordianException {
138         /* Derive the certificate request message */
139         final CertRequest myCertReq = pRequest.getCertReq();
140         final ProofOfPossession myProof = pRequest.getPop();
141         final AttributeTypeAndValue[] myAttrs = pRequest.getRegInfo();
142         final CertTemplate myTemplate = myCertReq.getCertTemplate();
143         final X500Name mySubject = myTemplate.getSubject();
144         final SubjectPublicKeyInfo myPublic = myTemplate.getPublicKey();
145         final GordianKeyPairUsage myUsage = GordianCertUtils.determineUsage(myTemplate.getExtensions());
146 
147         /* Check the PKMacValue */
148         checkPKMACValue(mySubject, myAttrs, myPublic);
149 
150         /* Derive keyPair and create certificate chain */
151         final GordianKeyPair myPair = deriveKeyPair(myProof, myCertReq, mySubject, myPublic);
152         final GordianBaseKeyStoreManager myKeyStoreMgr = (GordianBaseKeyStoreManager) theGateway.getKeyStoreManager();
153         final GordianKeyStorePair mySigner = theGateway.getSigner();
154         if (mySigner == null) {
155             throw new GordianLogicException("Null keyPairSigner");
156         }
157         return myKeyStoreMgr.signKeyPair(myPair, mySubject, myUsage, mySigner);
158     }

160     /**
161      * process a certificate response.
162      *
163      * @param pResponse the certificate response
164      * @param pKeyPair  the keyPair
165      */
166     public void processCertificateResponse(final GordianCertResponseASN1 pResponse,
167                                            final GordianKeyStorePair pKeyPair) throws GordianException {
168         /* Decrypt if necessary */
169         if (pResponse.isEncrypted()) {
170             final GordianCRMEncryptor myEncryptor = theGateway.getEncryptor();
171             pResponse.decryptCertificate(myEncryptor, pKeyPair);
172         }
173 
174         /* Check the MACValue */
175         final GordianCoreCertificate myCert = (GordianCoreCertificate) pKeyPair.getCertificateChain().get(0);
176         final X500Name myName = myCert.getSubjectName();
177         final byte[] myMACSecret = theGateway.getMACSecret(myName);
178         final ASN1Object myMACData = pResponse.getMACData();
179         final PKMACValue mySent = pResponse.getMACValue();
180 
181         /* If we have a mismatch on security */
182         if ((myMACSecret == null) != (mySent == null)) {
183             throw new GordianDataException("Mismatch on PKMAC Security");
184         }
185 
186         /* If we have a MACValue */
187         if (mySent != null) {
188             /* Calculate the PKMACValue and compare with the value that was sent */
189             theBuilder.checkPKMACValue(myMACSecret, myMACData, mySent);
190         }
191 
192         /* Check that the publicKey matches */
193         final GordianCoreCertificate myNewCert = pResponse.getCertificate(theGateway.getEncryptor());
194         if (!myNewCert.checkMatchingPublicKey(pKeyPair.getKeyPair())) {
195             throw new GordianDataException("Mismatch on publicKey");
196         }
197 
198         /* Check that subjectName matches */
199         final GordianCoreCertificate myOldCert = (GordianCoreCertificate) pKeyPair.getCertificateChain().get(0);
200         if (!myNewCert.getSubjectName().equals(myOldCert.getSubjectName())) {
201             throw new GordianDataException("Mismatch on subjectName");
202         }
203     }

205     /**
206      * Derive the privateKey.
207      *
208      * @param pProof   the proof of possession
209      * @param pSubject the subject name
210      * @return the PKCS8Encoded privateKey
211      * @throws GordianException on error
212      */
213     PKCS8EncodedKeySpec derivePrivateKey(final ProofOfPossession pProof,
214                                          final X500Name pSubject) throws GordianException {
215         /* Protect against exceptions */
216         try {
217             /* Extract details */
218             final POPOPrivKey myEncrypt = (POPOPrivKey) pProof.getObject();
219             final EnvelopedData myData = (EnvelopedData) myEncrypt.getValue();
220             final RecipientInfo myRecipient = RecipientInfo.getInstance(myData.getRecipientInfos().getObjectAt(0));
221             final EncryptedContentInfo myContent = myData.getEncryptedContentInfo();
222             final byte[] myEncryptedPrivKey = myContent.getEncryptedContent().getOctets();
223             final KeyTransRecipientInfo myRecInfo = (KeyTransRecipientInfo) myRecipient.getInfo();
224 
225             /* Derive the keySet */
226             final GordianKeySet myKeySet = deriveKeySetFromRecInfo(myRecInfo);
227 
228             /* Decrypt the privateKey/ID */
229             final EncKeyWithID myKeyWithId = EncKeyWithID.getInstance(myKeySet.decryptBytes(myEncryptedPrivKey));
230 
231             /* Check that the ID matches */
232             final X500Name myName = X500Name.getInstance(GeneralName.getInstance(myKeyWithId.getIdentifier()).getName());
233             if (!myName.equals(pSubject)) {
234                 throw new GordianDataException("Mismatch on subjectID");
235             }
236 
237             /* myName and Subject should be identical */
238             return new PKCS8EncodedKeySpec(myKeyWithId.getPrivateKey().getEncoded());
239 
240         } catch (IOException e) {
241             throw new GordianIOException("Failed to derive encrypted privateKey", e);
242         }
243     }

245     /**
246      * Derive the keySet via a keyPairSet issuer.
247      *
248      * @param pRecInfo the recipient info
249      * @return the keySet
250      * @throws GordianException on error
251      */
252     private GordianKeySet deriveKeySetFromRecInfo(final KeyTransRecipientInfo pRecInfo) throws GordianException {
253         /* Access issuer details */
254         final IssuerAndSerialNumber myIssId = (IssuerAndSerialNumber) pRecInfo.getRecipientIdentifier().getId();
255 
256         /* Locate issuer */
257         final GordianBaseKeyStore myKeyStore = (GordianBaseKeyStore) theGateway.getKeyStore();
258         final String myAlias = myKeyStore.findIssuerCert(myIssId);
259         final Function<String, char[]> myResolver = theGateway.getPasswordResolver();
260         final char[] myPassword = myResolver.apply(myAlias);
261         if (myPassword == null) {
262             throw new GordianDataException("No password available for issuer");
263         }
264         final GordianKeyStoreEntry myIssuerEntry = myKeyStore.getEntry(myAlias, myPassword);
265         Arrays.fill(myPassword, (char) 0);
266 
267         /* Access details */
268         final GordianKeyStorePair myIssuer = (GordianKeyStorePair) myIssuerEntry;
269         final GordianCertificate myCert = myIssuer.getCertificateChain().get(0);
270         final GordianCRMEncryptor myEncryptor = theGateway.getEncryptor();
271         return myEncryptor.deriveKeySetFromRecInfo(pRecInfo, myCert, myIssuer.getKeyPair());
272     }

274     /**
275      * Derive and check the keyPair.
276      *
277      * @param pProof     the proof of possession
278      * @param pCertReq   the certificate request
279      * @param pSubject   the subject name
280      * @param pPublicKey the publicKey
281      * @return the keyPair
282      * @throws GordianException on error
283      */
284     private GordianKeyPair deriveKeyPair(final ProofOfPossession pProof,
285                                          final CertRequest pCertReq,
286                                          final X500Name pSubject,
287                                          final SubjectPublicKeyInfo pPublicKey) throws GordianException {
288         /* Handle signed keyPair */
289         switch (pProof.getType()) {
290             case ProofOfPossession.TYPE_SIGNING_KEY:
291                 return deriveSignedKeyPair(pProof, pCertReq, pPublicKey);
292             case ProofOfPossession.TYPE_KEY_ENCIPHERMENT:
293                 return deriveEncryptedKeyPair(pProof, pSubject, pPublicKey);
294             default:
295                 throw new GordianDataException("Unsupported proof type");
296         }
297     }

299     /**
300      * Derive a signed keyPair.
301      *
302      * @param pProof     the proof of possession
303      * @param pCertReq   the certificate request
304      * @param pPublicKey the publicKey
305      * @return the keyPair
306      * @throws GordianException on error
307      */
308     private GordianKeyPair deriveSignedKeyPair(final ProofOfPossession pProof,
309                                                final CertRequest pCertReq,
310                                                final SubjectPublicKeyInfo pPublicKey) throws GordianException {
311         /* Protect against exceptions */
312         try {
313             /* Derive the public Key */
314             final GordianAsyncFactory myFactory = theGateway.getFactory().getAsyncFactory();
315             final GordianKeyPairFactory myKPFactory = myFactory.getKeyPairFactory();
316             final X509EncodedKeySpec myX509Spec = new X509EncodedKeySpec(pPublicKey.getEncoded());
317             final GordianKeyPairSpec myKeySpec = myKPFactory.determineKeyPairSpec(myX509Spec);
318             final GordianKeyPairGenerator myGenerator = myKPFactory.getKeyPairGenerator(myKeySpec);
319             final GordianKeyPair myKeyPair = myGenerator.derivePublicOnlyKeyPair(myX509Spec);
320 
321             /* Access the verifier */
322             final POPOSigningKey mySigning = (POPOSigningKey) pProof.getObject();
323             final AlgorithmIdentifier myAlgId = mySigning.getAlgorithmIdentifier();
324             final GordianCoreSignatureFactory mySignFactory = (GordianCoreSignatureFactory) myFactory.getSignatureFactory();
325             final GordianSignatureSpec mySignSpec = mySignFactory.getSpecForIdentifier(myAlgId);
326             final GordianSignature myVerifier = mySignFactory.createSigner(mySignSpec);
327 
328             /* Verify the signature */
329             final byte[] mySignature = mySigning.getSignature().getBytes();
330             myVerifier.initForVerify(GordianSignParams.keyPair(myKeyPair));
331             myVerifier.update(pCertReq.getEncoded());
332             if (!myVerifier.verify(mySignature)) {
333                 throw new GordianDataException("Verification of keyPair failed");
334             }
335             return myKeyPair;
336 
337         } catch (IOException e) {
338             throw new GordianIOException("Failed to derive signed keyPair", e);
339         }
340     }

342     /**
343      * Derive an encrypted keyPair.
344      *
345      * @param pProof     the proof of possession
346      * @param pSubject   the subject name
347      * @param pPublicKey the publicKey
348      * @return the keyPair
349      * @throws GordianException on error
350      */
351     private GordianKeyPair deriveEncryptedKeyPair(final ProofOfPossession pProof,
352                                                   final X500Name pSubject,
353                                                   final SubjectPublicKeyInfo pPublicKey) throws GordianException {
354         /* Protect against exceptions */
355         try {
356             /* Access the generator */
357             final GordianKeyPairFactory myFactory = theGateway.getFactory().getAsyncFactory().getKeyPairFactory();
358             final X509EncodedKeySpec myX509Spec = new X509EncodedKeySpec(pPublicKey.getEncoded());
359             final GordianKeyPairSpec myKeySpec = myFactory.determineKeyPairSpec(myX509Spec);
360             final GordianKeyPairGenerator myGenerator = myFactory.getKeyPairGenerator(myKeySpec);
361 
362             /* Determine type of proof of Possession */
363             final POPOPrivKey myPOP = POPOPrivKey.getInstance(pProof.getObject());
364             final int myProofType = myPOP.getType();
365 
366             /* Handle encryptedKey */
367             if (myProofType == POPOPrivKey.encryptedKey) {
368                 /* derive the privateKey and full keyPair */
369                 final PKCS8EncodedKeySpec myPKCS8Spec = derivePrivateKey(pProof, pSubject);
370                 final GordianKeyPair myKeyPair = myGenerator.deriveKeyPair(myX509Spec, myPKCS8Spec);
371 
372                 /* Check that the privateKey matches the publicKey */
373                 checkPrivateKey(myKeyPair);
374             } else if (myProofType != POPOPrivKey.subsequentMessage
375                     || ASN1Integer.getInstance(myPOP.getValue()).intValueExact() != SubsequentMessage.encrCert.intValueExact()) {
376                 throw new GordianDataException("Unsupported ProofType");
377             }
378 
379             /* Return the public only value */
380             return myGenerator.derivePublicOnlyKeyPair(myX509Spec);
381 
382         } catch (IOException e) {
383             throw new GordianIOException("Failed to derive encrypted keyPair", e);
384         }
385     }

387     /**
388      * Check PrivateKey.
389      *
390      * @param pKeyPair the keyPair
391      * @throws GordianException on error
392      */
393     private void checkPrivateKey(final GordianKeyPair pKeyPair) throws GordianException {
394         /* Access details */
395         final GordianAsyncFactory myFactory = theGateway.getFactory().getAsyncFactory();
396         final GordianKeyPairSpec mySpec = pKeyPair.getKeyPairSpec();
397 
398         /* Check for encryption private key */
399         final GordianEncryptorSpec myEncSpec = myFactory.getEncryptorFactory().defaultForKeyPair(mySpec);
400         if (myEncSpec != null) {
401             checkEncryptionPrivateKey(pKeyPair);
402             return;
403         }
404 
405         /* Check for agreement private key */
406         final GordianAgreementSpec myAgreeSpec = myFactory.getAgreementFactory().defaultForKeyPair(mySpec);
407         if (myAgreeSpec != null) {
408             checkAgreementPrivateKey(pKeyPair);
409             return;
410         }
411 
412         /* Reject the request */
413         throw new GordianDataException("Unable to verify privateKey");
414     }

416     /**
417      * Check Encryption PrivateKey.
418      *
419      * @param pKeyPair the keyPair
420      * @throws GordianException on error
421      */
422     private void checkEncryptionPrivateKey(final GordianKeyPair pKeyPair) throws GordianException {
423         /* Create the data to encrypt */
424         final byte[] mySrc = new byte[TESTLEN];
425         final GordianBaseFactory myFactory = theGateway.getFactory();
426         myFactory.getRandomSource().getRandom().nextBytes(mySrc);
427 
428         /* Access details */
429         final GordianEncryptorFactory myEncFactory = myFactory.getAsyncFactory().getEncryptorFactory();
430         final GordianKeyPairSpec mySpec = pKeyPair.getKeyPairSpec();
431         final GordianEncryptorSpec myEncSpec = myEncFactory.defaultForKeyPair(mySpec);
432 
433         /* Create and initialise encryptors */
434         final GordianEncryptor mySender = myEncFactory.createEncryptor(myEncSpec);
435         final GordianEncryptor myReceiver = myEncFactory.createEncryptor(myEncSpec);
436 
437         /* Handle Initialisation */
438         mySender.initForEncrypt(pKeyPair);
439         myReceiver.initForDecrypt(pKeyPair);
440 
441         /* Perform the encryption and decryption for all zeros */
442         final byte[] myEncrypted = mySender.encrypt(mySrc);
443         final byte[] myResult = myReceiver.decrypt(myEncrypted);
444 
445         /* Check the decryption */
446         if (!org.bouncycastle.util.Arrays.areEqual(mySrc, myResult)) {
447             throw new GordianDataException("Private key failed validation");
448         }
449     }

451     /**
452      * Check Agreement PrivateKey.
453      *
454      * @param pKeyPair the keyPair
455      * @throws GordianException on error
456      */
457     private void checkAgreementPrivateKey(final GordianKeyPair pKeyPair) throws GordianException {
458         /* Access details */
459         final GordianBaseFactory myFactory = theGateway.getFactory();
460         final GordianAgreementFactory myAgreeFactory = myFactory.getAsyncFactory().getAgreementFactory();
461         final GordianKeyPairSpec mySpec = pKeyPair.getKeyPairSpec();
462         final GordianAgreementSpec myAgreeSpec = myAgreeFactory.defaultForKeyPair(mySpec);
463 
464         /* Create agreement */
465         final GordianCertificate myCert = myAgreeFactory.newMiniCertificate(GordianCRMEncryptor.SERVER, pKeyPair,
466                 new GordianKeyPairUsage(GordianKeyPairUse.AGREEMENT));
467         GordianAgreementParams myParams = myAgreeFactory.newAgreementParams(myAgreeSpec, GordianLength.LEN_256.getByteLength())
468                 .setServerCertificate(myCert);
469         final GordianAgreement mySender = myAgreeFactory.createAgreement(myParams);
470         final byte[] myClientHello = mySender.nextMessage();
471         final GordianAgreement myResponder = myAgreeFactory.parseAgreementMessage(myClientHello);
472         myParams = myResponder.getAgreementParams().setServerCertificate(myCert);
473         myResponder.updateParams(myParams);
474 
475         /* Check the agreements */
476         final byte[] myFirst = (byte[]) mySender.getResult();
477         final byte[] mySecond = (byte[]) myResponder.getResult();
478         if (!Arrays.equals(myFirst, mySecond)) {
479             throw new GordianDataException("Private key failed validation");
480         }
481     }

483     /**
484      * Check PKMacValue.
485      *
486      * @param pSubject   the subject name
487      * @param pAttrs     the attributes
488      * @param pPublicKey the public key
489      * @throws GordianException on error
490      */
491     private void checkPKMACValue(final X500Name pSubject,
492                                  final AttributeTypeAndValue[] pAttrs,
493                                  final SubjectPublicKeyInfo pPublicKey) throws GordianException {
494         /* Loop through the Attrs */
495         AttributeTypeAndValue myAttr = null;
496         if (pAttrs != null) {
497             for (AttributeTypeAndValue myCurr : pAttrs) {
498                 if (GordianCRMBuilder.MACVALUEATTROID.equals(myCurr.getType())) {
499                     myAttr = myCurr;
500                     break;
501                 }
502             }
503         }
504 
505         /* If we have a mismatch on security */
506         final byte[] myMACSecret = theGateway.getMACSecret(pSubject);
507         if ((myMACSecret == null) != (myAttr == null)) {
508             throw new GordianDataException("Mismatch on PKMAC Security");
509         }
510 
511         /* If we have a MACValue */
512         if (myAttr != null) {
513             /* Calculate the PKMACValue and compare with the value that was sent */
514             final PKMACValue mySent = PKMACValue.getInstance(myAttr.getValue());
515             theBuilder.checkPKMACValue(myMACSecret, pPublicKey, mySent);
516         }
517     }
