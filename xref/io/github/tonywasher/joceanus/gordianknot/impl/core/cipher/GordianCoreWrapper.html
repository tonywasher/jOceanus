43  /**
44   * GordianKnot base for WrapCipher.
45   * <p>
46   * This class uses a variant of RFC5649 to wrap keyData. It has been modified so that it does not
47   * require a 128-block cipher.
48   */
49  public class GordianCoreWrapper
50          implements GordianWrapper {
51      /**
52       * Wrap repeat count.
53       */
54      private static final int WRAP_COUNT = 6;
55  
56      /**
57       * Integrity value.
58       */
59      private static final byte INTEGRITY_VALUE1 = (byte) 0xA6;

61      /**
62       * Integrity value.

64      private static final byte INTEGRITY_VALUE2 = (byte) 0x59;

66      /**

69      private static final int INTEGRITY_MODULO = 4;

101         theFactory = pFactory;
102         theKey = pKey;
103         theCipher = pCipher;
104         theBlockLen = getKeySpec().getBlockLength().getByteLength() >> 1;

135     @Override
136     public byte[] secureKey(final GordianKey<?> pKeyToSecure) throws GordianException {
137         /* Secure the bytes */
138         final byte[] myBytes = secureBytes(((GordianCoreKey<?>) pKeyToSecure).getKeyBytes());
139 
140         /* Create the ASN1 */
141         final AlgorithmIdentifier myAlgId = theFactory.getIdentifierForSpec(pKeyToSecure.getKeyType());
142         final GordianWrappedKeyASN1 myASN1 = new GordianWrappedKeyASN1(myAlgId, myBytes);
143         return myASN1.getEncodedBytes();
144     }

146     @Override
147     public <T extends GordianKeySpec> GordianKey<T> deriveKey(final byte[] pSecuredKey,
148                                                               final T pKeyType) throws GordianException {
149         /* Parse the ASN1 */
150         final GordianWrappedKeyASN1 myASN1 = GordianWrappedKeyASN1.getInstance(pSecuredKey);
151         final AlgorithmIdentifier myAlgId = myASN1.getKeySpecId();
152         final byte[] myWrappedKey = myASN1.getWrappedKey();
153 
154         /* Check the algorithmId */
155         final GordianKeySpec mySpec = theFactory.getKeySpecForIdentifier(myAlgId);
156         if (mySpec == null || !mySpec.equals(pKeyType)) {
157             throw new GordianDataException("Incorrect KeySpec");
158         }
159 
160         /* Unwrap the bytes */
161         final byte[] myBytes = deriveBytes(myWrappedKey);
162 
163         /* Access the relevant keyGenerator */
164         final GordianCoreKeyGenerator<T> myGenerator = pKeyType instanceof GordianMacSpec
165                 ? (GordianCoreKeyGenerator<T>) theFactory.getMacFactory().getKeyGenerator(pKeyType)
166                 : (GordianCoreKeyGenerator<T>) theFactory.getCipherFactory().getKeyGenerator(pKeyType);
167 
168         /* Generate the key */
169         return myGenerator.buildKeyFromBytes(myBytes);
170     }

172     @Override
173     public byte[] securePrivateKey(final GordianKeyPair pKeyPairToSecure) throws GordianException {
174         /* Access the KeyPair Generator */
175         final GordianKeyPairFactory myFactory = theFactory.getAsyncFactory().getKeyPairFactory();
176         final GordianCoreKeyPairGenerator myGenerator = (GordianCoreKeyPairGenerator) myFactory.getKeyPairGenerator(pKeyPairToSecure.getKeyPairSpec());
177         final PKCS8EncodedKeySpec myPKCS8Key = myGenerator.getPKCS8Encoding(pKeyPairToSecure);
178         return secureBytes(myPKCS8Key.getEncoded());
179     }

181     @Override
182     public GordianKeyPair deriveKeyPair(final X509EncodedKeySpec pPublicKeySpec,
183                                         final byte[] pSecuredPrivateKey) throws GordianException {
184         /* Access the PKCS8Encoding */
185         final PKCS8EncodedKeySpec myPrivate = derivePrivateKeySpec(pSecuredPrivateKey);
186 
187         /* Determine and check the keyPairSpec */
188         final GordianKeyPairFactory myFactory = theFactory.getAsyncFactory().getKeyPairFactory();
189         final GordianKeyPairSpec myKeySpec = myFactory.determineKeyPairSpec(pPublicKeySpec);
190         if (!myKeySpec.equals(myFactory.determineKeyPairSpec(myPrivate))) {
191             throw new GordianLogicException("Mismatch on keySpecs");
192         }
193 
194         /* Derive the keyPair */
195         final GordianCoreKeyPairGenerator myGenerator = (GordianCoreKeyPairGenerator) myFactory.getKeyPairGenerator(myKeySpec);
196         return myGenerator.deriveKeyPair(pPublicKeySpec, myPrivate);
197     }

212     @Override
213     public byte[] secureBytes(final byte[] pBytesToSecure) throws GordianException {
214         /* Determine number of blocks */
215         final int myDataLen = pBytesToSecure.length;
216         int myNumBlocks = myDataLen
217                 / theBlockLen;
218 
219         /* Calculate padding length */
220         final int myTrueLen;
221         if (myDataLen % theBlockLen == 0) {
222             myTrueLen = myDataLen;
223         } else {
224             myNumBlocks++;
225             myTrueLen = myNumBlocks
226                     * theBlockLen;
227         }
228 
229         /* Determine semantics of the initial block */
230         final byte[] myByteLen = GordianDataConverter.integerToByteArray(myDataLen);
231         int myCheckLen = theBlockLen - Integer.BYTES;
232         int myHdrLen = theBlockLen;
233 
234         /* Handle 64-bit ciphers */
235         if (myCheckLen == 0) {
236             myCheckLen = theBlockLen;
237             myHdrLen += Integer.BYTES;
238             myNumBlocks++;
239         }
240 
241         /* Add a block for random data */
242         myNumBlocks++;
243         myHdrLen += theBlockLen;
244 
245         /* Allocate buffer for data and encryption */
246         final int myBufferLen = theBlockLen << 1;
247         final byte[] myData = new byte[myTrueLen
248                 + myHdrLen];
249         final byte[] myBuffer = new byte[myBufferLen];
250         final byte[] myResult = new byte[myBufferLen];
251         final byte[] myRandom = new byte[theBlockLen];
252 
253         /* Build the initial block */
254         for (int i = 0; i < myCheckLen; i++) {
255             myData[i] = getIntegrityValue(i);
256         }
257         System.arraycopy(myByteLen, 0, myData, myCheckLen, Integer.BYTES);
258         theFactory.getRandomSource().getRandom().nextBytes(myRandom);
259         System.arraycopy(myRandom, 0, myData, myCheckLen + Integer.BYTES, theBlockLen);
260         System.arraycopy(pBytesToSecure, 0, myData, myHdrLen, myDataLen);
261 
262         /* Initialise the cipher */
263         initCipherForWrapping();
264 
265         /* Loop WRAP_COUNT times */
266         int myCount = 1;
267         for (int myCycle = 0; myCycle < WRAP_COUNT; myCycle++) {
268             /* Loop through the data blocks */
269             for (int myBlock = 1, myOffset = theBlockLen; myBlock <= myNumBlocks; myBlock++, myOffset += theBlockLen) {
270                 /* Build the data to be encrypted */
271                 System.arraycopy(myData, 0, myBuffer, 0, theBlockLen);
272                 System.arraycopy(myData, myOffset, myBuffer, theBlockLen, theBlockLen);
273 
274                 /* Encrypt the byte array */
275                 iterateCipher(myBuffer, myBufferLen, myResult);
276 
277                 /* Adjust the result using the count as a mask */
278                 for (int myMask = myCount++, myIndex = myBufferLen - 1; myMask != 0; myMask >>>= Byte.SIZE, myIndex--) {
279                     myResult[myIndex] ^= (byte) myMask;
280                 }
281 
282                 /* Restore encrypted data */
283                 System.arraycopy(myResult, 0, myData, 0, theBlockLen);
284                 System.arraycopy(myResult, theBlockLen, myData, myOffset, theBlockLen);
285             }
286         }
287 
288         /* Return the wrapped data */
289         return myData;
290     }

292     @Override
293     public byte[] deriveBytes(final byte[] pSecuredBytes,
294                               final int pOffset) throws GordianException {
295         /* Determine number of blocks */
296         int myDataLen = pSecuredBytes.length
297                 - theBlockLen - pOffset;
298         final int myNumBlocks = myDataLen
299                 / theBlockLen;
300 
301         /* Data must be a multiple of BlockLength */
302         if (myNumBlocks * theBlockLen != myDataLen) {
303             throw new GordianDataException("Invalid data length");
304         }
305 
306         /* Allocate buffers for data and encryption */
307         final int myBufferLen = theBlockLen << 1;
308         final byte[] myData = new byte[pSecuredBytes.length - pOffset];
309         System.arraycopy(pSecuredBytes, pOffset, myData, 0, pSecuredBytes.length - pOffset);
310         final byte[] myBuffer = new byte[myBufferLen];
311         final byte[] myResult = new byte[myBufferLen];
312 
313         /* Initialise the cipher */
314         initCipherForUnwrapping();
315 
316         /* Loop WRAP_COUNT times */
317         int myCount = myNumBlocks * WRAP_COUNT;
318         for (int myCycle = WRAP_COUNT; myCycle > 0; myCycle--) {
319             /* Loop through the data blocks */
320             for (int myBlock = myNumBlocks, myOffset = theBlockLen
321                     * myBlock; myBlock >= 1; myBlock--, myOffset -= theBlockLen) {
322                 /* Build the data to be decrypted */
323                 System.arraycopy(myData, 0, myBuffer, 0, theBlockLen);
324                 System.arraycopy(myData, myOffset, myBuffer, theBlockLen, theBlockLen);
325 
326                 /* Adjust the buffer using the count as a mask */
327                 for (int myMask = myCount--, myIndex = myBufferLen - 1; myMask != 0; myMask >>>= Byte.SIZE, myIndex--) {
328                     myBuffer[myIndex] ^= (byte) myMask;
329                 }
330 
331                 /* Decrypt the byte array */
332                 iterateCipher(myBuffer, myBufferLen, myResult);
333 
334                 /* Restore decrypted data */
335                 System.arraycopy(myResult, 0, myData, 0, theBlockLen);
336                 System.arraycopy(myResult, theBlockLen, myData, myOffset, theBlockLen);
337             }
338         }
339 
340         /* Handle 64-bit ciphers */
341         int myCheckLen = theBlockLen - Integer.BYTES;
342         if (myCheckLen == 0) {
343             myCheckLen = theBlockLen;
344             myDataLen -= theBlockLen;
345         }
346 
347         /* Adjust for random data */
348         myDataLen -= theBlockLen;
349 
350         /* Check initialisation value */
351         int checkFail = 0;
352         for (int myInit = 0; myInit < myCheckLen; myInit++) {
353             checkFail |= myData[myInit] ^ getIntegrityValue(myInit);
354         }
355 
356         /* Obtain encoded length */
357         final byte[] myByteLen = Arrays.copyOfRange(myData, myCheckLen, myCheckLen + Integer.BYTES);
358         final int myEncodedLen = GordianDataConverter.byteArrayToInteger(myByteLen);
359 
360         /* Obtain zeroLen and check that it is valid */
361         final int myZeroLen = myDataLen - myEncodedLen;
362         checkFail |= myZeroLen < 0 ? 1 : 0;
363         checkFail |= myZeroLen >= theBlockLen ? 2 : 0;
364 
365         /* Check trailing bytes */
366         for (int myZeros = myZeroLen, myLoc = myData.length - 1; myZeros > 0 && myLoc > 0; myZeros--, myLoc--) {
367             /* Check that byte is zero */
368             checkFail |= myData[myLoc];
369         }
370 
371         /* Return unwrapped data */
372         if (checkFail == 0) {
373             return Arrays.copyOfRange(myData, myCheckLen + Integer.BYTES + theBlockLen, myData.length
374                     - myZeroLen);
375         }
376 
377         /* Reject if checks fail */
378         throw new GordianDataException("Integrity checks failed");
379     }

381     /**
382      * Determine integrity value for position.
383      *
384      * @param pIndex the index
385      * @return the integrity value
386      */
387     private static byte getIntegrityValue(final int pIndex) {
388         return (pIndex + 1) % INTEGRITY_MODULO < 2
389                 ? INTEGRITY_VALUE1
390                 : INTEGRITY_VALUE2;
391     }

425     @Override
426     public int getKeyWrapLength(final GordianKey<?> pKey) {
427         /* Obtain the id of the keySpec */
428         final GordianKeySpec mySpec = pKey.getKeyType();
429         final AlgorithmIdentifier myAlgId = theFactory.getIdentifierForSpec(mySpec);
430 
431         /* Determine wrapped key length */
432         final int myDataLen = getDataWrapLength(mySpec.getKeyLength().getByteLength());
433 
434         /* return the calculated length */
435         return GordianWrappedKeyASN1.getEncodedLength(myAlgId, myDataLen);
436     }

438     @Override
439     public int getDataWrapLength(final int pDataLength) {
440         final GordianLength myBlockLen = getKeySpec().getBlockLength();
441         return getKeyWrapLength(pDataLength, myBlockLen)
442                 + getKeyWrapExpansion(myBlockLen);
443     }

446     public int getPrivateKeyWrapLength(final GordianKeyPair pKeyPair) throws GordianException {
447         /* Determine and check the keySpec */

454     /**
455      * Obtain initial wrapLength for a particular dataLength and BlockSize.
456      *
457      * @param pDataLength the data length
458      * @param pBlockLen   the number of bits in the blockLen
459      * @return the keyWrap expansion
460      */
461     public static int getKeyWrapLength(final int pDataLength,
462                                        final GordianLength pBlockLen) {
463         final int myBlockLen = pBlockLen.getByteLength() >> 1;
464         final int myNumBlocks = (pDataLength + myBlockLen - 1) / myBlockLen;
465         return myNumBlocks * myBlockLen;
466     }
467 
468     /**
469      * Obtain keyWrapExpansion for a blockLen.
470      *
471      * @param pBlockLen the number of bits in the blockLen
472      * @return the keyWrap expansion
473      */
474     public static int getKeyWrapExpansion(final GordianLength pBlockLen) {
475         final int myBlockLen = pBlockLen.getByteLength() >> 1;
476         final int myNumBlocks = 1 + (myBlockLen <= Integer.BYTES
477                 ? 2
478                 : 1);
479         return myNumBlocks * myBlockLen;
480     }
