135      * @param pFactory the factory

138     public GordianCoreRandomFactory(final GordianBaseFactory pFactory) throws GordianException {
139         /* Access the initial stringRandom */
140         theFactory = pFactory;
141         theRandomSource = theFactory.getRandomSource();
142         theRandom = theRandomSource.getRandom();
143 
144         /* Store parameters and create an entropy provider */
145         theEntropyProvider = new BasicEntropySourceProvider(theRandom, true);
146 
147         /* Create a random combinedRandom and register it */
148         theRandomSource.setRandom(generateRandomCombined());
149     }

151     @Override
152     public GordianSecureRandom createRandom(final GordianRandomSpec pRandomSpec) throws GordianException {
153         /* Check validity of RandomSpec */
154         if (!supportedRandomSpecs().test(pRandomSpec)) {
155             throw new GordianDataException(GordianBaseData.getInvalidText(pRandomSpec));
156         }
157 
158         /* Access factories */
159         final GordianCoreDigestFactory myDigests = (GordianCoreDigestFactory) theFactory.getDigestFactory();
160         final GordianCoreCipherFactory myCiphers = (GordianCoreCipherFactory) theFactory.getCipherFactory();
161         final GordianCoreMacFactory myMacs = (GordianCoreMacFactory) theFactory.getMacFactory();
162 
163         /* Access the digestSpec */
164         final GordianDigestSpec myDigest = pRandomSpec.getDigestSpec();
165         final boolean isResistent = pRandomSpec.isPredictionResistant();
166         switch (pRandomSpec.getRandomType()) {
167             case HASH:
168                 return buildHash(myDigests.createDigest(myDigest), isResistent);
169             case HMAC:
170                 final GordianMacSpec myMacSpec = GordianMacSpecBuilder.hMac(myDigest);
171                 return buildHMAC(myMacs.createMac(myMacSpec), isResistent);
172             case CTR:
173                 GordianSymCipherSpec myCipherSpec = GordianSymCipherSpecBuilder.ecb(pRandomSpec.getSymKeySpec(), GordianPadding.NONE);
174                 return buildCTR(myCiphers.createSymKeyCipher(myCipherSpec), isResistent);
175             case X931:
176                 myCipherSpec = GordianSymCipherSpecBuilder.ecb(pRandomSpec.getSymKeySpec(), GordianPadding.NONE);
177                 return buildX931(myCiphers.createSymKeyCipher(myCipherSpec), isResistent);
178             default:
179                 throw new GordianDataException(GordianBaseData.getInvalidText(pRandomSpec));
180         }
181     }

183     @Override
184     public GordianCombinedRandom createRandom(final GordianRandomSpec pCtrSpec,
185                                               final GordianRandomSpec pHashSpec) throws GordianException {
186         /* Check validity of ctrSpecs */
187         if (!validCombinedSpec(pCtrSpec, pHashSpec)) {
188             throw new GordianDataException(GordianBaseData.getInvalidText(pCtrSpec)
189                     + "-" + pHashSpec);
190         }
191 
192         /* Build pair of randoms */
193         final GordianSecureRandom myCtr = createRandom(pCtrSpec);
194         final GordianSecureRandom myHash = createRandom(pHashSpec);
195 
196         /* return the combined random */
197         return new GordianCombinedRandom(myCtr, myHash);
198     }

210     /**
211      * Check RandomSpec.
212      *
213      * @param pRandomSpec the randomSpec
214      * @return true/false
215      */
216     private boolean validRandomSpec(final GordianRandomSpec pRandomSpec) {
217         /* Reject invalid randomSpec */
218         if (pRandomSpec == null || !pRandomSpec.isValid()) {
219             return false;
220         }
221 
222         /* Access details */
223         final GordianRandomType myType = pRandomSpec.getRandomType();
224         final GordianDigestSpec myDigest = pRandomSpec.getDigestSpec();
225         final GordianSymKeySpec mySymKey = pRandomSpec.getSymKeySpec();
226 
227         /* Check that the randomType is supported */
228         switch (myType) {
229             case HASH:
230                 final GordianCoreDigestFactory myDigests = (GordianCoreDigestFactory) theFactory.getDigestFactory();
231                 return myDigest != null && myDigests.validDigestSpec(myDigest);
232             case HMAC:
233                 final GordianCoreMacFactory myMacs = (GordianCoreMacFactory) theFactory.getMacFactory();
234                 return myDigest != null && myMacs.validHMacSpec(myDigest);
235             case CTR:
236             case X931:
237                 final GordianCoreCipherFactory myCiphers = (GordianCoreCipherFactory) theFactory.getCipherFactory();
238                 return mySymKey != null && myCiphers.validSymKeySpec(mySymKey);
239             default:
240                 return false;
241         }
242     }

244     /**
245      * Check combined RandomSpec.
246      *
247      * @param pCtrSpec  the Counter Spec.
248      * @param pHashSpec the Hash Spec
249      * @return true/false
250      */
251     private boolean validCombinedSpec(final GordianRandomSpec pCtrSpec,
252                                       final GordianRandomSpec pHashSpec) {
253         /* Check validity of ctrSpecs */
254         if (!supportedRandomSpecs().test(pCtrSpec)
255                 || pCtrSpec.getRandomType() != GordianRandomType.CTR
256                 || pCtrSpec.getSymKeySpec().getKeyLength() != GordianLength.LEN_128
257                 || pCtrSpec.getSymKeySpec().getBlockLength() != GordianLength.LEN_128) {
258             return false;
259         }
260 
261         /* Validate the hashSpec */
262         return supportedRandomSpecs().test(pHashSpec)
263                 && pHashSpec.getRandomType() == GordianRandomType.HASH
264                 && pHashSpec.getDigestSpec().getDigestLength() == GordianLength.LEN_512;
265     }

267     /**
268      * Create a random combinedRandom.
269      *
270      * @return the combinedRandom
271      * @throws GordianException on error
272      */
273     GordianCombinedRandom generateRandomCombined() throws GordianException {
274         /* Create a random ctrSpec */
275         final GordianSymKeySpec mySymKeySpec = generateRandomSymKeySpec();
276         final GordianRandomSpec myCtrSpec = GordianRandomSpecBuilder.ctr(mySymKeySpec);
277 
278         /* Create a random hashSpec */
279         final GordianDigestSpec myDigestSpec = generateRandomDigestSpec();
280         final GordianRandomSpec myHashSpec = GordianRandomSpecBuilder.hash(myDigestSpec);
281 
282         /* Build the combinedRandom */
283         return createRandom(myCtrSpec, myHashSpec);
284     }

286     /**
287      * Obtain random SymKeySpec with blockLength/keySize of 128 bits.
288      *
289      * @return the random symKeySpec
290      */
291     private GordianSymKeySpec generateRandomSymKeySpec() {
292         /* Access the list of symKeySpecs and unique symKeyTypes */
293         final GordianCoreCipherFactory myCiphers = (GordianCoreCipherFactory) theFactory.getCipherFactory();
294         final List<GordianSymKeySpec> mySpecs = myCiphers.listAllSupportedSymKeySpecs(GordianLength.LEN_128);
295 
296         /* Remove the specs that are wrong block size and obtain keyTypes */
297         mySpecs.removeIf(s -> s.getBlockLength() != GordianLength.LEN_128);
298         final List<GordianSymKeyType> myTypes
299                 = mySpecs.stream().map(GordianSymKeySpec::getSymKeyType).collect(Collectors.toCollection(ArrayList::new));
300 
301         /* Determine a random index into the list and obtain the symKeyType */
302         int myIndex = theRandom.nextInt(myTypes.size());
303         final GordianSymKeyType myKeyType = myTypes.get(myIndex);
304 
305         /* Select from among possible keySpecs of this type */
306         mySpecs.removeIf(s -> s.getSymKeyType() != myKeyType);
307         myIndex = theRandom.nextInt(mySpecs.size());
308         return mySpecs.get(myIndex);
309     }

311     /**
312      * Obtain random DigestSpec for a large data output length of 512-bits.
313      *
314      * @return the random digestSpec
315      */
316     private GordianDigestSpec generateRandomDigestSpec() {
317         /* Access the list to select from */
318         final GordianCoreDigestFactory myDigests = (GordianCoreDigestFactory) theFactory.getDigestFactory();
319         final List<GordianDigestSpec> mySpecs = myDigests.listAllSupportedSpecs();
320         mySpecs.removeIf(s -> !s.getDigestType().supportsLargeData()
321                 || s.getDigestLength() != GordianLength.LEN_512);
322         final List<GordianDigestType> myTypes
323                 = mySpecs.stream().map(GordianDigestSpec::getDigestType).collect(Collectors.toCollection(ArrayList::new));
324 
325         /* Determine a random index into the list and obtain the digestType */
326         final SecureRandom myRandom = theFactory.getRandomSource().getRandom();
327         int myIndex = myRandom.nextInt(myTypes.size());
328         final GordianDigestType myDigestType = myTypes.get(myIndex);
329 
330         /* Select from among possible digestSpecs of this type */
331         mySpecs.removeIf(s -> s.getDigestType() != myDigestType);
332         myIndex = myRandom.nextInt(mySpecs.size());
333         return mySpecs.get(myIndex);
334     }

348     public GordianMac generateRandomMac(final GordianLength pKeyLen,
349                                         final boolean pLargeData) throws GordianException {
350         /* Access Mac Factory and IdManager */
351         final GordianMacFactory myMacs = theFactory.getMacFactory();
352         final GordianIdManager myIds = theFactory.getIdManager();
353 
354         /* Determine a random specification */
355         final GordianMacSpec mySpec = myIds.generateRandomMacSpec(pKeyLen, pLargeData);
356 
357         /* Determine a random key */
358         final GordianKeyGenerator<GordianMacSpec> myGenerator = myMacs.getKeyGenerator(mySpec);
359         final GordianKey<GordianMacSpec> myKey = myGenerator.generateKey();
360 
361         /* Create and initialise the MAC */
362         final GordianMac myMac = myMacs.createMac(mySpec);
363         myMac.init(GordianMacParameters.keyWithRandomNonce(myKey));
364 
365         /* Return it */
366         return myMac;
367     }

398     /**
399      * Build a SecureRandom based on a SP 800-90A Hash DRBG.
400      *
401      * @param pDigest               digest to use in the DRBG underneath the SecureRandom.
402      * @param isPredictionResistant specify whether the underlying DRBG in the resulting
403      *                              SecureRandom should re-seed on each request for bytes.
404      * @return a SecureRandom supported by a Hash DRBG.
405      */
406     private GordianSecureRandom buildHash(final GordianDigest pDigest,
407                                           final boolean isPredictionResistant) {
408         /* Create initVector */
409         final byte[] myInit = theRandom.generateSeed(NUM_ENTROPY_BYTES_REQUIRED);
410 
411         /* Build DRBG */
412         final EntropySource myEntropy = theEntropyProvider.get(NUM_ENTROPY_BITS_REQUIRED);
413         final GordianSP800HashDRBG myProvider = new GordianSP800HashDRBG(pDigest, myEntropy, theRandomSource.defaultPersonalisation(), myInit);
414         return new GordianSecureRandom(myProvider, theRandom, myEntropy, isPredictionResistant);
415     }

459     /**
460      * Build a SecureRandom based on a X931 Cipher DRBG.
461      *
462      * @param pCipher               ctr cipher to use in the DRBG underneath the SecureRandom.
463      * @param isPredictionResistant specify whether the underlying DRBG in the resulting
464      *                              SecureRandom should re-seed on each request for bytes.
465      * @return a SecureRandom supported by a HMAC DRBG.
466      * @throws GordianException on error
467      */
468     private GordianSecureRandom buildX931(final GordianSymCipher pCipher,
469                                           final boolean isPredictionResistant) throws GordianException {
470         /* Initialise the cipher with a random key */
471         final GordianCipherFactory myCiphers = theFactory.getCipherFactory();
472         final GordianKeyGenerator<GordianSymKeySpec> myGenerator = myCiphers.getKeyGenerator(pCipher.getKeyType());
473         final GordianKey<GordianSymKeySpec> myKey = myGenerator.generateKey();
474         pCipher.initForEncrypt(GordianCipherParameters.key(myKey));
475 
476         /* Build DRBG */
477         final EntropySource myEntropy = theEntropyProvider.get(pCipher.getKeyType().getBlockLength().getLength());
478         final GordianX931CipherDRBG myProvider = new GordianX931CipherDRBG(pCipher, myEntropy, theRandomSource.defaultPersonalisation());
479         return new GordianSecureRandom(myProvider, theRandom, myEntropy, isPredictionResistant);
480     }

499     @Override
500     public List<GordianRandomSpec> listAllSupportedRandomSpecs(final GordianRandomType pType,
501                                                                final GordianLength pKeyLen) {
502         return listAllPossibleSpecs()
503                 .stream()
504                 .filter(s -> s.getRandomType().equals(pType))
505                 .filter(s -> s.getRandomType().hasSymKeySpec())
506                 .filter(s -> s.getSymKeySpec().getKeyLength() == pKeyLen)
507                 .filter(supportedRandomSpecs())
508                 .toList();

511     /**
512      * List all possible randomSpecs.
513      *
514      * @return the list
515      */
516     private List<GordianRandomSpec> listAllPossibleSpecs() {
517         /* Create the array list */
518         final List<GordianRandomSpec> myList = new ArrayList<>();
519 
520         /* For each digestSpec */
521         for (final GordianDigestSpec mySpec : theFactory.getDigestFactory().listAllPossibleSpecs()) {
522             /* Add a hash random */
523             myList.add(GordianRandomSpecBuilder.hash(mySpec));
524             myList.add(GordianRandomSpecBuilder.hashResist(mySpec));
525 
526             /* Add an hMac random */
527             myList.add(GordianRandomSpecBuilder.hMac(mySpec));
528             myList.add(GordianRandomSpecBuilder.hMacResist(mySpec));
529         }
530 
531         /* For each KeyLength */
532         final Iterator<GordianLength> myIterator = GordianKeyLengths.iterator();
