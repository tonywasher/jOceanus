167     /**
168      * Process cipherParameters.
169      *
170      * @param pParams the cipher parameters
171      * @throws GordianException on error
172      */
173     void processParameters(final GordianCipherParameters pParams) throws GordianException {
174         /* If the cipher parameters are PBE */
175         if (pParams instanceof GordianPBECipherParameters myPBEParams) {
176             /* Process separately */
177             processPBEParameters(myPBEParams);
178 
179             /* else standard parameters */
180         } else {
181             /* Show that we are not PBE */
182             thePBESpec = null;
183             thePBESalt = null;
184 
185             /* Access the key details */
186             theKey = obtainKeyFromParameters(pParams);
187             theInitVector = obtainNonceFromParameters(pParams, false);
188             theInitialAEAD = obtainInitialAEADFromParameters(pParams);
189         }
190     }
191 
192     /**
193      * Process cipherParameters.
194      *
195      * @param pParams the cipher parameters
196      * @throws GordianException on error
197      */
198     private void processPBEParameters(final GordianPBECipherParameters pParams) throws GordianException {
199         /* Check that the PBE parameters are supported */
200         final GordianPBESpec myPBESpec = pParams.getPBESpec();
201         final GordianPBECipherSpec<T> myPBECipherSpec = new GordianPBECipherSpec<>(myPBESpec, theSpec);
202         final GordianBaseCipherFactory myCipherFactory = (GordianBaseCipherFactory) theFactory.getCipherFactory();
203         if (!myCipherFactory.supportedPBECipherSpecs().test(myPBECipherSpec)) {
204             throw new GordianDataException(GordianBaseData.getInvalidText(myPBECipherSpec));
205         }
206 
207         /* Access PBE details */
208         thePBESpec = myPBESpec;
209         thePBESalt = obtainNonceFromParameters(pParams, true);
210 
211         /* Switch on the PBE type */
212         CipherParameters myParams;
213         switch (thePBESpec.getPBEType()) {
214             case PBKDF2:
215                 myParams = derivePBKDF2Parameters(pParams.getPassword());
216                 break;
217             case PKCS12:
218                 myParams = derivePKCS12Parameters(pParams.getPassword());
219                 break;
220             case SCRYPT:
221                 myParams = deriveSCRYPTParameters(pParams.getPassword());
222                 break;
223             case ARGON2:
224             default:
225                 myParams = deriveArgon2Parameters(pParams.getPassword());
226                 break;
227         }
228 
229         /* Store details */
230         theInitialAEAD = null;
231         theInitVector = null;
232         if (myParams instanceof ParametersWithIV myIVParams) {
233             theInitVector = myIVParams.getIV();
234             myParams = myIVParams.getParameters();
235         }
236         theKey = buildKeyFromBytes(((KeyParameter) myParams).getKey());
237     }

239     /**
240      * Build a key from bytes.
241      *
242      * @param pKeyBytes the bytes to use
243      * @return the key
244      * @throws GordianException on error
245      */
246     GordianKey<T> buildKeyFromBytes(final byte[] pKeyBytes) throws GordianException {
247         /* Create generator if needed */
248         if (theGenerator == null) {
249             final GordianCipherFactory myFactory = theFactory.getCipherFactory();
250             theGenerator = (GordianCoreKeyGenerator<T>) myFactory.getKeyGenerator(theSpec.getKeyType());
251         }
252 
253         /* Create the key */
254         return theGenerator.buildKeyFromBytes(pKeyBytes);
255     }

257     /**
258      * Obtain Key from CipherParameters.
259      *
260      * @param pParams parameters
261      * @return the key
262      */
263     @SuppressWarnings("unchecked")
264     private GordianKey<T> obtainKeyFromParameters(final GordianCipherParameters pParams) {
265         /* If we have specified IV */
266         if (pParams instanceof GordianKeyCipherParameters<?> myParams) {
267             return (GordianKey<T>) myParams.getKey();
268         }
269 
270         /* No key */
271         return null;
272     }

274     /**
275      * Obtain Nonce from CipherParameters.
276      *
277      * @param pParams  parameters
278      * @param pPBESalt is this a PBESalt
279      * @return the nonce
280      */
281     private byte[] obtainNonceFromParameters(final GordianCipherParameters pParams,
282                                              final boolean pPBESalt) {
283         /* Default IV is null */
284         byte[] myIV = null;
285 
286         /* If we have specified IV */
287         if (pParams instanceof GordianNonceParameters myParams) {
288             /* If we have an explicit Nonce */
289             if (!myParams.randomNonce()) {
290                 /* access the nonce */
291                 myIV = Arrays.clone(myParams.getNonce());
292 
293                 /* Else if we actually need a nonce */
294             } else if (pPBESalt || theSpec.needsIV()) {
295                 /* Create a random IV */
296                 final int myLen = pPBESalt ? PBESALTLEN : theSpec.getIVLength();
297                 myIV = new byte[myLen];
298                 theRandom.getRandom().nextBytes(myIV);
299             }
300         }
301 
302         /* return the IV */
303         return myIV;
304     }

325     /**
326      * derive PBKDF2 key and IV.
327      *
328      * @param pPassword the password
329      * @return the parameters
330      */
331     private CipherParameters derivePBKDF2Parameters(final char[] pPassword) {
332         /* Protect password bytes */
333         byte[] myPassword = null;
334         try {
335             /* Create the digest */
336             final GordianPBEDigestAndCountSpec mySpec = (GordianPBEDigestAndCountSpec) thePBESpec;
337             final Digest myDigest = new SHA512Digest();
338 
339             /* Create the generator and initialise it */
340             final PKCS5S2ParametersGenerator myGenerator = new PKCS5S2ParametersGenerator(myDigest);
341             myPassword = PBEParametersGenerator.PKCS5PasswordToUTF8Bytes(pPassword);
342             myGenerator.init(myPassword, thePBESalt, mySpec.getIterationCount());
343 
344             /* Generate the parameters */
345             final int myKeyLen = theSpec.getKeyType().getKeyLength().getLength();
346             final int myIVLen = theSpec.needsIV() ? Byte.SIZE * theSpec.getIVLength() : 0;
347             return myIVLen == 0
348                     ? myGenerator.generateDerivedParameters(myKeyLen)
349                     : myGenerator.generateDerivedParameters(myKeyLen, myIVLen);
350         } finally {
351             if (myPassword != null) {
352                 Arrays.fill(myPassword, (byte) 0);
353             }
354         }
355     }

360      * @param pPassword the password

363     private CipherParameters derivePKCS12Parameters(final char[] pPassword) {

377             final int myKeyLen = theSpec.getKeyType().getKeyLength().getLength();
378             final int myIVLen = theSpec.needsIV() ? Byte.SIZE * theSpec.getIVLength() : 0;
379             return myIVLen == 0
380                     ? myGenerator.generateDerivedParameters(myKeyLen)
381                     : myGenerator.generateDerivedParameters(myKeyLen, myIVLen);

389     /**
390      * derive SCRYPT key and IV.
391      *
392      * @param pPassword the password
393      * @return the parameters
394      */
395     private CipherParameters deriveSCRYPTParameters(final char[] pPassword) {
396         /* Protect password bytes */
397         byte[] myPassword = null;
398         try {
399             /* Access the password as bytes */
400             final GordianPBESCryptSpec mySpec = (GordianPBESCryptSpec) thePBESpec;
401             myPassword = PBEParametersGenerator.PKCS5PasswordToUTF8Bytes(pPassword);
402 
403             /* Generate the bytes */
404             final int myKeyLen = theSpec.getKeyType().getKeyLength().getByteLength();
405             final int myIVLen = theSpec.needsIV() ? theSpec.getIVLength() : 0;
406             final int myBufLen = myIVLen + myKeyLen;
407             final byte[] myBuffer = SCrypt.generate(myPassword, thePBESalt, mySpec.getCost(),
408                     mySpec.getBlockSize(), mySpec.getParallel(), myBufLen);
409 
410             /* Convert to parameters */
411             final KeyParameter myKeyParm = new KeyParameter(myBuffer, 0, myKeyLen);
412             return myIVLen == 0
413                     ? myKeyParm
414                     : new ParametersWithIV(myKeyParm, myBuffer, myKeyLen, myIVLen);
415         } finally {
416             if (myPassword != null) {
417                 Arrays.fill(myPassword, (byte) 0);
418             }
419         }
420     }

422     /**
423      * derive Argon2 key and IV.
424      *
425      * @param pPassword the password
426      * @return the parameters
427      */
428     private CipherParameters deriveArgon2Parameters(final char[] pPassword) {
429         /* Create the parameters */
430         final GordianPBEArgon2Spec mySpec = (GordianPBEArgon2Spec) thePBESpec;
431         final Argon2Parameters.Builder myBuilder = new Argon2Parameters.Builder();
432         myBuilder.withSalt(thePBESalt);
433         myBuilder.withIterations(mySpec.getIterationCount());
434         myBuilder.withParallelism(mySpec.getLanes());
435         myBuilder.withMemoryAsKB(mySpec.getMemory());
436         final Argon2Parameters myParams = myBuilder.build();
437 
438         /* Generate the bytes */
439         final Argon2BytesGenerator myGenerator = new Argon2BytesGenerator();
440         myGenerator.init(myParams);
441         final int myKeyLen = theSpec.getKeyType().getKeyLength().getByteLength();
442         final int myIVLen = theSpec.needsIV() ? theSpec.getIVLength() : 0;
443         final int myBufLen = myIVLen + myKeyLen;
444         final byte[] myBuffer = new byte[myBufLen];
445         myGenerator.generateBytes(pPassword, myBuffer);
446 
447         /* Convert to parameters */
448         final KeyParameter myKeyParm = new KeyParameter(myBuffer, 0, myKeyLen);
449         return myIVLen == 0
450                 ? myKeyParm
451                 : new ParametersWithIV(myKeyParm, myBuffer, myKeyLen, myIVLen);
452     }
