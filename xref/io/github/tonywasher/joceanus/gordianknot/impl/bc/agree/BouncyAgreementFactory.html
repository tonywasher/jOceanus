31  /**
32   * Bouncy Agreement Factory.
33   */
34  public class BouncyAgreementFactory
35          extends GordianCoreAgreementFactory {
36      /**
37       * Constructor.
38       *
39       * @param pFactory the factory
40       */
41      public BouncyAgreementFactory(final GordianBaseFactory pFactory) {
42          super(pFactory);
43      }
44  
45      @Override
46      public GordianCoreAgreementEngine createEngine(final GordianAgreementSpec pSpec) throws GordianException {
47          final GordianCoreAgreementSpec mySpec = (GordianCoreAgreementSpec) pSpec;
48          switch (pSpec.getKeyPairSpec().getKeyPairType()) {
49              case RSA:
50                  return new BouncyRSAAgreementEngine(this, mySpec);
51              case EC, GOST, DSTU, SM2:
52                  return getBCECEngine(mySpec);
53              case DH:
54                  return getBCDHEngine(mySpec);
55              case NEWHOPE:
56                  return new BouncyNewHopeAgreementEngine(this, mySpec);
57              case CMCE:
58                  return new BouncyCMCEAgreementEngine(this, mySpec);
59              case FRODO:
60                  return new BouncyFrodoAgreementEngine(this, mySpec);
61              case SABER:
62                  return new BouncySABERAgreementEngine(this, mySpec);
63              case MLKEM:
64                  return new BouncyMLKEMAgreementEngine(this, mySpec);
65              case HQC:
66                  return new BouncyHQCAgreementEngine(this, mySpec);
67              case BIKE:
68                  return new BouncyBIKEAgreementEngine(this, mySpec);
69              case NTRU:
70                  return new BouncyNTRUAgreementEngine(this, mySpec);
71              case NTRUPLUS:
72                  return new BouncyNTRUPlusAgreementEngine(this, mySpec);
73              case NTRUPRIME:
74                  final GordianCoreKeyPairSpec myKeySpec = (GordianCoreKeyPairSpec) pSpec.getKeyPairSpec();
75                  final GordianCoreNTRUPrimeSpec myPrimeSpec = myKeySpec.getNTRUPrimeSpec();
76                  return myPrimeSpec.getType() == GordianNTRUPrimeType.NTRUL
77                          ? new BouncyNTRULPrimeAgreementEngine(this, mySpec)
78                          : new BouncySNTRUPrimeAgreementEngine(this, mySpec);
79              case XDH:
80                  return getBCXDHEngine(mySpec);
81              case COMPOSITE:
82              default:
83                  return super.createEngine(pSpec);
84          }
85      }
86  
87      /**
88       * Create the BouncyCastle EC Agreement.
89       *
90       * @param pSpec the agreementSpec
91       * @return the Agreement
92       * @throws GordianException on error
93       */
94      private GordianCoreAgreementEngine getBCECEngine(final GordianCoreAgreementSpec pSpec) throws GordianException {
95          return switch (pSpec.getAgreementType()) {
96              case KEM -> new BouncyECIESAgreementEngine(this, pSpec);
97              case ANON -> new BouncyECAnonAgreementEngine(this, pSpec);
98              case SIGNED, BASIC -> new BouncyECBasicAgreementEngine(this, pSpec);
99              case MQV -> new BouncyECMQVAgreementEngine(this, pSpec);
100             case UNIFIED -> new BouncyECUnifiedAgreementEngine(this, pSpec);
101             case SM2 -> new BouncySM2AgreementEngine(this, pSpec);
102             default -> throw new GordianDataException(GordianBaseData.getInvalidText(pSpec));
103         };
104     }
105 
106     /**
107      * Create the BouncyCastle DH Agreement.
108      *
109      * @param pSpec the agreementSpec
110      * @return the Agreement
111      * @throws GordianException on error
112      */
113     private GordianCoreAgreementEngine getBCDHEngine(final GordianCoreAgreementSpec pSpec) throws GordianException {
114         return switch (pSpec.getAgreementType()) {
115             case ANON -> new BouncyDHAnonAgreementEngine(this, pSpec);
116             case SIGNED, BASIC -> new BouncyDHBasicAgreementEngine(this, pSpec);
117             case MQV -> new BouncyDHMQVAgreementEngine(this, pSpec);
118             case UNIFIED -> new BouncyDHUnifiedAgreementEngine(this, pSpec);
119             default -> throw new GordianDataException(GordianBaseData.getInvalidText(pSpec));
120         };
121     }
122 
123     /**
124      * Create the BouncyCastle XDH Agreement.
125      *
126      * @param pSpec the agreementSpec
127      * @return the Agreement
128      * @throws GordianException on error
129      */
130     private GordianCoreAgreementEngine getBCXDHEngine(final GordianCoreAgreementSpec pSpec) throws GordianException {
131         return switch (pSpec.getAgreementType()) {
132             case ANON -> new BouncyXDHAnonAgreementEngine(this, pSpec);
133             case SIGNED, BASIC -> new BouncyXDHBasicAgreementEngine(this, pSpec);
134             case UNIFIED -> new BouncyXDHUnifiedAgreementEngine(this, pSpec);
135             default -> throw new GordianDataException(GordianBaseData.getInvalidText(pSpec));
136         };
137     }
138 }
