18  package io.github.tonywasher.joceanus.gordianknot.impl.bc.encrypt;
19  
20  import io.github.tonywasher.joceanus.gordianknot.api.base.GordianException;
21  import io.github.tonywasher.joceanus.gordianknot.impl.bc.keypair.BouncyRSAKeyPair.BouncyRSAPrivateKey;
22  import io.github.tonywasher.joceanus.gordianknot.impl.bc.keypair.BouncyRSAKeyPair.BouncyRSAPublicKey;
23  import io.github.tonywasher.joceanus.gordianknot.impl.core.base.GordianBaseFactory;
24  import io.github.tonywasher.joceanus.gordianknot.impl.core.spec.encrypt.GordianCoreEncryptorSpec;
25  import org.bouncycastle.crypto.engines.RSABlindedEngine;
26  
27  /**
28   * RSA Encryptor.
29   */
30  public class BouncyRSAEncryptor
31          extends BouncyCoreEncryptor {
32      /**
33       * Constructor.
34       *
35       * @param pFactory the factory
36       * @param pSpec    the encryptorSpec
37       * @throws GordianException on error
38       */
39      BouncyRSAEncryptor(final GordianBaseFactory pFactory,
40                         final GordianCoreEncryptorSpec pSpec) throws GordianException {
41          /* Initialise underlying cipher */
42          super(pFactory, pSpec, new RSABlindedEngine());
43      }
44  
45      @Override
46      protected BouncyRSAPublicKey getPublicKey() {
47          return (BouncyRSAPublicKey) super.getPublicKey();
48      }
49  
50      @Override
51      protected BouncyRSAPrivateKey getPrivateKey() {
52          return (BouncyRSAPrivateKey) super.getPrivateKey();
53      }
54  }
