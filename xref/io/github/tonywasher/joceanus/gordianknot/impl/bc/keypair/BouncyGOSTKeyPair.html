36  /**
37   * GOST KeyPair classes.
38   */
39  public final class BouncyGOSTKeyPair {
40      /**
41       * Private constructor.
42       */
43      private BouncyGOSTKeyPair() {
44      }
45  
46      /**
47       * BouncyCastle GOST KeyPair generator.
48       */
49      public static class BouncyGOSTKeyPairGenerator
50              extends BouncyKeyPairGenerator<ECPrivateKeyParameters, ECPublicKeyParameters> {
51          /**
52           * Constructor.
53           *
54           * @param pFactory the Security Factory
55           * @param pKeySpec the keySpec
56           */
57          BouncyGOSTKeyPairGenerator(final GordianBaseFactory pFactory,
58                                     final GordianKeyPairSpec pKeySpec) {
59              /* Initialise underlying class */
60              super(pFactory, pKeySpec);
61  
62              /* Determine domain */
63              final GordianCoreKeyPairSpec myKeySpec = (GordianCoreKeyPairSpec) pKeySpec;
64              final String myCurve = myKeySpec.getElliptic().getCurveName();
65              final GOST3410ParameterSpec mySpec = new GOST3410ParameterSpec(myCurve);
66              final X9ECParameters x9 = ECGOST3410NamedCurves.getByNameX9(myCurve);
67              final ECDomainParameters myDomain = new ECNamedDomainParameters(mySpec.getPublicKeyParamSet(), x9);
68  
69              /* initialise */
70              final ECKeyGenerationParameters myParams = new ECKeyGenerationParameters(
71                      new ECGOST3410Parameters(myDomain, mySpec.getPublicKeyParamSet(), mySpec.getDigestParamSet(),
72                              mySpec.getEncryptionParamSet()), getRandom());
73  
74              /* Create and initialise the generator */
75              setGenerator(new ECKeyPairGenerator(), myParams);
76              setFactorySet(BouncyStdKeyFactorySet.INSTANCE);
77          }
78  
79          @Override
80          BouncyECPrivateKey newPrivateKey(final AsymmetricKeyParameter pThat) {
81              return new BouncyECPrivateKey(getKeySpec(), (ECPrivateKeyParameters) pThat);
82          }
83  
84          @Override
85          BouncyECPublicKey newPublicKey(final AsymmetricKeyParameter pThat) {
86              return new BouncyECPublicKey(getKeySpec(), (ECPublicKeyParameters) pThat);
87          }
88      }
89  }
