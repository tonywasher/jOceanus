45  public class JcaMacFactory
46          extends GordianCoreMacFactory {
47      /**
48       * CMAC algorithm name.
49       */
50      private static final String CMAC_ALGORITHM = "CMAC";
51  
52      /**
53       * GOST algorithm name.
54       */
55      private static final String GOST_ALGORITHM = "GOST28147";
56  
57      /**
58       * ZUC-128 algorithm name.
59       */
60      private static final String ZUC256_ALGORITHM = "ZUC-256";
61  
62      /**
63       * KeyGenerator Cache.
64       */
65      private final Map<GordianKeySpec, JcaKeyGenerator<? extends GordianKeySpec>> theCache;
66  
67      /**
68       * Constructor.
69       *
70       * @param pFactory the factory
71       */
72      JcaMacFactory(final GordianBaseFactory pFactory) {
73          /* Initialise underlying class */
74          super(pFactory);
75  
76          /* Create the cache */
77          theCache = new HashMap<>();
78      }
79  
80      @Override
81      @SuppressWarnings("unchecked")
82      public <T extends GordianKeySpec> GordianKeyGenerator<T> getKeyGenerator(final T pMacSpec) throws GordianException {
83          /* Look up in the cache */
84          JcaKeyGenerator<T> myGenerator = (JcaKeyGenerator<T>) theCache.get(pMacSpec);
85          if (myGenerator == null) {
86              /* Check validity of MacSpec */
87              checkMacSpec(pMacSpec);
88  
89              /* Create the new generator */
90              final String myAlgorithm = getMacSpecAlgorithm((GordianCoreMacSpec) pMacSpec);
91              final KeyGenerator myJavaGenerator = getJavaKeyGenerator(myAlgorithm);
92              myGenerator = new JcaKeyGenerator<>(getFactory(), pMacSpec, myJavaGenerator);
93  
94              /* Add to cache */
95              theCache.put(pMacSpec, myGenerator);
96          }
97          return myGenerator;
98      }
99  
100     @Override
101     public JcaMac createMac(final GordianMacSpec pMacSpec) throws GordianException {
102         /* Check validity of MacSpec */
103         checkMacSpec(pMacSpec);
104 
105         /* Create Mac */
106         final Mac myJavaMac = getJavaMac((GordianCoreMacSpec) pMacSpec);
107         return new JcaMac(getFactory(), pMacSpec, myJavaMac);
108     }

110     @Override
111     protected boolean validMacType(final GordianMacType pMacType) {
112         return switch (pMacType) {
113             case BLAKE2, BLAKE3, KALYNA, CBCMAC, CFBMAC -> false;
114             default -> super.validMacType(pMacType);
115         };
116     }

118     /**
119      * Create the BouncyCastle MAC via JCA.
120      *
121      * @param pMacSpec the MacSpec
122      * @return the MAC
123      * @throws GordianException on error
124      */
125     private Mac getJavaMac(final GordianCoreMacSpec pMacSpec) throws GordianException {
126         return switch (pMacSpec.getMacType()) {
127             case HMAC, GMAC, CMAC, POLY1305, SKEIN, KUPYNA, SIPHASH, GOST, ZUC, KMAC ->
128                     getJavaMac(getMacSpecAlgorithm(pMacSpec));
129             case VMPC -> getJavaMac("VMPC-MAC");
130             default -> throw new GordianDataException(GordianBaseData.getInvalidText(pMacSpec));
131         };
132     }
133 
134     /**
135      * Create the BouncyCastle MAC via JCA.
136      *
137      * @param pAlgorithm the Algorithm
138      * @return the MAC
139      * @throws GordianException on error
140      */
141     private static Mac getJavaMac(final String pAlgorithm) throws GordianException {
142         /* Protect against exceptions */
143         try {
144             /* Return a MAC for the algorithm */
145             return Mac.getInstance(pAlgorithm, JcaProvider.BCPROV);
146 
147             /* Catch exceptions */
148         } catch (NoSuchAlgorithmException e) {
149             /* Throw the exception */
150             throw new GordianCryptoException("Failed to create Mac", e);
151         }
152     }
153 
154     /**
155      * Create the BouncyCastle KeyGenerator via JCA.
156      *
157      * @param pAlgorithm the Algorithm
158      * @return the KeyGenerator
159      * @throws GordianException on error
160      */
161     private static KeyGenerator getJavaKeyGenerator(final String pAlgorithm) throws GordianException {
162         /* Protect against exceptions */
163         try {
164             /* Massage the keyGenerator name */
165             String myAlgorithm = pAlgorithm;
166 
167             /* CMAC generators use the symKeyGenerator */
168             if (myAlgorithm.endsWith(CMAC_ALGORITHM)) {
169                 myAlgorithm = myAlgorithm.substring(0, myAlgorithm.length() - CMAC_ALGORITHM.length());
170             }
171 
172             /* GOST generators use the GOST28147 key generator */
173             if (myAlgorithm.startsWith(GOST_ALGORITHM)) {
174                 myAlgorithm = GOST_ALGORITHM;
175             }
176 
177             /* ZUC-256 generators use the ZUC-256 key generator */
178             if (myAlgorithm.startsWith(ZUC256_ALGORITHM)) {
179                 myAlgorithm = ZUC256_ALGORITHM;
180             }
181 
182             /* Return a KeyGenerator for the algorithm */
183             return KeyGenerator.getInstance(myAlgorithm, JcaProvider.BCPROV);
184 
185             /* Catch exceptions */
186         } catch (NoSuchAlgorithmException e) {
187             /* Throw the exception */
188             throw new GordianCryptoException("Failed to create KeyGenerator", e);
189         }
190     }

192     /**
193      * Obtain the MacSpec Key algorithm.
194      *
195      * @param pMacSpec the MacSpec
196      * @return the Algorithm
197      * @throws GordianException on error
198      */
199     private static String getMacSpecAlgorithm(final GordianCoreMacSpec pMacSpec) throws GordianException {
200         return switch (pMacSpec.getMacType()) {
201             case HMAC -> getHMacAlgorithm(pMacSpec.getDigestSpec());
202             case GMAC -> getGMacAlgorithm(pMacSpec.getSymKeySpec());
203             case CMAC -> getCMacAlgorithm(pMacSpec.getSymKeySpec());
204             case POLY1305 -> getPoly1305Algorithm(pMacSpec.getSymKeySpec());
205             case SKEIN -> getSkeinMacAlgorithm(pMacSpec.getDigestSpec());
206             case KUPYNA -> getKupynaMacAlgorithm(pMacSpec.getDigestSpec());
207             case ZUC -> getZucMacAlgorithm(pMacSpec);
208             case SIPHASH -> pMacSpec.toString();
209             case KMAC -> pMacSpec.getMacType().toString() + pMacSpec.getDigestSpec().getCoreDigestState();
210             case GOST -> "GOST28147MAC";
211             case VMPC -> "VMPC-KSA3";
212             default -> throw new GordianDataException(GordianBaseData.getInvalidText(pMacSpec));
213         };
214     }

251     /**
252      * Obtain the Poly1305 algorithm.
253      *
254      * @param pKeySpec the symmetric keySpec
255      * @return the algorithm
256      * @throws GordianException on error
257      */
258     private static String getPoly1305Algorithm(final GordianSymKeySpec pKeySpec) throws GordianException {
259         /* Return algorithm name */
260         return pKeySpec == null
261                 ? "POLY1305"
262                 : "POLY1305-" + JcaCipherFactory.getSymKeyAlgorithm(pKeySpec);
263     }

265     /**
266      * Obtain the Skein MAC algorithm.
267      *
268      * @param pSpec the digestSpec
269      * @return the algorithm
270      */
271     private static String getSkeinMacAlgorithm(final GordianDigestSpec pSpec) {
272         return "Skein-MAC-"
273                 + ((GordianCoreDigestSpec) pSpec).getCoreDigestState()
274                 + '-'
275                 + pSpec.getDigestLength();
276     }
277 
278     /**
279      * Obtain the Skein MAC algorithm.
280      *
281      * @param pSpec the digestSpec
282      * @return the algorithm
283      */
284     private static String getZucMacAlgorithm(final GordianCoreMacSpec pSpec) {
285         final String myKeyLen = Integer.toString(pSpec.getKeyLength().getLength());
286         final String myMacLen = Integer.toString(pSpec.getMacLength().getLength());
287         final StringBuilder myBuilder = new StringBuilder();
288         myBuilder.append("ZUC-")
289                 .append(myKeyLen);
290         if (GordianLength.LEN_256 == pSpec.getKeyLength()) {
291             myBuilder.append('-')
292                     .append(myMacLen);
293         }
294         return myBuilder.toString();
295     }

305         /* For some reason this is accessed as HMAC !!! */
306         return getHMacAlgorithm(pSpec);

309     @Override
310     protected boolean validCMacSymKeySpec(final GordianSymKeySpec pKeySpec) {
311         return switch (pKeySpec.getSymKeyType()) {
312             case AES, DESEDE, BLOWFISH, SHACAL2, THREEFISH, SEED, SM4 -> super.validCMacSymKeySpec(pKeySpec);
313             default -> false;
314         };
315     }

317     @Override
318     protected boolean validGMacSymKeySpec(final GordianSymKeySpec pKeySpec) {
319         return switch (pKeySpec.getSymKeyType()) {
320             case KUZNYECHIK, KALYNA -> false;
321             default -> super.validGMacSymKeySpec(pKeySpec);
322         };
323     }

325     @Override
326     protected boolean validPoly1305SymKeySpec(final GordianSymKeySpec pKeySpec) {
327         if (GordianSymKeyType.KALYNA.equals(pKeySpec.getSymKeyType())) {
328             return false;
329         }
330         return super.validPoly1305SymKeySpec(pKeySpec);
