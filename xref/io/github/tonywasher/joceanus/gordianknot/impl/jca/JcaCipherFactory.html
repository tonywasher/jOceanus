59   * Jca Cipher Factory.

61  public class JcaCipherFactory
62          extends GordianCoreCipherFactory {
63      /**
64       * Cipher Algorithm Separator.
65       */
66      private static final Character ALGO_SEP = '/';
67  
68      /**
69       * Kalyna algorithm name.
70       */
71      private static final String KALYNA_ALGORITHM = "DSTU7624";
72  
73      /**
74       * KeyGenerator Cache.
75       */
76      private final Map<GordianKeySpec, JcaKeyGenerator<? extends GordianKeySpec>> theCache;
77  
78      /**
79       * Constructor.
80       *
81       * @param pFactory the factory
82       */
83      JcaCipherFactory(final GordianBaseFactory pFactory) {
84          /* Initialise underlying class */
85          super(pFactory);
86  
87          /* Create the cache */
88          theCache = new HashMap<>();
89      }
90  
91      @Override
92      @SuppressWarnings("unchecked")
93      public <T extends GordianKeySpec> GordianKeyGenerator<T> getKeyGenerator(final T pKeySpec) throws GordianException {
94          /* Look up in the cache */
95          JcaKeyGenerator<T> myGenerator = (JcaKeyGenerator<T>) theCache.get(pKeySpec);
96          if (myGenerator == null) {
97              /* Check the KeySpec */
98              checkKeySpec(pKeySpec);
99  
100             /* Create the new generator */
101             final String myAlgorithm = getKeyAlgorithm(pKeySpec);
102             final KeyGenerator myJavaGenerator = getJavaKeyGenerator(myAlgorithm);
103             myGenerator = new JcaKeyGenerator<>(getFactory(), pKeySpec, myJavaGenerator);
104 
105             /* Add to cache */
106 
107             theCache.put(pKeySpec, myGenerator);
108         }
109         return myGenerator;
110     }

112     @Override
113     public GordianSymCipher createSymKeyCipher(final GordianSymCipherSpec pCipherSpec) throws GordianException {
114         /* Check validity of SymKeySpec */
115         checkSymCipherSpec(pCipherSpec);
116         final GordianCoreSymCipherSpec mySpec = (GordianCoreSymCipherSpec) pCipherSpec;
117 
118         /* If this is an AAD cipher */
119         if (mySpec.isAAD()) {
120             /* Create the cipher */
121             final Cipher myBCCipher = getJavaCipher(pCipherSpec);
122             return new JcaSymAEADCipher(getFactory(), mySpec, myBCCipher);
123 
124             /* else create the standard cipher */
125         } else {
126             /* Create the cipher */
127             final Cipher myBCCipher = getJavaCipher(pCipherSpec);
128             return new JcaSymCipher(getFactory(), mySpec, myBCCipher);
129         }
130     }
131 
132     @Override
133     public GordianStreamCipher createStreamKeyCipher(final GordianStreamCipherSpec pCipherSpec) throws GordianException {
134         /* Check validity of StreamKeySpec */
135         checkStreamCipherSpec(pCipherSpec);
136         final GordianCoreStreamCipherSpec mySpec = (GordianCoreStreamCipherSpec) pCipherSpec;
137 
138         final Cipher myJCACipher = getJavaCipher(pCipherSpec);
139         return pCipherSpec.isAEAD()
140                 ? new JcaStreamAEADCipher(getFactory(), mySpec, myJCACipher)
141                 : new JcaStreamCipher(getFactory(), mySpec, myJCACipher);
142     }
143 
144     @Override
145     public GordianWrapper createKeyWrapper(final GordianKey<GordianSymKeySpec> pKey) throws GordianException {
146         /* Create the cipher */
147         final JcaKey<GordianSymKeySpec> myKey = JcaKey.accessKey(pKey);
148         final GordianSymCipherSpecBuilder myBuilder = GordianCoreSymCipherSpecBuilder.newInstance();
149         final GordianSymCipherSpec mySpec = myBuilder.ecb(myKey.getKeyType(), GordianPadding.NONE);
150         final JcaSymCipher myJcaCipher = (JcaSymCipher) createSymKeyCipher(mySpec);
151         return createKeyWrapper(myKey, myJcaCipher);
152     }

154     /**
155      * Obtain the algorithm for the keySpec.
156      *
157      * @param pKeySpec the keySpec
158      * @param <T>      the SpecType
159      * @return the name of the algorithm
160      * @throws GordianException on error
161      */
162     private static <T extends GordianKeySpec> String getKeyAlgorithm(final T pKeySpec) throws GordianException {
163         if (pKeySpec instanceof GordianStreamKeySpec mySpec) {
164             return getStreamKeyAlgorithm(mySpec);
165         }
166         if (pKeySpec instanceof GordianSymKeySpec mySpec) {
167             return getSymKeyAlgorithm(mySpec);
168         }
169         throw new GordianDataException(GordianBaseData.getInvalidText(pKeySpec));
170     }

172     /**
173      * Create the BouncyCastle KeyGenerator via JCA.
174      *
175      * @param pAlgorithm the Algorithm
176      * @return the KeyGenerator
177      * @throws GordianException on error
178      */
179     private static KeyGenerator getJavaKeyGenerator(final String pAlgorithm) throws GordianException {
180         /* Protect against exceptions */
181         try {
182             /* Massage the keyGenerator name */
183             String myAlgorithm = pAlgorithm;
184 
185             /* Note that DSTU7624 has only a single keyGenerator */
186             if (myAlgorithm.startsWith(KALYNA_ALGORITHM)) {
187                 myAlgorithm = KALYNA_ALGORITHM;
188             }
189 
190             /* Return a KeyGenerator for the algorithm */
191             return KeyGenerator.getInstance(myAlgorithm, JcaProvider.BCPROV);
192 
193             /* Catch exceptions */
194         } catch (NoSuchAlgorithmException e) {
195             /* Throw the exception */
196             throw new GordianCryptoException("Failed to create KeyGenerator", e);
197         }
198     }
199 
200     /**
201      * Create the BouncyCastle SymKey Cipher via JCA.
202      *
203      * @param pCipherSpec the cipherSpec
204      * @return the Cipher
205      * @throws GordianException on error
206      */
207     private static Cipher getJavaCipher(final GordianSymCipherSpec pCipherSpec) throws GordianException {
208         final String myAlgo = getSymKeyAlgorithm(pCipherSpec.getKeySpec())
209                 + ALGO_SEP
210                 + getCipherModeAlgorithm(pCipherSpec)
211                 + ALGO_SEP
212                 + getPaddingAlgorithm(pCipherSpec.getPadding());
213         return getJavaCipher(myAlgo);
214     }
215 
216     /**
217      * Create the BouncyCastle StreamKey Cipher via JCA.
218      *
219      * @param pCipherSpec the StreamCipherSpec
220      * @return the Cipher
221      * @throws GordianException on error
222      */
223     private static Cipher getJavaCipher(final GordianStreamCipherSpec pCipherSpec) throws GordianException {
224         final GordianStreamKeySpec myKeySpec = pCipherSpec.getKeySpec();
225         String myAlgo = getStreamKeyAlgorithm(myKeySpec);
226         if (pCipherSpec.isAEAD()
227                 && GordianStreamKeyType.CHACHA20 == myKeySpec.getStreamKeyType()) {
228             myAlgo = "CHACHA20-POLY1305";
229         }
230         return getJavaCipher(myAlgo);
231     }
232 
233     /**
234      * Create the StreamKey Cipher via JCA.
235      *
236      * @param pAlgorithm the Algorithm
237      * @return the KeyGenerator
238      * @throws GordianException on error
239      */
240     private static Cipher getJavaCipher(final String pAlgorithm) throws GordianException {
241         /* Protect against exceptions */
242         try {
243             /* Return a Cipher for the algorithm */
244             return Cipher.getInstance(pAlgorithm, JcaProvider.BCPROV);
245 
246             /* Catch exceptions */
247         } catch (NoSuchAlgorithmException
248                  | NoSuchPaddingException e) {
249             /* Throw the exception */
250             throw new GordianCryptoException("Failed to create Cipher", e);
251         }
252     }

254     /**
255      * Obtain the SymKey algorithm.
256      *
257      * @param pKeySpec the keySpec
258      * @return the Algorithm
259      * @throws GordianException on error
260      */
261     static String getSymKeyAlgorithm(final GordianSymKeySpec pKeySpec) throws GordianException {
262         return switch (pKeySpec.getSymKeyType()) {
263             case TWOFISH -> "TwoFish";
264             case SERPENT -> "Serpent";
265             case THREEFISH -> "ThreeFish-" + pKeySpec.getBlockLength().getLength();
266             case GOST -> "GOST28147";
267             case SHACAL2 -> "Shacal-2";
268             case KUZNYECHIK -> "GOST3412-2015";
269             case KALYNA -> KALYNA_ALGORITHM + "-" + pKeySpec.getBlockLength().getLength();
270             case RC5 -> GordianLength.LEN_128.equals(pKeySpec.getBlockLength())
271                     ? "RC5-64"
272                     : "RC5";
273             case AES, CAMELLIA, CAST6, RC6, ARIA, NOEKEON, SM4, SEED, SKIPJACK, TEA, XTEA, IDEA, RC2, CAST5, BLOWFISH,
274                  DESEDE -> pKeySpec.getSymKeyType().name();
275             default -> throw new GordianDataException(GordianBaseData.getInvalidText(pKeySpec));
276         };
277     }
278 
279     /**
280      * Obtain the CipherMode algorithm.
281      *
282      * @param pSpec the cipherSpec
283      * @return the Algorithm
284      * @throws GordianException on error
285      */
286     private static String getCipherModeAlgorithm(final GordianSymCipherSpec pSpec) throws GordianException {
287         final GordianCipherMode myMode = pSpec.getCipherMode();
288         return switch (pSpec.getCipherMode()) {
289             case ECB, SIC, EAX, CCM, GCM, OCB, GOFB, GCFB, CFB8, OFB8 -> myMode.name();
290             case CBC, G3413CBC -> GordianCipherMode.CBC.name();
291             case CFB, G3413CFB -> "CFB";
292             case OFB, G3413OFB -> GordianCipherMode.OFB.name();
293             case KCTR, G3413CTR -> "CTR";
294             case KCCM -> GordianCipherMode.CCM.name();
295             case KGCM -> GordianCipherMode.GCM.name();
296             default -> throw new GordianDataException(GordianBaseData.getInvalidText(myMode));
297         };
298     }
299 
300     /**
301      * Obtain the Padding algorithm.
302      *
303      * @param pPadding use padding true/false
304      * @return the Algorithm
305      */
306     private static String getPaddingAlgorithm(final GordianPadding pPadding) {
307         return switch (pPadding) {
308             case CTS -> "withCTS";
309             case X923 -> "X923Padding";
310             case PKCS7 -> "PKCS7Padding";
311             case ISO7816D4 -> "ISO7816-4Padding";
312             case TBC -> "TBCPadding";
313             default -> "NoPadding";
314         };
315     }

317     /**
318      * Obtain the StreamKey algorithm.
319      *
320      * @param pKeySpec the keySpec
321      * @return the Algorithm
322      * @throws GordianException on error
323      */
324     private static String getStreamKeyAlgorithm(final GordianStreamKeySpec pKeySpec) throws GordianException {
325         return switch (pKeySpec.getStreamKeyType()) {
326             case HC -> GordianLength.LEN_128 == pKeySpec.getKeyLength()
327                     ? "HC128"
328                     : "HC256";
329             case ZUC -> GordianLength.LEN_128 == pKeySpec.getKeyLength()
330                     ? "ZUC-128"
331                     : "ZUC-256";
332             case CHACHA20 -> pKeySpec.getSubKeyType() == GordianChaCha20Key.STD
333                     ? "CHACHA"
334                     : "CHACHA7539";
335             case SALSA20 -> pKeySpec.getSubKeyType() == GordianSalsa20Key.STD
336                     ? pKeySpec.getStreamKeyType().name()
337                     : "XSALSA20";
338             case VMPC -> pKeySpec.getSubKeyType() == GordianVMPCKey.STD
339                     ? pKeySpec.getStreamKeyType().name()
340                     : "VMPC-KSA3";
341             case GRAIN -> "Grain128";
342             case ISAAC, RC4 -> pKeySpec.getStreamKeyType().name();
343             default -> throw new GordianDataException(GordianBaseData.getInvalidText(pKeySpec));
344         };
345     }

347     @Override
348     protected boolean validStreamKeySpec(final GordianStreamKeySpec pKeySpec) {
349         /* Check basic validity */
350         if (!super.validStreamKeySpec(pKeySpec)) {
351             return false;
352         }
353 
354         /* Reject XChaCha20 */
355         return pKeySpec.getStreamKeyType() != GordianStreamKeyType.CHACHA20
356                 || pKeySpec.getSubKeyType() != GordianChaCha20Key.XCHACHA;
357     }
358 
359     @Override
360     protected boolean validSymCipherSpec(final GordianSymCipherSpec pCipherSpec) {
361         /* Check standard features */
362         if (!super.validSymCipherSpec(pCipherSpec)) {
363             return false;
364         }
365 
366         /* Disallow GCM-SIV */
367         final GordianCipherMode myMode = pCipherSpec.getCipherMode();
368         if (GordianCipherMode.GCMSIV.equals(myMode)) {
369             return false;
370         }
371 
372         /* Additional Checks */
373         return switch (pCipherSpec.getKeySpec().getSymKeyType()) {
374             case KALYNA ->
375                 /* Disallow OCB, CCM and GCM */
376                     !GordianCipherMode.OCB.equals(myMode)
377                             && !GordianCipherMode.KCCM.equals(myMode)
378                             && !GordianCipherMode.KGCM.equals(myMode)
379                             && !GordianCipherMode.CCM.equals(myMode)
380                             && !GordianCipherMode.GCM.equals(myMode);
381             case GOST ->
382                 /* Disallow OFB and CFB */
383                     !GordianCipherMode.OFB.equals(myMode)
384                             && !GordianCipherMode.CFB.equals(myMode);
385             case KUZNYECHIK ->
386                 /* Disallow OCB, OFB, CFB and CBC */
387                     !GordianCipherMode.OCB.equals(myMode)
388                             && !GordianCipherMode.OFB.equals(myMode)
389                             && !GordianCipherMode.CFB.equals(myMode)
390                             && !GordianCipherMode.CBC.equals(myMode);
391             default -> true;
392         };
393     }
