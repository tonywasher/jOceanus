17  package io.github.tonywasher.joceanus.gordianknot.api.base;
18  
19  import java.lang.ref.WeakReference;
20  import java.util.Map;
21  import java.util.ResourceBundle;
22  
23  /**
24   * Bundle Loader. (Cloned from OceanusBundleLoader)
25   */
26  public final class GordianBundleLoader {
27      /**
28       * Interface for bundleIds.
29       */
30      public interface GordianBundleId {
31          /**
32           * Get Key Name.
33           *
34           * @return the key name.
35           */
36          String getKeyName();
37  
38          /**
39           * Get NameSpace.
40           *
41           * @return the nameSpace.
42           */
43          String getNameSpace();
44  
45          /**
46           * Get Value.
47           *
48           * @return the value.
49           */
50          String getValue();
51      }
52  
53      /**
54       * String Builder.
55       */
56      private final StringBuilder theBuilder = new StringBuilder();
57  
58      /**
59       * BundleName.
60       */
61      private final String theBundleName;
62  
63      /**
64       * ResourceBundle.
65       */
66      private WeakReference<ResourceBundle> theBundle;
67  
68      /**
69       * Standard constructor.
70       *
71       * @param pBundleName the Bundle Name
72       */
73      private GordianBundleLoader(final String pBundleName) {
74          theBundleName = pBundleName;
75      }
76  
77      /**
78       * Obtain standard loader.
79       *
80       * @param pBundleName the Bundle Name
81       * @return the loader
82       */
83      public static GordianBundleLoader getLoader(final String pBundleName) {
84          return new GordianBundleLoader(pBundleName);
85      }
86  
87      /**
88       * Build name of resource for Key.
89       *
90       * @param <K>  the resource key data type
91       * @param pKey the resource Key
92       * @return the resourceName
93       */
94      public <K extends Enum<K> & GordianBundleId> String getValue(final K pKey) {
95          /* Build the required name */
96          theBuilder.setLength(0);
97          theBuilder.append(pKey.getNameSpace());
98          theBuilder.append('.');
99          theBuilder.append(pKey.getKeyName());
100 
101         /* Access the cached bundle */
102         ResourceBundle myBundle = theBundle == null
103                 ? null
104                 : theBundle.get();
105 
106         /* If the bundle is not cached */
107         if (myBundle == null) {
108             /* Access and cache the bundle */
109             myBundle = ResourceBundle.getBundle(theBundleName);
110             theBundle = new WeakReference<>(myBundle);
111         }
112 
113         /* Access the required resource */
114         return myBundle.getString(theBuilder.toString());
115     }
116 
117     /**
118      * Obtain key for enum.
119      *
120      * @param <E>    the enum type
121      * @param pMap   the map
122      * @param pValue the enum value
123      * @return the resource key
124      */
125     public static <E extends Enum<E>> GordianBundleId getKeyForEnum(final Map<E, GordianBundleId> pMap,
126                                                                     final E pValue) {
127         final GordianBundleId myId = pMap.get(pValue);
128         if (myId == null) {
129             throw new IllegalArgumentException(getErrorNoResource(pValue));
130         }
131         return myId;
132     }
133 
134     /**
135      * Error for missing resource.
136      *
137      * @param pId the missing id
138      * @return the error message
139      */
140     private static String getErrorNoResource(final Enum<?> pId) {
141         return "Missing Resource: "
142                 + pId.getClass().getName()
143                 + ':'
144                 + pId.name();
145     }
146 }
