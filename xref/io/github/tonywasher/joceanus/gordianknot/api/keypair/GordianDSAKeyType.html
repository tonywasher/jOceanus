17  package io.github.tonywasher.joceanus.gordianknot.api.keypair;
18  
19  import org.bouncycastle.asn1.x509.DSAParameter;
20  
21  /**
22   * DSA KeyTypes.
23   */
24  public enum GordianDSAKeyType {
25      /**
26       * 2048.
27       */
28      MOD2048(2048, 256),
29  
30      /**
31       * 1024, 160.
32       */
33      MOD1024(1024, 160),
34  
35      /**
36       * 3072.
37       */
38      MOD3072(3072, 256);
39  
40      /**
41       * The keySize.
42       */
43      private final int theKeySize;
44  
45      /**
46       * The hashSize.
47       */
48      private final int theHashSize;
49  
50      /**
51       * Constructor.
52       *
53       * @param pKeySize  the keySize
54       * @param pHashSize the hashSize
55       */
56      GordianDSAKeyType(final int pKeySize,
57                        final int pHashSize) {
58          theKeySize = pKeySize;
59          theHashSize = pHashSize;
60      }
61  
62      /**
63       * Obtain the keySize of the keyType.
64       *
65       * @return the keySize
66       */
67      public int getKeySize() {
68          return theKeySize;
69      }
70  
71      /**
72       * Obtain the hashSize of the keyType.
73       *
74       * @return the hashSize
75       */
76      public int getHashSize() {
77          return theHashSize;
78      }
79  
80      /**
81       * Obtain the DSAKeyType for DSAParameters.
82       *
83       * @param pParams the parameters
84       * @return the DSAKeyType
85       */
86      public static GordianDSAKeyType getDSATypeForParms(final DSAParameter pParams) {
87          /* Loop through the values */
88          final int myLen = pParams.getP().bitLength();
89          final int myHashSize = pParams.getQ().bitLength();
90          for (GordianDSAKeyType myType : values()) {
91              if (myType.getKeySize() == myLen
92                      && myType.getHashSize() == myHashSize) {
93                  return myType;
94              }
95          }
96          return null;
97      }
98  
99      @Override
100     public String toString() {
101         return name();
102     }
103 }
