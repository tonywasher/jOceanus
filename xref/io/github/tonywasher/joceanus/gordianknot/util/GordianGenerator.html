17  package io.github.tonywasher.joceanus.gordianknot.util;
18  
19  import io.github.tonywasher.joceanus.gordianknot.api.base.GordianException;
20  import io.github.tonywasher.joceanus.gordianknot.api.factory.GordianFactory;
21  import io.github.tonywasher.joceanus.gordianknot.api.factory.GordianFactoryType;
22  import io.github.tonywasher.joceanus.gordianknot.impl.bc.BouncyFactory;
23  import io.github.tonywasher.joceanus.gordianknot.impl.core.base.GordianFactoryGenerator;
24  import io.github.tonywasher.joceanus.gordianknot.impl.core.base.GordianParameters;
25  import io.github.tonywasher.joceanus.gordianknot.impl.jca.JcaFactory;
26  
27  /**
28   * Factory generator.
29   */
30  public final class GordianGenerator {
31      /**
32       * Private  Constructor.
33       */
34      private GordianGenerator() {
35      }
36  
37      /**
38       * Create a new factory instance.
39       *
40       * @param pFactoryType the factoryType
41       * @return the new factory
42       * @throws GordianException on error
43       */
44      public static GordianFactory createFactory(final GordianFactoryType pFactoryType) throws GordianException {
45          /* Create a factory with null security phrase */
46          return createFactory(pFactoryType, null);
47      }
48  
49      /**
50       * Create a new factory instance.
51       *
52       * @param pFactoryType    the factoryType
53       * @param pSecurityPhrase the securityPhrase
54       * @return the new factory
55       * @throws GordianException on error
56       */
57      public static GordianFactory createFactory(final GordianFactoryType pFactoryType,
58                                                 final char[] pSecurityPhrase) throws GordianException {
59          /* Allocate a generator and the parameters */
60          final GordianFactoryGenerator myGenerator = new GordianUtilGenerator();
61          final GordianParameters myParams = new GordianParameters(pFactoryType, pSecurityPhrase);
62          return myGenerator.newFactory(myParams);
63      }
64  
65      /**
66       * Create a new random bouncyCastle factory instance.
67       *
68       * @param pFactoryType the factory type
69       * @return the new factory
70       * @throws GordianException on error
71       */
72      public static GordianFactory createRandomFactory(final GordianFactoryType pFactoryType) throws GordianException {
73          /* Allocate a generator and the parameters */
74          final GordianFactoryGenerator myGenerator = new GordianUtilGenerator();
75          final GordianParameters myParams = GordianParameters.randomParams(pFactoryType);
76          return myGenerator.newFactory(myParams);
77      }
78  
79      /**
80       * True Factory generator.
81       */
82      static class GordianUtilGenerator
83              implements GordianFactoryGenerator {
84          /**
85           * Constructor.
86           */
87          GordianUtilGenerator() {
88          }
89  
90          @Override
91          public GordianFactory newFactory(final GordianParameters pParameters) throws GordianException {
92              /* Allocate the factory */
93              return GordianFactoryType.BC.equals(pParameters.getFactoryType())
94                      ? new BouncyFactory(this, pParameters)
95                      : new JcaFactory(this, pParameters);
96          }
97      }
98  }
