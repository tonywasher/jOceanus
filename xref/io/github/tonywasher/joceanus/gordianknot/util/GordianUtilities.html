72      /**
73       * Obtain Maximum KeyWrapLength.
74       *
75       * @return the maximum keyWrap size
76       */
77      public static int getMaximumKeyWrapLength() {
78          return GordianCoreKeySet.getDataWrapLength(GordianLength.LEN_256.getByteLength());
79      }
80  
81      /**
82       * Obtain Maximum KeyWrapLength.
83       *
84       * @return the maximum keyWrap size
85       */
86      public static int getMaximumKeySetWrapLength() {
87          final int my128 = GordianCoreKeySet.getKeySetWrapLength(GordianLength.LEN_128);
88          final int my256 = GordianCoreKeySet.getKeySetWrapLength(GordianLength.LEN_256);
89          return Math.max(my128, my256);
90      }
91  
92      /**
93       * Obtain FactoryLockLen.
94       *
95       * @return the factoryLock length
96       */
97      public static int getFactoryLockLen() {
98          return GordianFactoryLockImpl.getEncodedLength();
99      }
