17  package io.github.tonywasher.joceanus.moneywise.threads;

19  import io.github.tonywasher.joceanus.oceanus.base.OceanusException;
20  import io.github.tonywasher.joceanus.moneywise.archive.MoneyWiseArchiveLoader;

24  import io.github.tonywasher.joceanus.prometheus.preference.PrometheusBackup.PrometheusBackupPreferences;
25  import io.github.tonywasher.joceanus.prometheus.preference.PrometheusPreferenceManager;
26  import io.github.tonywasher.joceanus.prometheus.toolkit.PrometheusToolkit;
27  import io.github.tonywasher.joceanus.tethys.api.thread.TethysUIThread;
28  import io.github.tonywasher.joceanus.tethys.api.thread.TethysUIThreadManager;
29  
30  /**
31   * LoaderThread extension to load an archive spreadsheet.
32   */
33  public class MoneyWiseThreadLoadArchive
34          implements TethysUIThread<MoneyWiseDataSet> {
35      /**
36       * Data Control.
37       */
38      private final MoneyWiseView theView;
39  
40      /**
41       * Constructor (Event Thread).
42       *
43       * @param pView the view
44       */
45      public MoneyWiseThreadLoadArchive(final MoneyWiseView pView) {
46          theView = pView;
47      }
48  
49      @Override
50      public String getTaskName() {
51          return MoneyWiseThreadId.LOADARCHIVE.toString();
52      }
53  
54      @Override
55      public MoneyWiseDataSet performTask(final TethysUIThreadManager pManager) throws OceanusException {
56          /* Initialise the status window */
57          pManager.initTask(getTaskName());
58  
59          /* Load workbook */
60          final PrometheusToolkit myPromToolkit = (PrometheusToolkit) pManager.getThreadData();
61          final PrometheusPreferenceManager myMgr = myPromToolkit.getPreferenceManager();
62          final MoneyWiseDataSet myData = theView.getNewData();
63          final MoneyWiseArchiveLoader myLoader = new MoneyWiseArchiveLoader(myData);
64          myLoader.loadArchive(pManager, myMgr.getPreferenceSet(PrometheusBackupPreferences.class));
65  
66          /* Initialise the status window */
67          pManager.initTask("Analysing Data");
68  
69          /* Analyse the Data to ensure that close dates are updated */
70          theView.analyseData(myData);
71  
72          /* Initialise the status window */
73          pManager.initTask("Accessing DataStore");
74  
75          /* Create interface */
76          final PrometheusDataStore myDatabase = theView.getDatabase();
77  
78          /* Protect against failures */
79          try {
80              /* Load underlying database */
81              final MoneyWiseDataSet myStore = theView.getNewData();
82              myDatabase.loadDatabase(pManager, myStore);
83  
84              /* Check security on the database */
85              myStore.checkSecurity(pManager);
86              if (myStore.hasUpdates()) {
87                  /* Store any updates */
88                  myDatabase.updateDatabase(pManager, myStore);
89              }
90  
91              /* Initialise the security, either from database or with a new security control */
92              myData.initialiseSecurity(pManager, myStore);
93  
94              /* Re-base the loaded spreadsheet onto the database image */
95              myData.reBase(pManager, myStore);
96  
97              /* State that we have completed */
98              pManager.setCompletion();
99  
100             /* Return the loaded data */
101             return myData;
102 
103             /* Make sure that the database is closed */
104         } finally {
105             /* Close the database */
106             myDatabase.close();
107         }
108     }
109 
110     @Override
111     public void processResult(final MoneyWiseDataSet pResult) {
112         theView.setData(pResult);
113     }
114 }
