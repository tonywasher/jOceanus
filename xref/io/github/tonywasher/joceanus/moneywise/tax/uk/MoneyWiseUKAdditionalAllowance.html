17  package io.github.tonywasher.joceanus.moneywise.tax.uk;
18  
19  import io.github.tonywasher.joceanus.oceanus.decimal.OceanusMoney;
20  import io.github.tonywasher.joceanus.oceanus.format.OceanusDataFormatter;
21  import io.github.tonywasher.joceanus.metis.field.MetisFieldSet;
22  import io.github.tonywasher.joceanus.moneywise.tax.MoneyWiseTaxResource;
23  
24  /**
25   * Additional Band UK Tax Allowance.
26   */
27  public class MoneyWiseUKAdditionalAllowance
28          extends MoneyWiseUKAgeAllowance {
29      /**
30       * Local Report fields.
31       */
32      private static final MetisFieldSet<MoneyWiseUKAdditionalAllowance> FIELD_DEFS = MetisFieldSet.newFieldSet(MoneyWiseUKAdditionalAllowance.class);
33  
34      /*
35       * Declare Fields.
36       */
37      static {
38          FIELD_DEFS.declareLocalField(MoneyWiseTaxResource.LIMIT_ADDALLOWANCE, MoneyWiseUKAdditionalAllowance::getAdditionalAllowanceLimit);
39      }
40  
41      /**
42       * IncomeBoundary.
43       */
44      private final OceanusMoney theAddAllowLimit;
45  
46      /**
47       * Constructor.
48       *
49       * @param pAllowance         the allowance
50       * @param pRentalAllowance   the rental allowance
51       * @param pCapitalAllowance  the capital allowance
52       * @param pLoAgeAllowance    the low age allowance
53       * @param pHiAgeAllowance    the high age allowance
54       * @param pAgeAllowanceLimit the age allowance limit
55       * @param pAddAllowLimit     the additional allowance limit
56       */
57      protected MoneyWiseUKAdditionalAllowance(final OceanusMoney pAllowance,
58                                               final OceanusMoney pRentalAllowance,
59                                               final OceanusMoney pCapitalAllowance,
60                                               final OceanusMoney pLoAgeAllowance,
61                                               final OceanusMoney pHiAgeAllowance,
62                                               final OceanusMoney pAgeAllowanceLimit,
63                                               final OceanusMoney pAddAllowLimit) {
64          super(pAllowance, pRentalAllowance, pCapitalAllowance, pLoAgeAllowance, pHiAgeAllowance, pAgeAllowanceLimit);
65          theAddAllowLimit = pAddAllowLimit;
66      }
67  
68      /**
69       * Obtain the additional Allowance limit.
70       *
71       * @return the Limit
72       */
73      protected OceanusMoney getAdditionalAllowanceLimit() {
74          return theAddAllowLimit;
75      }
76  
77      @Override
78      protected OceanusMoney calculateBasicAllowance(final MoneyWiseUKTaxConfig pConfig) {
79          /* Determine AgeRelated allowance */
80          OceanusMoney myAllowance = super.calculateBasicAllowance(pConfig);
81  
82          /* If we have additional tax possible and we are above the allowance limit */
83          final OceanusMoney myGross = pConfig.getGrossTaxable();
84          if (myGross.compareTo(theAddAllowLimit) > 0) {
85              /* Calculate and apply the reduction */
86              final OceanusMoney myReduction = getMarginalReduction().calculateReduction(myGross, theAddAllowLimit);
87              myAllowance = new OceanusMoney(myAllowance);
88              myAllowance.subtractAmount(myReduction);
89  
90              /* If we have reduced below zero */
91              if (!myAllowance.isPositive()) {
92                  /* Set the allowance to zero */
93                  myAllowance.setZero();
94              }
95          }
96  
97          /* Return the allowance */
98          return myAllowance;
99      }
100 
101     @Override
102     public MetisFieldSet<MoneyWiseUKAdditionalAllowance> getDataFieldSet() {
103         return FIELD_DEFS;
104     }
105 
106     @Override
107     public String formatObject(final OceanusDataFormatter pFormatter) {
108         return FIELD_DEFS.getName();
109     }
110 }
