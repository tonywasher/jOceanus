521         /**
522          * Constructor.
523          */
524         private TransactionDetail() {
525             /* Store detail */
526             theAccount = theCurrent.getAccount();
527             thePartner = theCurrent.getPartner();
528             theDirection = theCurrent.getDirection();
529             theCategory = theCurrent.getCategory();
530             theTaxCredit = theCurrent.getTaxCredit();
531             theEmployerNatIns = theCurrent.getEmployerNatIns();
532             theEmployeeNatIns = theCurrent.getEmployeeNatIns();
533             theBenefit = theCurrent.getDeemedBenefit();
534             theWithheld = theCurrent.getWithheld();
535             theReturnedCashAccount = theCurrent.getReturnedCashAccount();
536             theAccountUnits = theCurrent.getAccountDeltaUnits();
537             thePartnerUnits = theCurrent.getPartnerDeltaUnits();
538             theDilution = theCurrent.getDilution();
539 
540             /* Obtain the amounts */
541             final OceanusMoney myAmount = theCurrent.getAmount();
542             final OceanusMoney myPartnerAmount = theCurrent.getPartnerAmount();
543             final OceanusMoney myReturnedCash = theCurrent.getReturnedCash();
544 
545             /* Determine account prices */
546             theAccountPrice = theAccount instanceof MoneyWiseSecurityHolding
547                     ? thePriceCursor.getSecurityPrice(((MoneyWiseSecurityHolding) theAccount).getSecurity(), theDate)
548                     : null;
549             thePartnerPrice = thePartner instanceof MoneyWiseSecurityHolding
550                     ? thePriceCursor.getSecurityPrice(((MoneyWiseSecurityHolding) thePartner).getSecurity(), theDate)
551                     : null;
552 
553             /* Determine foreign account detail */
554             MoneyWiseCurrency myActCurrency = theAccount.getAssetCurrency();
555             theForeignAccount = MetisDataDifference.isEqual(myActCurrency, theCurrency)
556                     ? null
557                     : new ForeignAccountDetail(this, myActCurrency, myAmount);
558 
559             /* If we have a partner amount */
560             myActCurrency = thePartner.getAssetCurrency();
561             theForeignPartner = myActCurrency == null
562                     || MetisDataDifference.isEqual(myActCurrency, theCurrency)
563                     ? null
564                     : new ForeignPartnerDetail(myActCurrency, myPartnerAmount);
565 
566             /* If we have a returnedCash account */
567             if (theReturnedCashAccount != null) {
568                 /* Determine foreign returnedCash detail */
569                 myActCurrency = theReturnedCashAccount.getAssetCurrency();
570                 theForeignReturnedCash = MetisDataDifference.isEqual(myActCurrency, theCurrency)
571                         ? null
572                         : new ForeignPartnerDetail(myActCurrency, myReturnedCash);
573             } else {
574                 theForeignReturnedCash = null;
575             }
576 
577             /* Determine the local amounts */
578             theAmount = theForeignAccount == null
579                     ? myAmount
580                     : theForeignPartner == null
581                     ? myPartnerAmount
582                     : theForeignAccount.theAmount;
583             theReturnedCash = theForeignReturnedCash == null
584                     ? myReturnedCash
585                     : theForeignReturnedCash.theAmount;
586         }
