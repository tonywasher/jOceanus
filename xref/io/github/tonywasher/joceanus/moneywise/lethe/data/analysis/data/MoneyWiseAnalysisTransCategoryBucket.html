474     /**
475      * Add transaction to totals.
476      *
477      * @param pTrans the transaction helper
478      * @return isIncome true/false
479      */
480     boolean adjustValues(final MoneyWiseAnalysisTransactionHelper pTrans) {
481         /* Analyse the event */
482         final MoneyWiseTransCategoryClass myClass = pTrans.getCategoryClass();
483         MoneyWiseAssetDirection myDir = pTrans.getDirection();
484         boolean isExpense = myClass.isExpense();
485         final OceanusMoney myAmount = new OceanusMoney(pTrans.getLocalAmount());
486 
487         /* If this is an expense */
488         if (isExpense) {
489             /* Adjust for TaxCredit */
490             final OceanusMoney myTaxCredit = pTrans.getTaxCredit();
491             if (myTaxCredit != null) {
492                 myAmount.addAmount(myTaxCredit);
493             }
494 
495             /* if we have a non-zero amount */
496             if (myAmount.isNonZero()) {
497                 /* If this is a recovered expense */
498                 if (myDir.isFrom()) {
499                     /* Add as income */
500                     adjustCounter(MoneyWiseAnalysisTransAttr.INCOME, myAmount);
501                     isExpense = false;
502 
503                     /* else its standard expense */
504                 } else {
505                     /* Add as expense */
506                     adjustCounter(MoneyWiseAnalysisTransAttr.EXPENSE, myAmount);
507                 }
508             }
509 
510             /* else this is an income */
511         } else {
512             /* Adjust for TaxCredit */
513             final OceanusMoney myTaxCredit = pTrans.getTaxCredit();
514             if (myTaxCredit != null) {
515                 myAmount.addAmount(myTaxCredit);
516             }
517 
518             /* Adjust for Withheld */
519             final OceanusMoney myWithheld = pTrans.getWithheld();
520             if (myWithheld != null) {
521                 myAmount.addAmount(myWithheld);
522             }
523 
524             /* If we need to switch direction */
525             if (myClass.isSwitchDirection()) {
526                 /* switch the direction */
527                 myDir = myDir.reverse();
528             }
529 
530             /* if we have a non-zero amount */
531             if (myAmount.isNonZero()) {
532                 /* If this is a returned income */
533                 if (myDir.isTo()) {
534                     /* Add as expense */
535                     adjustCounter(MoneyWiseAnalysisTransAttr.EXPENSE, myAmount);
536 
537                     /* else standard income */
538                 } else {
539                     /* Add as income */
540                     adjustCounter(MoneyWiseAnalysisTransAttr.INCOME, myAmount);
541                 }
542             }
543         }
544 
545         /* Register the transaction in the history */
546         registerTransaction(pTrans);
547 
548         /* Return the income flag */
549         return !isExpense;
550     }

960         /**
961          * Adjust category buckets.
962          *
963          * @param pTrans    the transaction helper
964          * @param pCategory primary category
965          */
966         public void adjustCategories(final MoneyWiseAnalysisTransactionHelper pTrans,
967                                      final MoneyWiseTransCategory pCategory) {
968             /* Adjust the primary category bucket */
969             final MoneyWiseAnalysisTransCategoryBucket myCatBucket = getBucket(pCategory);
970 
971             /* Adjust for Tax Credit */
972             OceanusMoney myTaxCredit = pTrans.getTaxCredit();
973             if (myTaxCredit != null) {
974                 if (pCategory.isCategoryClass(MoneyWiseTransCategoryClass.LOANINTERESTCHARGED)) {
975                     theTaxRelief.addIncome(pTrans, myTaxCredit);
976                     myTaxCredit = new OceanusMoney(myTaxCredit);
977                     myTaxCredit.negate();
978                     theTaxBasis.adjustValue(pTrans, MoneyWiseTaxClass.VIRTUAL, myTaxCredit);
979                 } else {
980                     theTaxCredit.addExpense(pTrans, myTaxCredit);
981                     theTaxBasis.adjustGrossValue(pTrans, MoneyWiseTaxClass.TAXPAID, myTaxCredit);
982                 }
983             }
984 
985             /* Adjust for EmployeeNatInsurance */
986             final boolean isPension = pCategory.isCategoryClass(MoneyWiseTransCategoryClass.PENSIONCONTRIB);
987             OceanusMoney myNatIns = pTrans.getEmployeeNatIns();
988             if (myNatIns != null && myNatIns.isNonZero()) {
989                 if (!isPension) {
990                     theEmployeeNatIns.addIncome(pTrans, myNatIns);
991                 } else {
992                     myCatBucket.addIncome(myNatIns);
993                 }
994                 myNatIns = new OceanusMoney(myNatIns);
995                 myNatIns.negate();
996                 theTaxBasis.adjustNettValue(pTrans, MoneyWiseTaxClass.VIRTUAL, myNatIns);
997             }
998 
999             /* Adjust for EmployerNatInsurance */
1000             myNatIns = pTrans.getEmployerNatIns();
1001             if (myNatIns != null) {
1002                 if (!isPension) {
1003                     theEmployerNatIns.addIncome(pTrans, myNatIns);
1004                 } else {
1005                     myCatBucket.addIncome(myNatIns);
1006                 }
1007                 theTaxBasis.adjustValue(pTrans, MoneyWiseTaxClass.TAXFREE, myNatIns);
1008             }
1009 
1010             /* Adjust for DeemedBenefit */
1011             final OceanusMoney myBenefit = pTrans.getDeemedBenefit();
1012             if (myBenefit != null) {
1013                 theDeemedBenefit.addIncome(pTrans, myBenefit);
1014                 theWithheld.addExpense(pTrans, myBenefit);
1015                 theTaxBasis.adjustGrossValue(pTrans, MoneyWiseTaxClass.VIRTUAL, myBenefit);
1016             }
1017 
1018             /* Adjust for Withheld */
1019             final OceanusMoney myWithheld = pTrans.getWithheld();
1020             if (myWithheld != null) {
1021                 theWithheld.addExpense(pTrans, myWithheld);
1022                 theTaxBasis.adjustValue(pTrans, MoneyWiseTaxClass.VIRTUAL, myWithheld);
1023             }
1024 
1025             /* Adjust the category bucket */
1026             myCatBucket.adjustValues(pTrans);
1027 
1028             /* Adjust tax basis */
1029             theTaxBasis.adjustBasis(pTrans, pCategory);
1030         }

1032         /**
1033          * Adjust for Standard Gains.
1034          *
1035          * @param pTrans  the transaction helper
1036          * @param pSource the source security holding
1037          * @param pGains  the gains
1038          */
1039         public void adjustStandardGain(final MoneyWiseAnalysisTransactionHelper pTrans,
1040                                        final MoneyWiseSecurityHolding pSource,
1041                                        final OceanusMoney pGains) {
1042             /* Access security and portfolio */
1043             final MoneyWiseSecurity mySecurity = pSource.getSecurity();
1044             final MoneyWisePortfolio myPortfolio = pSource.getPortfolio();
1045 
1046             final boolean bTaxFreeGains = myPortfolio.isTaxFree()
1047                     || !mySecurity.getCategoryClass().isCapitalGains();
1048             final MoneyWiseAnalysisTransCategoryBucket myCategory = bTaxFreeGains
1049                     ? theTaxFreeGains
1050                     : theCapitalGains;
1051             final MoneyWiseTaxClass myTaxBasis = bTaxFreeGains
1052                     ? MoneyWiseTaxClass.TAXFREE
1053                     : MoneyWiseTaxClass.CAPITALGAINS;
1054 
1055             /* Add to Capital Gains income/expense */
1056             if (pGains.isPositive()) {
1057                 myCategory.addIncome(pTrans, pGains);
1058             } else {
1059                 myCategory.subtractExpense(pTrans, pGains);
1060             }
1061             theTaxBasis.adjustValue(pTrans, myTaxBasis, pGains);
1062         }

1065          * Adjust for Chargeable Gains.
1066          *
1067          * @param pTrans     the transaction helper
1068          * @param pReduction the income reduction
1069          */
1070         public void adjustChargeableGain(final MoneyWiseAnalysisTransactionHelper pTrans,
1071                                          final OceanusMoney pReduction) {
1072             /* Adjust Taxable Gains */
1073             theChargeableGains.subtractIncome(pReduction);
1074             theChargeableGains.adjustValues(pTrans);
1075 
1076             /* Obtain normalised value */
1077             final OceanusMoney myGains = new OceanusMoney(pTrans.getLocalAmount());
1078             myGains.subtractAmount(pReduction);
1079 
1080             /* Adjust for Tax Credit */
1081             OceanusMoney myTaxCredit = pTrans.getTaxCredit();
1082             if (myTaxCredit != null) {
1083                 /* Adjust the taxCredit category */
1084                 theTaxCredit.addExpense(pTrans, myTaxCredit);
1085 
1086                 /* Adjust tax basis */
1087                 theTaxBasis.adjustGrossValue(pTrans, MoneyWiseTaxClass.TAXPAID, myTaxCredit);
1088                 myTaxCredit = new OceanusMoney(myTaxCredit);
1089                 myTaxCredit.negate();
1090                 theTaxBasis.adjustGrossValue(pTrans, MoneyWiseTaxClass.VIRTUAL, myTaxCredit);
1091             }
1092 
1093             /* Adjust tax basis */
1094             theTaxBasis.adjustValue(pTrans, MoneyWiseTaxClass.CHARGEABLEGAINS, myGains);
1095         }

1097         /**
1098          * Produce totals for the TransactionCategories.
1099          */
1100         void produceTotals() {
1101             /* Create a list of new buckets */
1102             final MetisListIndexed<MoneyWiseAnalysisTransCategoryBucket> myTotals = new MetisListIndexed<>();
1103 
1104             /* Loop through the buckets */
1105             Iterator<MoneyWiseAnalysisTransCategoryBucket> myIterator = iterator();
1106             while (myIterator.hasNext()) {
1107                 final MoneyWiseAnalysisTransCategoryBucket myCurr = myIterator.next();
1108 
1109                 /* Obtain category and parent category */
1110                 final MoneyWiseTransCategory myCategory = myCurr.getTransactionCategory();
1111                 final MoneyWiseTransCategory myParent = myCategory.getParentCategory();
1112 
1113                 /* Access parent bucket */
1114                 MoneyWiseAnalysisTransCategoryBucket myTotal = findItemById(myParent.getIndexedId());
1115 
1116                 /* Calculate the delta */
1117                 myCurr.calculateDelta();
1118 
1119                 /* If the bucket does not exist */
1120                 if (myTotal == null) {
1121                     /* Look for bucket in the new list */
1122                     myTotal = myTotals.getItemById(myParent.getIndexedId());
1123 
1124                     /* If the bucket is completely new */
1125                     if (myTotal == null) {
1126                         /* Create the new bucket and add to new list */
1127                         myTotal = new MoneyWiseAnalysisTransCategoryBucket(theAnalysis, myParent);
1128                         myTotals.add(myTotal);
1129                     }
1130                 }
1131 
1132                 /* Add to totals bucket */
1133                 myTotal.addValues(myCurr);
1134                 theTotals.addValues(myCurr);
1135             }
1136 
1137             /* Loop through the new totals */
1138             myIterator = myTotals.listIterator();
1139             while (myIterator.hasNext()) {
1140                 final MoneyWiseAnalysisTransCategoryBucket myCurr = myIterator.next();
1141 
1142                 /* Calculate delta for the category total */
1143                 myCurr.calculateDelta();
1144 
1145                 /* Add it to the list */
1146                 theList.add(myCurr);
1147             }
1148 
1149             /* Sort the list */
1150             theList.getUnderlyingList().sort(Comparator.comparing(MoneyWiseAnalysisTransCategoryBucket::getTransactionCategory));
1151 
1152             /* Calculate delta for the totals */
1153             if (theTotals != null) {
1154                 theTotals.calculateDelta();
1155             }
1156         }
