17  package io.github.tonywasher.joceanus.moneywise.lethe.data.analysis.data;
18  
19  import io.github.tonywasher.joceanus.oceanus.decimal.OceanusMoney;
20  import io.github.tonywasher.joceanus.moneywise.data.statics.MoneyWiseCurrency;
21  import io.github.tonywasher.joceanus.moneywise.data.statics.MoneyWisePayeeClass;
22  import io.github.tonywasher.joceanus.moneywise.data.statics.MoneyWiseTransCategoryClass;
23  import io.github.tonywasher.joceanus.moneywise.lethe.data.analysis.data.MoneyWiseAnalysisPayeeBucket.MoneyWiseAnalysisPayeeBucketList;
24  import io.github.tonywasher.joceanus.moneywise.lethe.data.analysis.data.MoneyWiseAnalysisTaxBasisBucket.MoneyWiseAnalysisTaxBasisBucketList;
25  import io.github.tonywasher.joceanus.moneywise.lethe.data.analysis.data.MoneyWiseAnalysisTransCategoryBucket.MoneyWiseAnalysisTransCategoryBucketList;
26  import io.github.tonywasher.joceanus.moneywise.lethe.data.analysis.values.MoneyWiseAnalysisAccountAttr;
27  import io.github.tonywasher.joceanus.moneywise.lethe.data.analysis.values.MoneyWiseAnalysisAccountValues;
28  import io.github.tonywasher.joceanus.moneywise.lethe.data.analysis.values.MoneyWiseAnalysisSecurityAttr;
29  import io.github.tonywasher.joceanus.moneywise.lethe.data.analysis.values.MoneyWiseAnalysisSecurityValues;
30  
31  import java.util.Currency;
32  
33  /**
34   * Market analysis.
35   */
36  public class MoneyWiseAnalysisMarket {
37      /**
38       * Analysis.
39       */
40      private final MoneyWiseAnalysis theAnalysis;
41  
42      /**
43       * Market Payee Income.
44       */
45      private final OceanusMoney theMarketIncome;
46  
47      /**
48       * Market Payee Expense.
49       */
50      private final OceanusMoney theMarketExpense;
51  
52      /**
53       * MarketGrowth Income.
54       */
55      private final OceanusMoney theGrowthIncome;
56  
57      /**
58       * MarketGrowth Expense.
59       */
60      private final OceanusMoney theGrowthExpense;
61  
62      /**
63       * CurrencyFluctuation Income.
64       */
65      private final OceanusMoney theFluctIncome;
66  
67      /**
68       * CurrencyFluctuation Expense.
69       */
70      private final OceanusMoney theFluctExpense;
71  
72      /**
73       * Constructor.
74       *
75       * @param pAnalysis the analysis.
76       */
77      protected MoneyWiseAnalysisMarket(final MoneyWiseAnalysis pAnalysis) {
78          /* Store parameters */
79          theAnalysis = pAnalysis;
80  
81          /* Determine the currency */
82          final MoneyWiseCurrency myCurr = pAnalysis.getCurrency();
83          final Currency myCurrency = myCurr == null
84                  ? MoneyWiseAnalysisAccountBucket.DEFAULT_CURRENCY
85                  : myCurr.getCurrency();
86  
87          /* Create buckets */
88          theMarketIncome = new OceanusMoney(myCurrency);
89          theMarketExpense = new OceanusMoney(myCurrency);
90          theGrowthIncome = new OceanusMoney(myCurrency);
91          theGrowthExpense = new OceanusMoney(myCurrency);
92          theFluctIncome = new OceanusMoney(myCurrency);
93          theFluctExpense = new OceanusMoney(myCurrency);
94      }
95  
96      /**
97       * Process account bucket.
98       *
99       * @param pBucket the account bucket.
100      */
101     protected void processAccount(final MoneyWiseAnalysisAccountBucket<?> pBucket) {
102         /* Access market and gains */
103         final MoneyWiseAnalysisAccountValues myValues = pBucket.getValues();
104         final OceanusMoney myFluct = myValues.getMoneyValue(MoneyWiseAnalysisAccountAttr.CURRENCYFLUCT);
105 
106         /* If there are fluctuations */
107         if (myFluct != null) {
108             final MoneyWiseAnalysisAccountValues myBaseValues = pBucket.getBaseValues();
109             final OceanusMoney myBaseFluct = myBaseValues.getMoneyValue(MoneyWiseAnalysisAccountAttr.CURRENCYFLUCT);
110             final OceanusMoney myPeriodFluct = new OceanusMoney(myFluct);
111             myPeriodFluct.subtractAmount(myBaseFluct);
112 
113             /* Add to CurrencyFluctuation income/expense */
114             if (myPeriodFluct.isPositive()) {
115                 theFluctIncome.addAmount(myPeriodFluct);
116                 theMarketIncome.addAmount(myPeriodFluct);
117             } else {
118                 theFluctExpense.subtractAmount(myPeriodFluct);
119                 theMarketExpense.subtractAmount(myPeriodFluct);
120             }
121         }
122     }
123 
124     /**
125      * Process security bucket.
126      *
127      * @param pBucket the security bucket.
128      */
129     protected void processSecurity(final MoneyWiseAnalysisSecurityBucket pBucket) {
130         /* Access market and gains */
131         final MoneyWiseAnalysisSecurityValues myValues = pBucket.getValues();
132         final OceanusMoney myMarket = myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.MARKETGROWTH);
133         final OceanusMoney myGains = myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.REALISEDGAINS);
134         final OceanusMoney myFluct = myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.CURRENCYFLUCT);
135 
136         /* If there are gains in the period */
137         if (myGains.isNonZero()) {
138             /* Subtract them from the market movement */
139             myMarket.subtractAmount(myGains);
140 
141             /* These gains have been allocated separately */
142             if (myGains.isPositive()) {
143                 /* Adjust market account */
144                 theMarketIncome.addAmount(myGains);
145             } else {
146                 /* Adjust market account */
147                 theMarketExpense.subtractAmount(myGains);
148             }
149         }
150 
151         /* If there is growth in the period */
152         if (myMarket.isNonZero()) {
153             /* Add to MarketGrowth income/expense */
154             if (myMarket.isPositive()) {
155                 theGrowthIncome.addAmount(myMarket);
156                 theMarketIncome.addAmount(myMarket);
157             } else {
158                 theGrowthExpense.subtractAmount(myMarket);
159                 theMarketExpense.subtractAmount(myMarket);
160             }
161         }
162 
163         /* If there are fluctuations in the period */
164         if (myFluct != null && myFluct.isNonZero()) {
165             /* Add to CurrencyFluctuation income/expense */
166             if (myFluct.isPositive()) {
167                 theFluctIncome.addAmount(myFluct);
168                 theMarketIncome.addAmount(myFluct);
169             } else {
170                 theFluctExpense.subtractAmount(myFluct);
171                 theMarketExpense.subtractAmount(myFluct);
172             }
173         }
174     }
175 
176     /**
177      * Propagate totals.
178      */
179     protected void propagateTotals() {
180         /* Access lists */
181         final MoneyWiseAnalysisPayeeBucketList myPayees = theAnalysis.getPayees();
182         final MoneyWiseAnalysisTaxBasisBucketList myTaxBasis = theAnalysis.getTaxBasis();
183         final MoneyWiseAnalysisTransCategoryBucketList myCategories = theAnalysis.getTransCategories();
184 
185         /* If we have market income/expense */
186         if (theMarketIncome.isNonZero()
187                 || theMarketExpense.isNonZero()) {
188             /* Access market payee */
189             final MoneyWiseAnalysisPayeeBucket myMarket = myPayees.getBucket(MoneyWisePayeeClass.MARKET);
190 
191             /* Adjust totals */
192             myMarket.addIncome(theMarketIncome);
193             myMarket.addExpense(theMarketExpense);
194         }
195 
196         /* If we have marketGrowth */
197         if (theGrowthIncome.isNonZero()
198                 || theGrowthExpense.isNonZero()) {
199             /* Access marketGrowth category */
200             final MoneyWiseAnalysisTransCategoryBucket myGrowth = myCategories.getBucket(MoneyWiseTransCategoryClass.MARKETGROWTH);
201 
202             /* Adjust totals */
203             myGrowth.addIncome(theGrowthIncome);
204             myGrowth.addExpense(theGrowthExpense);
205 
206             /* Adjust tax basis */
207             myTaxBasis.adjustMarket(theGrowthIncome, theGrowthExpense);
208         }
209 
210         /* If we have currencyFluctuation */
211         if (theFluctIncome.isNonZero()
212                 || theFluctExpense.isNonZero()) {
213             /* Access currecyFluctuation category */
214             final MoneyWiseAnalysisTransCategoryBucket myFluct = myCategories.getBucket(MoneyWiseTransCategoryClass.CURRENCYFLUCTUATION);
215 
216             /* Adjust totals */
217             myFluct.addIncome(theFluctIncome);
218             myFluct.addExpense(theFluctExpense);
219 
220             /* Adjust tax basis */
221             myTaxBasis.adjustMarket(theFluctIncome, theFluctExpense);
222         }
223     }
224 }
