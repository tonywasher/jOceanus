113     public MoneyWiseMainTab(final TethysUIFactory<?> pFactory) throws OceanusException {
114         /* Create prometheus toolkit */
115         final PrometheusToolkit myToolkit = new PrometheusToolkit(pFactory);
116 
117         /* create the view */
118         theView = new MoneyWiseView(myToolkit, new MoneyWiseUKTaxYearCache());

120         /* Build the main window */
121         buildMainWindow(theView, myToolkit);
122 
123         /* Initialise visibility */
124         setVisibility();
125     }

141     @Override
142     protected TethysUIComponent buildMainPanel() throws OceanusException {
143         /* Obtain the active profile */
144         OceanusProfile myTask = theView.getActiveTask();
145         myTask = myTask.startTask("buildMain");
146 
147         /* Create the Tabbed Pane */
148         final TethysUIFactory<?> myFactory = theView.getGuiFactory();
149         theTabs = myFactory.paneFactory().newTabPane();
150 
151         /* Create the Report Tab */
152         myTask.startTask(TITLE_REPORT);
153         final MoneyWiseReportTab myReports = new MoneyWiseReportTab(theView);
154         theTabs.addTabItem(TITLE_REPORT, myReports);
155 
156         /* Create the Register Tab */
157         myTask.startTask(TITLE_REGISTER);
158         theRegister = new MoneyWiseStatementPanel(theView);
159         theTabs.addTabItem(TITLE_REGISTER, theRegister);
160 
161         /* Create the Market Tab */
162         myTask.startTask(TITLE_MARKET);
163         theMarket = new MoneyWiseMarketTabs(theView);
164         theTabs.addTabItem(TITLE_MARKET, theMarket);
165 
166         /* Create the Maintenance Tab */
167         myTask.startTask(TITLE_MAINT);
168         theMaint = new MoneyWiseMaintenance(theView);
169         theTabs.addTabItem(TITLE_MAINT, theMaint);
170 
171         /* Create the log tab */
172         final TethysUILogTextArea myLog = myFactory.getLogSink();
173         final TethysUITabItem myLogTab = theTabs.addTabItem(MoneyWiseUIResource.MAIN_LOG.getValue(), myLog);
174         myLog.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> myLogTab.setVisible(true));
175         myLog.getEventRegistrar().addEventListener(TethysUIEvent.WINDOWCLOSED, e -> myLogTab.setVisible(false));
176         myLogTab.setVisible(myLog.isActive());
177 
178         /* Create listeners */
179         theTabs.getEventRegistrar().addEventListener(e -> determineFocus());
180         theView.getEventRegistrar().addEventListener(e -> setVisibility());
181         myReports.getEventRegistrar().addEventListener(this::handleGoToEvent);
182         theMarket.getEventRegistrar().addEventListener(e -> setVisibility());
183         setChildListeners(theRegister.getEventRegistrar());
184         setChildListeners(theMaint.getEventRegistrar());
185 
186         /* Create listener and initialise focus */
187         determineFocus();
188 
189         /* Complete task */
190         myTask.end();
191 
192         /* Return the panel */
193         return theTabs;
194     }

196     /**
197      * setChildListeners.
198      *
199      * @param pRegistrar the registrar
200      */
201     private void setChildListeners(final OceanusEventRegistrar<PrometheusDataEvent> pRegistrar) {
202         pRegistrar.addEventListener(PrometheusDataEvent.ADJUSTVISIBILITY, e -> setVisibility());
203         pRegistrar.addEventListener(PrometheusDataEvent.GOTOWINDOW, this::handleGoToEvent);
204     }

221     @Override
222     public final boolean hasUpdates() {
223         /* Determine whether we have edit session updates */
224         boolean hasUpdates = theRegister.hasUpdates();
225         if (!hasUpdates) {
226             hasUpdates = theMarket.hasUpdates();
227         }
228         if (!hasUpdates) {
229             hasUpdates = theMaint.hasUpdates();
230         }
231         return hasUpdates;
232     }

240         /* Determine whether we have edit session updates */

243             hasSession = theMarket.hasSession();

245         if (!hasSession) {
246             hasSession = theMaint.hasSession();

295     /**
296      * Goto the specific tab.
297      *
298      * @param pTabName the tab name

301         /* Look up item and select it */

308     @Override
309     public final void setVisibility() {
310         /* Sort out underlying visibility */
311         super.setVisibility();
312 
313         /* Determine whether we have any session focus */
314         final boolean hasSession = hasSession();
315 
316         /* Note whether we have a worker thread */
317         final boolean hasWorker = hasWorker();
318 
319         /* Note whether we have data */
320         final boolean hasControl = theView.getData().getControl() != null;
321 
322         /* Obtain the menuBar */
323         final TethysUIMenuBarManager myMenuBar = getMenuBar();
324 
325         /* Disable menus if we have no data */
326         myMenuBar.setEnabled(MoneyWiseThreadId.CREATEQIF, !hasWorker && hasControl);
327 
328         /* Enable/Disable the reports tab */
329         boolean doEnabled = !hasWorker && !hasSession;
330         theTabs.enableItemByName(TITLE_REPORT, doEnabled);
331 
332         /* Enable/Disable the register tab */
333         doEnabled = !hasWorker && (!hasSession || theRegister.hasSession());
334         theTabs.enableItemByName(TITLE_REGISTER, doEnabled);
335 
336         /* Enable/Disable/Hide the spotPrices tab */
337         doEnabled = !hasWorker && (!hasSession || theMarket.hasSession());
338         final TethysUITabItem myItem = theTabs.findItemByName(TITLE_MARKET);
339         myItem.setEnabled(doEnabled);
340         myItem.setVisible(theView.hasActiveSecurities() || theView.hasMultipleCurrencies());
341 
342         /* Enable/Disable the maintenance tab */
343         doEnabled = !hasWorker && (!hasSession || theMaint.hasSession());
344         theTabs.enableItemByName(TITLE_MAINT, doEnabled);
345 
346         /* Enable/Disable the tabs */
347         theTabs.setEnabled(!hasWorker);
348 
349         /* If we have updates disable the load backup/database option */
350         myMenuBar.setEnabled(MoneyWiseThreadId.LOADARCHIVE, !hasSession);
351     }

353     /**
354      * Determine focus.
355      */
356     private void determineFocus() {
357         /* Access the selected component */
358         final TethysUITabItem myItem = theTabs.getSelectedTab();
359         final Integer myId = myItem.getId();
360 
361         /* If the selected component is Register */
362         if (myId.equals(theRegister.getId())) {
363             /* Determine focus of Register */
364             theRegister.determineFocus();
365 
366             /* If the selected component is Market */
367         } else if (myId.equals(theMarket.getId())) {
368             /* Determine focus of Market */
369             theMarket.determineFocus();
370 
371             /* If the selected component is Maintenance */
372         } else if (myId.equals(theMaint.getId())) {
373             /* Determine focus of maintenance */
374             theMaint.determineFocus();
375         }
376     }

387     /**
388      * handle GoTo Event.
389      *
390      * @param pEvent the event
391      */
392     private void handleGoToEvent(final OceanusEvent<PrometheusDataEvent> pEvent) {
393         /* Access details */
394         @SuppressWarnings("unchecked") final PrometheusGoToEvent<MoneyWiseGoToId> myEvent = pEvent.getDetails(PrometheusGoToEvent.class);
395 
396         /* Access event and obtain details */
397         switch (myEvent.getId()) {
398             /* View the requested statement */
399             case STATEMENT:
400                 final MoneyWiseStatementSelect mySelect = myEvent.getDetails(MoneyWiseStatementSelect.class);
401                 selectStatement(mySelect);
402                 break;
403 
404             /* Access maintenance */
405             case ACCOUNT:
406             case CATEGORY:
407             case REGION:
408             case TAG:
409             case STATIC:
410                 selectMaintenance(myEvent);
411                 break;
412             default:
413                 break;
414         }
415     }
