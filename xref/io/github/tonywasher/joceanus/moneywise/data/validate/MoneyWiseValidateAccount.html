30  /**
31   * Validator for assetBase.
32   *
33   * @param <T> the asset type
34   */
35  public abstract class MoneyWiseValidateAccount<T extends MoneyWiseAssetBase>
36          implements MoneyWiseDataValidatorAccount<T> {
37      /**
38       * Bad category error.
39       */
40      static final String ERROR_BADCATEGORY = MoneyWiseBasicResource.ASSET_ERROR_BADCAT.getValue();
41  
42      /**
43       * Bad parent error.
44       */
45      static final String ERROR_BADPARENT = MoneyWiseBasicResource.ASSET_ERROR_BADPARENT.getValue();
46  
47      /**
48       * Parent Closed Error Text.
49       */
50      static final String ERROR_PARCLOSED = MoneyWiseBasicResource.ASSET_ERROR_PARENTCLOSED.getValue();
51  
52      /**
53       * Reserved name error.
54       */
55      static final String ERROR_RESERVED = MoneyWiseBasicResource.ASSET_ERROR_RESERVED.getValue();
56  
57      /**
58       * Set the editSet.
59       */
60      private PrometheusEditSet theEditSet;
61  
62      @Override
63      public void setEditSet(final PrometheusEditSet pEditSet) {
64          theEditSet = pEditSet;
65      }
66  
67      /**
68       * Obtain the editSet.
69       *
70       * @return the editSet
71       */
72      PrometheusEditSet getEditSet() {
73          if (theEditSet == null) {
74              throw new IllegalStateException("editSet not set up");
75          }
76          return theEditSet;
77      }
78  
79      @Override
80      public void validate(final PrometheusDataItem pAsset) {
81          final MoneyWiseAssetBase myAsset = (MoneyWiseAssetBase) pAsset;
82          final String myName = myAsset.getName();
83          final String myDesc = myAsset.getDesc();
84  
85          /* Name must be non-null */
86          if (myName == null) {
87              pAsset.addError(PrometheusDataItem.ERROR_MISSING, PrometheusDataResource.DATAITEM_FIELD_NAME);
88  
89              /* Check that the name is unique */
90          } else {
91              /* Validate the name */
92              validateName(myAsset, myName);
93          }
94  
95          /* Check description length */
96          if (myDesc != null
97                  && myDesc.length() > PrometheusDataItem.DESCLEN) {
98              pAsset.addError(PrometheusDataItem.ERROR_LENGTH, PrometheusDataResource.DATAITEM_FIELD_DESC);
99          }
100     }
101 
102     /**
103      * Validate the name.
104      *
105      * @param pAsset the asset
106      * @param pName  the name
107      */
108     public void validateName(final MoneyWiseAssetBase pAsset,
109                              final String pName) {
110         /* Access the list */
111         final MoneyWiseAssetBaseList<?> myList = pAsset.getList();
112 
113         /* The name must not be too long */
114         if (pName.length() > PrometheusDataItem.NAMELEN) {
115             pAsset.addError(PrometheusDataItem.ERROR_LENGTH, PrometheusDataResource.DATAITEM_FIELD_NAME);
116         }
117 
118         /* Check name count */
119         if (!myList.validNameCount(pName)) {
120             pAsset.addError(PrometheusDataItem.ERROR_DUPLICATE, PrometheusDataResource.DATAITEM_FIELD_NAME);
121         }
122 
123         /* Check that the name does not contain invalid characters */
124         if (pName.contains(MoneyWiseSecurityHolding.SECURITYHOLDING_SEP)) {
125             pAsset.addError(PrometheusDataItem.ERROR_INVALIDCHAR, PrometheusDataResource.DATAITEM_FIELD_NAME);
126         }
127     }
128 
129     /**
130      * Obtain the reporting currency.
131      *
132      * @return the reporting currency
133      */
134     MoneyWiseCurrency getReportingCurrency() {
135         final MoneyWiseDataSet myDataSet = (MoneyWiseDataSet) getEditSet().getDataSet();
136         return myDataSet.getReportingCurrency();
137     }
138 
139     /**
140      * Obtain unique name for new account.
141      *
142      * @param pList the list
143      * @param pBase the base name
144      * @return The new name
145      */
146     public String getUniqueName(final MoneyWiseAssetBaseList<T> pList,
147                                 final String pBase) {
148         /* Set up base constraints */
149         int iNextId = 1;
150 
151         /* Loop until we found a name */
152         String myName = pBase;
153         while (true) {
154             /* try out the name */
155             if (pList.checkAvailableName(myName)) {
156                 return myName;
157             }
158 
159             /* Build next name */
160             myName = pBase.concat(Integer.toString(iNextId++));
161         }
162     }
163 }
