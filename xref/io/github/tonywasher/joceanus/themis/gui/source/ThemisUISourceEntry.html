52  /**
53   * Source Panel Tree Entry.
54   */
55  public class ThemisUISourceEntry {
56      /**
57       * Entry name prefix.
58       */
59      private static final String ENTRY_PREFIX = "TreeItem";
60  
61      /**
62       * The Next entryId.
63       */
64      private static final AtomicInteger NEXT_ENTRY_ID = new AtomicInteger(1);
65  
66      /**
67       * The Parent.
68       */
69      private final ThemisUISourceEntry theParent;
70  
71      /**
72       * The id of the entry.
73       */
74      private final int theId;
75  
76      /**
77       * The Child List.
78       */
79      private List<ThemisUISourceEntry> theChildList;
80  
81      /**
82       * The unique name of the entry.
83       */
84      private final String theUniqueName;
85  
86      /**
87       * The display name of the entry.
88       */
89      private final String theDisplayName;
90  
91      /**
92       * The icon of the entry.
93       */
94      private final TethysUIIconId theIcon;
95  
96      /**
97       * The object for the entry.
98       */
99      private final ThemisInstance theObject;
100 
101     /**
102      * Constructor.
103      *
104      * @param pElement the sourceElement
105      */
106     ThemisUISourceEntry(final ThemisInstance pElement) {
107         this(null, pElement);
108     }
109 
110     /**
111      * Constructor.
112      *
113      * @param pParent  the parent entry
114      * @param pElement the sourceElement
115      */
116     ThemisUISourceEntry(final ThemisUISourceEntry pParent,
117                         final ThemisInstance pElement) {
118         /* Store parameters */
119         theParent = pParent;
120         theObject = pElement;
121 
122         /* Allocate id and unique name */
123         theId = NEXT_ENTRY_ID.getAndIncrement();
124         theUniqueName = ENTRY_PREFIX + theId;
125         theDisplayName = getElementDisplay(pElement);
126         theIcon = ThemisUISourceIcon.getElementIcon(pElement);
127 
128         /* If we have a parent */
129         if (pParent != null) {
130             /* Add the entry to the child list */
131             pParent.addChild(this);
132         }
133     }
134

176     /**
177      * Get child iterator.
178      *
179      * @return the iterator
180      */
181     Iterator<ThemisUISourceEntry> childIterator() {
182         return theChildList == null
183                 ? Collections.emptyIterator()
184                 : theChildList.iterator();
185     }
186 
187     /**
188      * Add child.
189      *
190      * @param pChild the child to add
191      */
192     private void addChild(final ThemisUISourceEntry pChild) {
193         if (theChildList == null) {
194             theChildList = new ArrayList<>();
195         }
196         theChildList.add(pChild);
197     }
198 
199     @Override
200     public boolean equals(final Object pThat) {
201         /* Handle trivial cases */
202         if (this == pThat) {
203             return true;
204         }
205         if (pThat == null) {
206             return false;
207         }
208 
209         /* Check class */
210         if (!(pThat instanceof ThemisUISourceEntry myThat)) {
211             return false;
212         }
213 
214         /* Must have same id */
215         return theId == myThat.theId;
216     }
217 
218     @Override
219     public int hashCode() {
220         return Objects.hashCode(theId);
221     }

223     /**
224      * Obtain the displayName for the element.
225      *
226      * @param pElement the element
227      * @return the displayName
228      */
229     static String getElementDisplay(final ThemisInstance pElement) {
230         /* Switch on the element type */
231         return switch (pElement) {
232             case ThemisDeclarationInstance myDecl -> getDeclarationDisplay(myDecl);
233             case ThemisNodeInstance myNode -> getNodeDisplay(myNode);
234             case ThemisExpressionInstance myExpr -> getExpressionDisplay(myExpr);
235             case ThemisStatementInstance myStmt -> getStatementDisplay(myStmt);
236             case ThemisTypeInstance myType -> getTypeDisplay(myType);
237             default -> throw new IllegalArgumentException("Unknown ThemisInstance");
238         };
239     }
240 
241     /**
242      * Obtain the displayName for the declaration.
243      *
244      * @param pElement the element
245      * @return the icon
246      */
247     private static String getDeclarationDisplay(final ThemisDeclarationInstance pElement) {
248         /* Switch on the id */
249         return switch ((ThemisDeclaration) pElement.getId()) {
250             case ANNOTATION -> ThemisUIResource.SOURCEDECL_ANNOT.getValue();
251             case ANNOTATIONMEMBER -> ThemisUIResource.SOURCEDECL_ANNOTMEMBER.getValue();
252             case CLASSINTERFACE -> ((ThemisDeclClassInterface) pElement).getName();
253             case COMPACT -> ThemisUIResource.SOURCEDECL_COMPACT.getValue();
254             case CONSTRUCTOR -> ThemisUIResource.SOURCEDECL_CONSTRUCT.getValue();
255             case ENUM -> ((ThemisDeclEnum) pElement).getName();
256             case ENUMVALUE -> ((ThemisDeclEnumValue) pElement).getNode().getNameAsString();
257             case FIELD -> ThemisUIResource.SOURCEDECL_FIELD.getValue();
258             case INITIALIZER -> ThemisUIResource.SOURCEDECL_INIT.getValue();
259             case METHOD -> ((ThemisDeclMethod) pElement).getName();
260             case RECORD -> ((ThemisDeclRecord) pElement).getName();
261             default -> throw new IllegalArgumentException("Unknown ThemisDeclaration");
262         };
263     }
