116     /**
117      * Constructor.
118      *
119      * @param pFactory the GuiFactory
120      * @throws OceanusException on error
121      */
122     public ThemisUIMainPanel(final TethysUIFactory<?> pFactory) throws OceanusException {
123         /* Store guiFactory */
124         theGuiFactory = pFactory;
125 
126         /* Create the subPanels */
127         theSource = new ThemisUISourcePanel(theGuiFactory);
128         theStats = new ThemisUIStatsPanel(theGuiFactory);
129         theRefs = new ThemisUIRefPanel(theGuiFactory);
130 
131         /* Create the tabs */
132         final TethysUIPaneFactory myPanes = theGuiFactory.paneFactory();
133         final TethysUITabPaneManager myTabs = myPanes.newTabPane();
134         theSourceTab = myTabs.addTabItem(ThemisUIResource.TAB_SOURCE.getValue(), theSource);
135         theRefsTab = myTabs.addTabItem(ThemisUIResource.TAB_REFERENCES.getValue(), theRefs);
136         theStatsTab = myTabs.addTabItem(ThemisUIResource.TAB_STATS.getValue(), theStats);
137 
138         /* Hide tabs */
139         theSourceTab.setVisible(false);
140         theStatsTab.setVisible(false);
141         theRefsTab.setVisible(false);
142 
143         /* Create the log tab */
144         theLogSink = theGuiFactory.getLogSink();
145         theLogTab = myTabs.addTabItem(ThemisUIResource.TAB_LOG.getValue(), theLogSink);
146         theLogSink.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> theLogTab.setVisible(true));
147         theLogSink.getEventRegistrar().addEventListener(TethysUIEvent.WINDOWCLOSED, e -> theLogTab.setVisible(false));
148         theLogTab.setVisible(theLogSink.isActive());
149 
150         /* Create the project selection panel */
151         final TethysUIControlFactory myControls = theGuiFactory.controlFactory();
152         final TethysUIButtonFactory<?> myButtons = theGuiFactory.buttonFactory();
153         final TethysUIBoxPaneManager myProjectSelect = myPanes.newHBoxPane();
154         myProjectSelect.addNode(myControls.newLabel(ThemisUIResource.PROMPT_PROJECT.getValue()));
155         theProjectButton = myButtons.newButton();
156         theProjectButton.setTextOnly();
157         myProjectSelect.addNode(theProjectButton);
158         theProjectButton.getEventRegistrar().addEventListener(e -> selectProject());
159         theProjectButton.setText("None");
160 
161         /* create the overall project select panel */
162         final TethysUIBoxPaneManager myProjectControl = myPanes.newHBoxPane();
163         myProjectControl.addSpacer();
164         myProjectControl.addNode(myProjectSelect);
165         myProjectControl.addSpacer();
166 
167         /* create the overall panel */
168         thePanel = myPanes.newBorderPane();
169         thePanel.setNorth(myProjectControl);
170         thePanel.setCentre(myTabs);
171 
172         /* Handle the default location */
173         final File myLocation = getDefaultLocation();
174         if (myLocation != null) {
175             handleNewProject(myLocation);
176         }
177     }
178 
179     @Override
180     public TethysUIComponent getComponent() {
181         return thePanel;
182     }

184     /**
185      * Handle select project.
186      */
187     void selectProject() {
188         /* Determine initial directory */
189         final File myInit = new File(System.getProperty("user.home"));
190 
191         /* Determine the name of the directory to load */
192         final TethysUIDialogFactory myDialogs = theGuiFactory.dialogFactory();
193         final TethysUIDirectorySelector myDialog = myDialogs.newDirectorySelector();
194         myDialog.setTitle(ThemisUIResource.SELECT_PROJECT.getValue());
195         myDialog.setInitialDirectory(myInit);
196         final File myFile = myDialog.selectDirectory();
197 
198         /* If we selected a file */
199         if (myFile != null) {
200             /* Handle the new project */
201             handleNewProject(myFile);
202         }
203     }
204 
205     /**
206      * Handle the new project.
207      *
208      * @param pProjectDir the new project directory
209      */
210     private void handleNewProject(final File pProjectDir) {
211         /* Hide tabs */
212         theSourceTab.setVisible(false);
213         theRefsTab.setVisible(false);
214         theStatsTab.setVisible(false);
215 
216         /* Save details */
217         OceanusException myError = storeDefaultLocation(pProjectDir);
218         if (myError != null) {
219             writeErrorToLog(myError);
220             return;
221         }
222 
223         /* Parse the project */
224         final ThemisParser myParser = new ThemisParser(pProjectDir);
225         myError = myParser.getError();
226         if (myError == null) {
227             final ThemisProject myProject = myParser.getProject();
228             theSource.setCurrentProject(myProject);
229             theProjectButton.setText(myProject.toString());
230             theSourceTab.setVisible(true);
231         }
232 
233         /* If we parsed successfully */
234         if (myError == null) {
235             /* Resolve references */
236             final ThemisSolverProject myProject = new ThemisSolverProject(myParser);
237             final ThemisSolver mySolver = new ThemisSolver(myProject);
238             myError = mySolver.getError();
239             if (myError == null) {
240                 theRefs.setCurrentProject(myProject);
241                 theRefsTab.setVisible(true);
242             }
243         }
244 
245         /* If we parsed successfully */
246         if (myError == null) {
247             /* Resolve stats */
248             final ThemisStatsProject myStats = new ThemisStatsProject(myParser);
249             myError = myStats.getError();
250             if (myError == null) {
251                 theStats.setCurrentProject(myStats);
252                 theStatsTab.setVisible(true);
253             }
254         }
255 
256         /* Display any error */
257         if (myError != null) {
258             writeErrorToLog(myError);
259         }
260     }
261 
262     /**
263      * Write error to log.
264      *
265      * @param pError the error
266      */
267     private void writeErrorToLog(final OceanusException pError) {
268         theLogSink.writeLogMessage(pError.getMessage());
269         theLogTab.setVisible(true);
270     }

272     /**
273      * Obtain the default location.
274      *
275      * @return the default location
276      */
277     private File getDefaultLocation() {
278         final Preferences myPreferences = deriveHandle();
279         final String myLocation = myPreferences.get(PREF_PROJECT, null);
280         return myLocation == null ? null : new File(myLocation);
281     }
282 
283     /**
284      * Store the default location.
285      *
286      * @param pLocation the default location
287      * @return exception or null
288      */
289     private OceanusException storeDefaultLocation(final File pLocation) {
290         /* Protect against exceptions */
291         try {
292             final Preferences myPreferences = deriveHandle();
293             myPreferences.put(PREF_PROJECT, pLocation.getAbsolutePath());
294             myPreferences.flush();
295             return null;
296         } catch (BackingStoreException e) {
297             return new ThemisIOException("Failed to save preference", e);
298         }
299     }
300 
301     /**
302      * Derive handle for node.
303      *
304      * @return the class name
305      */
306     private Preferences deriveHandle() {
307         /* Obtain the class name */
308         final Class<?> myClass = this.getClass();
309         String myName = myClass.getCanonicalName();
310 
311         /* Obtain the package name */
312         final String myPackage = myClass.getPackage().getName();
313 
314         /* Strip off the package name */
315         myName = myName.substring(myPackage.length() + 1);
316 
317         /* Derive the handle */
318         final Preferences myHandle = Preferences.userNodeForPackage(myClass);
319         return myHandle.node(myName);
320     }
