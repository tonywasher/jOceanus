178     /**
179      * Constructor.
180      *
181      * @param pFactory the gui factory
182      * @throws OceanusException on error
183      */
184     public ThemisUIMainPanel(final TethysUIFactory<?> pFactory) throws OceanusException {
185         /* Store guiFactory */
186         theGuiFactory = pFactory;
187         final MetisToolkit myToolkit = new MetisToolkit(pFactory, false);
188 
189         /* Create the subPanels */
190         theRefs = new ThemisUIRefPanel(theGuiFactory);
191         theStats = new ThemisUIStatsPanel(theGuiFactory);
192         theSource = new ThemisUISourcePanel(theGuiFactory);
193 
194         /* Create the tabs */
195         final TethysUIPaneFactory myPanes = theGuiFactory.paneFactory();
196         final TethysUITabPaneManager myTabs = myPanes.newTabPane();
197         theRefsTab = myTabs.addTabItem(ThemisUIResource.TAB_REFERENCES.getValue(), theRefs);
198         theStatsTab = myTabs.addTabItem(ThemisUIResource.TAB_STATS.getValue(), theStats);
199         theSourceTab = myTabs.addTabItem(ThemisUIResource.TAB_SOURCE.getValue(), theSource);
200 
201         /* Hide tabs */
202         theRefsTab.setVisible(false);
203         theStatsTab.setVisible(false);
204         theSourceTab.setVisible(false);
205 
206         /* Create the log tab */
207         theLogSink = theGuiFactory.getLogSink();
208         theLogTab = myTabs.addTabItem(ThemisUIResource.TAB_LOG.getValue(), theLogSink);
209         theLogSink.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> theLogTab.setVisible(true));
210         theLogSink.getEventRegistrar().addEventListener(TethysUIEvent.WINDOWCLOSED, e -> theLogTab.setVisible(false));
211         theLogTab.setVisible(theLogSink.isActive());
212 
213         /* Create the project selection panel */
214         final TethysUIControlFactory myControls = theGuiFactory.controlFactory();
215         final TethysUIButtonFactory<?> myButtons = theGuiFactory.buttonFactory();
216         final TethysUIBoxPaneManager myProjectSelect = myPanes.newHBoxPane();
217         myProjectSelect.addNode(myControls.newLabel(ThemisUIResource.PROMPT_PROJECT.getValue()));
218         theProjectButton = myButtons.newButton();
219         theProjectButton.setTextOnly();
220         myProjectSelect.addNode(theProjectButton);
221         theProjectButton.getEventRegistrar().addEventListener(e -> selectProject());
222         theProjectButton.setText(ThemisUIResource.PROMPT_NONE.getValue());
223 
224         /* Configure refresh button */
225         final TethysUIButton myRefreshButton = myButtons.newButton();
226         MetisIcon.configureButton(myRefreshButton);
227         myRefreshButton.setIcon(ThemisUIIcon.REFRESH);
228         myRefreshButton.setToolTip(ThemisUIResource.TOOLTIP_REFRESH.getValue());
229         myRefreshButton.getEventRegistrar().addEventListener(TethysUIEvent.PRESSED,
230                 e -> handleNewProject(getDefaultLocation()));
231 
232         /* create the overall project select panel */
233         theProjectControl = myPanes.newHBoxPane();
234         theProjectControl.addSpacer();
235         theProjectControl.addNode(myProjectSelect);
236         theProjectControl.addSpacer();
237         theProjectControl.addNode(myRefreshButton);
238         theProjectControl.addStrut();
239 
240         /* Access the status bar and set to invisible */
241         theThreadMgr = myToolkit.getThreadManager();
242         theStatusBar = theThreadMgr.getStatusManager();
243         theStatusBar.setVisible(false);
244         theThreadMgr.getEventRegistrar().addEventListener(TethysUIThreadEvent.THREADEND, e -> setVisibility(false));
245 
246         /* create the overall project select panel */
247         final TethysUIBoxPaneManager myBanner = myPanes.newVBoxPane();
248         myBanner.addNode(theProjectControl);
249         myBanner.addNode(theStatusBar);
250 
251         /* create the overall panel */
252         thePanel = myPanes.newBorderPane();
253         thePanel.setNorth(myBanner);
254         thePanel.setCentre(myTabs);
255 
256         /* Create the menu bar */
257         theMenuBar = theGuiFactory.menuFactory().newMenuBar();
258         final TethysUIMenuSubMenu myMenu = theMenuBar.newSubMenu(ThemisUIMenuId.HELP);
259         myMenu.newMenuItem(ThemisUIMenuId.DATAVIEWER, e -> displayViewerMgr());
260         myMenu.newMenuItem(ThemisUIMenuId.ABOUT, e -> displayAbout());
261 
262         /* Create the data window */
263         theDataWdw = myToolkit.newViewerWindow();
264         theDataWdw.getEventRegistrar().addEventListener(TethysUIEvent.WINDOWCLOSED,
265                 e -> theMenuBar.setEnabled(ThemisUIMenuId.DATAVIEWER, true));
266 
267         /* Create viewer locations */
268         final MetisViewerManager myViewer = myToolkit.getViewerManager();
269         theErrorEntry = myViewer.getStandardEntry(MetisViewerStandardEntry.ERROR);
270         theSolverEntry = myViewer.newEntry(ThemisDataResource.DATA_SOLVER.getValue());
271         theStatsEntry = myViewer.newEntry(ThemisDataResource.DATA_STATS.getValue());
272         theSourceEntry = myViewer.newEntry(ThemisDataResource.DATA_SOURCE.getValue());
273 
274         /* Handle the default location */
275         final File myLocation = getDefaultLocation();
276         if (myLocation != null) {
277             handleNewProject(myLocation);
278         }
279     }

291     /**
292      * Handle select project.
293      */
294     void selectProject() {
295         /* Determine initial directory */
296         final File myInit = new File(System.getProperty("user.home"));
297 
298         /* Determine the name of the directory to load */
299         final TethysUIDialogFactory myDialogs = theGuiFactory.dialogFactory();
300         final TethysUIDirectorySelector myDialog = myDialogs.newDirectorySelector();
301         myDialog.setTitle(ThemisUIResource.SELECT_PROJECT.getValue());
302         myDialog.setInitialDirectory(myInit);
303         final File myFile = myDialog.selectDirectory();
304 
305         /* If we selected a file */
306         if (myFile != null) {
307             /* Handle the new project */
308             handleNewProject(myFile);
309         }
310     }
311 
312     /**
313      * Handle the new project.
314      *
315      * @param pProjectDir the new project directory
316      */
317     private void handleNewProject(final File pProjectDir) {
318         /* Adjust visibility */
319         setVisibility(true);
320 
321         /* Create and start thread */
322         final ThemisUIThread myLoader = new ThemisUIThread(this, pProjectDir);
323         theThreadMgr.startThread(myLoader);
324     }

326     /**
327      * Set panel visibility.
328      *
329      * @param pLoading are we loading?
330      */
331     private void setVisibility(final boolean pLoading) {
332         /* Set tabs visibility */
333         theRefsTab.setVisible(!pLoading);
334         theStatsTab.setVisible(!pLoading);
335         theSourceTab.setVisible(!pLoading);
336 
337         /* Update selection bar and menuBar */
338         theProjectControl.setVisible(!pLoading);
339         theStatusBar.setVisible(pLoading);
340         theMenuBar.setEnabled(ThemisUIMenuId.HELP, !pLoading);
341 
342         /* Set error entry visibility */
343         theErrorEntry.setVisible(theErrorEntry.getObject() != null);
344     }

346     @Override
347     public void setNewData(final ThemisUIData pData) {
348         /* Update source */
349         final ThemisProject myProject = pData.getParsedProject();
350         theSource.setCurrentProject(myProject);
351         theProjectButton.setText(myProject.toString());
352         theSourceEntry.setObject(myProject);
353 
354         /* Update references */
355         final ThemisSolverProject mySolved = pData.getSolvedProject();
356         theRefs.setCurrentProject(mySolved);
357         theSolverEntry.setObject(mySolved);
358 
359         /* Resolve stats */
360         final ThemisStatsProject myStats = pData.getProjectStats();
361         theStats.setCurrentProject(myStats);
362         theStatsEntry.setObject(myStats);
363 
364         /* Save details */
365         final OceanusException myError = storeDefaultLocation(pData.getProjectDir());
366         if (myError != null) {
367             writeErrorToLog(myError);
368         }
369     }

371     /**

373      *
374      * @param pError the error

376     private void writeErrorToLog(final OceanusException pError) {
377         theLogSink.writeLogMessage(pError.getMessage());
378         theLogTab.setVisible(true);

381     /**
382      * Obtain the default location.
383      *
384      * @return the default location
385      */
386     private File getDefaultLocation() {
387         final Preferences myPreferences = deriveHandle();
388         final String myLocation = myPreferences.get(PREF_PROJECT, null);
389         return myLocation == null ? null : new File(myLocation);
390     }
391 
392     /**
393      * Store the default location.
394      *
395      * @param pLocation the default location
396      * @return exception or null
397      */
398     private OceanusException storeDefaultLocation(final File pLocation) {
399         /* Protect against exceptions */
400         try {
401             final Preferences myPreferences = deriveHandle();
402             myPreferences.put(PREF_PROJECT, pLocation.getAbsolutePath());
403             myPreferences.flush();
404             return null;
405         } catch (BackingStoreException e) {
406             return new ThemisIOException("Failed to save preference", e);
407         }
408     }
409 
410     /**
411      * Derive handle for node.
412      *
413      * @return the class name
414      */
415     private Preferences deriveHandle() {
416         /* Obtain the class name */
417         final Class<?> myClass = this.getClass();
418         String myName = myClass.getCanonicalName();
419 
420         /* Obtain the package name */
421         final String myPackage = myClass.getPackage().getName();
422 
423         /* Strip off the package name */
424         myName = myName.substring(myPackage.length() + 1);
425 
426         /* Derive the handle */
427         final Preferences myHandle = Preferences.userNodeForPackage(myClass);
428         return myHandle.node(myName);
429     }
