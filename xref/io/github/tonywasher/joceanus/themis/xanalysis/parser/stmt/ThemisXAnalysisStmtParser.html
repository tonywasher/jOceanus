17  package io.github.tonywasher.joceanus.themis.xanalysis.parser.stmt;
18  
19  import com.github.javaparser.ast.stmt.Statement;
20  import io.github.tonywasher.joceanus.oceanus.base.OceanusException;
21  import io.github.tonywasher.joceanus.themis.xanalysis.parser.base.ThemisXAnalysisInstance.ThemisXAnalysisStatementInstance;
22  import io.github.tonywasher.joceanus.themis.xanalysis.parser.base.ThemisXAnalysisParserDef;
23  
24  /**
25   * Analysis Statement Parser.
26   */
27  public final class ThemisXAnalysisStmtParser {
28      /**
29       * Private Constructor.
30       */
31      private ThemisXAnalysisStmtParser() {
32      }
33  
34      /**
35       * Parse a statement.
36       *
37       * @param pParser    the parser
38       * @param pStatement the statement
39       * @return the parsed statement
40       * @throws OceanusException on error
41       */
42      public static ThemisXAnalysisStatementInstance parseStatement(final ThemisXAnalysisParserDef pParser,
43                                                                    final Statement pStatement) throws OceanusException {
44          /* Handle null Statement */
45          if (pStatement == null) {
46              return null;
47          }
48  
49          /* Allocate correct Statement */
50          switch (ThemisXAnalysisStatement.determineStatement(pParser, pStatement)) {
51              case ASSERT:
52                  return new ThemisXAnalysisStmtAssert(pParser, pStatement.asAssertStmt());
53              case BLOCK:
54                  return new ThemisXAnalysisStmtBlock(pParser, pStatement.asBlockStmt());
55              case BREAK:
56                  return new ThemisXAnalysisStmtBreak(pParser, pStatement.asBreakStmt());
57              case CONSTRUCTOR:
58                  return new ThemisXAnalysisStmtConstructor(pParser, pStatement.asExplicitConstructorInvocationStmt());
59              case CONTINUE:
60                  return new ThemisXAnalysisStmtContinue(pParser, pStatement.asContinueStmt());
61              case DO:
62                  return new ThemisXAnalysisStmtDo(pParser, pStatement.asDoStmt());
63              case EMPTY:
64                  return new ThemisXAnalysisStmtEmpty(pParser, pStatement.asEmptyStmt());
65              case EXPRESSION:
66                  return new ThemisXAnalysisStmtExpression(pParser, pStatement.asExpressionStmt());
67              case FOR:
68                  return new ThemisXAnalysisStmtFor(pParser, pStatement.asForStmt());
69              case FOREACH:
70                  return new ThemisXAnalysisStmtForEach(pParser, pStatement.asForEachStmt());
71              case IF:
72                  return new ThemisXAnalysisStmtIf(pParser, pStatement.asIfStmt());
73              case LABELED:
74                  return new ThemisXAnalysisStmtLabeled(pParser, pStatement.asLabeledStmt());
75              case LOCALCLASS:
76                  return new ThemisXAnalysisStmtClass(pParser, pStatement.asLocalClassDeclarationStmt());
77              case LOCALRECORD:
78                  return new ThemisXAnalysisStmtRecord(pParser, pStatement.asLocalRecordDeclarationStmt());
79              case RETURN:
80                  return new ThemisXAnalysisStmtReturn(pParser, pStatement.asReturnStmt());
81              case SWITCH:
82                  return new ThemisXAnalysisStmtSwitch(pParser, pStatement.asSwitchStmt());
83              case SYNCHRONIZED:
84                  return new ThemisXAnalysisStmtSynch(pParser, pStatement.asSynchronizedStmt());
85              case THROW:
86                  return new ThemisXAnalysisStmtThrow(pParser, pStatement.asThrowStmt());
87              case TRY:
88                  return new ThemisXAnalysisStmtTry(pParser, pStatement.asTryStmt());
89              case WHILE:
90                  return new ThemisXAnalysisStmtWhile(pParser, pStatement.asWhileStmt());
91              case YIELD:
92                  return new ThemisXAnalysisStmtYield(pParser, pStatement.asYieldStmt());
93              default:
94                  throw pParser.buildException("Unsupported Statement Type", pStatement);
95          }
96      }
97  }
