17  package io.github.tonywasher.joceanus.themis.xanalysis.parser.expr;
18  
19  import com.github.javaparser.ast.expr.Expression;
20  import io.github.tonywasher.joceanus.oceanus.base.OceanusException;
21  import io.github.tonywasher.joceanus.themis.xanalysis.parser.base.ThemisXAnalysisInstance.ThemisXAnalysisExpressionInstance;
22  import io.github.tonywasher.joceanus.themis.xanalysis.parser.base.ThemisXAnalysisParserDef;
23  
24  /**
25   * Analysis Expression Parser.
26   */
27  public final class ThemisXAnalysisExprParser {
28      /**
29       * Private Constructor.
30       */
31      private ThemisXAnalysisExprParser() {
32      }
33  
34      /**
35       * Parse an expression.
36       *
37       * @param pParser the parser
38       * @param pExpr   the expression
39       * @return the parsed expression
40       * @throws OceanusException on error
41       */
42      public static ThemisXAnalysisExpressionInstance parseExpression(final ThemisXAnalysisParserDef pParser,
43                                                                      final Expression pExpr) throws OceanusException {
44          /* Handle null Expression */
45          if (pExpr == null) {
46              return null;
47          }
48  
49          /* Allocate correct Expression */
50          switch (ThemisXAnalysisExpression.determineExpression(pParser, pExpr)) {
51              case ARRAYACCESS:
52                  return new ThemisXAnalysisExprArrayAccess(pParser, pExpr.asArrayAccessExpr());
53              case ARRAYCREATION:
54                  return new ThemisXAnalysisExprArrayCreation(pParser, pExpr.asArrayCreationExpr());
55              case ARRAYINIT:
56                  return new ThemisXAnalysisExprArrayInit(pParser, pExpr.asArrayInitializerExpr());
57              case ASSIGN:
58                  return new ThemisXAnalysisExprAssign(pParser, pExpr.asAssignExpr());
59              case BINARY:
60                  return new ThemisXAnalysisExprBinary(pParser, pExpr.asBinaryExpr());
61              case BOOLEAN:
62                  return new ThemisXAnalysisExprBooleanLit(pParser, pExpr.asBooleanLiteralExpr());
63              case CAST:
64                  return new ThemisXAnalysisExprCast(pParser, pExpr.asCastExpr());
65              case CHAR:
66                  return new ThemisXAnalysisExprCharLit(pParser, pExpr.asCharLiteralExpr());
67              case CLASS:
68                  return new ThemisXAnalysisExprClass(pParser, pExpr.asClassExpr());
69              case CONDITIONAL:
70                  return new ThemisXAnalysisExprConditional(pParser, pExpr.asConditionalExpr());
71              case DOUBLE:
72                  return new ThemisXAnalysisExprDoubleLit(pParser, pExpr.asDoubleLiteralExpr());
73              case ENCLOSED:
74                  return new ThemisXAnalysisExprEnclosed(pParser, pExpr.asEnclosedExpr());
75              case FIELDACCESS:
76                  return new ThemisXAnalysisExprFieldAccess(pParser, pExpr.asFieldAccessExpr());
77              case INSTANCEOF:
78                  return new ThemisXAnalysisExprInstanceOf(pParser, pExpr.asInstanceOfExpr());
79              case INTEGER:
80                  return new ThemisXAnalysisExprIntegerLit(pParser, pExpr.asIntegerLiteralExpr());
81              case LAMBDA:
82                  return new ThemisXAnalysisExprLambda(pParser, pExpr.asLambdaExpr());
83              case LONG:
84                  return new ThemisXAnalysisExprLongLit(pParser, pExpr.asLongLiteralExpr());
85              case MARKER:
86                  return new ThemisXAnalysisExprMarkerAnnotation(pParser, pExpr.asMarkerAnnotationExpr());
87              case METHODCALL:
88                  return new ThemisXAnalysisExprMethodCall(pParser, pExpr.asMethodCallExpr());
89              case METHODREFERENCE:
90                  return new ThemisXAnalysisExprMethodRef(pParser, pExpr.asMethodReferenceExpr());
91              case NAME:
92                  return new ThemisXAnalysisExprName(pParser, pExpr.asNameExpr());
93              case NORMAL:
94                  return new ThemisXAnalysisExprNormalAnnotation(pParser, pExpr.asNormalAnnotationExpr());
95              case NULL:
96                  return new ThemisXAnalysisExprNullLit(pParser, pExpr.asNullLiteralExpr());
97              case OBJECTCREATE:
98                  return new ThemisXAnalysisExprObjectCreate(pParser, pExpr.asObjectCreationExpr());
99              case RECORDPATTERN:
100                 return new ThemisXAnalysisExprRecordPattern(pParser, pExpr.asRecordPatternExpr());
101             case SINGLEMEMBER:
102                 return new ThemisXAnalysisExprSingleMemberAnnotation(pParser, pExpr.asSingleMemberAnnotationExpr());
103             case STRING:
104                 return new ThemisXAnalysisExprStringLit(pParser, pExpr.asStringLiteralExpr());
105             case SUPER:
106                 return new ThemisXAnalysisExprSuper(pParser, pExpr.asSuperExpr());
107             case SWITCH:
108                 return new ThemisXAnalysisExprSwitch(pParser, pExpr.asSwitchExpr());
109             case TEXTBLOCK:
110                 return new ThemisXAnalysisExprTextBlockLit(pParser, pExpr.asTextBlockLiteralExpr());
111             case THIS:
112                 return new ThemisXAnalysisExprThis(pParser, pExpr.asThisExpr());
113             case TYPE:
114                 return new ThemisXAnalysisExprType(pParser, pExpr.asTypeExpr());
115             case TYPEPATTERN:
116                 return new ThemisXAnalysisExprTypePattern(pParser, pExpr.asTypePatternExpr());
117             case UNARY:
118                 return new ThemisXAnalysisExprUnary(pParser, pExpr.asUnaryExpr());
119             case VARIABLE:
120                 return new ThemisXAnalysisExprVarDecl(pParser, pExpr.asVariableDeclarationExpr());
121             default:
122                 throw pParser.buildException("Unsupported Expression Type", pExpr);
123         }
124     }
125 }
