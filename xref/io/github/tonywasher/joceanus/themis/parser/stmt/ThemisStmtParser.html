17  package io.github.tonywasher.joceanus.themis.parser.stmt;
18  
19  import com.github.javaparser.ast.stmt.Statement;
20  import io.github.tonywasher.joceanus.oceanus.base.OceanusException;
21  import io.github.tonywasher.joceanus.themis.parser.base.ThemisInstance.ThemisStatementInstance;
22  import io.github.tonywasher.joceanus.themis.parser.base.ThemisParserDef;
23  
24  /**
25   * Analysis Statement Parser.
26   */
27  public final class ThemisStmtParser {
28      /**
29       * Private Constructor.
30       */
31      private ThemisStmtParser() {
32      }
33  
34      /**
35       * Parse a statement.
36       *
37       * @param pParser    the parser
38       * @param pStatement the statement
39       * @return the parsed statement
40       * @throws OceanusException on error
41       */
42      public static ThemisStatementInstance parseStatement(final ThemisParserDef pParser,
43                                                           final Statement pStatement) throws OceanusException {
44          /* Handle null Statement */
45          if (pStatement == null) {
46              return null;
47          }
48  
49          /* Allocate correct Statement */
50          return switch (ThemisStatement.determineStatement(pParser, pStatement)) {
51              case ASSERT -> new ThemisStmtAssert(pParser, pStatement.asAssertStmt());
52              case BLOCK -> new ThemisStmtBlock(pParser, pStatement.asBlockStmt());
53              case BREAK -> new ThemisStmtBreak(pParser, pStatement.asBreakStmt());
54              case CONSTRUCTOR -> new ThemisStmtConstructor(pParser, pStatement.asExplicitConstructorInvocationStmt());
55              case CONTINUE -> new ThemisStmtContinue(pParser, pStatement.asContinueStmt());
56              case DO -> new ThemisStmtDo(pParser, pStatement.asDoStmt());
57              case EMPTY -> new ThemisStmtEmpty(pParser, pStatement.asEmptyStmt());
58              case EXPRESSION -> new ThemisStmtExpression(pParser, pStatement.asExpressionStmt());
59              case FOR -> new ThemisStmtFor(pParser, pStatement.asForStmt());
60              case FOREACH -> new ThemisStmtForEach(pParser, pStatement.asForEachStmt());
61              case IF -> new ThemisStmtIf(pParser, pStatement.asIfStmt());
62              case LABELED -> new ThemisStmtLabeled(pParser, pStatement.asLabeledStmt());
63              case LOCALCLASS -> new ThemisStmtClass(pParser, pStatement.asLocalClassDeclarationStmt());
64              case LOCALRECORD -> new ThemisStmtRecord(pParser, pStatement.asLocalRecordDeclarationStmt());
65              case RETURN -> new ThemisStmtReturn(pParser, pStatement.asReturnStmt());
66              case SWITCH -> new ThemisStmtSwitch(pParser, pStatement.asSwitchStmt());
67              case SYNCHRONIZED -> new ThemisStmtSynch(pParser, pStatement.asSynchronizedStmt());
68              case THROW -> new ThemisStmtThrow(pParser, pStatement.asThrowStmt());
69              case TRY -> new ThemisStmtTry(pParser, pStatement.asTryStmt());
70              case WHILE -> new ThemisStmtWhile(pParser, pStatement.asWhileStmt());
71              case YIELD -> new ThemisStmtYield(pParser, pStatement.asYieldStmt());
72              default -> throw pParser.buildException("Unsupported Statement Type", pStatement);
73          };
74      }
75  }
