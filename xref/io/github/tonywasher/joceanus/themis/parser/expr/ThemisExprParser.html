17  package io.github.tonywasher.joceanus.themis.parser.expr;
18  
19  import com.github.javaparser.ast.expr.Expression;
20  import io.github.tonywasher.joceanus.oceanus.base.OceanusException;
21  import io.github.tonywasher.joceanus.themis.parser.base.ThemisInstance.ThemisExpressionInstance;
22  import io.github.tonywasher.joceanus.themis.parser.base.ThemisParserDef;
23  
24  /**
25   * Analysis Expression Parser.
26   */
27  public final class ThemisExprParser {
28      /**
29       * Private Constructor.
30       */
31      private ThemisExprParser() {
32      }
33  
34      /**
35       * Parse an expression.
36       *
37       * @param pParser the parser
38       * @param pExpr   the expression
39       * @return the parsed expression
40       * @throws OceanusException on error
41       */
42      public static ThemisExpressionInstance parseExpression(final ThemisParserDef pParser,
43                                                             final Expression pExpr) throws OceanusException {
44          /* Handle null Expression */
45          if (pExpr == null) {
46              return null;
47          }
48  
49          /* Allocate correct Expression */
50          return switch (ThemisExpression.determineExpression(pParser, pExpr)) {
51              case ARRAYACCESS -> new ThemisExprArrayAccess(pParser, pExpr.asArrayAccessExpr());
52              case ARRAYCREATION -> new ThemisExprArrayCreation(pParser, pExpr.asArrayCreationExpr());
53              case ARRAYINIT -> new ThemisExprArrayInit(pParser, pExpr.asArrayInitializerExpr());
54              case ASSIGN -> new ThemisExprAssign(pParser, pExpr.asAssignExpr());
55              case BINARY -> new ThemisExprBinary(pParser, pExpr.asBinaryExpr());
56              case BOOLEAN -> new ThemisExprBooleanLit(pParser, pExpr.asBooleanLiteralExpr());
57              case CAST -> new ThemisExprCast(pParser, pExpr.asCastExpr());
58              case CHAR -> new ThemisExprCharLit(pParser, pExpr.asCharLiteralExpr());
59              case CLASS -> new ThemisExprClass(pParser, pExpr.asClassExpr());
60              case CONDITIONAL -> new ThemisExprConditional(pParser, pExpr.asConditionalExpr());
61              case DOUBLE -> new ThemisExprDoubleLit(pParser, pExpr.asDoubleLiteralExpr());
62              case ENCLOSED -> new ThemisExprEnclosed(pParser, pExpr.asEnclosedExpr());
63              case FIELDACCESS -> new ThemisExprFieldAccess(pParser, pExpr.asFieldAccessExpr());
64              case INSTANCEOF -> new ThemisExprInstanceOf(pParser, pExpr.asInstanceOfExpr());
65              case INTEGER -> new ThemisExprIntegerLit(pParser, pExpr.asIntegerLiteralExpr());
66              case LAMBDA -> new ThemisExprLambda(pParser, pExpr.asLambdaExpr());
67              case LONG -> new ThemisExprLongLit(pParser, pExpr.asLongLiteralExpr());
68              case MARKER -> new ThemisExprMarkerAnnotation(pParser, pExpr.asMarkerAnnotationExpr());
69              case METHODCALL -> new ThemisExprMethodCall(pParser, pExpr.asMethodCallExpr());
70              case METHODREFERENCE -> new ThemisExprMethodRef(pParser, pExpr.asMethodReferenceExpr());
71              case NAME -> new ThemisExprName(pParser, pExpr.asNameExpr());
72              case NORMAL -> new ThemisExprNormalAnnotation(pParser, pExpr.asNormalAnnotationExpr());
73              case NULL -> new ThemisExprNullLit(pParser, pExpr.asNullLiteralExpr());
74              case OBJECTCREATE -> new ThemisExprObjectCreate(pParser, pExpr.asObjectCreationExpr());
75              case RECORDPATTERN -> new ThemisExprRecordPattern(pParser, pExpr.asRecordPatternExpr());
76              case SINGLEMEMBER -> new ThemisExprSingleMemberAnnotation(pParser, pExpr.asSingleMemberAnnotationExpr());
77              case STRING -> new ThemisExprStringLit(pParser, pExpr.asStringLiteralExpr());
78              case SUPER -> new ThemisExprSuper(pParser, pExpr.asSuperExpr());
79              case SWITCH -> new ThemisExprSwitch(pParser, pExpr.asSwitchExpr());
80              case TEXTBLOCK -> new ThemisExprTextBlockLit(pParser, pExpr.asTextBlockLiteralExpr());
81              case THIS -> new ThemisExprThis(pParser, pExpr.asThisExpr());
82              case TYPE -> new ThemisExprType(pParser, pExpr.asTypeExpr());
83              case TYPEPATTERN -> new ThemisExprTypePattern(pParser, pExpr.asTypePatternExpr());
84              case UNARY -> new ThemisExprUnary(pParser, pExpr.asUnaryExpr());
85              case VARIABLE -> new ThemisExprVarDecl(pParser, pExpr.asVariableDeclarationExpr());
86              default -> throw pParser.buildException("Unsupported Expression Type", pExpr);
87          };
88      }
89  }
