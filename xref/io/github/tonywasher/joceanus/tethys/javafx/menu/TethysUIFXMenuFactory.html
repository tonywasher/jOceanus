17  package io.github.tonywasher.joceanus.tethys.javafx.menu;
18  
19  import io.github.tonywasher.joceanus.tethys.api.menu.TethysUIMenuBarManager;
20  import io.github.tonywasher.joceanus.tethys.api.menu.TethysUIMenuFactory;
21  import io.github.tonywasher.joceanus.tethys.api.menu.TethysUIScrollMenu;
22  import io.github.tonywasher.joceanus.tethys.api.menu.TethysUIToolBarManager;
23  import io.github.tonywasher.joceanus.tethys.core.factory.TethysUICoreFactory;
24  
25  /**
26   * FX Menu Factory.
27   */
28  public class TethysUIFXMenuFactory
29          implements TethysUIMenuFactory {
30      /**
31       * The Factory.
32       */
33      private final TethysUICoreFactory<?> theFactory;
34  
35      /**
36       * Constructor.
37       *
38       * @param pFactory the factory.
39       */
40      public TethysUIFXMenuFactory(final TethysUICoreFactory<?> pFactory) {
41          /* Store parameters */
42          theFactory = pFactory;
43      }
44  
45      @Override
46      public <T> TethysUIScrollMenu<T> newContextMenu() {
47          return new TethysUIFXScrollMenu<>();
48      }
49  
50      @Override
51      public TethysUIMenuBarManager newMenuBar() {
52          return new TethysUIFXMenuBarManager();
53      }
54  
55      @Override
56      public TethysUIToolBarManager newToolBar() {
57          return new TethysUIFXToolBarManager(theFactory);
58      }
59  }
