150     /**
151      * Constructor.
152      *
153      * @param pFactory     the GUI factory
154      * @param pEditControl the edit Control
155      */
156     TethysUIFXDataTextField(final TethysUICoreFactory<?> pFactory,
157                             final Node pEditControl) {
158         /* Initialise the underlying class */
159         super(pFactory);
160 
161         /* Create resources */
162         thePane = new BorderPane();
163         theNode = new TethysUIFXNode(thePane);
164         theLabel = new Label();
165         theEditControl = (Control) pEditControl;
166 
167         /* Declare the label and edit control to be dataField */
168         theLabel.getStyleClass().add(TethysUIFXUtils.CSS_STYLE_BASE);
169         theLabel.getStyleClass().add(STYLE_FIELD);
170         theEditControl.getStyleClass().add(TethysUIFXUtils.CSS_STYLE_BASE);
171         theEditControl.getStyleClass().add(STYLE_FIELD);
172 
173         /* Set maximum widths for fields */
174         theLabel.setMaxWidth(Integer.MAX_VALUE);
175         theEditControl.setMaxWidth(Integer.MAX_VALUE);
176 
177         /* Create the command button */
178         theCmdButton = new Button();
179         theCmdButton.setGraphic(TethysUIFXArrowIcon.DOWN.getArrow());
180         theCmdButton.setFocusTraversable(false);
181 
182         /* declare the menu */
183         declareCmdMenu(pFactory.menuFactory().newContextMenu());
184 
185         /* Default to readOnly */
186         thePane.setCenter(theLabel);
187 
188         /* handle command button action */
189         theCmdButton.setOnAction(e -> handleCmdMenuRequest());
190 
191         /* Set context menu listener */
192         getCmdMenu().getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE,
193                 e -> handleCmdMenuClosed());
194     }

262     @Override
263     public void showCmdButton(final boolean pShow) {
264         /* Remove any button that is displaying */
265         thePane.setRight(null);
266         doShowCmdButton = pShow;
267 
268         /* If we have a button to display */
269         if (isEditable() && doShowCmdButton) {
270             thePane.setRight(theCmdButton);
271         }
272     }

274     @Override
275     public void setEditable(final boolean pEditable) {
276         /* Obtain current setting */
277         final boolean isEditable = isEditable();
278 
279         /* If we are changing */
280         if (pEditable != isEditable) {
281             /* If we are setting editable */
282             if (pEditable) {
283                 thePane.setCenter(theEditControl);
284                 if (doShowCmdButton) {
285                     thePane.setRight(theCmdButton);
286                 }
287             } else {
288                 thePane.setCenter(theLabel);
289                 thePane.setRight(null);
290             }
291 
292             /* Pass call on */
293             super.setEditable(pEditable);
294         }
295     }

334     private static String getStyleForAttribute(final TethysUIFieldAttribute pAttr) {

345                 return STYLE_FIELDSET;
346             case ALTERNATE:
347             default:
348                 return STYLE_ALTERNATE;
349         }

352     /**
353      * TextField class.
354      *
355      * @param <T> the data type
356      * @param <F> the field type
357      */
358     public abstract static class TethysUIFXTextEditField<T, F extends TextInputControl>
359             extends TethysUIFXDataTextField<T>
360             implements TethysUIValidatedEditField<T> {
361         /**
362          * The error pseudoClass.
363          */
364         private static final PseudoClass ERROR_CLASS = PseudoClass.getPseudoClass(STYLE_ERROR);
365 
366         /**
367          * The converterControl.
368          */
369         private final TethysUIDataEditTextFieldControl<T> theControl;
370 
371         /**
372          * The textField.
373          */
374         private final F theTextField;
375 
376         /**
377          * The error text.
378          */
379         private String theErrorText;
380 
381         /**
382          * Are we editing a cell?
383          */
384         private boolean isCellEditing;
385 
386         /**
387          * Constructor.
388          *
389          * @param pFactory   the GUI factory
390          * @param pConverter the text converter
391          * @param pField     the edit field
392          */
393         TethysUIFXTextEditField(final TethysUICoreFactory<?> pFactory,
394                                 final TethysUICoreDataEditConverter<T> pConverter,
395                                 final F pField) {
396             /* Initialise underlying class */
397             super(pFactory, pField);
398 
399             /* Create the converter control */
400             theControl = new TethysUIDataEditTextFieldControl<>(this, pConverter);
401 
402             /* Access the fields */
403             final Label myLabel = getLabel();
404             theTextField = getEditControl();
405 
406             /* Set Padding */
407             myLabel.setPadding(new Insets(PADDING, PADDING, PADDING, PADDING));
408 
409             /* Set alignment */
410             final Pos myAlignment = pConverter.rightAlignFields()
411                     ? Pos.CENTER_RIGHT
412                     : Pos.CENTER_LEFT;
413             myLabel.setAlignment(myAlignment);
414             if (theTextField instanceof TextField myField) {
415                 myField.setAlignment(myAlignment);
416             }
417             theTextField.setEditable(true);
418 
419 
420             /* Add listener to handle change of focus */
421             theTextField.focusedProperty().addListener((v, o, n) -> handleFocusChange(n));
422 
423             /* handle enter/escape keys */
424             theTextField.setOnKeyPressed(this::handleKeyPressed);
425         }
426 
427         @Override
428         public void setValidator(final Function<T, String> pValidator) {
429             theControl.setValidator(pValidator);
430         }
431 
432         @Override
433         public void setReporter(final Consumer<String> pReporter) {
434             theControl.setReporter(pReporter);
435         }
436 
437         /**
438          * handle focusChange.
439          *
440          * @param pFocused is the field focused?
441          */
442         private void handleFocusChange(final boolean pFocused) {
443             if (pFocused) {
444                 handleFocusGained();
445             } else {
446                 handleEscapeKey();
447             }
448         }
449 
450         /**
451          * handle keyPressed.
452          *
453          * @param pEvent the event
454          */
455         private void handleKeyPressed(final KeyEvent pEvent) {
456             switch (pEvent.getCode()) {
457                 case ENTER:
458                     handleEnterKey();
459                     break;
460                 case ESCAPE:
461                     handleEscapeKey();
462                     break;
463                 default:
464                     break;
465             }
466         }
467 
468         /**
469          * handle enterKey.
470          */
471         private void handleEnterKey() {
472             processValue();
473             if (!isAttributeSet(TethysUIFieldAttribute.ERROR)) {
474                 haltCellEditing();
475             }
476         }
477 
478         /**
479          * handle escapeKey.
480          */
481         private void handleEscapeKey() {
482             theTextField.setText(theControl.getEditText());
483             clearError();
484             haltCellEditing();
485         }
486 
487         @Override
488         public void startCellEditing(final Node pCell) {
489             isCellEditing = true;
490             setEditable(true);
491             theControl.clearNewValue();
492             theTextField.requestFocus();
493         }
494 
495         /**
496          * Halt cell editing.
497          */
498         private void haltCellEditing() {
499             if (isCellEditing) {
500                 setEditable(false);
501                 if (!theControl.parsedNewValue()) {
502                     fireEvent(TethysUIEvent.EDITFOCUSLOST);
503                 }
504             }
505             isCellEditing = false;
506         }
507 
508         @Override
509         @SuppressWarnings("unchecked")
510         protected F getEditControl() {
511             return (F) super.getEditControl();
512         }
513 
514         /**
515          * Obtain the converter.
516          *
517          * @return the converter.
518          */
519         protected TethysUICoreDataEditConverter<T> getConverter() {
520             return theControl.getConverter();
521         }
522 
523         /**
524          * Process value.
525          */
526         private void processValue() {
527             /* If we failed to process the value */
528             final String myText = theTextField.getText();
529             if (!theControl.processValue(myText)) {
530                 /* Set toolTip, save error text and retain the focus */
531                 theTextField.setTooltip(new Tooltip(theControl.getErrorText()));
532                 theErrorText = myText;
533 
534                 /* add an error style */
535                 theTextField.pseudoClassStateChanged(ERROR_CLASS, true);
536                 setTheAttribute(TethysUIFieldAttribute.ERROR);
537                 theTextField.requestFocus();
538 
539                 /* else value was OK */
540             } else {
541                 /* Clear error indications */
542                 clearError();
543                 getLabel().setText(theControl.getDisplayText());
544             }
545         }
546 
547         /**
548          * Clear error indication.
549          */
550         private void clearError() {
551             theTextField.setTooltip(null);
552             theErrorText = null;
553             theControl.clearErrorReport();
554             theTextField.pseudoClassStateChanged(ERROR_CLASS, false);
555             clearTheAttribute(TethysUIFieldAttribute.ERROR);
556         }
557 
558         /**
559          * Handle focusGained.
560          */
561         void handleFocusGained() {
562             theTextField.setText(theErrorText == null
563                     ? theControl.getEditText()
564                     : theErrorText);
565             theTextField.selectAll();
566         }
567 
568         @Override
569         public void setValue(final T pValue) {
570             /* Store the value */
571             theControl.setValue(pValue);
572 
573             /* Update nodes */
574             getLabel().setText(theControl.getDisplayText());
575             theTextField.setText(theTextField.isFocused()
576                     ? theControl.getEditText()
577                     : theControl.getDisplayText());
578         }
579     }
