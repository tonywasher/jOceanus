33  /**
34   * Thread TextArea Status Manager.
35   */
36  public abstract class TethysUICoreThreadTextAreaStatus
37          extends TethysUICoreComponent
38          implements TethysUIThreadStatusManager {
39      /**
40       * Cancel button text.
41       */
42      private static final String NLS_CANCEL = TethysUIThreadResource.STATUSBAR_BUTTON_CANCEL.getValue();
43  
44      /**
45       * Clear button text.
46       */
47      private static final String NLS_CLEAR = TethysUIThreadResource.STATUSBAR_BUTTON_CLEAR.getValue();
48  
49      /**
50       * Succeeded message.
51       */
52      private static final String NLS_SUCCEEDED = TethysUIThreadResource.STATUSBAR_STATUS_SUCCESS.getValue();
53  
54      /**
55       * Cancelled message.
56       */
57      private static final String NLS_CANCELLED = TethysUIThreadResource.STATUSBAR_STATUS_CANCEL.getValue();
58  
59      /**
60       * Failed message.
61       */
62      private static final String NLS_FAILED = TethysUIThreadResource.STATUSBAR_STATUS_FAIL.getValue();
63  
64      /**
65       * The Thread Manager.
66       */
67      private final TethysUICoreThreadManager theThreadManager;
68  
69      /**
70       * Pane.
71       */
72      private final TethysUIBorderPaneManager thePane;
73  
74      /**
75       * TextArea.
76       */
77      private final TethysUITextArea theTextArea;
78  
79      /**
80       * Cancel Button.
81       */
82      private final TethysUIButton theCancelButton;
83  
84      /**
85       * The current status.
86       */
87      private TethysUIThreadStatus theStatus;
88  
89      /**
90       * The TextPosition.
91       */
92      private int thePosition;
93  
94      /**
95       * The StatusPosition.
96       */
97      private int theStatusPosition;
98  
99      /**
100      * Constructor.
101      *
102      * @param pManager the Thread Manager
103      * @param pFactory the GUI factory
104      */
105     protected TethysUICoreThreadTextAreaStatus(final TethysUICoreThreadManager pManager,
106                                                final TethysUICoreFactory<?> pFactory) {
107         /* Store parameters */
108         theThreadManager = pManager;
109 
110         /* Create textArea */
111         final TethysUIControlFactory myControls = pFactory.controlFactory();
112         theTextArea = myControls.newTextArea();
113 
114         /* Create buttons */
115         final TethysUIButtonFactory<?> myButtons = pFactory.buttonFactory();
116         theCancelButton = myButtons.newButton();
117         theCancelButton.setTextOnly();
118         theCancelButton.setText(NLS_CANCEL);
119         final TethysUIButton theClearButton = myButtons.newButton();
120         theClearButton.setTextOnly();
121         theClearButton.setText(NLS_CLEAR);
122         theCancelButton.setVisible(false);
123 
124         /* Create a scroll manager */
125         final TethysUIPaneFactory myPanes = pFactory.paneFactory();
126         final TethysUIScrollPaneManager myScroll = myPanes.newScrollPane();
127         myScroll.setContent(theTextArea);
128 
129         /* Create a new subPanel for the buttons */
130         final TethysUIBoxPaneManager myButtonPanel = myPanes.newHBoxPane();
131         myButtonPanel.addNode(theClearButton);
132         myButtonPanel.addNode(theCancelButton);
133 
134         /* Add the components */
135         thePane = myPanes.newBorderPane();
136         thePane.setCentre(myScroll);
137         thePane.setSouth(myButtonPanel);
138 
139         /* Initialise the status */
140         theStatus = new TethysUICoreThreadStatus();
141 
142         /* Add button listeners */
143         theClearButton.getEventRegistrar().addEventListener(e -> handleClear());
144         theCancelButton.getEventRegistrar().addEventListener(e -> handleCancel());
145     }
146 
147     @Override
148     public Integer getId() {
149         return thePane.getId();
150     }
151 
152     @Override
153     public TethysUINode getNode() {
154         return thePane.getNode();
155     }
156 
157     @Override
158     public void setEnabled(final boolean pEnabled) {
159         thePane.setEnabled(pEnabled);
160     }
161 
162     @Override
163     public void setVisible(final boolean pVisible) {
164         thePane.setVisible(pVisible);
165     }
166 
167     @Override
168     public void setProgress(final TethysUIThreadStatus pStatus) {
169         /* Set new status */
170         final TethysUIThreadStatus myOld = theStatus;
171         theStatus = pStatus;
172 
173         /* Handle new task */
174         final String myNewTask = pStatus.getTask();
175         if (myNewTask != null
176                 && !myNewTask.equals(myOld.getTask())) {
177             setNewStage(myNewTask);
178             return;
179         }
180 
181         /* Handle new stage */
182         final int myStagesDone = pStatus.getStagesDone();
183         if (myStagesDone != myOld.getStagesDone()) {
184             setNewStage(pStatus.getStage());
185             return;
186         }
187 
188         /* Handle new step */
189         final int myStepsDone = pStatus.getStepsDone();
190         if (myStepsDone != myOld.getStepsDone()) {
191             final String myStep = pStatus.getStep() == null
192                     ? "" : (": " + pStatus.getStep());
193             final String myDone = (myStepsDone + 1)
194                     + " of "
195                     + pStatus.getNumSteps()
196                     + myStep;
197             setNewStep(myDone);
198         }
199 
200         /* Show the cancel button */
201         theCancelButton.setVisible(true);
202     }
203 
204     /**
205      * Set new stage.
206      *
207      * @param pStage the stage to set
208      */
209     private void setNewStage(final String pStage) {
210         /* Handle null stage */
211         final String myStage = pStage == null
212                 ? ""
213                 : pStage;
214 
215         /* Add status to text area */
216         theTextArea.replaceText(myStage, theStatusPosition, thePosition);
217         theTextArea.appendText("\n");
218         thePosition = theStatusPosition + myStage.length() + 1;
219         theStatusPosition = thePosition;
220         theTextArea.setCaretPosition(thePosition);
221     }
222 
223     /**
224      * Set new step.
225      *
226      * @param pStep the step to set
227      */
228     protected void setNewStep(final String pStep) {
229         /* Replace any existing status */
230         theTextArea.replaceText(pStep, theStatusPosition, thePosition);
231         thePosition = theStatusPosition + pStep.length();
232         theTextArea.appendText("\n");
233         thePosition++;
234         theTextArea.setCaretPosition(thePosition);
235     }
236 
237     /**
238      * Complete the status.
239      *
240      * @param pStatus the final status
241      */
242     private void completeStatus(final String pStatus) {
243         /* Append the text */
244         theTextArea.replaceText(pStatus, theStatusPosition, thePosition);
245         thePosition = theStatusPosition + pStatus.length();
246         theTextArea.appendText("\n");
247         thePosition++;
248         theStatusPosition = thePosition;
249         theTextArea.setCaretPosition(thePosition);
250 
251         /* Hide the cancel button */
252         theCancelButton.setVisible(false);
253         theThreadManager.threadCompleted();
254         theStatus = new TethysUICoreThreadStatus();
255     }
256 
257     @Override
258     public void setCompletion() {
259         /* Obtain the task name */
260         final String myTask = theThreadManager.getTaskName();
261 
262         /* Initialise the message */
263         final String myText = myTask + " " + NLS_SUCCEEDED;
264 
265         /* Complete the status */
266         completeStatus(myText);
267     }
268 
269     @Override
270     public void setFailure(final Throwable pException) {
271         /* Obtain the task name */
272         final String myTask = theThreadManager.getTaskName();
273 
274         /* Initialise the message */
275         String myText = myTask + " " + NLS_FAILED;
276 
277         /* Add the first error detail */
278         myText += " " + pException.getMessage();
279 
280         /* Complete the status */
281         completeStatus(myText);
282     }
283 
284     @Override
285     public void setCancelled() {
286         /* Obtain the task name */
287         final String myTask = theThreadManager.getTaskName();
288 
289         /* Initialise the message */
290         final String myText = myTask + " " + NLS_CANCELLED;
291 
292         /* Complete the status */
293         completeStatus(myText);
294     }
295 
296     /**
297      * handle cancel.
298      */
299     protected void handleCancel() {
300         theThreadManager.cancelWorker();
301     }
302 
303     /**
304      * Handle clear request.
305      */
306     protected void handleClear() {
307         /* Clear contents */
308         theTextArea.setText(null);
309         thePosition = 0;
310         theStatusPosition = 0;
311     }
312 }
