43  /**
44   * Column Definition.
45   *
46   * @param <T> the data type
47   * @param <C> the column identity
48   * @param <R> the row type
49   */
50  public interface TethysUITableColumn<T, C, R>
51          extends OceanusEventProvider<TethysUIEvent> {
52      /**
53       * Obtain the table manager.
54       *
55       * @return the table manager
56       */
57      TethysUITableManager<C, R> getTable();
58  
59      /**
60       * Obtain the id of the column.
61       *
62       * @return the column id
63       */
64      C getId();
65  
66      /**
67       * Obtain the type of the column.
68       *
69       * @return the column type
70       */
71      TethysUIFieldType getCellType();
72  
73      /**
74       * Obtain the name of the column.
75       *
76       * @return the column name
77       */
78      String getName();
79  
80      /**
81       * Set the name of the column.
82       *
83       * @param pName the column name
84       * @return the column
85       */
86      TethysUITableColumn<T, C, R> setName(String pName);
87  
88      /**
89       * Set the column width.
90       *
91       * @param pWidth the width
92       * @return the column
93       */
94      TethysUITableColumn<T, C, R> setColumnWidth(int pWidth);
95  
96      /**
97       * Is the column visible?
98       *
99       * @return true/false
100      */
101     boolean isVisible();
102 
103     /**
104      * Set the visibility of the column.
105      *
106      * @param pVisible true/false
107      * @return the column
108      */
109     TethysUITableColumn<T, C, R> setVisible(boolean pVisible);
110 
111     /**
112      * Is the column editable?
113      *
114      * @return true/false
115      */
116     boolean isEditable();
117 
118     /**
119      * Set the edit-ability of the column.
120      *
121      * @param pEditable true/false
122      * @return the column
123      */
124     TethysUITableColumn<T, C, R> setEditable(boolean pEditable);
125 
126     /**
127      * Set cell value Factory.
128      *
129      * @param pFactory the cell factory
130      * @return the column
131      */
132     TethysUITableColumn<T, C, R> setCellValueFactory(Function<R, T> pFactory);
133 
134     /**
135      * Obtain value for row.
136      *
137      * @param pRow the row
138      * @return the value
139      */
140     T getValueForRow(R pRow);
141 
142     /**
143      * Set the cell-editable tester.
144      *
145      * @param pEditable the editable tester
146      * @return the column
147      */
148     TethysUITableColumn<T, C, R> setCellEditable(Predicate<R> pEditable);
149 
150     /**
151      * Get the cell-editable tester.
152      *
153      * @return the current tester
154      */
155     Predicate<R> getCellEditable();
156 
157     /**
158      * Set the on-commit consumer.
159      *
160      * @param pOnCommit the consumer
161      * @return the column
162      */
163     TethysUITableColumn<T, C, R> setOnCommit(TethysUIOnCellCommit<R, T> pOnCommit);
164 
165     /**
166      * do we rePaintColumn on commit?
167      *
168      * @return true/false
169      */
170     boolean doRePaintColumnOnCommit();
171 
172     /**
173      * Set repaintColumn on Commit.
174      *
175      * @param pRePaint the flag
176      * @return the column
177      */
178     TethysUITableColumn<T, C, R> setRepaintColumnOnCommit(boolean pRePaint);
179 
180     /**
181      * get the column id which forces a rePaint.
182      *
183      * @return the column id
184      */
185     C getRePaintColumnId();
186 
187     /**
188      * Set repaintColumnId.
189      *
190      * @param pRePaintId the repaint id
191      * @return the column
192      */
193     TethysUITableColumn<T, C, R> setRepaintColumnId(C pRePaintId);
194 
195     /**
196      * OnCell commit callback.
197      *
198      * @param <R> the row type
199      * @param <T> the value type
200      */
201     @FunctionalInterface
202     interface TethysUIOnCellCommit<R, T> {
203         /**
204          * CallBack on a columnCommit.
205          *
206          * @param pRow   the row that is being committed
207          * @param pValue the new value
208          * @throws OceanusException on error
209          */
210         void commitCell(R pRow, T pValue) throws OceanusException;
211     }
