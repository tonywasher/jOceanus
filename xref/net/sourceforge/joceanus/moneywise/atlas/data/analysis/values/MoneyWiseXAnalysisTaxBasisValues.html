17  package net.sourceforge.joceanus.moneywise.atlas.data.analysis.values;
18  
19  import java.util.Currency;
20  
21  import net.sourceforge.joceanus.moneywise.atlas.data.analysis.base.MoneyWiseXAnalysisValues;
22  import net.sourceforge.joceanus.oceanus.decimal.OceanusMoney;
23  
24  /**
25   * TaxBasisValues class.
26   */
27  public final class MoneyWiseXAnalysisTaxBasisValues
28          extends MoneyWiseXAnalysisValues<MoneyWiseXAnalysisTaxBasisValues, MoneyWiseXAnalysisTaxBasisAttr> {
29      /**
30       * Constructor.
31       * @param pCurrency the reporting currency
32       */
33      public MoneyWiseXAnalysisTaxBasisValues(final Currency pCurrency) {
34          /* Initialise class */
35          super(MoneyWiseXAnalysisTaxBasisAttr.class);
36  
37          /* Create all possible values */
38          super.setValue(MoneyWiseXAnalysisTaxBasisAttr.GROSS, new OceanusMoney(pCurrency));
39          super.setValue(MoneyWiseXAnalysisTaxBasisAttr.NETT, new OceanusMoney(pCurrency));
40          super.setValue(MoneyWiseXAnalysisTaxBasisAttr.TAXCREDIT, new OceanusMoney(pCurrency));
41      }
42  
43      /**
44       * Constructor.
45       * @param pSource the source map.
46       */
47      private MoneyWiseXAnalysisTaxBasisValues(final MoneyWiseXAnalysisTaxBasisValues pSource) {
48          /* Initialise class */
49          super(pSource);
50      }
51  
52      @Override
53      protected MoneyWiseXAnalysisTaxBasisValues newSnapShot() {
54          return new MoneyWiseXAnalysisTaxBasisValues(this);
55      }
56  
57      @Override
58      public void adjustToBaseValues(final MoneyWiseXAnalysisTaxBasisValues pBase) {
59          /* Adjust gross/net/tax values */
60          adjustMoneyToBase(pBase, MoneyWiseXAnalysisTaxBasisAttr.GROSS);
61          adjustMoneyToBase(pBase, MoneyWiseXAnalysisTaxBasisAttr.NETT);
62          adjustMoneyToBase(pBase, MoneyWiseXAnalysisTaxBasisAttr.TAXCREDIT);
63      }
64  
65      @Override
66      public void resetBaseValues() {
67          /* Create a zero value in the correct currency */
68          OceanusMoney myValue = super.getMoneyValue(MoneyWiseXAnalysisTaxBasisAttr.GROSS);
69          myValue = new OceanusMoney(myValue);
70          myValue.setZero();
71  
72          /* Reset Income and expense values */
73          super.setValue(MoneyWiseXAnalysisTaxBasisAttr.GROSS, myValue);
74          super.setValue(MoneyWiseXAnalysisTaxBasisAttr.NETT, new OceanusMoney(myValue));
75          super.setValue(MoneyWiseXAnalysisTaxBasisAttr.TAXCREDIT, new OceanusMoney(myValue));
76      }
77  
78      /**
79       * Are the values?
80       * @return true/false
81       */
82      public boolean isActive() {
83          final OceanusMoney myGross = super.getMoneyValue(MoneyWiseXAnalysisTaxBasisAttr.GROSS);
84          final OceanusMoney myNet = super.getMoneyValue(MoneyWiseXAnalysisTaxBasisAttr.NETT);
85          final OceanusMoney myTax = super.getMoneyValue(MoneyWiseXAnalysisTaxBasisAttr.TAXCREDIT);
86          return myGross.isNonZero() || myNet.isNonZero() || myTax.isNonZero();
87      }
88  }
