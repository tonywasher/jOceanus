106         theTax.declareSecurityAnalyser(theSecurity);

117     /**
118      * Process transaction event.
119      * @param pEvent the event
120      * @throws OceanusException on error
121      */
122     public void processTransaction(final MoneyWiseXAnalysisEvent pEvent) throws OceanusException {
123         /* Parse the event */
124         theTransaction = new MoneyWiseXAnalysisTransaction(pEvent);
125         final MoneyWiseTransaction myTrans = theTransaction.getTransaction();
126 
127         /* Ignore header transactions */
128         if (!myTrans.isHeader()) {
129             /* Touch underlying items */
130             myTrans.touchUnderlyingItems();
131 
132             /* Process tags */
133             final List<MoneyWiseTransTag> myTags = myTrans.getTransactionTags();
134             if (myTags != null) {
135                 /* Process the transaction tags */
136                 theAnalysis.getTransactionTags().processEvent(pEvent, myTags.iterator());
137             }
138 
139             /* Process the transaction */
140             processTransaction();
141         }
142     }

144     /**
145      * Process transaction.
146      * @throws OceanusException on error
147      */
148     private void processTransaction() throws OceanusException {
149         /* Access account and partner */
150         final MoneyWiseTransaction myTrans = theTransaction.getTransaction();
151         final MoneyWiseTransAsset myDebit = theTransaction.getDebitAccount();
152         final MoneyWiseTransAsset myCredit = theTransaction.getCreditAccount();
153 
154         /* If the event relates to a security holding account, split out the workings */
155         if (isSecurityHolding(myDebit)) {
156             /* Process as a Security transaction */
157             theSecurity.processDebitSecurity(theTransaction);
158 
159             /* If the event relates to a security holding partner, split out the workings */
160         } else if (isSecurityHolding(myCredit)) {
161             /* Process as a Security transaction */
162             theSecurity.processCreditSecurity(theTransaction);
163 
164             /* If the event is portfolioXfer, split out the workings */
165         } else if (isPortfolioXfer(theTransaction.getCategoryClass())) {
166             /* Process the portfolioXfer */
167             thePortfolioXfer.processPortfolioXfer(theTransaction);
168 
169             /* Else handle the event normally */
170         } else if (myDebit instanceof MoneyWiseAssetBase
171                 && myCredit instanceof MoneyWiseAssetBase) {
172             processNonSecurity();
173 
174             /* else reject */
175         } else {
176             throw new MoneyWiseLogicException("Invalid Asset Pair: "
177                     + myTrans.getAccount().getAssetType()
178                     + " "
179                     + myTrans.getDirection()
180                     + " "
181                     + myTrans.getPartner().getAssetType());
182         }
183 
184         /* adjust category bucket */
185         adjustCategoryBucket();
186 
187         /* Process taxBasis */
188         theTax.adjustTaxBasis(theTransaction);
189 
190         /* Register the eventBuckets */
191         final MoneyWiseXAnalysisEvent myEvent = theTransaction.getEvent();
192         theMarket.adjustMarketTotals(myEvent);
193         theState.registerInterestedBucketsForEvent(myEvent);
194     }

196     /**
197      * process non-security transaction.
198      */
199     private void processNonSecurity() {
200         /* Adjust parent details */
201         theTransaction.adjustParent();
202 
203         /* Process Asset accounts */
204         processAssets();
205 
206         /* Process Auto-expense accounts */
207         processAutoExpense();
208 
209         /* Process Payee accounts */
210         processPayees();
211     }

213     /**
214      * process assets.
215      */
216     private void processAssets() {
217         /* Access debit and credit accounts and amounts */
218         final MoneyWiseAssetBase myDebit = (MoneyWiseAssetBase) theTransaction.getDebitAccount();
219         final MoneyWiseAssetBase myCredit = (MoneyWiseAssetBase) theTransaction.getCreditAccount();
220 
221         /* If the debit account is an asset */
222         OceanusMoney myDebitAmount = theTransaction.getDebitAmount();
223         if (isAsset(myDebit)) {
224             /* Process the debit asset bucket */
225             myDebitAmount = processDebitAsset(myDebit);
226         }
227 
228         /* If the credit account is an asset */
229         OceanusMoney myCreditAmount = theTransaction.getCreditAmount();
230         if (isAsset(myCredit)) {
231             /* Process the credit asset bucket */
232             myCreditAmount = processCreditAsset(myCredit);
233 
234             /* Reload the debitAmount which changes if debit is payee and credit is foreign */
235             myDebitAmount = theTransaction.getDebitAmount();
236         }
237 
238         /* Adjust for currencyFluctuation */
239         final OceanusMoney myFluctuation = new OceanusMoney(myDebitAmount);
240         myFluctuation.addAmount(myCreditAmount);
241         if (myFluctuation.isNonZero()) {
242             theMarket.adjustTotalsForCurrencyFluctuation(theTransaction.getEvent(), myFluctuation);
243         }
244     }

246     /**
247      * process debit Asset.
248      * @param pDebit the debit asset
249      * @return the debitAmount in reporting currency
250      */
251     OceanusMoney processDebitAsset(final MoneyWiseAssetBase pDebit) {
252         /* Adjust the debit asset bucket */
253         final MoneyWiseXAnalysisAccountBucket<?> myBucket = getAccountBucket(pDebit);
254         OceanusMoney myDebitAmount = theTransaction.getDebitAmount();
255         myBucket.addToBalance(myDebitAmount);
256         myBucket.adjustValuation();
257         theState.registerBucketInterest(myBucket);
258 
259         /* If the asset is foreign, convert debit amount to reporting currency */
260         if (pDebit.isForeign()) {
261             /* convert debit amount to reporting currency */
262             myDebitAmount = myBucket.getDeltaValuation();
263             theTransaction.setDebitAmount(myDebitAmount);
264         }
265 
266         /* Return the debit amount */
267         return myDebitAmount;
268     }

295     /**
296      * process autoExpense assets.
297      */
298     private void processAutoExpense() {
299         /* Access debit and credit accounts and amounts */
300         final MoneyWiseAssetBase myDebit = (MoneyWiseAssetBase) theTransaction.getDebitAccount();
301         final MoneyWiseAssetBase myCredit = (MoneyWiseAssetBase) theTransaction.getCreditAccount();
302 
303         /* If the debit account is auto-Expense */
304         if (isAutoExpense(myDebit)) {
305             /* Access debit as cashPayee */
306             processDebitAutoExpense((MoneyWiseCash) myDebit);
307         }
308 
309         /* If the credit account is auto-Expense */
310         if (isAutoExpense(myCredit)) {
311             /* Access credit as cashPayee */
312             processCreditAutoExpense((MoneyWiseCash) myCredit);
313         }
314     }

316     /**
317      * process debit autoExpense asset.
318      * @param pDebit the debit asset
319      */
320     private void processDebitAutoExpense(final MoneyWiseCash pDebit) {
321         /* Access debit Payee/Category auto-expense */
322         final MoneyWiseTransCategory myAuto = pDebit.getAutoExpense();
323         final MoneyWisePayee myDebit = pDebit.getAutoPayee();
324 
325         /* If the debit is foreign and the credit is a payee */
326         if (pDebit.isForeign()) {
327             final MoneyWiseTransAsset myCredit = theTransaction.getCreditAccount();
328             if (myCredit instanceof MoneyWisePayee) {
329                 /* Convert the debit amount to reporting currency */
330                 final MoneyWiseXAnalysisCursor myCursor = theAnalysis.getCursor();
331                 final OceanusRatio myRate = myCursor.getCurrentXchgRate(pDebit.getAssetCurrency());
332                 final OceanusMoney myAmount = theTransaction.getDebitAmount();
333                 theTransaction.setDebitAmount(myAmount.convertCurrency(theAnalysis.getCurrency().getCurrency(), myRate));
334             }
335         }
336 
337         /* Adjust expense for autoPayee bucket */
338         final OceanusMoney myAmount = theTransaction.getDebitAmount();
339         final MoneyWiseXAnalysisPayeeBucket myPayee = theAnalysis.getPayees().getBucket(myDebit);
340         myPayee.addExpense(myAmount);
341         theState.registerBucketInterest(myPayee);
342 
343         /* Adjust expense for Category bucket */
344         final MoneyWiseXAnalysisTransCategoryBucket myCategory = theAnalysis.getTransCategories().getBucket(myAuto);
345         myCategory.addExpense(myAmount);
346         theState.registerBucketInterest(myCategory);
347 
348         /* Adjust expense taxBasis */
349         theTax.processAutoExpense(myAmount);
350     }

389     /**
390      * process payee assets.
391      */
392     private void processPayees() {
393         /* Access debit and credit accounts and amounts */
394         final MoneyWiseAssetBase myDebit = (MoneyWiseAssetBase) theTransaction.getDebitAccount();
395         final MoneyWiseAssetBase myCredit = (MoneyWiseAssetBase) theTransaction.getCreditAccount();
396 
397         /* If the debit account is payee */
398         if (isPayee(myDebit)) {
399             /* process debit as Payee */
400             processDebitPayee((MoneyWisePayee) myDebit);
401         }
402 
403         /* If the credit account is payee */
404         if (isPayee(myCredit)) {
405             /* process credit as Payee */
406             processCreditPayee((MoneyWisePayee) myCredit);
407         }
408     }

410     /**
411      * process debit payee asset.
412      * @param pDebit the debit asset
413      */
414     void processDebitPayee(final MoneyWisePayee pDebit) {
415         /* Determine income/expense */
416         final boolean isExpense = theTransaction.isExpenseCategory();
417 
418         /* Adjust expense for Payee bucket */
419         final OceanusMoney myAmount = theTransaction.getDebitAmount();
420         final MoneyWiseXAnalysisPayeeBucket myPayeeBucket = theAnalysis.getPayees().getBucket(pDebit);
421         if (isExpense) {
422             myPayeeBucket.addExpense(myAmount);
423         } else {
424             myPayeeBucket.subtractIncome(myAmount);
425         }
426         theState.registerBucketInterest(myPayeeBucket);
427 
428         /* Record payee bucket */
429         theTax.recordPayeeBucket(myPayeeBucket);
430     }

454     /**
455      * Adjust category buckets.
456      */
457     void adjustCategoryBucket() {
458         /* Access the credit amount and category */
459         final OceanusMoney myAmount = theTransaction.isTo()
460                 ? theTransaction.getCreditAmount()
461                 : theTransaction.getDebitAmount();
462         final MoneyWiseTransCategory myCategory = theTransaction.getCategory();
463 
464         /* Ignore transfers */
465         if (myCategory.isTransfer()) {
466             return;
467         }
468 
469         /* Adjust income/expense for Category bucket */
470         final MoneyWiseXAnalysisTransCategoryBucket myCatBucket = theAnalysis.getTransCategories().getBucket(myCategory);
471         if (theTransaction.isExpenseCategory()) {
472             myCatBucket.addExpense(myAmount);
473         } else if (theTransaction.isIncomeCategory()) {
474             myCatBucket.subtractIncome(myAmount);
475         }
476         theState.registerBucketInterest(myCatBucket);
477     }

479     /**
480      * adjustForeignDebit.
481      * @param pExchangeRate the exchangeRate
482      * @return the adjusted debitAmount
483      */
484     OceanusMoney adjustForeignAssetDebit(final OceanusRatio pExchangeRate) {
485         /* Calculate the value in the local currency */
486         OceanusMoney myAmount = theTransaction.getDebitAmount();
487         myAmount = myAmount.convertCurrency(theCurrency.getCurrency(), pExchangeRate);
488         theTransaction.setDebitAmount(myAmount);
489 
490         /* Adjust for currencyFluctuation */
491         final OceanusMoney myCreditAmount = theTransaction.getCreditAmount();
492         final OceanusMoney myFluctuation = new OceanusMoney(myCreditAmount);
493         myFluctuation.addAmount(myAmount);
494         if (myFluctuation.isNonZero()) {
495             theMarket.adjustTotalsForCurrencyFluctuation(theTransaction.getEvent(), myFluctuation);
496         }
497 
498         /* return the new debit amount */
499         return myAmount;
500     }

525     /**
526      * Obtain Account bucket for asset.
527      * @param pAsset the asset
528      * @return the bucket
529      */
530     MoneyWiseXAnalysisAccountBucket<?> getAccountBucket(final MoneyWiseAssetBase pAsset) {
531         switch (pAsset.getAssetType()) {
532             case DEPOSIT:
533                 return theAnalysis.getDeposits().getBucket((MoneyWiseDeposit) pAsset);
534             case CASH:
535                 return theAnalysis.getCash().getBucket((MoneyWiseCash) pAsset);
536             case LOAN:
537                 return theAnalysis.getLoans().getBucket((MoneyWiseLoan) pAsset);
538             case PORTFOLIO:
539                 return theAnalysis.getPortfolios().getCashBucket((MoneyWisePortfolio) pAsset);
540             default:
541                 throw new IllegalArgumentException();
542         }
543     }

545     /**
546      * is the account an asset?
547      * @param pAccount the account
548      * @return true/false
549      */
550     boolean isAsset(final MoneyWiseAssetBase pAccount) {
551         switch (pAccount.getAssetType()) {
552             case DEPOSIT:
553             case LOAN:
554             case CASH:
555             case PORTFOLIO:
556                 return true;
557             case SECURITY:
558             case PAYEE:
559             case SECURITYHOLDING:
560             case AUTOEXPENSE:
561             default:
562                 return false;
563         }
564     }
