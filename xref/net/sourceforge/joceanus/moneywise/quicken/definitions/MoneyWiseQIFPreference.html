17  package net.sourceforge.joceanus.moneywise.quicken.definitions;
18  
19  import net.sourceforge.joceanus.metis.preference.MetisPreferenceKey;
20  import net.sourceforge.joceanus.metis.preference.MetisPreferenceManager;
21  import net.sourceforge.joceanus.metis.preference.MetisPreferenceSet;
22  import net.sourceforge.joceanus.oceanus.base.OceanusException;
23  import net.sourceforge.joceanus.oceanus.date.OceanusDate;
24  
25  /**
26   * Quicken Preferences.
27   */
28  public final class MoneyWiseQIFPreference {
29      /**
30       * Constructor.
31       */
32      private MoneyWiseQIFPreference() {
33      }
34  
35      /**
36       * QIFPreferenceKeys.
37       */
38      public enum MoneyWiseQIFPreferenceKey implements MetisPreferenceKey {
39          /**
40           * QIF directory.
41           */
42          QIFDIR("QIFDir", "Output Directory"),
43  
44          /**
45           * QIF Type.
46           */
47          QIFTYPE("QIFType", "Output Type"),
48  
49          /**
50           * Database Driver.
51           */
52          LASTEVENT("LastEvent", "Last Event");
53  
54          /**
55           * The name of the Preference.
56           */
57          private final String theName;
58  
59          /**
60           * The display string.
61           */
62          private final String theDisplay;
63  
64          /**
65           * Constructor.
66           * @param pName the name
67           * @param pDisplay the display string;
68           */
69          MoneyWiseQIFPreferenceKey(final String pName,
70                                    final String pDisplay) {
71              theName = pName;
72              theDisplay = pDisplay;
73          }
74  
75          @Override
76          public String getName() {
77              return theName;
78          }
79  
80          @Override
81          public String getDisplay() {
82              return theDisplay;
83          }
84      }
85  
86      /**
87       * QIF Preferences.
88       */
89      public static class MoneyWiseQIFPreferences
90              extends MetisPreferenceSet {
91          /**
92           * Constructor.
93           * @param pManager the preference manager
94           * @throws OceanusException on error
95           */
96          public MoneyWiseQIFPreferences(final MetisPreferenceManager pManager) throws OceanusException {
97              super(pManager, "QIF Preferences");
98          }
99  
100         @Override
101         protected void definePreferences() throws OceanusException {
102             defineDirectoryPreference(MoneyWiseQIFPreferenceKey.QIFDIR);
103             defineEnumPreference(MoneyWiseQIFPreferenceKey.QIFTYPE, MoneyWiseQIFType.class);
104             defineDatePreference(MoneyWiseQIFPreferenceKey.LASTEVENT);
105         }
106 
107         @Override
108         public void autoCorrectPreferences() {
109             /* Make sure that the directory is specified */
110             final MetisStringPreference myDirPref = getStringPreference(MoneyWiseQIFPreferenceKey.QIFDIR);
111             if (!myDirPref.isAvailable()) {
112                 myDirPref.setValue(System.getProperty("user.home"));
113             }
114 
115             /* Make sure that the QIFType is specified */
116             final MetisEnumPreference<MoneyWiseQIFType> myTypePref = getEnumPreference(MoneyWiseQIFPreferenceKey.QIFTYPE, MoneyWiseQIFType.class);
117             if (!myTypePref.isAvailable()) {
118                 myTypePref.setValue(MoneyWiseQIFType.ACEMONEY);
119             }
120 
121             /* Make sure that the eventDate is specified */
122             final MetisDatePreference myPref = getDatePreference(MoneyWiseQIFPreferenceKey.LASTEVENT);
123             if (!myPref.isAvailable()) {
124                 myPref.setValue(new OceanusDate());
125             }
126         }
127     }
128 }
