38  /**
39   * CashFlow report builder.
40   */
41  public class MoneyWiseReportCashFlow
42          extends MetisReportBase<MoneyWiseAnalysis, MoneyWiseAnalysisFilter<?, ?>> {
43      /**
44       * The Title text.
45       */
46      private static final String TEXT_TITLE = MoneyWiseReportResource.CASHFLOW_TITLE.getValue();
47  
48      /**
49       * HTML builder.
50       */
51      private final MetisReportHTMLBuilder theBuilder;
52  
53      /**
54       * The Formatter.
55       */
56      private final OceanusDataFormatter theFormatter;
57  
58      /**
59       * Constructor.
60       * @param pManager the Report Manager
61       */
62      protected MoneyWiseReportCashFlow(final MetisReportManager<MoneyWiseAnalysisFilter<?, ?>> pManager) {
63          /* Access underlying utilities */
64          theBuilder = pManager.getBuilder();
65          theFormatter = theBuilder.getDataFormatter();
66      }
67  
68      @Override
69      public Document createReport(final MoneyWiseAnalysis pAnalysis) {
70          /* Access the bucket lists */
71          final MoneyWiseAnalysisPayeeBucketList myPayees = pAnalysis.getPayees();
72          final OceanusDateRange myRange = pAnalysis.getDateRange();
73  
74          /* Obtain the totals bucket */
75          final MoneyWiseAnalysisPayeeBucket myTotals = myPayees.getTotals();
76  
77          /* Start the report */
78          final Element myBody = theBuilder.startReport();
79          theBuilder.makeTitle(myBody, TEXT_TITLE, theFormatter.formatObject(myRange));
80  
81          /* Initialise the table */
82          final MetisHTMLTable myTable = theBuilder.startTable(myBody);
83          theBuilder.startHdrRow(myTable);
84          theBuilder.makeTotalCell(myTable);
85          theBuilder.makeTitleCell(myTable, MoneyWiseReportBuilder.TEXT_INCOME);
86          theBuilder.makeTitleCell(myTable, MoneyWiseReportBuilder.TEXT_EXPENSE);
87          theBuilder.makeTitleCell(myTable, MoneyWiseReportBuilder.TEXT_PROFIT);
88  
89          /* Loop through the Payee Buckets */
90          final Iterator<MoneyWiseAnalysisPayeeBucket> myIterator = myPayees.iterator();
91          while (myIterator.hasNext()) {
92              final MoneyWiseAnalysisPayeeBucket myBucket = myIterator.next();
93  
94              /* Access bucket name */
95              final String myName = myBucket.getName();
96  
97              /* Access values */
98              final MoneyWiseAnalysisPayeeValues myValues = myBucket.getValues();
99  
100             /* Format the detail */
101             theBuilder.startRow(myTable);
102             theBuilder.makeFilterLinkCell(myTable, myName);
103             theBuilder.makeValueCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisPayeeAttr.INCOME));
104             theBuilder.makeValueCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisPayeeAttr.EXPENSE));
105             theBuilder.makeValueCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisPayeeAttr.PROFIT));
106 
107             /* Record the filter */
108             setFilterForId(myName, myBucket);
109         }
110 
111         /* Access values */
112         final MoneyWiseAnalysisPayeeValues myValues = myTotals.getValues();
113 
114         /* Format the total */
115         theBuilder.startTotalRow(myTable);
116         theBuilder.makeTitleCell(myTable, MoneyWiseReportBuilder.TEXT_TOTAL);
117         theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisPayeeAttr.INCOME));
118         theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisPayeeAttr.EXPENSE));
119         theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisPayeeAttr.PROFIT));
120 
121         /* Return the document */
122         return theBuilder.getDocument();
123     }
124 
125     @Override
126     public MoneyWiseAnalysisPayeeFilter processFilter(final Object pSource) {
127         /* If this is an PayeeBucket */
128         if (pSource instanceof MoneyWiseAnalysisPayeeBucket mySource) {
129             /* Create the new filter */
130             return new MoneyWiseAnalysisPayeeFilter(mySource);
131         }
132         return null;
133     }
134 
135     @Override
136     public MetisHTMLTable createDelayedTable(final DelayedTable pTable) {
137         return null;
138     }
139 }
