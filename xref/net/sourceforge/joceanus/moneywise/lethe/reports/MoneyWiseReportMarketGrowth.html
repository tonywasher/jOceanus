124     @Override
125     public Document createReport(final MoneyWiseAnalysis pAnalysis) {
126         /* Access the bucket lists */
127         final MoneyWiseAnalysisPortfolioBucketList myPortfolios = pAnalysis.getPortfolios();
128         hasForeign = myPortfolios.haveForeignCurrency();
129 
130         /* Access the totals */
131         final MoneyWiseAnalysisPortfolioBucket myTotals = myPortfolios.getTotals();
132         final OceanusDateRange myRange = pAnalysis.getDateRange();
133 
134         /* Start the report */
135         final Element myBody = theBuilder.startReport();
136         theBuilder.makeTitle(myBody, TEXT_TITLE, theFormatter.formatObject(myRange));
137 
138         /* Initialise the table */
139         final MetisHTMLTable myTable = theBuilder.startTable(myBody);
140         theBuilder.startHdrRow(myTable);
141         theBuilder.makeTitleCell(myTable);
142         theBuilder.makeTitleCell(myTable, TEXT_VALUE);
143         theBuilder.makeTitleCell(myTable, TEXT_BASE);
144         theBuilder.makeTitleCell(myTable, TEXT_INVEST);
145         theBuilder.makeTitleCell(myTable, TEXT_ADJUST);
146         theBuilder.makeTitleCell(myTable, TEXT_GAINS);
147         theBuilder.makeTitleCell(myTable, TEXT_GROWTH);
148         if (hasForeign) {
149             theBuilder.makeTitleCell(myTable, TEXT_CURRENCY);
150         }
151         theBuilder.makeTitleCell(myTable, TEXT_PROFIT);
152 
153         /* Loop through the Portfolio Buckets */
154         final Iterator<MoneyWiseAnalysisPortfolioBucket> myIterator = myPortfolios.iterator();
155         while (myIterator.hasNext()) {
156             final MoneyWiseAnalysisPortfolioBucket myBucket = myIterator.next();
157 
158             /* Access bucket name */
159             final String myName = myBucket.getName();
160 
161             /* Access values */
162             final MoneyWiseAnalysisSecurityValues myValues = myBucket.getValues();
163 
164             /* Only declare the entry if we have securities */
165             if (myBucket.securityIterator().hasNext()) {
166                 /* Format the Asset */
167                 theBuilder.startRow(myTable);
168                 theBuilder.makeDelayLinkCell(myTable, myName);
169                 theBuilder.makeTotalCell(myTable, myBucket.getNonCashValue(false));
170                 theBuilder.makeTotalCell(myTable, myBucket.getNonCashValue(true));
171                 theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.INVESTED));
172                 theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.GROWTHADJUST));
173                 theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.REALISEDGAINS));
174                 theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.MARKETGROWTH));
175                 if (hasForeign) {
176                     theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.CURRENCYFLUCT));
177                 }
178                 theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.MARKETPROFIT));
179                 checkPortfolioGrowth(myBucket);
180 
181                 /* Note the delayed subTable */
182                 setDelayedTable(myName, myTable, myBucket);
183             }
184         }
185 
186         /* Access values */
187         final MoneyWiseAnalysisSecurityValues myValues = myTotals.getValues();
188 
189         /* Create the total row */
190         theBuilder.startTotalRow(myTable);
191         theBuilder.makeTitleCell(myTable, MoneyWiseReportBuilder.TEXT_TOTAL);
192         theBuilder.makeTotalCell(myTable, myTotals.getNonCashValue(false));
193         theBuilder.makeTotalCell(myTable, myTotals.getNonCashValue(true));
194         theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.INVESTED));
195         theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.GROWTHADJUST));
196         theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.REALISEDGAINS));
197         theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.MARKETGROWTH));
198         if (hasForeign) {
199             theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.CURRENCYFLUCT));
200         }
201         theBuilder.makeTotalCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.MARKETPROFIT));
202         checkPortfolioGrowth(myTotals);
203 
204         /* Return the document */
205         return theBuilder.getDocument();
206     }

208     @Override
209     public MetisHTMLTable createDelayedTable(final DelayedTable pTable) {
210         /* Access the source */
211         final Object mySource = pTable.getSource();
212         if (mySource instanceof MoneyWiseAnalysisPortfolioBucket mySourceBucket) {
213             return createDelayedPortfolio(pTable.getParent(), mySourceBucket);
214         }
215 
216         /* Return the null table */
217         return null;
218     }

220     /**
221      * Create a delayed portfolio table.
222      * @param pParent the parent table
223      * @param pSource the source bucket
224      * @return the new document fragment
225      */
226     private MetisHTMLTable createDelayedPortfolio(final MetisHTMLTable pParent,
227                                                   final MoneyWiseAnalysisPortfolioBucket pSource) {
228         /* Access the securities */
229         final MoneyWiseAnalysisSecurityBucketList mySecurities = pSource.getSecurities();
230 
231         /* Create a new table */
232         final MetisHTMLTable myTable = theBuilder.createEmbeddedTable(pParent);
233 
234         /* Loop through the Security Buckets */
235         final Iterator<MoneyWiseAnalysisSecurityBucket> myIterator = mySecurities.iterator();
236         while (myIterator.hasNext()) {
237             final MoneyWiseAnalysisSecurityBucket myBucket = myIterator.next();
238 
239             /* Access bucket name */
240             final String myName = myBucket.getSecurityName();
241             String myFullName = myBucket.getDecoratedName();
242             myFullName = myFullName.replace(':', '-');
243 
244             /* Access values */
245             final MoneyWiseAnalysisSecurityValues myValues = myBucket.getValues();
246             final MoneyWiseAnalysisSecurityValues myBaseValues = myBucket.getBaseValues();
247 
248             /* Create the detail row */
249             theBuilder.startRow(myTable);
250             theBuilder.makeFilterLinkCell(myTable, myFullName, myName);
251             theBuilder.makeValueCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.VALUATION));
252             theBuilder.makeValueCell(myTable, myBaseValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.VALUATION));
253             theBuilder.makeValueCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.INVESTED));
254             theBuilder.makeValueCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.GROWTHADJUST));
255             theBuilder.makeValueCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.REALISEDGAINS));
256             theBuilder.makeValueCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.MARKETGROWTH));
257             if (hasForeign) {
258                 theBuilder.makeValueCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.CURRENCYFLUCT));
259             }
260             theBuilder.makeValueCell(myTable, myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.MARKETPROFIT));
261             checkSecurityGrowth(myBucket);
262 
263             /* Record the filter */
264             setFilterForId(myFullName, myBucket);
265         }
266 
267         /* Return the table */
268         return myTable;
269     }

271     /**
272      * Check portfolio growth calculation.
273      * @param pBucket the portfolio bucket
274      */
275     private static void checkPortfolioGrowth(final MoneyWiseAnalysisPortfolioBucket pBucket) {
276         /* Check market profit */
277         final MoneyWiseAnalysisSecurityValues myValues = pBucket.getValues();
278         final OceanusMoney myAdjust = myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.GROWTHADJUST);
279         final OceanusMoney myCalcGrowth = pBucket.getNonCashValue(false);
280         myCalcGrowth.subtractAmount(pBucket.getNonCashValue(true));
281         myCalcGrowth.subtractAmount(myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.INVESTED));
282         myCalcGrowth.addAmount(myAdjust);
283         final OceanusMoney myProfit = myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.MARKETPROFIT);
284         if (!myProfit.equals(myCalcGrowth)) {
285             LOGGER.error("Incorrect profit calculation for security <%s> of <%s>", pBucket.getName(), myCalcGrowth);
286         }
287 
288         /* Check market growth */
289         myCalcGrowth.subtractAmount(myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.REALISEDGAINS));
290         myCalcGrowth.subtractAmount(myAdjust);
291         myCalcGrowth.subtractAmount(myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.CURRENCYFLUCT));
292         final OceanusMoney myGrowth = myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.MARKETGROWTH);
293         if (!myGrowth.equals(myCalcGrowth)) {
294             LOGGER.error("Incorrect growth calculation for portfolio <%s> of <%s>", pBucket.getName(), myCalcGrowth);
295         }
296     }

298     /**
299      * Check security portfolio profit calculation.
300      * @param pBucket the security bucket
301      */
302     private static void checkSecurityGrowth(final MoneyWiseAnalysisSecurityBucket pBucket) {
303         /* Check market profit */
304         final MoneyWiseAnalysisSecurityValues myValues = pBucket.getValues();
305         final MoneyWiseAnalysisSecurityValues myBaseValues = pBucket.getBaseValues();
306         final OceanusMoney myAdjust = myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.GROWTHADJUST);
307         final OceanusMoney myCalcGrowth = new OceanusMoney(myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.VALUATION));
308         myCalcGrowth.subtractAmount(myBaseValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.VALUATION));
309         myCalcGrowth.subtractAmount(myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.INVESTED));
310         myCalcGrowth.addAmount(myAdjust);
311         final OceanusMoney myProfit = myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.MARKETPROFIT);
312         if (!myProfit.equals(myCalcGrowth)) {
313             LOGGER.error("Incorrect profit calculation for security <%s> of <%s>", pBucket.getDecoratedName(), myCalcGrowth);
314         }
315 
316         /* Check market growth */
317         myCalcGrowth.subtractAmount(myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.REALISEDGAINS));
318         myCalcGrowth.subtractAmount(myAdjust);
319         final OceanusMoney myFluct = myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.CURRENCYFLUCT);
320         if (myFluct != null) {
321             myCalcGrowth.subtractAmount(myFluct);
322         }
323         final OceanusMoney myGrowth = myValues.getMoneyValue(MoneyWiseAnalysisSecurityAttr.MARKETGROWTH);
324         if (!myGrowth.equals(myCalcGrowth)) {
325             LOGGER.error("Incorrect growth calculation for security <%s> of <%s>", pBucket.getDecoratedName(), myCalcGrowth);
326         }
327     }
