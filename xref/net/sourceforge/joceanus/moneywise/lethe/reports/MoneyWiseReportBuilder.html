19  import java.util.EnumMap;
20  import java.util.Map;
21  
22  import org.w3c.dom.Document;
23  
24  import net.sourceforge.joceanus.metis.report.MetisReportBase;
25  import net.sourceforge.joceanus.metis.report.MetisReportManager;
26  import net.sourceforge.joceanus.moneywise.lethe.data.analysis.data.MoneyWiseAnalysis;
27  import net.sourceforge.joceanus.moneywise.lethe.data.analysis.data.MoneyWiseAnalysisDataResource;
28  import net.sourceforge.joceanus.moneywise.lethe.data.analysis.data.MoneyWiseAnalysisSecurityBucket;
29  import net.sourceforge.joceanus.moneywise.lethe.data.analysis.values.MoneyWiseAnalysisValuesResource;
30  import net.sourceforge.joceanus.moneywise.lethe.views.MoneyWiseAnalysisFilter;
31  
32  /**
33   * Report Classes.
34   * @author Tony Washer
35   */
36  public class MoneyWiseReportBuilder {
37      /**
38       * The Total text.
39       */
40      protected static final String TEXT_TOTAL = MoneyWiseAnalysisDataResource.ANALYSIS_TOTALS.getValue();
41  
42      /**
43       * The Profit text.
44       */
45      protected static final String TEXT_PROFIT = MoneyWiseAnalysisValuesResource.SECURITYATTR_PROFIT.getValue();
46  
47      /**
48       * The Income text.
49       */
50      protected static final String TEXT_INCOME = MoneyWiseAnalysisValuesResource.PAYEEATTR_INCOME.getValue();
51  
52      /**
53       * The Expense text.
54       */
55      protected static final String TEXT_EXPENSE = MoneyWiseAnalysisValuesResource.PAYEEATTR_EXPENSE.getValue();
56  
57      /**
58       * The Report Manager.
59       */
60      private final MetisReportManager<MoneyWiseAnalysisFilter<?, ?>> theManager;
61  
62      /**
63       * Map of allocated reports.
64       */
65      private final Map<MoneyWiseReportType, MetisReportBase<MoneyWiseAnalysis, MoneyWiseAnalysisFilter<?, ?>>> theReportMap;
66  
67      /**
68       * Constructor.
69       * @param pManager the report manager
70       */
71      public MoneyWiseReportBuilder(final MetisReportManager<MoneyWiseAnalysisFilter<?, ?>> pManager) {
72          /* Record the details */
73          theManager = pManager;
74  
75          /* Allocate map */
76          theReportMap = new EnumMap<>(MoneyWiseReportType.class);
77      }
78  
79      /**
80       * Build a report of the appropriate type.
81       * @param pAnalysis the analysis
82       * @param pType the report type
83       * @param pSecurity the security
84       * @return the Web document
85       */
86      public Document createReport(final MoneyWiseAnalysis pAnalysis,
87                                   final MoneyWiseReportType pType,
88                                   final MoneyWiseAnalysisSecurityBucket pSecurity) {
89          /* Access existing report */
90          MetisReportBase<MoneyWiseAnalysis, MoneyWiseAnalysisFilter<?, ?>> myReport = theReportMap.get(pType);
91  
92          /* If we have not previously allocated this report */
93          if (myReport == null) {
94              /* Switch on the report type */
95              switch (pType) {
96                  case NETWORTH:
97                      myReport = new MoneyWiseReportNetWorth(theManager);
98                      break;
99                  case BALANCESHEET:
100                     myReport = new MoneyWiseReportBalanceSheet(theManager);
101                     break;
102                 case CASHFLOW:
103                     myReport = new MoneyWiseReportCashFlow(theManager);
104                     break;
105                 case INCOMEEXPENSE:
106                     myReport = new MoneyWiseReportIncomeExpense(theManager);
107                     break;
108                 case PORTFOLIO:
109                     myReport = new MoneyWiseReportPortfolioView(theManager);
110                     break;
111                 case MARKETGROWTH:
112                     myReport = new MoneyWiseReportMarketGrowth(theManager);
113                     break;
114                 case TAXBASIS:
115                     myReport = new MoneyWiseReportTaxationBasis(theManager);
116                     break;
117                 case TAXCALC:
118                     myReport = new MoneyWiseReportTaxCalculation(theManager);
119                     break;
120                 case ASSETGAINS:
121                     myReport = new MoneyWiseReportAssetGains(theManager);
122                     break;
123                 case CAPITALGAINS:
124                     myReport = new MoneyWiseReportCapitalGains(theManager);
125                     break;
126                 default:
127                     return null;
128             }
129 
130             /* Store allocated report */
131             theReportMap.put(pType, myReport);
132         }
133 
134         /* If the report requires the security */
135         if (myReport instanceof MoneyWiseReportCapitalGains) {
136             ((MoneyWiseReportCapitalGains) myReport).setSecurity(pSecurity);
137         }
138 
139         /* Set up the report */
140         theManager.setReport(myReport);
141 
142         /* Create the report */
143         return myReport.createReport(pAnalysis);
144     }
145 }
