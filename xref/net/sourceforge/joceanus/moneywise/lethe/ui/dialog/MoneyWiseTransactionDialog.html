174     /**
175      * Build Main subPanel.
176      * @param pFactory the GUI factory
177      */
178     private void buildMainPanel(final TethysUIFactory<?> pFactory) {
179         /* Allocate fields */
180         final TethysUIFieldFactory myFields = pFactory.fieldFactory();
181         final TethysUIMoneyEditField myAmount = myFields.newMoneyField();
182 
183         /* Create the buttons */
184         final TethysUIDateButtonField myDateButton = myFields.newDateField();
185         final TethysUIScrollButtonField<MoneyWiseTransAsset> myAccountButton = myFields.newScrollField(MoneyWiseTransAsset.class);
186         final TethysUIScrollButtonField<MoneyWiseTransAsset> myPartnerButton = myFields.newScrollField(MoneyWiseTransAsset.class);
187         final TethysUIScrollButtonField<MoneyWiseTransCategory> myCategoryButton = myFields.newScrollField(MoneyWiseTransCategory.class);
188         final TethysUIIconButtonField<Boolean> myReconciledButton = myFields.newIconField(Boolean.class);
189         final TethysUIIconButtonField<MoneyWiseAssetDirection> myDirectionButton = myFields.newIconField(MoneyWiseAssetDirection.class);
190 
191         /* Assign the fields to the panel */
192         theFieldSet.addField(MoneyWiseBasicResource.MONEYWISEDATA_FIELD_DATE, myDateButton, MoneyWiseTransaction::getDate);
193         theFieldSet.addField(MoneyWiseBasicResource.TRANSACTION_ACCOUNT, myAccountButton, MoneyWiseTransaction::getAccount);
194         theFieldSet.addField(MoneyWiseBasicDataType.TRANSCATEGORY, myCategoryButton, MoneyWiseTransaction::getCategory);
195         theFieldSet.addField(MoneyWiseBasicResource.TRANSACTION_DIRECTION, myDirectionButton, MoneyWiseTransaction::getDirection);
196         theFieldSet.addField(MoneyWiseBasicResource.TRANSACTION_PARTNER, myPartnerButton, MoneyWiseTransaction::getPartner);
197         theFieldSet.addField(MoneyWiseBasicResource.TRANSACTION_AMOUNT, myAmount, MoneyWiseTransaction::getAmount);
198         theFieldSet.addField(MoneyWiseBasicResource.TRANSACTION_RECONCILED, myReconciledButton, MoneyWiseTransaction::isReconciled);
199 
200         /* Configure the menuBuilders */
201         myDateButton.setDateConfigurator(this::handleDateConfig);
202         myAccountButton.setMenuConfigurator(c -> buildAccountMenu(c, getItem()));
203         myCategoryButton.setMenuConfigurator(c -> buildCategoryMenu(c, getItem()));
204         myPartnerButton.setMenuConfigurator(c -> buildPartnerMenu(c, getItem()));
205         final Map<Boolean, TethysUIIconMapSet<Boolean>> myRecMapSets = MoneyWiseIcon.configureReconciledIconButton(pFactory);
206         myReconciledButton.setIconMapSet(() -> myRecMapSets.get(theReconciledState));
207         final Map<Boolean, TethysUIIconMapSet<MoneyWiseAssetDirection>> myDirMapSets = MoneyWiseIcon.configureDirectionIconButton(pFactory);
208         myDirectionButton.setIconMapSet(() -> myDirMapSets.get(theDirectionState));
209         myAmount.setDeemedCurrency(() -> getItem().getAccount().getCurrency());
210     }

331     @Override
332     public void refreshData() {
333         /* If we have an item */
334         final MoneyWiseTransaction myItem = getItem();
335         if (myItem != null) {
336             final MoneyWiseTransactionList myTrans = getDataList(MoneyWiseBasicDataType.TRANSACTION, MoneyWiseTransactionList.class);
337             setItem(myTrans.findItemById(myItem.getIndexedId()));
338         }
339 
340         /* Make sure that the item is not editable */
341         setEditable(false);
342     }

514

735                                                                       final MoneyWiseAssetBaseList<T> pList,
736                                                                       final boolean pIsAccount,
737                                                                       final MoneyWiseTransaction pTrans) {
738         /* Record active item */
739         final MoneyWiseTransAsset myAccount = pTrans.getAccount();
740         final MoneyWiseTransCategory myCategory = pTrans.getCategory();
741         final MoneyWiseDataValidatorTrans<?> myValidator = pTrans.getList().getValidator();
742         final MoneyWiseTransAsset myCurr = pIsAccount
743                 ? myAccount
744                 : pTrans.getPartner();
745         TethysUIScrollItem<MoneyWiseTransAsset> myActive = null;
746         TethysUIScrollSubMenu<MoneyWiseTransAsset> myMenu = null;

752

755 
756             /* Check whether the asset is allowable for the owner */
757             bIgnore |= !(pIsAccount
758                     ? myValidator.isValidAccount(myAsset)
759                     : myValidator.isValidPartner(myAccount, myCategory, myAsset));
760             if (bIgnore) {
761                 continue;
762             }
763 
764             /* If this the first item */
765             if (myMenu == null) {
766                 /* Create a new subMenu and add it to the popUp */
767                 myMenu = pMenu.addSubMenu(pList.getItemType().getItemName());
768             }
769 
770             /* Create a new MenuItem and add it to the popUp */
771             final TethysUIScrollItem<MoneyWiseTransAsset> myItem = myMenu.getSubMenu().addItem(myAsset);
772 
773             /* If this is the active category */
774             if (myAsset.equals(myCurr)) {
775                 /* Record it */
776                 myActive = myItem;
777             }
778         }
779 
780         /* Ensure active item is visible */
781         if (myActive != null) {
782             myActive.scrollToItem();
783         }
