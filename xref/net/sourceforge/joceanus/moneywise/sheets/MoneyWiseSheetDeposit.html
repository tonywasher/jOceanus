17  package net.sourceforge.joceanus.moneywise.sheets;
18  
19  import net.sourceforge.joceanus.moneywise.data.basic.MoneyWiseBasicResource;
20  import net.sourceforge.joceanus.moneywise.data.basic.MoneyWiseDataSet;
21  import net.sourceforge.joceanus.moneywise.data.basic.MoneyWiseDeposit;
22  import net.sourceforge.joceanus.moneywise.data.statics.MoneyWiseStaticDataType;
23  import net.sourceforge.joceanus.oceanus.base.OceanusException;
24  import net.sourceforge.joceanus.prometheus.data.PrometheusDataResource;
25  import net.sourceforge.joceanus.prometheus.data.PrometheusDataValues;
26  import net.sourceforge.joceanus.prometheus.sheets.PrometheusSheetEncrypted;
27  
28  /**
29   * SheetDataItem extension for Deposit.
30   * @author Tony Washer
31   */
32  public final class MoneyWiseSheetDeposit
33          extends PrometheusSheetEncrypted<MoneyWiseDeposit> {
34      /**
35       * NamedArea for Deposits.
36       */
37      private static final String AREA_DEPOSITS = MoneyWiseDeposit.LIST_NAME;
38  
39      /**
40       * Name column.
41       */
42      private static final int COL_NAME = COL_KEYSETID + 1;
43  
44      /**
45       * Category column.
46       */
47      private static final int COL_CATEGORY = COL_NAME + 1;
48  
49      /**
50       * Parent column.
51       */
52      private static final int COL_PARENT = COL_CATEGORY + 1;
53  
54      /**
55       * Description column.
56       */
57      private static final int COL_DESC = COL_PARENT + 1;
58  
59      /**
60       * Currency column.
61       */
62      private static final int COL_CURRENCY = COL_DESC + 1;
63  
64      /**
65       * Closed column.
66       */
67      private static final int COL_CLOSED = COL_CURRENCY + 1;
68  
69      /**
70       * Constructor for loading a spreadsheet.
71       * @param pReader the spreadsheet reader
72       */
73      MoneyWiseSheetDeposit(final MoneyWiseReader pReader) {
74          /* Call super constructor */
75          super(pReader, AREA_DEPOSITS);
76  
77          /* Access the Deposits list */
78          final MoneyWiseDataSet myData = (MoneyWiseDataSet) pReader.getData();
79          setDataList(myData.getDeposits());
80      }
81  
82      /**
83       * Constructor for creating a spreadsheet.
84       * @param pWriter the spreadsheet writer
85       */
86      MoneyWiseSheetDeposit(final MoneyWiseWriter pWriter) {
87          /* Call super constructor */
88          super(pWriter, AREA_DEPOSITS);
89  
90          /* Access the Deposits list */
91          final MoneyWiseDataSet myData = (MoneyWiseDataSet) pWriter.getData();
92          setDataList(myData.getDeposits());
93      }
94  
95      @Override
96      protected PrometheusDataValues loadSecureValues() throws OceanusException {
97          /* Build data values */
98          final PrometheusDataValues myValues = getRowValues(MoneyWiseDeposit.OBJECT_NAME);
99          myValues.addValue(MoneyWiseBasicResource.CATEGORY_NAME, loadInteger(COL_CATEGORY));
100         myValues.addValue(MoneyWiseBasicResource.ASSET_PARENT, loadInteger(COL_PARENT));
101         myValues.addValue(MoneyWiseStaticDataType.CURRENCY, loadInteger(COL_CURRENCY));
102         myValues.addValue(PrometheusDataResource.DATAITEM_FIELD_NAME, loadBytes(COL_NAME));
103         myValues.addValue(PrometheusDataResource.DATAITEM_FIELD_DESC, loadBytes(COL_DESC));
104         myValues.addValue(MoneyWiseBasicResource.ASSET_CLOSED, loadBoolean(COL_CLOSED));
105 
106         /* Return the values */
107         return myValues;
108     }
109 
110     @Override
111     protected void insertSecureItem(final MoneyWiseDeposit pItem) throws OceanusException {
112         /* Set the fields */
113         super.insertSecureItem(pItem);
114         writeInteger(COL_CATEGORY, pItem.getCategoryId());
115         writeInteger(COL_PARENT, pItem.getParentId());
116         writeInteger(COL_CURRENCY, pItem.getAssetCurrencyId());
117         writeBytes(COL_NAME, pItem.getNameBytes());
118         writeBytes(COL_DESC, pItem.getDescBytes());
119         writeBoolean(COL_CLOSED, pItem.isClosed());
120     }
121 
122     @Override
123     protected int getLastColumn() {
124         /* Return the last column */
125         return COL_CLOSED;
126     }
127 }
