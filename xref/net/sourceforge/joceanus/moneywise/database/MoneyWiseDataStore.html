17  package net.sourceforge.joceanus.moneywise.database;
18  
19  import net.sourceforge.joceanus.moneywise.data.basic.MoneyWiseBasicDataType;
20  import net.sourceforge.joceanus.moneywise.data.statics.MoneyWiseStaticDataType;
21  import net.sourceforge.joceanus.prometheus.database.PrometheusDBConfig;
22  import net.sourceforge.joceanus.prometheus.database.PrometheusDataStore;
23  import net.sourceforge.joceanus.prometheus.database.PrometheusTableDataItem;
24  import net.sourceforge.joceanus.oceanus.base.OceanusException;
25  
26  /**
27   * Database extension for MoneyWiseData.
28   * @author Tony Washer
29   */
30  public class MoneyWiseDataStore
31          extends PrometheusDataStore {
32      /**
33       * Construct a new Database class for load.
34       * @param pDatabase the database
35       * @param pConfig the config
36       * @throws OceanusException on error
37       */
38      public MoneyWiseDataStore(final String pDatabase,
39                                final PrometheusDBConfig pConfig) throws OceanusException {
40          /* Call super-constructor */
41          super(pDatabase, pConfig);
42  
43          /* Loop through the static types */
44          for (MoneyWiseStaticDataType myType : MoneyWiseStaticDataType.values()) {
45              /* Create the table */
46              addTable(newTable(myType));
47          }
48  
49          /* Loop through the basic types */
50          for (MoneyWiseBasicDataType myType : MoneyWiseBasicDataType.values()) {
51              /* Create the table */
52              addTable(newTbble(myType));
53          }
54      }
55  
56      /**
57       * Construct a new Database class for create database.
58       * @param pConfig the config
59       * @throws OceanusException on error
60       */
61      public MoneyWiseDataStore(final PrometheusDBConfig pConfig) throws OceanusException {
62          /* Call super-constructor */
63          super(pConfig);
64  
65          /* Loop through the static types */
66          for (MoneyWiseStaticDataType myType : MoneyWiseStaticDataType.values()) {
67              /* Create the table */
68              addTable(newTable(myType));
69          }
70  
71          /* Loop through the basic types */
72          for (MoneyWiseBasicDataType myType : MoneyWiseBasicDataType.values()) {
73              /* Create the table */
74              addTable(newTbble(myType));
75          }
76      }
77  
78      /**
79       * Create new table of required type.
80       * @param pDataType the data type
81       * @return the new table
82       */
83      private PrometheusTableDataItem<?> newTable(final MoneyWiseStaticDataType pDataType) {
84          /* Switch on data Type */
85          switch (pDataType) {
86              case DEPOSITTYPE:
87                  return new MoneyWiseTableDepositCategoryType(this);
88              case CASHTYPE:
89                  return new MoneyWiseTableCashCategoryType(this);
90              case LOANTYPE:
91                  return new MoneyWiseTableLoanCategoryType(this);
92              case PORTFOLIOTYPE:
93                  return new MoneyWiseTablePortfolioType(this);
94              case PAYEETYPE:
95                  return new MoneyWiseTablePayeeType(this);
96              case SECURITYTYPE:
97                  return new MoneyWiseTableSecurityType(this);
98              case TRANSTYPE:
99                  return new MoneyWiseTableTransCategoryType(this);
100             case ACCOUNTINFOTYPE:
101                 return new MoneyWiseTableAccountInfoType(this);
102             case TRANSINFOTYPE:
103                 return new MoneyWiseTableTransInfoType(this);
104             case CURRENCY:
105                 return new MoneyWiseTableCurrency(this);
106             case TAXBASIS:
107                 return new MoneyWiseTableTaxBasis(this);
108             default:
109                 throw new IllegalArgumentException(pDataType.toString());
110         }
111     }
112 
113     /**
114      * Create new table of required type.
115      * @param pDataType the data type
116      * @return the new table
117      */
118     private PrometheusTableDataItem<?> newTbble(final MoneyWiseBasicDataType pDataType) {
119         /* Switch on data Type */
120         switch (pDataType) {
121             case TRANSTAG:
122                 return new MoneyWiseTableTransTag(this);
123             case REGION:
124                 return new MoneyWiseTableRegion(this);
125             case DEPOSITCATEGORY:
126                 return new MoneyWiseTableDepositCategory(this);
127             case CASHCATEGORY:
128                 return new MoneyWiseTableCashCategory(this);
129             case LOANCATEGORY:
130                 return new MoneyWiseTableLoanCategory(this);
131             case TRANSCATEGORY:
132                 return new MoneyWiseTableTransCategory(this);
133             case EXCHANGERATE:
134                 return new MoneyWiseTableExchangeRate(this);
135             case PAYEE:
136                 return new MoneyWiseTablePayee(this);
137             case PAYEEINFO:
138                 return new MoneyWiseTablePayeeInfo(this);
139             case SECURITY:
140                 return new MoneyWiseTableSecurity(this);
141             case SECURITYPRICE:
142                 return new MoneyWiseTableSecurityPrice(this);
143             case SECURITYINFO:
144                 return new MoneyWiseTableSecurityInfo(this);
145             case DEPOSIT:
146                 return new MoneyWiseTableDeposit(this);
147             case DEPOSITRATE:
148                 return new MoneyWiseTableDepositRate(this);
149             case DEPOSITINFO:
150                 return new MoneyWiseTableDepositInfo(this);
151             case CASH:
152                 return new MoneyWiseTableCash(this);
153             case CASHINFO:
154                 return new MoneyWiseTableCashInfo(this);
155             case LOAN:
156                 return new MoneyWiseTableLoan(this);
157             case LOANINFO:
158                 return new MoneyWiseTableLoanInfo(this);
159             case PORTFOLIO:
160                 return new MoneyWiseTablePortfolio(this);
161             case PORTFOLIOINFO:
162                 return new MoneyWiseTablePortfolioInfo(this);
163             case TRANSACTION:
164                 return new MoneyWiseTableTransaction(this);
165             case TRANSACTIONINFO:
166                 return new MoneyWiseTableTransInfo(this);
167             default:
168                 throw new IllegalArgumentException(pDataType.toString());
169         }
170     }
171 }
