42  /**
43   * Validate CashInfoSet.
44   */
45  public class MoneyWiseValidateCashInfoSet
46          extends PrometheusValidateInfoSet<MoneyWiseCashInfo> {
47      /**
48       * ClosedPayee Error string.
49       */
50      private static final String ERROR_PAYEECLOSED = "AutoPayee is closed for non-closed autoCash";
51  
52      @Override
53      public MoneyWiseCash getOwner() {
54          return (MoneyWiseCash) super.getOwner();
55      }
56  
57      @Override
58      public MoneyWiseCashInfoSet getInfoSet() {
59          return (MoneyWiseCashInfoSet) super.getInfoSet();
60      }
61  
62      @Override
63      public MetisFieldRequired isClassRequired(final PrometheusDataInfoClass pClass) {
64          /* Access details about the Cash */
65          final MoneyWiseCash myCash = getOwner();
66          final MoneyWiseCashCategory myCategory = myCash.getCategory();
67  
68          /* If we have no Category, no class is allowed */
69          if (myCategory == null) {
70              return MetisFieldRequired.NOTALLOWED;
71          }
72  
73          /* Switch on class */
74          switch ((MoneyWiseAccountInfoClass) pClass) {
75              /* Allowed set */
76              case NOTES:
77                  return MetisFieldRequired.CANEXIST;
78  
79              case OPENINGBALANCE:
80                  return myCash.isAutoExpense()
81                          ? MetisFieldRequired.NOTALLOWED
82                          : MetisFieldRequired.CANEXIST;
83  
84              case AUTOPAYEE:
85              case AUTOEXPENSE:
86                  return myCash.isAutoExpense()
87                          ? MetisFieldRequired.MUSTEXIST
88                          : MetisFieldRequired.NOTALLOWED;
89  
90              /* Disallowed Set */
91              case SORTCODE:
92              case ACCOUNT:
93              case REFERENCE:
94              case WEBSITE:
95              case CUSTOMERNO:
96              case USERID:
97              case PASSWORD:
98              case MATURITY:
99              case SYMBOL:
100             case REGION:
101             case UNDERLYINGSTOCK:
102             case OPTIONPRICE:
103             default:
104                 return MetisFieldRequired.NOTALLOWED;
105         }
106     }
107 
108     @Override
109     public void validateClass(final MoneyWiseCashInfo pInfo,
110                               final PrometheusDataInfoClass pClass) {
111         /* Switch on class */
112         switch ((MoneyWiseAccountInfoClass) pClass) {
113             case OPENINGBALANCE:
114                 validateOpeningBalance(pInfo);
115                 break;
116             case AUTOEXPENSE:
117                 validateAutoExpense(pInfo);
118                 break;
119             case AUTOPAYEE:
120                 validateAutoPayee(pInfo);
121                 break;
122             case NOTES:
123                 validateNotes(pInfo);
124                 break;
125             default:
126                 break;
127         }
128     }
129 
130     /**
131      * Validate the opening balance.
132      * @param pInfo the info
133      */
134     private void validateOpeningBalance(final MoneyWiseCashInfo pInfo) {
135         final OceanusMoney myBalance = pInfo.getValue(OceanusMoney.class);
136         if (!myBalance.getCurrency().equals(getOwner().getCurrency())) {
137             getOwner().addError(MoneyWiseDepositInfoSet.ERROR_CURRENCY, MoneyWiseCashInfoSet.getFieldForClass(MoneyWiseAccountInfoClass.OPENINGBALANCE));
138         }
139     }
140 
141     /**
142      * Validate the autoExpense info.
143      * @param pInfo the info
144      */
145     private void validateAutoExpense(final MoneyWiseCashInfo pInfo) {
146         final MoneyWiseTransCategory myExpense = pInfo.getEventCategory();
147         final MoneyWiseTransCategoryClass myCatClass = myExpense.getCategoryTypeClass();
148         if (!myCatClass.isExpense() || myCatClass.canParentCategory()) {
149             getOwner().addError(MoneyWiseCashInfoSet.ERROR_AUTOEXP, MoneyWiseCashInfoSet.getFieldForClass(MoneyWiseAccountInfoClass.AUTOEXPENSE));
150         }
151     }
152 
153     /**
154      * Validate the autoPayee info.
155      * @param pInfo the info
156      */
157     private void validateAutoPayee(final MoneyWiseCashInfo pInfo) {
158         final MoneyWisePayee myPayee = pInfo.getPayee();
159         if (myPayee.isClosed() && !getOwner().isClosed()) {
160             getOwner().addError(ERROR_PAYEECLOSED, MoneyWiseCashInfoSet.getFieldForClass(MoneyWiseAccountInfoClass.AUTOPAYEE));
161         }
162     }
163 
164     /**
165      * Validate the Notes info.
166      * @param pInfo the info
167      */
168     private void validateNotes(final MoneyWiseCashInfo pInfo) {
169         final char[] myArray = pInfo.getValue(char[].class);
170         if (myArray.length > MoneyWiseAccountInfoClass.NOTES.getMaximumLength()) {
171             getOwner().addError(PrometheusDataItem.ERROR_LENGTH, MoneyWiseCashInfoSet.getFieldForClass(MoneyWiseAccountInfoClass.NOTES));
172         }
173     }
174 
175     @Override
176     protected void setDefault(final PrometheusDataInfoClass pClass) throws OceanusException {
177         /* Switch on the class */
178         switch ((MoneyWiseAccountInfoClass) pClass) {
179             case AUTOEXPENSE:
180                 getInfoSet().setValue(pClass, getDefaultAutoExpense());
181                 break;
182             case AUTOPAYEE:
183                 getInfoSet().setValue(pClass, getDefaultAutoPayee());
184                 break;
185             default:
186                 break;
187         }
188     }
189 
190     /**
191      * Obtain default expense for autoExpense cash.
192      * @return the default expense
193      */
194     private MoneyWiseTransCategory getDefaultAutoExpense() {
195         /* Access the category list */
196         final MoneyWiseTransCategoryList myCategories
197                 = getEditSet().getDataList(MoneyWiseBasicDataType.TRANSCATEGORY, MoneyWiseTransCategoryList.class);
198 
199         /* loop through the categories */
200         final Iterator<MoneyWiseTransCategory> myIterator = myCategories.iterator();
201         while (myIterator.hasNext()) {
202             final MoneyWiseTransCategory myCategory = myIterator.next();
203 
204             /* Ignore deleted categories */
205             if (myCategory.isDeleted()) {
206                 continue;
207             }
208 
209             /* Ignore categories that are the wrong class */
210             final MoneyWiseTransCategoryClass myCatClass = myCategory.getCategoryTypeClass();
211             if (myCatClass.isExpense() && !myCatClass.canParentCategory()) {
212                 return myCategory;
213             }
214         }
215 
216         /* Return no category */
217         return null;
218     }
219 
220     /**
221      * Obtain default payee for autoExpense cash.
222      * @return the default payee
223      */
224     private MoneyWisePayee getDefaultAutoPayee() {
225         /* Access the payee list */
226         final MoneyWisePayeeList myPayees
227                 = getEditSet().getDataList(MoneyWiseBasicDataType.PAYEE, MoneyWisePayeeList.class);
228 
229         /* loop through the payees */
230         final Iterator<MoneyWisePayee> myIterator = myPayees.iterator();
231         while (myIterator.hasNext()) {
232             final MoneyWisePayee myPayee = myIterator.next();
233 
234             /* Ignore deleted and closed payees */
235             if (!myPayee.isDeleted() && Boolean.TRUE.equals(!myPayee.isClosed())) {
236                 return myPayee;
237             }
238         }
239 
240         /* Return no payee */
241         return null;
242     }
243 
244     @Override
245     protected void autoCorrect(final PrometheusDataInfoClass pClass) throws OceanusException {
246         /* If the info is Opening balance */
247         if (MoneyWiseAccountInfoClass.OPENINGBALANCE.equals(pClass)) {
248             /* Access the value */
249             final MoneyWiseCash myOwner = getOwner();
250             OceanusMoney myOpening = myOwner.getOpeningBalance();
251             final MoneyWiseCurrency myAssetCurrency = myOwner.getAssetCurrency();
252             final Currency myCurrency = myAssetCurrency.getCurrency();
253 
254             /* If we need to change currency */
255             if (!myCurrency.equals(myOpening.getCurrency())) {
256                 myOpening = myOpening.changeCurrency(myCurrency);
257                 getInfoSet().setValue(pClass, myOpening);
258             }
259         }
260     }
