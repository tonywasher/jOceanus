199     @Override
200     protected void resolveEditSetLinks() throws OceanusException {
201         /* Resolve parent within list */
202         resolveDataLink(PrometheusDataResource.DATAGROUP_PARENT, getList());
203 
204         /* Resolve StaticType if required */
205         final PrometheusEditSet myEditSet = getList().getEditSet();
206         if (myEditSet.hasDataType(MoneyWiseStaticDataType.LOANTYPE)) {
207             resolveDataLink(MoneyWiseStaticDataType.LOANTYPE, myEditSet.getDataList(MoneyWiseStaticDataType.LOANTYPE, MoneyWiseLoanCategoryTypeList.class));
208         }
209     }

216     /**
217      * Update base category from an edited category.
218      * @param pCategory the edited category
219      * @return whether changes have been made
220      */
221     @Override
222     public boolean applyChanges(final PrometheusDataItem pCategory) {
223         /* Can only update from a loan category */
224         if (!(pCategory instanceof MoneyWiseLoanCategory)) {
225             return false;
226         }
227         final MoneyWiseLoanCategory myCategory = (MoneyWiseLoanCategory) pCategory;
228 
229         /* Store the current detail into history */
230         pushHistory();
231 
232         /* Apply basic changes */
233         applyBasicChanges(myCategory);
234 
235         /* Update the category type if required */
236         if (!MetisDataDifference.isEqual(getCategoryType(), myCategory.getCategoryType())) {
237             setValueType(myCategory.getCategoryType());
238         }
239 
240         /* Check for changes */
241         return checkForHistory();
242     }

305         /**
306          * Derive Edit list.
307          * @param pEditSet the editSet
308          * @return the edit list
309          * @throws OceanusException on error
310          */
311         public MoneyWiseLoanCategoryList deriveEditList(final PrometheusEditSet pEditSet) throws OceanusException {
312             /* Build an empty List */
313             final MoneyWiseLoanCategoryList myList = getEmptyList(PrometheusListStyle.EDIT);
314             myList.ensureMap();
315             pEditSet.setEditEntryList(MoneyWiseBasicDataType.LOANCATEGORY, myList);
316             myList.getValidator().setEditSet(pEditSet);
317 
318             /* Store the editSet */
319             myList.theEditSet = pEditSet;
320 
321             /* Loop through the categories */
322             final Iterator<MoneyWiseLoanCategory> myIterator = iterator();
323             while (myIterator.hasNext()) {
324                 final MoneyWiseLoanCategory myCurr = myIterator.next();
325 
326                 /* Ignore deleted events */
327                 if (myCurr.isDeleted()) {
328                     continue;
329                 }
330 
331                 /* Build the new linked category and add it to the list */
332                 final MoneyWiseLoanCategory myCategory = new MoneyWiseLoanCategory(myList, myCurr);
333                 myList.add(myCategory);
334                 myCategory.resolveEditSetLinks();
335 
336                 /* Adjust the map */
337                 myCategory.adjustMapForItem();
338             }
339 
340             /* Return the list */
341             return myList;
342         }

372         @Override
373         public MoneyWiseLoanCategory addValuesItem(final PrometheusDataValues pValues) throws OceanusException {
374             /* Create the category */
375             final MoneyWiseLoanCategory myCategory = new MoneyWiseLoanCategory(this, pValues);
376 
377             /* Check that this CategoryId has not been previously added */
378             if (!isIdUnique(myCategory.getIndexedId())) {
379                 myCategory.addError(ERROR_DUPLICATE, MetisDataResource.DATA_ID);
380                 throw new MoneyWiseDataException(myCategory, ERROR_VALIDATION);
381             }
382 
383             /* Add to the list */
384             add(myCategory);
385 
386             /* Return it */
387             return myCategory;
388         }
