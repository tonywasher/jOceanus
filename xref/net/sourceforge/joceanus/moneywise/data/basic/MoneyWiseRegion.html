79      /**
80       * Values constructor.
81       * @param pList the List to add to
82       * @param pValues the values constructor
83       * @throws OceanusException on error
84       */
85      private MoneyWiseRegion(final MoneyWiseRegionList pList,
86                              final PrometheusDataValues pValues) throws OceanusException {
87          /* Initialise the item */
88          super(pList, pValues);
89  
90          /* Protect against exceptions */
91          try {
92              /* Store the Name */
93              Object myValue = pValues.getValue(PrometheusDataResource.DATAITEM_FIELD_NAME);
94              if (myValue instanceof String s) {
95                  setValueName(s);
96              } else if (myValue instanceof byte[] ba) {
97                  setValueName(ba);
98              }
99  
100             /* Store the Description */
101             myValue = pValues.getValue(PrometheusDataResource.DATAITEM_FIELD_DESC);
102             if (myValue instanceof String s) {
103                 setValueDesc(s);
104             } else if (myValue instanceof byte[] ba) {
105                 setValueDesc(ba);
106             }
107 
108             /* Catch Exceptions */
109         } catch (OceanusException e) {
110             /* Pass on exception */
111             throw new MoneyWiseDataException(this, ERROR_CREATEITEM, e);
112         }
113     }

298     /**
299      * Update base tag from an edited tag.
300      * @param pTag the edited tag
301      * @return whether changes have been made
302      */
303     @Override
304     public boolean applyChanges(final PrometheusDataItem pTag) {
305         /* Can only update from a region */
306         if (!(pTag instanceof MoneyWiseRegion)) {
307             return false;
308         }
309         final MoneyWiseRegion myRegion = (MoneyWiseRegion) pTag;
310 
311         /* Store the current detail into history */
312         pushHistory();
313 
314         /* Update the Name if required */
315         if (!MetisDataDifference.isEqual(getName(), myRegion.getName())) {
316             setValueName(myRegion.getNameField());
317         }
318 
319         /* Update the description if required */
320         if (!MetisDataDifference.isEqual(getDesc(), myRegion.getDesc())) {
321             setValueDesc(myRegion.getDescField());
322         }
323 
324         /* Check for changes */
325         return checkForHistory();
326     }

400         /**
401          * Derive Edit list.
402          * @param pEditSet the editSet
403          * @return the edit list
404          */
405         public MoneyWiseRegionList deriveEditList(final PrometheusEditSet pEditSet) {
406             /* Build an empty List */
407             final MoneyWiseRegionList myList = getEmptyList(PrometheusListStyle.EDIT);
408             myList.ensureMap();
409             pEditSet.setEditEntryList(MoneyWiseBasicDataType.REGION, myList);
410             myList.getValidator().setEditSet(pEditSet);
411 
412             /* Loop through the regions */
413             final Iterator<MoneyWiseRegion> myIterator = iterator();
414             while (myIterator.hasNext()) {
415                 final MoneyWiseRegion myCurr = myIterator.next();
416 
417                 /* Ignore deleted regions */
418                 if (myCurr.isDeleted()) {
419                     continue;
420                 }
421 
422                 /* Build the new linked region and add it to the list */
423                 final MoneyWiseRegion myRegion = new MoneyWiseRegion(myList, myCurr);
424                 myList.add(myRegion);
425 
426                 /* Adjust the map */
427                 myRegion.adjustMapForItem();
428             }
429 
430             /* Return the list */
431             return myList;
432         }

468         @Override
469         public MoneyWiseRegion addValuesItem(final PrometheusDataValues pValues) throws OceanusException {
470             /* Create the region */
471             final MoneyWiseRegion myRegion = new MoneyWiseRegion(this, pValues);
472 
473             /* Check that this regionId has not been previously added */
474             if (!isIdUnique(myRegion.getIndexedId())) {
475                 myRegion.addError(ERROR_DUPLICATE, MetisDataResource.DATA_ID);
476                 throw new MoneyWiseDataException(myRegion, ERROR_VALIDATION);
477             }
478 
479             /* Add to the list */
480             add(myRegion);
481 
482             /* Return it */
483             return myRegion;
484         }
