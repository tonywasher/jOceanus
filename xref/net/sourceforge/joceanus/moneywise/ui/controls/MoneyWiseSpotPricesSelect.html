159      * Constructor.
160      * @param pFactory the GUI factory
161      * @param pView the data view
162      */
163     public MoneyWiseSpotPricesSelect(final TethysUIFactory<?> pFactory,
164                                      final MoneyWiseView pView) {
165         /* Store table and view details */
166         theView = pView;
167 
168         /* Create Event Manager */
169         theEventManager = new OceanusEventManager<>();
170 
171         /* Create Labels */
172         final TethysUIControlFactory myControls = pFactory.controlFactory();
173         final TethysUILabel myDate = myControls.newLabel(NLS_DATE);
174         final TethysUILabel myPort = myControls.newLabel(NLS_PORT);
175 
176         /* Create the check box */
177         theShowClosed = myControls.newCheckBox(NLS_CLOSED);
178 
179         /* Create the DateButton */
180         final TethysUIButtonFactory<?> myButtons = pFactory.buttonFactory();
181         theDateButton = myButtons.newDateButton();
182 
183         /* Create the Download Button */
184         theDownloadButton = myButtons.newButton();
185         MetisIcon.configureDownloadIconButton(theDownloadButton);
186 
187         /* Create the Buttons */
188         theNext = myButtons.newButton();
189         theNext.setIcon(TethysUIArrowIconId.RIGHT);
190         theNext.setToolTip(NLS_NEXTTIP);
191         thePrev = myButtons.newButton();
192         thePrev.setIcon(TethysUIArrowIconId.LEFT);
193         thePrev.setToolTip(NLS_PREVTIP);
194 
195         /* Create the portfolio button */
196         thePortButton = myButtons.newScrollButton(MoneyWiseAnalysisPortfolioBucket.class);
197 
198         /* Create initial state */
199         theState = new MoneyWiseSpotPricesState();
200 
201         /* Create the panel */
202         thePanel = pFactory.paneFactory().newHBoxPane();
203         thePanel.setBorderTitle(NLS_TITLE);
204 
205         /* Define the layout */
206         thePanel.addNode(myDate);
207         thePanel.addNode(thePrev);
208         thePanel.addNode(theDateButton);
209         thePanel.addNode(theNext);
210         thePanel.addSpacer();
211         thePanel.addNode(myPort);
212         thePanel.addNode(thePortButton);
213         thePanel.addSpacer();
214         thePanel.addNode(theShowClosed);
215         thePanel.addSpacer();
216         thePanel.addNode(theDownloadButton);
217 
218         /* Initialise the data from the view */
219         refreshData();
220 
221         /* Apply the current state */
222         theState.applyState();
223 
224         /* Access the menus */
225         thePortMenu = thePortButton.getMenu();
226 
227         /* Add the listeners */
228         final OceanusEventRegistrar<TethysUIEvent> myRegistrar = thePortButton.getEventRegistrar();
229         myRegistrar.addEventListener(TethysUIEvent.NEWVALUE, e -> handleNewPortfolio());
230         thePortButton.setMenuConfigurator(e -> buildPortfolioMenu());
231         theDownloadButton.getEventRegistrar().addEventListener(TethysUIEvent.PRESSED, e -> theEventManager.fireEvent(PrometheusDataEvent.DOWNLOAD));
232         theDateButton.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> handleNewDate());
233         theShowClosed.getEventRegistrar().addEventListener(e -> handleNewClosed());
234         theNext.getEventRegistrar().addEventListener(e -> {
235             theState.setNext();
236             theEventManager.fireEvent(PrometheusDataEvent.SELECTIONCHANGED);
237         });
238         thePrev.getEventRegistrar().addEventListener(e -> {
239             theState.setPrev();
240             theEventManager.fireEvent(PrometheusDataEvent.SELECTIONCHANGED);
241         });
242     }

281     /**
282      * Refresh data.
283      */
284     public final void refreshData() {
285         /* Access the data */
286         final OceanusDateRange myRange = theView.getRange();
287 
288         /* Set the range for the Date Button */
289         setRange(myRange);
290 
291         /* Access portfolio list */
292         final MoneyWiseAnalysisManager myManager = theView.getAnalysisManager();
293         final MoneyWiseAnalysis myAnalysis = myManager.getAnalysis();
294         thePortfolios = myAnalysis.getPortfolios();
295 
296         /* Note that we are refreshing data */
297         refreshingData = true;
298 
299         /* Obtain the current portfolio */
300         MoneyWiseAnalysisPortfolioBucket myPortfolio = theState.getPortfolio();
301 
302         /* Switch to portfolio in this analysis */
303         myPortfolio = myPortfolio != null
304                 ? thePortfolios.getMatchingPortfolio(myPortfolio.getPortfolio())
305                 : thePortfolios.getDefaultPortfolio();
306 
307         /* Set the portfolio */
308         theState.setPortfolio(myPortfolio);
309         theState.applyState();
310 
311         /* Note that we have finished refreshing data */
312         refreshingData = false;
313     }

377     /**
378      * Build the portfolio menu.
379      */
380     private void buildPortfolioMenu() {
381         /* Reset the popUp menu */
382         thePortMenu.removeAllItems();
383 
384         /* Record active item */
385         TethysUIScrollItem<MoneyWiseAnalysisPortfolioBucket> myActive = null;
386         final MoneyWiseAnalysisPortfolioBucket myCurr = theState.getPortfolio();
387 
388         /* Loop through the available portfolio values */
389         final Iterator<MoneyWiseAnalysisPortfolioBucket> myIterator = thePortfolios.iterator();
390         while (myIterator.hasNext()) {
391             final MoneyWiseAnalysisPortfolioBucket myBucket = myIterator.next();
392 
393             /* Skip if the portfolio is closed and we are not showing closed accounts */
394             if (!myBucket.isActive()
395                     && !theState.showClosed()) {
396                 continue;
397             }
398 
399             /* Create a new MenuItem and add it to the popUp */
400             final TethysUIScrollItem<MoneyWiseAnalysisPortfolioBucket> myItem = thePortMenu.addItem(myBucket);
401 
402             /* If this is the active bucket */
403             if (myBucket.equals(myCurr)) {
404                 /* Record it */
405                 myActive = myItem;
406             }
407         }
408 
409         /* Ensure active item is visible */
410         if (myActive != null) {
411             myActive.scrollToItem();
412         }
413     }

415     /**
416      * Handle new Date.
417      */
418     private void handleNewDate() {
419         /* Select the new date */
420         if (theState.setDate(theDateButton)) {
421             theEventManager.fireEvent(PrometheusDataEvent.SELECTIONCHANGED);
422         }
423     }
424 
425     /**
426      * Handle new Portfolio.
427      */
428     private void handleNewPortfolio() {
429         /* Select the new portfolio */
430         if (theState.setPortfolio(thePortButton.getValue())) {
431             theState.applyState();
432             theEventManager.fireEvent(PrometheusDataEvent.SELECTIONCHANGED);
433         }
434     }
435 
436     /**
437      * Handle new Closed.
438      */
439     private void handleNewClosed() {
440         if (!refreshingData) {
441             theState.setShowClosed(theShowClosed.isSelected());
442             theEventManager.fireEvent(PrometheusDataEvent.SELECTIONCHANGED);
443         }
444     }

449     private final class MoneyWiseSpotPricesState {
450         /**
451          * Portfolio.
452          */
453         private MoneyWiseAnalysisPortfolioBucket thePortfolio;
454 
455         /**
456          * Selected date.
457          */
458         private OceanusDate theDate;
459 
460         /**
461          * Next date.

463         private OceanusDate theNextDate;

465         /**
466          * Previous date.

469 
470         /**
471          * showClosed.
472          */
473         private boolean showClosed;
474 
475         /**
476          * Constructor.
477          */
478         private MoneyWiseSpotPricesState() {
479             theDate = new OceanusDate();
480         }
481 
482         /**
483          * Constructor.
484          * @param pState state to copy from
485          */
486         private MoneyWiseSpotPricesState(final MoneyWiseSpotPricesState pState) {
487             thePortfolio = pState.getPortfolio();
488             theDate = new OceanusDate(pState.getDate());
489             if (pState.getNextDate() != null) {
490                 theNextDate = new OceanusDate(pState.getNextDate());
491             }
492             if (pState.getPrevDate() != null) {
493                 thePrevDate = new OceanusDate(pState.getPrevDate());
494             }
495             showClosed = pState.showClosed();
496         }

538         /**
539          * Set new Portfolio.
540          * @param pPortfolio the Portfolio
541          * @return true/false did a change occur
542          */
543         private boolean setPortfolio(final MoneyWiseAnalysisPortfolioBucket pPortfolio) {
544             /* Adjust the selected portfolio */
545             if (!MetisDataDifference.isEqual(pPortfolio, thePortfolio)) {
546                 thePortfolio = pPortfolio;
547                 return true;
548             }
549             return false;
550         }
551 
552         /**
553          * Set new Date.
554          * @param pButton the Button with the new date
555          * @return true/false did a change occur
556          */
557         private boolean setDate(final TethysUIDateButtonManager pButton) {
558             /* Adjust the date and build the new range */
559             final OceanusDate myDate = new OceanusDate(pButton.getSelectedDate());
560             if (!MetisDataDifference.isEqual(myDate, theDate)) {
561                 theDate = myDate;
562                 return true;
563             }
564             return false;
565         }
566

568          * Set Next Date.
569          */
570         private void setNext() {
571             /* Copy date */
572             theDate = new OceanusDate(theNextDate);
573             applyState();
574         }
575 
576         /**
577          * Set Previous Date.
578          */
579         private void setPrev() {
580             /* Copy date */
581             theDate = new OceanusDate(thePrevDate);
582             applyState();
583         }
584 
585         /**
586          * Set showClosed.
587          * @param pShowClosed true/false
588          */
589         private void setShowClosed(final boolean pShowClosed) {
590             /* Set flag */
591             showClosed = pShowClosed;
592             applyState();
593         }
594 
595         /**
596          * Set Adjacent dates.
597          * @param pPrev the previous Date
598          * @param pNext the next Date
599          */
600         private void setAdjacent(final OceanusDate pPrev,
601                                  final OceanusDate pNext) {
602             /* Record the dates */
603             thePrevDate = pPrev;
604             theNextDate = pNext;
605 
606             /* Adjust values */
607             setEnabled(true);
608         }
609 
610         /**
611          * Apply the State.
612          */
613         private void applyState() {
614             /* Adjust the lock-down */
615             setEnabled(true);
616             theDateButton.setSelectedDate(theDate);
617             thePortButton.setValue(thePortfolio);
618             theShowClosed.setSelected(showClosed);
619 
620             /* Determine whether we are todays date */
621             final boolean isToday = MetisDataDifference.isEqual(theDate, new OceanusDate());
622             final boolean isActive = thePortfolio != null && thePortfolio.isActive();
623             theDownloadButton.setVisible(isToday && isActive);
624         }
625     }
