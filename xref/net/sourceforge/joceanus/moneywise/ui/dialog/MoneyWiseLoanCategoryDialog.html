44  /**
45   * Panel to display/edit/create a LoanCategory.
46   */
47  public class MoneyWiseLoanCategoryDialog
48          extends MoneyWiseItemPanel<MoneyWiseLoanCategory> {
49      /**
50       * Constructor.
51       * @param pFactory the GUI factory
52       * @param pEditSet the edit set
53       * @param pOwner the owning table
54       */
55      public MoneyWiseLoanCategoryDialog(final TethysUIFactory<?> pFactory,
56                                         final PrometheusEditSet pEditSet,
57                                         final MoneyWiseBaseTable<MoneyWiseLoanCategory> pOwner) {
58          /* Initialise the panel */
59          super(pFactory, pEditSet, pOwner);
60  
61          /* Create a new panel */
62          final PrometheusFieldSet<MoneyWiseLoanCategory> myFieldSet = getFieldSet();
63  
64          /* Create the text fields */
65          final TethysUIFieldFactory myFields = pFactory.fieldFactory();
66          final TethysUIStringEditField myName = myFields.newStringField();
67          final TethysUIStringEditField mySubName = myFields.newStringField();
68          final TethysUIStringEditField myDesc = myFields.newStringField();
69  
70          /* Create the buttons */
71          final TethysUIScrollButtonField<MoneyWiseLoanCategoryType> myTypeButton = myFields.newScrollField(MoneyWiseLoanCategoryType.class);
72          final TethysUIScrollButtonField<MoneyWiseLoanCategory> myParentButton = myFields.newScrollField(MoneyWiseLoanCategory.class);
73  
74          /* Assign the fields to the panel */
75          myFieldSet.addField(PrometheusDataResource.DATAITEM_FIELD_NAME, myName, MoneyWiseLoanCategory::getName);
76          myFieldSet.addField(MoneyWiseBasicResource.CATEGORY_SUBCAT, mySubName, MoneyWiseLoanCategory::getSubCategory);
77          myFieldSet.addField(PrometheusDataResource.DATAITEM_FIELD_DESC, myDesc, MoneyWiseLoanCategory::getDesc);
78          myFieldSet.addField(MoneyWiseStaticDataType.LOANTYPE, myTypeButton, MoneyWiseLoanCategory::getCategoryType);
79          myFieldSet.addField(PrometheusDataResource.DATAGROUP_PARENT, myParentButton, MoneyWiseLoanCategory::getParentCategory);
80  
81          /* Configure the menuBuilders */
82          myTypeButton.setMenuConfigurator(c -> buildCategoryTypeMenu(c, getItem()));
83          myParentButton.setMenuConfigurator(c -> buildParentMenu(c, getItem()));
84  
85          /* Configure name checks */
86          mySubName.setValidator(this::isValidName);
87          mySubName.setReporter(pOwner::showValidateError);
88  
89          /* Configure description checks */
90          myDesc.setValidator(this::isValidDesc);
91          myDesc.setReporter(pOwner::showValidateError);
92      }

94      @Override
95      public void refreshData() {
96          /* If we have an item */
97          final MoneyWiseLoanCategory myItem = getItem();
98          if (myItem != null) {
99              final MoneyWiseLoanCategoryList myCategories = getDataList(MoneyWiseBasicDataType.LOANCATEGORY, MoneyWiseLoanCategoryList.class);
100             setItem(myCategories.findItemById(myItem.getIndexedId()));
101         }
102 
103         /* Make sure that the item is not editable */
104         setEditable(false);
105     }
106 
107     @Override
108     protected void adjustFields(final boolean isEditable) {
109         /* Access the fieldSet */
110         final PrometheusFieldSet<MoneyWiseLoanCategory> myFieldSet = getFieldSet();
111 
112         /* Determine whether parent/full-name fields are visible */
113         final MoneyWiseLoanCategory myCategory = getItem();
114         final MoneyWiseLoanCategoryType myType = myCategory.getCategoryType();
115         final boolean isParent = myType.isLoanCategory(MoneyWiseLoanCategoryClass.PARENT);
116 
117         /* Determine whether the description field should be visible */
118         final boolean bShowDesc = isEditable || myCategory.getDesc() != null;
119         myFieldSet.setFieldVisible(PrometheusDataResource.DATAITEM_FIELD_DESC, bShowDesc);
120 
121         /* Set visibility */
122         myFieldSet.setFieldVisible(PrometheusDataResource.DATAGROUP_PARENT, !isParent);
123         myFieldSet.setFieldVisible(MoneyWiseBasicResource.CATEGORY_SUBCAT, !isParent);
124 
125         /* If the category is active then we cannot change the category type */
126         boolean canEdit = isEditable && !myCategory.isActive();
127 
128         /* We cannot change a parent category type */
129         canEdit &= !isParent;
130         myFieldSet.setFieldEditable(MoneyWiseStaticDataType.LOANTYPE, canEdit);
131 
132         /* If the category is not a parent then we cannot edit the full name */
133         myFieldSet.setFieldEditable(PrometheusDataResource.DATAITEM_FIELD_NAME, isEditable && isParent);
134     }

136     @Override
137     protected void updateField(final PrometheusFieldSetEvent pUpdate) throws OceanusException {
138         /* Access the field */
139         final MetisDataFieldId myField = pUpdate.getFieldId();
140         final MoneyWiseLoanCategory myCategory = getItem();
141 
142         /* Process updates */
143         if (PrometheusDataResource.DATAITEM_FIELD_NAME.equals(myField)) {
144             /* Update the SUBCATEGORY(!!) Name */
145             myCategory.setSubCategoryName(pUpdate.getValue(String.class));
146         } else if (MoneyWiseBasicResource.CATEGORY_SUBCAT.equals(myField)) {
147             /* Update the SubCategory */
148             myCategory.setSubCategoryName(pUpdate.getValue(String.class));
149         } else if (PrometheusDataResource.DATAGROUP_PARENT.equals(myField)) {
150             /* Update the Parent */
151             myCategory.setParentCategory(pUpdate.getValue(MoneyWiseLoanCategory.class));
152         } else if (PrometheusDataResource.DATAITEM_FIELD_DESC.equals(myField)) {
153             /* Update the Description */
154             myCategory.setDescription(pUpdate.getValue(String.class));
155         } else if (MoneyWiseStaticDataType.LOANTYPE.equals(myField)) {
156             /* Update the Category Type */
157             myCategory.setCategoryType(pUpdate.getValue(MoneyWiseLoanCategoryType.class));
158         }
159     }

161     @Override
162     protected void declareGoToItems(final boolean pUpdates) {
163         final MoneyWiseLoanCategory myItem = getItem();
164         final MoneyWiseLoanCategory myParent = myItem.getParentCategory();
165         if (!pUpdates) {
166             final MoneyWiseLoanCategoryType myType = myItem.getCategoryType();
167             declareGoToItem(myType);
168         }
169         declareGoToItem(myParent);
170     }

172     /**
173      * Build the category type menu for an item.
174      * @param pMenu the menu
175      * @param pCategory the category to build for
176      */
177     public void buildCategoryTypeMenu(final TethysUIScrollMenu<MoneyWiseLoanCategoryType> pMenu,
178                                       final MoneyWiseLoanCategory pCategory) {
179         /* Clear the menu */
180         pMenu.removeAllItems();
181 
182         /* Record active item */
183         final MoneyWiseLoanCategoryType myCurr = pCategory.getCategoryType();
184         TethysUIScrollItem<MoneyWiseLoanCategoryType> myActive = null;
185 
186         /* Access Loan Category types */
187         final MoneyWiseLoanCategoryTypeList myCategoryTypes = getDataList(MoneyWiseStaticDataType.LOANTYPE, MoneyWiseLoanCategoryTypeList.class);
188 
189         /* Loop through the LoanCategoryTypes */
190         final Iterator<MoneyWiseLoanCategoryType> myIterator = myCategoryTypes.iterator();
191         while (myIterator.hasNext()) {
192             final MoneyWiseLoanCategoryType myType = myIterator.next();
193 
194             /* Ignore deleted or disabled */
195             boolean bIgnore = myType.isDeleted() || !myType.getEnabled();
196 
197             /* Ignore category if it is a parent */
198             bIgnore |= myType.getLoanClass().isParentCategory();
199             if (bIgnore) {
200                 continue;
201             }
202 
203             /* Create a new action for the type */
204             final TethysUIScrollItem<MoneyWiseLoanCategoryType> myItem = pMenu.addItem(myType);
205 
206             /* If this is the active type */
207             if (myType.equals(myCurr)) {
208                 /* Record it */
209                 myActive = myItem;
210             }
211         }
212 
213         /* Ensure active item is visible */
214         if (myActive != null) {
215             myActive.scrollToItem();
216         }
217     }

219     /**
220      * Build the parent menu for an item.
221      * @param pMenu the menu
222      * @param pCategory the category to build for
223      */
224     private static void buildParentMenu(final TethysUIScrollMenu<MoneyWiseLoanCategory> pMenu,
225                                         final MoneyWiseLoanCategory pCategory) {
226         /* Clear the menu */
227         pMenu.removeAllItems();
228 
229         /* Record active item */
230         final MoneyWiseLoanCategory myCurr = pCategory.getParentCategory();
231         TethysUIScrollItem<MoneyWiseLoanCategory> myActive = null;
232 
233         /* Loop through the LoanCategories */
234         final MoneyWiseLoanCategoryList myCategories = pCategory.getList();
235         final Iterator<MoneyWiseLoanCategory> myIterator = myCategories.iterator();
236         while (myIterator.hasNext()) {
237             final MoneyWiseLoanCategory myCat = myIterator.next();
238 
239             /* Ignore deleted and non-parent items */
240             final MoneyWiseLoanCategoryClass myClass = myCat.getCategoryTypeClass();
241             if (myCat.isDeleted() || !myClass.isParentCategory()) {
242                 continue;
243             }
244 
245             /* Create a new action for the parent */
246             final TethysUIScrollItem<MoneyWiseLoanCategory> myItem = pMenu.addItem(myCat);
247 
248             /* If this is the active parent */
249             if (myCat.equals(myCurr)) {
250                 /* Record it */
251                 myActive = myItem;
252             }
253         }
254 
255         /* Ensure active item is visible */
256         if (myActive != null) {
257             myActive.scrollToItem();
258         }
259     }
