34  /**
35   * Panel to display/edit/create a Region.
36   */
37  public class MoneyWiseRegionDialog
38          extends MoneyWiseItemPanel<MoneyWiseRegion> {
39      /**
40       * Constructor.
41       * @param pFactory the GUI factory
42       * @param pEditSet the edit set
43       * @param pOwner the owning table
44       */
45      public MoneyWiseRegionDialog(final TethysUIFactory<?> pFactory,
46                                   final PrometheusEditSet pEditSet,
47                                   final MoneyWiseBaseTable<MoneyWiseRegion> pOwner) {
48          /* Initialise the panel */
49          super(pFactory, pEditSet, pOwner);
50  
51          /* Access the fieldSet */
52          final PrometheusFieldSet<MoneyWiseRegion> myFieldSet = getFieldSet();
53  
54          /* Create the text fields */
55          final TethysUIFieldFactory myFields = pFactory.fieldFactory();
56          final TethysUIStringEditField myName = myFields.newStringField();
57          final TethysUIStringEditField myDesc = myFields.newStringField();
58  
59          /* Assign the fields to the panel */
60          myFieldSet.addField(PrometheusDataResource.DATAITEM_FIELD_NAME, myName, MoneyWiseRegion::getName);
61          myFieldSet.addField(PrometheusDataResource.DATAITEM_FIELD_DESC, myDesc, MoneyWiseRegion::getDesc);
62  
63          /* Configure name checks */
64          myName.setValidator(this::isValidName);
65          myName.setReporter(pOwner::showValidateError);
66  
67          /* Configure description checks */
68          myDesc.setValidator(this::isValidDesc);
69          myDesc.setReporter(pOwner::showValidateError);
70      }
71  
72      @Override
73      public void refreshData() {
74          /* If we have an item */
75          final MoneyWiseRegion myItem = getItem();
76          if (myItem != null) {
77              final MoneyWiseRegionList myRegions = getDataList(MoneyWiseBasicDataType.REGION, MoneyWiseRegionList.class);
78              setItem(myRegions.findItemById(myItem.getIndexedId()));
79          }
80  
81          /* Make sure that the item is not editable */
82          setEditable(false);
83      }
84  
85      @Override
86      protected void adjustFields(final boolean isEditable) {
87          /* Access the fieldSet */
88          final PrometheusFieldSet<MoneyWiseRegion> myFieldSet = getFieldSet();
89  
90          /* Determine whether the description field should be visible */
91          final boolean bShowDesc = isEditable || getItem().getDesc() != null;
92          myFieldSet.setFieldVisible(PrometheusDataResource.DATAITEM_FIELD_DESC, bShowDesc);
93      }
94  
95      @Override
96      protected void updateField(final PrometheusFieldSetEvent pUpdate) throws OceanusException {
97          /* Access the field */
98          final MetisDataFieldId myField = pUpdate.getFieldId();
99          final MoneyWiseRegion myRegion = getItem();
100 
101         /* Process updates */
102         if (PrometheusDataResource.DATAITEM_FIELD_NAME.equals(myField)) {
103             /* Update the Name */
104             myRegion.setName(pUpdate.getValue(String.class));
105         } else if (PrometheusDataResource.DATAITEM_FIELD_DESC.equals(myField)) {
106             /* Update the Description */
107             myRegion.setDescription(pUpdate.getValue(String.class));
108         }
109     }
110 
111     @Override
112     protected void declareGoToItems(final boolean pUpdates) {
113         /* No GoTo items */
114     }
115 }
