47  /**
48   * MoneyWise Static Table.
49   * @param <L> the list type
50   * @param <T> the data type
51   */
52  public class MoneyWiseStaticTable<L extends PrometheusStaticList<T>, T extends PrometheusStaticDataItem>
53          extends MoneyWiseBaseTable<T> {
54      /**
55       * Class column width.
56       */
57      private static final int WIDTH_CLASS = 90;
58  
59      /**
60       * The list class.
61       */
62      private final Class<L> theClass;
63  
64      /**
65       * The enabled column.
66       */
67      private final TethysUITableColumn<Boolean, MetisDataFieldId, T> theEnabledColumn;
68  
69      /**
70       * The new button.
71       */
72      private final TethysUIScrollButtonManager<TethysUIGenericWrapper> theNewButton;
73  
74      /**
75       * The edit list.
76       */
77      private L theStatic;
78  
79      /**
80       * show disabled.
81       */
82      private boolean showAll;
83  
84      /**
85       * Constructor.
86       * @param pView the view
87       * @param pEditSet the editSet
88       * @param pError the error panel
89       * @param pDataType the dataType
90       * @param pListClass the listClass
91       */
92      MoneyWiseStaticTable(final MoneyWiseView pView,
93                           final PrometheusEditSet pEditSet,
94                           final MetisErrorPanel pError,
95                           final MoneyWiseStaticDataType pDataType,
96                           final Class<L> pListClass) {
97          /* Store parameters */
98          super(pView, pEditSet, pError, pDataType);
99          theClass = pListClass;
100 
101         /* Access the gui factory */
102         final TethysUIFactory<?> myGuiFactory = pView.getGuiFactory();
103         final TethysUITableManager<MetisDataFieldId, T> myTable = getTable();
104 
105         /* Create new button */
106         theNewButton = myGuiFactory.buttonFactory().newScrollButton(TethysUIGenericWrapper.class);
107         MetisIcon.configureNewScrollButton(theNewButton);
108 
109         /* Set table configuration */
110         myTable.setDisabled(PrometheusStaticDataItem::isDisabled)
111                 .setComparator(PrometheusStaticDataItem::compareTo);
112 
113         /* Create the class column */
114         myTable.declareStringColumn(PrometheusDataResource.STATICDATA_CLASS)
115                 .setCellValueFactory(r -> r.getStaticClass().toString())
116                 .setEditable(false)
117                 .setColumnWidth(WIDTH_CLASS);
118 
119         /* Create the name column */
120         myTable.declareStringColumn(PrometheusDataResource.DATAITEM_FIELD_NAME)
121                 .setValidator(this::isValidName)
122                 .setCellValueFactory(PrometheusStaticDataItem::getName)
123                 .setEditable(true)
124                 .setColumnWidth(WIDTH_NAME)
125                 .setOnCommit((r, v) -> updateField(PrometheusStaticDataItem::setName, r, v));
126 
127         /* Create the description column */
128         myTable.declareStringColumn(PrometheusDataResource.DATAITEM_FIELD_DESC)
129                 .setValidator(this::isValidDesc)
130                 .setCellValueFactory(PrometheusStaticDataItem::getDesc)
131                 .setEditable(true)
132                 .setColumnWidth(WIDTH_DESC)
133                 .setOnCommit((r, v) -> updateField(PrometheusStaticDataItem::setDescription, r, v));
134 
135         /* Create the enabled column */
136         final TethysUIIconMapSet<Boolean> myEnabledMapSet = PrometheusIcon.configureEnabledIconButton(myGuiFactory);
137         theEnabledColumn = myTable.declareIconColumn(PrometheusDataResource.STATICDATA_ENABLED, Boolean.class)
138                 .setIconMapSet(r -> myEnabledMapSet)
139                 .setCellValueFactory(PrometheusStaticDataItem::getEnabled)
140                 .setVisible(false)
141                 .setEditable(true)
142                 .setCellEditable(r -> !r.isActive())
143                 .setColumnWidth(WIDTH_ICON)
144                 .setOnCommit((r, v) -> updateField(PrometheusStaticDataItem::setEnabled, r, v));
145 
146         /* Create the Active column */
147         final TethysUIIconMapSet<MetisAction> myActionMapSet = MetisIcon.configureStatusIconButton(myGuiFactory);
148         myTable.declareIconColumn(PrometheusDataResource.DATAITEM_TOUCH, MetisAction.class)
149                 .setIconMapSet(r -> myActionMapSet)
150                 .setCellValueFactory(r -> r.isActive() ? MetisAction.ACTIVE : MetisAction.DELETE)
151                 .setName(MoneyWiseUIResource.STATICDATA_ACTIVE.getValue())
152                 .setEditable(true)
153                 .setCellEditable(r -> !r.isActive())
154                 .setColumnWidth(WIDTH_ICON)
155                 .setOnCommit((r, v) -> updateField(this::deleteRow, r, v));
156 
157         /* Add listeners */
158         final OceanusEventRegistrar<TethysUIEvent> myRegistrar = theNewButton.getEventRegistrar();
159         myRegistrar.addEventListener(TethysUIEvent.NEWVALUE, e -> handleNewClass());
160         theNewButton.setMenuConfigurator(e -> buildNewMenu());
161         setShowAll(false);
162     }
163 
164     /**
165      * Obtain the new button.
166      * @return the new Button
167      */
168     TethysUIScrollButtonManager<TethysUIGenericWrapper> getNewButton() {
169         return theNewButton;
170     }
171 
172     @Override
173     protected void refreshData() throws OceanusException {
174         final MoneyWiseDataSet myData = getView().getData();
175         final PrometheusStaticList<T> myStatic = myData.getDataList(theClass);
176         theStatic = theClass.cast(myStatic.deriveList(PrometheusListStyle.EDIT));
177         theStatic.mapData();
178         getTable().setItems(theStatic.getUnderlyingList());
179         getEditEntry().setDataList(theStatic);
180         restoreSelected();
181     }
182 
183     /**
184      * handle new static class.
185      */
186     private void handleNewClass() {
187         /* Create a new profile */
188         final OceanusProfile myTask = getView().getNewProfile("addNewClass");
189 
190         /* Access the new class */
191         cancelEditing();
192         final PrometheusStaticDataClass myClass = (PrometheusStaticDataClass) theNewButton.getValue().getData();
193 
194         /* Protect the action */
195         try {
196             /* Look to find a deleted value */
197             myTask.startTask("addToList");
198             T myValue = theStatic.findItemByClass(myClass);
199 
200             /* If we found a deleted value */
201             if (myValue != null) {
202                 /* reinstate it */
203                 myValue.setDeleted(false);
204 
205                 /* else we have no existing value */
206             } else {
207                 /* Create the new value */
208                 myValue = theStatic.addNewItem(myClass);
209                 myValue.setNewVersion();
210                 myValue.adjustMapForItem();
211             }
212 
213             /* Update the table */
214             myTask.startTask("incrementVersion");
215             getEditSet().incrementVersion();
216             updateTableData();
217             selectItem(myValue);
218             notifyChanges();
219 
220             /* Handle exceptions */
221         } catch (OceanusException e) {
222             setError(e);
223         }
224 
225         /* End the task */
226         myTask.end();
227     }
228 
229     /**
230      * Build the menu of available new items.
231      */
232     private void buildNewMenu() {
233         /* Reset the menu popUp */
234         final TethysUIScrollMenu<TethysUIGenericWrapper> myMenu = theNewButton.getMenu();
235         myMenu.removeAllItems();
236 
237         /* Loop through the missing classes */
238         for (PrometheusStaticDataClass myValue : theStatic.getMissingClasses()) {
239             /* Create a new MenuItem and add it to the popUp */
240             myMenu.addItem(new TethysUIGenericWrapper(myValue));
241         }
242     }
243 
244     /**
245      * Select static data.
246      * @param pStatic the static data
247      */
248     @SuppressWarnings("unchecked")
249     void selectStatic(final PrometheusStaticDataItem pStatic) {
250         getTable().selectRow((T) pStatic);
251     }
252 
253     /**
254      * Is the static table full?
255      * @return true/false
256      */
257     boolean isFull() {
258         return theStatic == null
259                 || theStatic.isFull();
260     }
261 
262     /**
263      * adjust showALL.
264      * @param pShow show disabled entries
265      */
266     void setShowAll(final boolean pShow) {
267         showAll = pShow;
268         cancelEditing();
269         getTable().setFilter(this::isFiltered);
270         theEnabledColumn.setVisible(showAll);
271         restoreSelected();
272     }
273 
274     @Override
275     protected boolean isFiltered(final T pRow) {
276         return super.isFiltered(pRow) && (pRow.getEnabled() || showAll);
277     }
