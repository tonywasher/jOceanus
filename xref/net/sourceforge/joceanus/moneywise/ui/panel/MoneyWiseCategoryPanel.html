170     /**
171      * Constructor.
172      * @param pView the data view
173      */
174     MoneyWiseCategoryPanel(final MoneyWiseView pView) {
175         /* Store details */
176         theView = pView;
177 
178         /* Access GUI Factory */
179         final TethysUIFactory<?> myFactory = pView.getGuiFactory();
180         final MetisViewerManager myViewer = pView.getViewerManager();
181 
182         /* Create the event manager */
183         theEventManager = new OceanusEventManager<>();
184 
185         /* Build the Update set */
186         theEditSet = new PrometheusEditSet(pView);
187 
188         /* Create the Panel */
189         final TethysUIPaneFactory myPanes = myFactory.paneFactory();
190         thePanel = myPanes.newBorderPane();
191 
192         /* Create the top level viewer entry for this view */
193         final MetisViewerEntry mySection = pView.getViewerEntry(PrometheusViewerEntryId.MAINTENANCE);
194         theViewerEntry = myViewer.newEntry(mySection, NLS_DATAENTRY);
195         theViewerEntry.setTreeObject(theEditSet);
196 
197         /* Create the error panel */
198         theError = pView.getToolkit().getToolkit().newErrorPanel(theViewerEntry);
199 
200         /* Create the action buttons panel */
201         theActionButtons = new PrometheusActionButtons(myFactory, theEditSet);
202 
203         /* Create the table panels */
204         theDepositTable = new MoneyWiseDepositCategoryTable(pView, theEditSet, theError);
205         theCashTable = new MoneyWiseCashCategoryTable(pView, theEditSet, theError);
206         theLoanTable = new MoneyWiseLoanCategoryTable(pView, theEditSet, theError);
207         theEventTable = new MoneyWiseTransCategoryTable(pView, theEditSet, theError);
208         theTagTable = new MoneyWiseTransTagTable(pView, theEditSet, theError);
209         theRegionTable = new MoneyWiseRegionTable(pView, theEditSet, theError);
210 
211         /* Create selection button and label */
212         final TethysUILabel myLabel = myFactory.controlFactory().newLabel(NLS_DATA);
213         theSelectButton = myFactory.buttonFactory().newScrollButton(PanelName.class);
214         buildSelectMenu();
215 
216         /* Create the card panel */
217         theCardPanel = myPanes.newCardPane();
218 
219         /* Add to the card panels */
220         theCardPanel.addCard(PanelName.DEPOSITS.toString(), theDepositTable);
221         theCardPanel.addCard(PanelName.CASH.toString(), theCashTable);
222         theCardPanel.addCard(PanelName.LOANS.toString(), theLoanTable);
223         theCardPanel.addCard(PanelName.EVENTS.toString(), theEventTable);
224         theCardPanel.addCard(PanelName.EVENTTAGS.toString(), theTagTable);
225         theCardPanel.addCard(PanelName.REGIONS.toString(), theRegionTable);
226         theActive = PanelName.DEPOSITS;
227         theSelectButton.setValue(theActive);
228 
229         /* Create the card panel */
230         theFilterCardPanel = myPanes.newCardPane();
231 
232         /* Add to the card panels */
233         theFilterCardPanel.addCard(PanelName.DEPOSITS.toString(), theDepositTable.getFilterPanel());
234         theFilterCardPanel.addCard(PanelName.CASH.toString(), theCashTable.getFilterPanel());
235         theFilterCardPanel.addCard(PanelName.LOANS.toString(), theLoanTable.getFilterPanel());
236         theFilterCardPanel.addCard(PanelName.EVENTS.toString(), theEventTable.getFilterPanel());
237         theFilterCardPanel.addCard(PanelName.EVENTTAGS.toString(), theTagTable.getFilterPanel());
238         theFilterCardPanel.addCard(PanelName.REGIONS.toString(), theRegionTable.getFilterPanel());
239 
240         /* Create the select prompt */
241         final TethysUIBoxPaneManager mySelect = myPanes.newHBoxPane();
242         mySelect.addNode(myLabel);
243         mySelect.addNode(theSelectButton);
244 
245         /* Create the selection panel */
246         theSelectPanel = myPanes.newBorderPane();
247         theSelectPanel.setBorderTitle(NLS_SELECT);
248         theSelectPanel.setWest(mySelect);
249         theSelectPanel.setCentre(theFilterCardPanel);
250 
251         /* Create the header panel */
252         final TethysUIBorderPaneManager myHeader = myPanes.newBorderPane();
253         myHeader.setCentre(theSelectPanel);
254         myHeader.setNorth(theError);
255         myHeader.setEast(theActionButtons);
256 
257         /* Now define the panel */
258         thePanel.setNorth(myHeader);
259         thePanel.setCentre(theCardPanel);
260 
261         /* Hide the action buttons initially */
262         theActionButtons.setVisible(false);
263 
264         /* Create the listeners */
265         theSelectButton.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> handleSelection());
266         theError.getEventRegistrar().addEventListener(e -> handleErrorPane());
267         theActionButtons.getEventRegistrar().addEventListener(this::handleActionButtons);
268         setChildListeners(theDepositTable.getEventRegistrar());
269         setChildListeners(theCashTable.getEventRegistrar());
270         setChildListeners(theLoanTable.getEventRegistrar());
271         setChildListeners(theEventTable.getEventRegistrar());
272         setChildListeners(theTagTable.getEventRegistrar());
273         setChildListeners(theRegionTable.getEventRegistrar());
274     }

286     /**

288      * @param pRegistrar the registrar
289      */
290     private void setChildListeners(final OceanusEventRegistrar<PrometheusDataEvent> pRegistrar) {
291         pRegistrar.addEventListener(PrometheusDataEvent.ADJUSTVISIBILITY, e -> {
292             if (!isRefreshing) {

294             }
295         });
296         pRegistrar.addEventListener(PrometheusDataEvent.GOTOWINDOW, this::handleGoToEvent);

325     /**
326      * Refresh data.
327      * @throws OceanusException on error
328      */
329     public void refreshData() throws OceanusException {
330         /* Obtain the active profile */
331         OceanusProfile myTask = theView.getActiveTask();
332         myTask = myTask.startTask("Categories");
333 
334         /* Note that we are refreshing */
335         isRefreshing = true;
336         theEditSet.setDataSet(theView.getData());
337 
338         /* Refresh the tables */
339         theDepositTable.refreshData();
340         theCashTable.refreshData();
341         theLoanTable.refreshData();
342         theEventTable.refreshData();
343         theTagTable.refreshData();
344         theRegionTable.refreshData();
345 
346         /* Clear refreshing flag */
347         isRefreshing = false;
348         setVisibility();
349 
350         /* Touch the updateSet */
351         theViewerEntry.setTreeObject(theEditSet);
352 
353         /* Complete the task */
354         myTask.end();
355     }

495      * Select category.
496      * @param pCategory the category to select

499         /* Determine which panel to show */
500         if (pCategory instanceof MoneyWiseDepositCategory myCategory) {
501             theDepositTable.selectCategory(myCategory);
502             showPanel(PanelName.DEPOSITS);

505             showPanel(PanelName.CASH);
506         } else if (pCategory instanceof MoneyWiseLoanCategory myCategory) {
507             theLoanTable.selectCategory(myCategory);
508             showPanel(PanelName.LOANS);
509         } else if (pCategory instanceof MoneyWiseTransCategory myCategory) {
510             theEventTable.selectCategory(myCategory);
511             showPanel(PanelName.EVENTS);
512         }
513     }

539     /**
540      * Show panel.
541      * @param pName the panel name
542      */
543     private void showPanel(final PanelName pName) {
544         /* Obtain name of panel */
545         final String myName = pName.toString();
546 
547         /* Move correct card to front */
548         theCardPanel.selectCard(myName);
549         theFilterCardPanel.selectCard(myName);
550 
551         /* Note the active panel */
552         theActive = pName;
553         theSelectButton.setFixedText(myName);
554 
555         /* Determine the focus */
556         determineFocus();
557     }

559     /**
560      * Set Visibility.
561      */
562     protected void setVisibility() {
563         /* Determine whether we have updates */
564         final boolean hasUpdates = hasUpdates();
565         final boolean isItemEditing = isItemEditing();
566 
567         /* Update the action buttons */
568         theActionButtons.setEnabled(true);
569         theActionButtons.setVisible(hasUpdates && !isItemEditing);
570 
571         /* Update the selection */
572         theSelectButton.setEnabled(!isItemEditing);
573         theFilterCardPanel.setEnabled(!isItemEditing);
574 
575         /* Alert listeners that there has been a change */
576         theEventManager.fireEvent(PrometheusDataEvent.ADJUSTVISIBILITY);
577     }

592     /**
593      * handleErrorPane.
594      */
595     private void handleErrorPane() {
596         /* Determine whether we have an error */
597         final boolean isError = theError.hasError();
598 
599         /* Hide selection panel on error */
600         theSelectPanel.setVisible(!isError);
601 
602         /* Lock card panel */

604 
605         /* Lock Action Buttons */
606         theActionButtons.setEnabled(!isError);
607     }

615

620     /**
621      * handle Action Buttons.

624     private void handleActionButtons(final OceanusEvent<PrometheusUIEvent> pEvent) {

626         cancelEditing();

629         theEditSet.processCommand(pEvent.getEventId(), theError);
630     }

632     /**
633      * handle GoTo Event.
634      * @param pEvent the event
635      */
636     private void handleGoToEvent(final OceanusEvent<PrometheusDataEvent> pEvent) {
637         /* Access details */
638         @SuppressWarnings("unchecked")
639         final PrometheusGoToEvent<MoneyWiseGoToId> myEvent = pEvent.getDetails(PrometheusGoToEvent.class);
640 
641         /* Access event and obtain details */
642         switch (myEvent.getId()) {
643             /* Pass through the event */
644             case STATEMENT:
645             case ACCOUNT:
646             case STATIC:
647                 theEventManager.cascadeEvent(pEvent);
648                 break;
649 
650             /* Access subPanels */
651             case CATEGORY:
652                 selectCategory(myEvent.getDetails());
653                 break;
654             case TAG:
655                 selectTag(pEvent.getDetails());
656                 break;
657             case REGION:
658                 selectRegion(pEvent.getDetails());
659                 break;
660             default:
661                 break;
662         }
663     }
