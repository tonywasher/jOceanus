57  /**
58   * MoneyWise SpotRates Table.
59   */
60  public class MoneyWiseMarketRatesTable
61          extends MoneyWiseBaseTable<MoneyWiseSpotExchangeRate> {
62      /**
63       * The SpotRates selection panel.
64       */
65      private final MoneyWiseSpotRatesSelect theSelect;
66  
67      /**
68       * The exchangeRates list.
69       */
70      private MoneyWiseSpotExchangeList theRates;
71  
72      /**
73       * The selected date.
74       */
75      private OceanusDate theDate;
76  
77      /**
78       * Constructor.
79       * @param pView the view
80       * @param pEditSet the editSet
81       * @param pError the error panel
82       */
83      MoneyWiseMarketRatesTable(final MoneyWiseView pView,
84                                final PrometheusEditSet pEditSet,
85                                final MetisErrorPanel pError) {
86          /* Store parameters */
87          super(pView, pEditSet, pError, MoneyWiseBasicDataType.EXCHANGERATE);
88  
89          /* Access Gui factory */
90          final TethysUIFactory<?> myGuiFactory = pView.getGuiFactory();
91          final TethysUITableManager<MetisDataFieldId, MoneyWiseSpotExchangeRate> myTable = getTable();
92  
93          /* Create new button */
94          final TethysUIButton myNewButton = myGuiFactory.buttonFactory().newButton();
95          MetisIcon.configureNewIconButton(myNewButton);
96  
97          /* Create a selection panel */
98          theSelect = new MoneyWiseSpotRatesSelect(myGuiFactory, pView);
99  
100         /* Set table configuration */
101         myTable.setDisabled(MoneyWiseSpotExchangeRate::isDisabled)
102                 .setComparator(MoneyWiseSpotExchangeRate::compareTo);
103 
104         /* Create the description column */
105         myTable.declareStringColumn(MoneyWiseBasicResource.XCHGRATE_FROM)
106                 .setCellValueFactory(r -> r.getToCurrency().getDesc())
107                 .setEditable(false)
108                 .setColumnWidth(WIDTH_NAME);
109 
110         /* Create the symbol column */
111         myTable.declareStringColumn(MoneyWiseBasicResource.XCHGRATE_TO)
112                 .setCellValueFactory(r -> r.getToCurrency().getName())
113                 .setName(MoneyWiseUIResource.SPOTRATE_COLUMN_SYMBOL.getValue())
114                 .setEditable(false)
115                 .setColumnWidth(WIDTH_NAME);
116 
117         /* Create the price column */
118         myTable.declareRatioColumn(MoneyWiseBasicResource.XCHGRATE_RATE)
119                 .setCellValueFactory(MoneyWiseSpotExchangeRate::getExchangeRate)
120                 .setEditable(true)
121                 .setCellEditable(r -> !r.isDisabled())
122                 .setColumnWidth(WIDTH_PRICE)
123                 .setOnCommit((r, v) -> updateField(MoneyWiseSpotExchangeRate::setExchangeRate, r, v));
124 
125         /* Create the previous rate column */
126         myTable.declareRatioColumn(MoneyWiseViewResource.SPOTRATE_PREVRATE)
127                 .setCellValueFactory(MoneyWiseSpotExchangeRate::getPrevRate)
128                 .setName(MoneyWiseViewResource.SPOTRATE_PREVRATE.getValue())
129                 .setEditable(false)
130                 .setColumnWidth(WIDTH_PRICE);
131 
132         /* Create the previous date column */
133         myTable.declareDateColumn(MoneyWiseViewResource.SPOTEVENT_PREVDATE)
134                 .setCellValueFactory(MoneyWiseSpotExchangeRate::getPrevDate)
135                 .setName(MoneyWiseViewResource.SPOTEVENT_PREVDATE.getValue())
136                 .setEditable(false)
137                 .setColumnWidth(WIDTH_DATE);
138 
139         /* Create the Active column */
140         final TethysUIIconMapSet<MetisAction> myActionMapSet = MetisIcon.configureStatusIconButton(myGuiFactory);
141         myTable.declareIconColumn(PrometheusDataResource.DATAITEM_TOUCH, MetisAction.class)
142                 .setIconMapSet(r -> myActionMapSet)
143                 .setCellValueFactory(r -> r.getExchangeRate() != null && !r.isDisabled() ? MetisAction.DELETE : MetisAction.DO)
144                 .setName(MoneyWiseUIResource.STATICDATA_ACTIVE.getValue())
145                 .setEditable(true)
146                 .setCellEditable(r -> r.getExchangeRate() != null && !r.isDisabled())
147                 .setColumnWidth(WIDTH_ICON)
148                 .setOnCommit((r, v) -> updateField(this::deleteRow, r, v));
149 
150         /* Add listeners */
151         theSelect.getEventRegistrar().addEventListener(PrometheusDataEvent.SELECTIONCHANGED, e -> handleNewSelection());
152         theSelect.getEventRegistrar().addEventListener(PrometheusDataEvent.DOWNLOAD, e -> downloadPrices());
153         pView.getEventRegistrar().addEventListener(e -> refreshData());
154         pEditSet.getEventRegistrar().addEventListener(e -> updateTableData());
155     }
156 
157     /**
158      * Obtain the selection panel.
159      * @return the select panel
160      */
161     MoneyWiseSpotRatesSelect getSelect() {
162         return theSelect;
163     }
164 
165     @Override
166     protected void refreshData() {
167         /* Obtain the active profile */
168         OceanusProfile myTask = getView().getActiveTask();
169         myTask = myTask.startTask("SpotRates1");
170 
171         /* Refresh the data */
172         theSelect.refreshData();
173 
174         /* Access the selection details */
175         setSelection(theSelect.getDate());
176 
177         /* Create SavePoint */
178         theSelect.createSavePoint();
179 
180         /* Complete the task */
181         myTask.end();
182     }
183 
184     /**
185      * Set Selection to the date.
186      * @param pDate the Date for the extract
187      */
188     public void setSelection(final OceanusDate pDate) {
189         /* Record selection */
190         theDate = pDate;
191 
192         /* If selection is valid */
193         if (theDate != null) {
194             /* Create the new list */
195             theRates = new MoneyWiseSpotExchangeList(getView(), pDate);
196 
197             /* Update Next/Previous values */
198             theSelect.setAdjacent(theRates.getPrev(), theRates.getNext());
199 
200             /* else invalid selection */
201         } else {
202             /* Set no selection */
203             theRates = null;
204             theSelect.setAdjacent(null, null);
205         }
206 
207         /* Update other details */
208         getTable().setItems(theRates.getUnderlyingList());
209         getEditEntry().setDataList(theRates);
210         theSelect.setEnabled(true);
211     }
212 
213     /**
214      * handle new selection.
215      */
216     private void handleNewSelection() {
217         /* Access selection */
218         final OceanusDate myDate = theSelect.getDate();
219 
220         /* If the selection differs */
221         if (!MetisDataDifference.isEqual(theDate, myDate)) {
222             /* Set selection */
223             setSelection(myDate);
224 
225             /* Create SavePoint */
226             theSelect.createSavePoint();
227         }
228     }
229

240     }

254     }
255 
256     /**
257      * Download prices.
258      */
259     private void downloadPrices() {
260         /* Cancel editing */
261         cancelEditing();
262 
263         /* Protect against exceptions */
264         try {
265             /* Download Rates */
266             if (MoneyWiseYQLDownloader.downloadRates(theRates)) {
267                 /* Increment data version */
268                 getEditSet().incrementVersion();
269 
270                 /* Update components to reflect changes */
271                 updateTableData();
272                 notifyChanges();
273             }
274         } catch (OceanusException e) {
275             /* Show the error */
276             setError(e);
277         }
278     }

280     /**
281      * SpotRates Panel.
282      */
283     public static class MoneyWiseSpotRatesPanel
284             implements TethysUIComponent, OceanusEventProvider<PrometheusDataEvent> {
285         /**
286          * Text for DataEntry Title.
287          */
288         private static final String NLS_DATAENTRY = MoneyWiseUIResource.RATES_DATAENTRY.getValue();
289 
290         /**
291          * The Event Manager.
292          */
293         private final OceanusEventManager<PrometheusDataEvent> theEventManager;
294 
295         /**
296          * The updateSet.
297          */
298         private final PrometheusEditSet theEditSet;
299 
300         /**
301          * The error panel.
302          */
303         private final MetisErrorPanel theError;
304 
305         /**
306          * The action buttons.
307          */
308         private final PrometheusActionButtons theActionButtons;
309 
310         /**
311          * The table.
312          */
313         private final MoneyWiseMarketRatesTable theTable;
314 
315         /**
316          * The panel.
317          */
318         private final TethysUIBorderPaneManager thePanel;
319 
320         /**
321          * The viewer entry.
322          */
323         private final MetisViewerEntry theViewerRate;
324 
325         /**
326          * Constructor.
327          *
328          * @param pView the data view
329          */
330         public MoneyWiseSpotRatesPanel(final MoneyWiseView pView) {
331             /* Build the Update set and entry */
332             theEditSet = new PrometheusEditSet(pView);
333 
334             /* Create the event manager */
335             theEventManager = new OceanusEventManager<>();
336 
337             /* Create the top level viewer entry for this view */
338             final MetisViewerEntry mySection = pView.getViewerEntry(PrometheusViewerEntryId.VIEW);
339             final MetisViewerManager myViewer = pView.getViewerManager();
340             theViewerRate = myViewer.newEntry(mySection, NLS_DATAENTRY);
341             theViewerRate.setTreeObject(theEditSet);
342 
343             /* Create the error panel for this view */
344             theError = pView.getToolkit().getToolkit().newErrorPanel(theViewerRate);
345 
346             /* Create the table */
347             theTable = new MoneyWiseMarketRatesTable(pView, theEditSet, theError);
348 
349             /* Create the action buttons */
350             final TethysUIFactory<?> myGuiFactory = pView.getGuiFactory();
351             theActionButtons = new PrometheusActionButtons(myGuiFactory, theEditSet);
352             theActionButtons.setVisible(false);
353 
354             /* Create the header panel */
355             final TethysUIPaneFactory myPanes = myGuiFactory.paneFactory();
356             final TethysUIBorderPaneManager myHeader = myPanes.newBorderPane();
357             myHeader.setCentre(theTable.getSelect());
358             myHeader.setNorth(theError);
359             myHeader.setEast(theActionButtons);
360 
361             /* Create the panel */
362             thePanel = myPanes.newBorderPane();
363             thePanel.setNorth(myHeader);
364             thePanel.setCentre(theTable);
365 
366             /* Add listeners */
367             theError.getEventRegistrar().addEventListener(e -> handleErrorPane());
368             theActionButtons.getEventRegistrar().addEventListener(this::handleActionButtons);
369             theTable.getEventRegistrar().addEventListener(e -> notifyChanges());
370         }
371 
372         @Override
373         public TethysUIComponent getUnderlying() {
374             return thePanel;
375         }
376 
377         @Override
378         public void setEnabled(final boolean pEnabled) {
379             thePanel.setEnabled(pEnabled);
380         }
381 
382         @Override
383         public void setVisible(final boolean pVisible) {
384             thePanel.setVisible(pVisible);
385         }
386 
387         @Override
388         public OceanusEventRegistrar<PrometheusDataEvent> getEventRegistrar() {
389             return theEventManager.getEventRegistrar();
390         }
391 
392         /**
393          * handleErrorPane.
394          */
395         private void handleErrorPane() {
396             /* Determine whether we have an error */
397             final boolean isError = theError.hasError();
398 
399             /* Hide selection panel on error */
400             theTable.getSelect().setVisible(!isError);
401 
402             /* Lock scroll area */
403             theTable.setEnabled(!isError);
404 
405             /* Lock Action Buttons */
406             theActionButtons.setEnabled(!isError);
407         }
408 
409         /**
410          * handle Action Buttons.
411          * @param pEvent the event
412          */
413         private void handleActionButtons(final OceanusEvent<PrometheusUIEvent> pEvent) {
414             /* Cancel editing */
415             theTable.cancelEditing();
416 
417             /* Perform the command */
418             theEditSet.processCommand(pEvent.getEventId(), theError);
419 
420             /* Adjust for changes */
421             theTable.notifyChanges();
422         }
423 
424         /**
425          * Determine Focus.
426          */
427         public void determineFocus() {
428             /* Request the focus */
429             theTable.determineFocus();
430 
431             /* Focus on the Data entry */
432             theViewerRate.setFocus();
433         }
434 
435         /**
436          * Call underlying controls to take notice of changes in view/selection.
437          */
438         private void notifyChanges() {
439             /* Determine whether we have updates */
440             final boolean hasUpdates = hasUpdates();
441 
442             /* Update the table buttons */
443             theActionButtons.setEnabled(true);
444             theActionButtons.setVisible(hasUpdates);
445 
446             /* Notify listeners */
447             theEventManager.fireEvent(PrometheusDataEvent.ADJUSTVISIBILITY);
448         }
449 
450         /**
451          * Does the panel have updates?
452          * @return true/false
453          */
454         public boolean hasUpdates() {
455             return theTable.hasUpdates();
456         }
457 
458         /**
459          * Does the panel have a session?
460          * @return true/false
461          */
462         public boolean hasSession() {
463             return theTable.hasUpdates();
464         }
465 
466         /**
467          * Does the panel have errors?
468          * @return true/false
469          */
470         public boolean hasErrors() {
471             return theTable.hasErrors();
472         }
473     }
