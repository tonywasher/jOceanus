18  
19  import net.sourceforge.joceanus.themis.xanalysis.parser.base.ThemisXAnalysisInstance.ThemisXAnalysisNodeInstance;
20  import net.sourceforge.joceanus.themis.xanalysis.parser.base.ThemisXAnalysisInstance.ThemisXAnalysisTypeInstance;
21  import net.sourceforge.joceanus.themis.xanalysis.parser.decl.ThemisXAnalysisDeclField;
22  import net.sourceforge.joceanus.themis.xanalysis.parser.expr.ThemisXAnalysisExprVarDecl;
23  import net.sourceforge.joceanus.themis.xanalysis.parser.node.ThemisXAnalysisNodeParameter;
24  import net.sourceforge.joceanus.themis.xanalysis.parser.node.ThemisXAnalysisNodeVariable;
25  import net.sourceforge.joceanus.themis.xanalysis.parser.type.ThemisXAnalysisTypeParameter;
26  
27  import java.util.HashMap;
28  import java.util.List;
29  import java.util.Map;
30  
31  /**
32   * Data Map.
33   */
34  public class ThemisXAnalysisSolverDataMap {
35      /**
36       * Parent map.
37       */
38      private final ThemisXAnalysisSolverDataMap theParent;
39  
40      /**
41       * The named variables map.
42       */
43      private final Map<String, ThemisXAnalysisTypeInstance> theNames;
44  
45      /**
46       * The type variables map.
47       */
48      private final Map<String, ThemisXAnalysisTypeParameter> theTypes;
49  
50      /**
51       * Constructor.
52       */
53      ThemisXAnalysisSolverDataMap() {
54          this(null);
55      }
56  
57      /**
58       * Constructor.
59       * @param pParent the parent
60       */
61      ThemisXAnalysisSolverDataMap(final ThemisXAnalysisSolverDataMap pParent) {
62          theParent = pParent;
63          theNames = new HashMap<>();
64          theTypes = new HashMap<>();
65      }
66  
67      /**
68       * Declare field.
69       * @param pField the field
70       */
71      void declareField(final ThemisXAnalysisDeclField pField) {
72          for (ThemisXAnalysisNodeInstance myNode : pField.getVariables()) {
73              final ThemisXAnalysisNodeVariable myVar = (ThemisXAnalysisNodeVariable) myNode;
74              theNames.put(myVar.getName().toString(), myVar.getType());
75          }
76      }
77  
78      /**
79       * Declare variables.
80       * @param pVar the variable declarator
81       */
82      void declareVars(final ThemisXAnalysisExprVarDecl pVar) {
83          for (ThemisXAnalysisNodeInstance myNode : pVar.getVariables()) {
84              final ThemisXAnalysisNodeVariable myVar = (ThemisXAnalysisNodeVariable) myNode;
85              theNames.put(myVar.getName().toString(), myVar.getType());
86          }
87      }
88  
89      /**
90       * Declare parameters.
91       * @param pParams the parameters
92       */
93      void declareParams(final List<ThemisXAnalysisNodeInstance> pParams) {
94          for (ThemisXAnalysisNodeInstance myNode : pParams) {
95              final ThemisXAnalysisNodeParameter myParam = (ThemisXAnalysisNodeParameter) myNode;
96              theNames.put(myParam.getName().toString(), myParam.getType());
97          }
98      }
99  
100     /**
101      * Declare type parameters.
102      * @param pParams the parameters
103      */
104     void declareTypeParams(final List<ThemisXAnalysisTypeInstance> pParams) {
105         for (ThemisXAnalysisTypeInstance myNode : pParams) {
106             final ThemisXAnalysisTypeParameter myParam = (ThemisXAnalysisTypeParameter) myNode;
107             theTypes.put(myParam.getName(), myParam);
108         }
109     }
110 
111     /**
112      * Look up name.
113      * @param pName the name of the variable
114      * @return the type
115      */
116     ThemisXAnalysisTypeInstance lookUpName(final String pName) {
117         /* Look up name in map */
118         final ThemisXAnalysisTypeInstance myType = theNames.get(pName);
119         if (myType != null) {
120             return myType;
121         }
122 
123         /* If we did not find the name, try the parent map */
124         return theParent == null ? null : theParent.lookUpName(pName);
125     }
126 
127     /**
128      * Look up type.
129      * @param pName the name of the type
130      * @return the type
131       */
132     ThemisXAnalysisTypeInstance lookUpType(final String pName) {
133         /* Look up name in map */
134         final ThemisXAnalysisTypeInstance myType = theTypes.get(pName);
135         if (myType != null) {
136             return myType;
137         }
138 
139         /* If we did not find the type, try the parent map */
140         return theParent == null ? null : theParent.lookUpType(pName);
141     }
142 }
