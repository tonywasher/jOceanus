41  public final class ThemisXAnalysisStatsStmtStmt {

44       */

48      /**
49       * Obtain the statement count for a statement.
50       *
51       * @param pCounter the counter
52       * @param pStatement the statement
53       * @return the count
54       */
55      static int count(final ThemisXAnalysisStatsStmtCounter pCounter,
56                       final ThemisXAnalysisStatementInstance pStatement) {
57          /* Handle null statement */
58          if (pStatement == null) {
59              return pCounter.fixedCount(0);
60          }
61  
62          /* Switch on statement id */
63          switch ((ThemisXAnalysisStatement) pStatement.getId()) {
64              case ASSERT:       return countAssert(pCounter, (ThemisXAnalysisStmtAssert) pStatement);
65              case BLOCK:        return countBlock(pCounter, (ThemisXAnalysisStmtBlock) pStatement);
66              case CONSTRUCTOR:  return countConstructor(pCounter, (ThemisXAnalysisStmtConstructor) pStatement);
67              case DO:           return countDo(pCounter, (ThemisXAnalysisStmtDo) pStatement);
68              case EXPRESSION:   return countExpression(pCounter, (ThemisXAnalysisStmtExpression) pStatement);
69              case FOR:          return countFor(pCounter, (ThemisXAnalysisStmtFor) pStatement);
70              case FOREACH:      return countForEach(pCounter, (ThemisXAnalysisStmtForEach) pStatement);
71              case IF:           return countIf(pCounter, (ThemisXAnalysisStmtIf) pStatement);
72              case LABELED:      return countLabeled(pCounter, (ThemisXAnalysisStmtLabeled) pStatement);
73              case RETURN:       return countReturn(pCounter, (ThemisXAnalysisStmtReturn) pStatement);
74              case SWITCH:       return countSwitch(pCounter, (ThemisXAnalysisStmtSwitch) pStatement);
75              case SYNCHRONIZED: return countSynch(pCounter, (ThemisXAnalysisStmtSynch) pStatement);
76              case THROW:        return countThrow(pCounter, (ThemisXAnalysisStmtThrow) pStatement);
77              case TRY:          return countTry(pCounter, (ThemisXAnalysisStmtTry) pStatement);
78              case WHILE:        return countWhile(pCounter, (ThemisXAnalysisStmtWhile) pStatement);
79              case YIELD:        return countYield(pCounter, (ThemisXAnalysisStmtYield) pStatement);
80              case BREAK:
81              case CONTINUE:     return pCounter.fixedCount(1);
82              case EMPTY:
83              case LOCALCLASS:
84              case LOCALRECORD:
85              default:           return pCounter.fixedCount(0);
86          }
87      }
88  
89      /**
90       * Obtain the statement count for an ASSERT statement.
91       *
92       * @param pCounter the counter
93       * @param pAssert the statement
94       * @return the count
95       */
96      private static int countAssert(final ThemisXAnalysisStatsStmtCounter pCounter,
97                                     final ThemisXAnalysisStmtAssert pAssert) {
98          int myCount = pCounter.countExpr(pAssert.getCheck());
99          myCount += pCounter.countExpr(pAssert.getMessage());
100         return myCount;
101     }
102 
103     /**
104      * Obtain the statement count for a BLOCK statement.
105      *
106      * @param pCounter the counter
107      * @param pBlock the statement
108      * @return the count
109      */
110     private static int countBlock(final ThemisXAnalysisStatsStmtCounter pCounter,
111                                   final ThemisXAnalysisStmtBlock pBlock) {
112         return pCounter.countStmtList(pBlock.getBody());
113     }
114 
115     /**
116      * Obtain the statement count for a CONSTRUCTOR statement.
117      *
118      * @param pCounter the counter
119      * @param pConstructor the statement
120      * @return the count
121      */
122     private static int countConstructor(final ThemisXAnalysisStatsStmtCounter pCounter,
123                                         final ThemisXAnalysisStmtConstructor pConstructor) {
124         return pCounter.countExpr(pConstructor.getExpression())
125                 + pCounter.countExprList(pConstructor.getArguments());
126     }
127 
128     /**
129      * Obtain the statement count for a DO statement.
130      *
131      * @param pCounter the counter
132      * @param pDo the statement
133      * @return the count
134      */
135     private static int countDo(final ThemisXAnalysisStatsStmtCounter pCounter,
136                                final ThemisXAnalysisStmtDo pDo) {
137         return pCounter.countExpr(pDo.getCondition())
138                 + pCounter.countStmt(pDo.getBody());
139     }
140 
141     /**
142      * Obtain the statement count for an EXPRESSION statement.
143      *
144      * @param pCounter the counter
145      * @param pExpression the statement
146      * @return the count
147      */
148     private static int countExpression(final ThemisXAnalysisStatsStmtCounter pCounter,
149                                        final ThemisXAnalysisStmtExpression pExpression) {
150         return pCounter.countExpr(pExpression.getExpression());
151     }
152 
153     /**
154      * Obtain the statement count for a FOR statement.
155      *
156      * @param pCounter the counter
157      * @param pFor the statement
158      * @return the count
159      */
160     private static int countFor(final ThemisXAnalysisStatsStmtCounter pCounter,
161                                 final ThemisXAnalysisStmtFor pFor) {
162         return pCounter.countExprList(pFor.getInit())
163                 + pCounter.countExpr(pFor.getCompare())
164                 + pCounter.countExprList(pFor.getUpdates())
165                 + pCounter.countStmt(pFor.getBody());
166     }
167 
168     /**
169      * Obtain the statement count for a FOREACH statement.
170      *
171      * @param pCounter the counter
172      * @param pFor the statement
173      * @return the count
174      */
175     private static int countForEach(final ThemisXAnalysisStatsStmtCounter pCounter,
176                                     final ThemisXAnalysisStmtForEach pFor) {
177         return pCounter.countNode(pFor.getVariableDeclarator())
178                 + pCounter.countStmt(pFor.getBody());
179     }
180 
181     /**
182      * Obtain the statement count for an IF statement.
183      *
184      * @param pCounter the counter
185      * @param pIf the statement
186      * @return the count
187      */
188     private static int countIf(final ThemisXAnalysisStatsStmtCounter pCounter,
189                                final ThemisXAnalysisStmtIf pIf) {
190         return pCounter.countExpr(pIf.getCondition())
191                 + pCounter.countStmt(pIf.getThen())
192                 + pCounter.countStmt(pIf.getElse());
193     }
194 
195     /**
196      * Obtain the statement count for a LABELED statement.
197      *
198      * @param pCounter the counter
199      * @param pLabeled the statement
200      * @return the count
201      */
202     private static int countLabeled(final ThemisXAnalysisStatsStmtCounter pCounter,
203                                     final ThemisXAnalysisStmtLabeled pLabeled) {
204         return pCounter.countStmt(pLabeled.getLabeled());
205     }

207     /**
208      * Obtain the statement count for a RETURN statement.
209      *
210      * @param pCounter the counter
211      * @param pReturn the statement
212      * @return the count
213      */
214     private static int countReturn(final ThemisXAnalysisStatsStmtCounter pCounter,
215                                    final ThemisXAnalysisStmtReturn pReturn) {
216         return pCounter.fixedCount(1)
217                 + pCounter.countExpr(pReturn.getExpression());
218     }

227     private static int countSwitch(final ThemisXAnalysisStatsStmtCounter pCounter,
228                                    final ThemisXAnalysisStmtSwitch pSwitch) {
229         return pCounter.countExpr(pSwitch.getSelector())
230                 + pCounter.countNodeList(pSwitch.getCases());
231     }

236      * @param pCounter the counter

240     private static int countSynch(final ThemisXAnalysisStatsStmtCounter pCounter,

242         return pCounter.countExpr(pSync.getSynched())
243                 + pCounter.countStmt(pSync.getBody());
244     }

246     /**
247      * Obtain the statement count for a THROW statement.
248      *
249      * @param pCounter the counter
250      * @param pThrow the statement
251      * @return the count
252      */
253     private static int countThrow(final ThemisXAnalysisStatsStmtCounter pCounter,

255         return pCounter.fixedCount(1)
256                 + pCounter.countExpr(pThrow.getThrown());
257     }
258 
259     /**
260      * Obtain the statement count for a TRY statement.
261      *
262      * @param pCounter the counter
263      * @param pTry the statement
264      * @return the count
265      */
266     private static int countTry(final ThemisXAnalysisStatsStmtCounter pCounter,
267                                 final ThemisXAnalysisStmtTry pTry) {
268         return pCounter.countExprList(pTry.getResources())
269                 + pCounter.countStmt(pTry.getTry())
270                 + pCounter.countNodeList(pTry.getCatches())
271                 + pCounter.countStmt(pTry.getFinally());
272     }

274     /**
275      * Obtain the statement count for a WHILE statement.
276      *
277      * @param pCounter the counter
278      * @param pWhile the statement

280      */
281     private static int countWhile(final ThemisXAnalysisStatsStmtCounter pCounter,

283         return pCounter.countExpr(pWhile.getCondition())
284                 + pCounter.countStmt(pWhile.getBody());
285     }

288      * Obtain the statement count for a YIELD statement.

294     private static int countYield(final ThemisXAnalysisStatsStmtCounter pCounter,
295                                   final ThemisXAnalysisStmtYield pYield) {
296         return pCounter.fixedCount(1)
297                 + pCounter.countExpr(pYield.getExpression());
298     }
