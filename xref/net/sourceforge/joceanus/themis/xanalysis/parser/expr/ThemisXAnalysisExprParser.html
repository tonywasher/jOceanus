17  package net.sourceforge.joceanus.themis.xanalysis.parser.expr;
18  
19  import com.github.javaparser.ast.expr.Expression;
20  import net.sourceforge.joceanus.oceanus.base.OceanusException;
21  import net.sourceforge.joceanus.themis.xanalysis.parser.base.ThemisXAnalysisInstance.ThemisXAnalysisExpressionInstance;
22  import net.sourceforge.joceanus.themis.xanalysis.parser.base.ThemisXAnalysisParserDef;
23  
24  /**
25   * Analysis Expression Parser.
26   */
27  public final class ThemisXAnalysisExprParser {
28      /**
29       * Private Constructor.
30       */
31      private ThemisXAnalysisExprParser() {
32      }
33  
34      /**
35       * Parse an expression.
36       * @param pParser the parser
37       * @param pExpr the expression
38       * @return the parsed expression
39       * @throws OceanusException on error
40       */
41      public static ThemisXAnalysisExpressionInstance parseExpression(final ThemisXAnalysisParserDef pParser,
42                                                                      final Expression pExpr) throws OceanusException {
43          /* Handle null Expression */
44          if (pExpr == null) {
45              return null;
46          }
47  
48          /* Allocate correct Expression */
49          switch (ThemisXAnalysisExpression.determineExpression(pParser, pExpr)) {
50              case ARRAYACCESS:     return new ThemisXAnalysisExprArrayAccess(pParser, pExpr.asArrayAccessExpr());
51              case ARRAYCREATION:   return new ThemisXAnalysisExprArrayCreation(pParser, pExpr.asArrayCreationExpr());
52              case ARRAYINIT:       return new ThemisXAnalysisExprArrayInit(pParser, pExpr.asArrayInitializerExpr());
53              case ASSIGN:          return new ThemisXAnalysisExprAssign(pParser, pExpr.asAssignExpr());
54              case BINARY:          return new ThemisXAnalysisExprBinary(pParser, pExpr.asBinaryExpr());
55              case BOOLEAN:         return new ThemisXAnalysisExprBooleanLit(pParser, pExpr.asBooleanLiteralExpr());
56              case CAST:            return new ThemisXAnalysisExprCast(pParser, pExpr.asCastExpr());
57              case CHAR:            return new ThemisXAnalysisExprCharLit(pParser, pExpr.asCharLiteralExpr());
58              case CLASS:           return new ThemisXAnalysisExprClass(pParser, pExpr.asClassExpr());
59              case CONDITIONAL:     return new ThemisXAnalysisExprConditional(pParser, pExpr.asConditionalExpr());
60              case DOUBLE:          return new ThemisXAnalysisExprDoubleLit(pParser, pExpr.asDoubleLiteralExpr());
61              case ENCLOSED:        return new ThemisXAnalysisExprEnclosed(pParser, pExpr.asEnclosedExpr());
62              case FIELDACCESS:     return new ThemisXAnalysisExprFieldAccess(pParser, pExpr.asFieldAccessExpr());
63              case INSTANCEOF:      return new ThemisXAnalysisExprInstanceOf(pParser, pExpr.asInstanceOfExpr());
64              case INTEGER:         return new ThemisXAnalysisExprIntegerLit(pParser, pExpr.asIntegerLiteralExpr());
65              case LAMBDA:          return new ThemisXAnalysisExprLambda(pParser, pExpr.asLambdaExpr());
66              case LONG:            return new ThemisXAnalysisExprLongLit(pParser, pExpr.asLongLiteralExpr());
67              case MARKER:          return new ThemisXAnalysisExprMarkerAnnotation(pParser, pExpr.asMarkerAnnotationExpr());
68              case METHODCALL:      return new ThemisXAnalysisExprMethodCall(pParser, pExpr.asMethodCallExpr());
69              case METHODREFERENCE: return new ThemisXAnalysisExprMethodRef(pParser, pExpr.asMethodReferenceExpr());
70              case NAME:            return new ThemisXAnalysisExprName(pParser, pExpr.asNameExpr());
71              case NORMAL:          return new ThemisXAnalysisExprNormalAnnotation(pParser, pExpr.asNormalAnnotationExpr());
72              case NULL:            return new ThemisXAnalysisExprNullLit(pParser, pExpr.asNullLiteralExpr());
73              case OBJECTCREATE:    return new ThemisXAnalysisExprObjectCreate(pParser, pExpr.asObjectCreationExpr());
74              case RECORDPATTERN:   return new ThemisXAnalysisExprRecordPattern(pParser, pExpr.asRecordPatternExpr());
75              case SINGLEMEMBER:    return new ThemisXAnalysisExprSingleMemberAnnotation(pParser, pExpr.asSingleMemberAnnotationExpr());
76              case STRING:          return new ThemisXAnalysisExprStringLit(pParser, pExpr.asStringLiteralExpr());
77              case SUPER:           return new ThemisXAnalysisExprSuper(pParser, pExpr.asSuperExpr());
78              case SWITCH:          return new ThemisXAnalysisExprSwitch(pParser, pExpr.asSwitchExpr());
79              case TEXTBLOCK:       return new ThemisXAnalysisExprTextBlockLit(pParser, pExpr.asTextBlockLiteralExpr());
80              case THIS:            return new ThemisXAnalysisExprThis(pParser, pExpr.asThisExpr());
81              case TYPE:            return new ThemisXAnalysisExprType(pParser, pExpr.asTypeExpr());
82              case TYPEPATTERN:     return new ThemisXAnalysisExprTypePattern(pParser, pExpr.asTypePatternExpr());
83              case UNARY:           return new ThemisXAnalysisExprUnary(pParser, pExpr.asUnaryExpr());
84              case VARIABLE:        return new ThemisXAnalysisExprVarDecl(pParser, pExpr.asVariableDeclarationExpr());
85              default:              throw pParser.buildException("Unsupported Expression Type", pExpr);
86          }
87      }
88  }
