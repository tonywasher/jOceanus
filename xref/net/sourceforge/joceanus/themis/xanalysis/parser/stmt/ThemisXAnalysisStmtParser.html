17  package net.sourceforge.joceanus.themis.xanalysis.parser.stmt;
18  
19  import com.github.javaparser.ast.stmt.Statement;
20  import net.sourceforge.joceanus.oceanus.base.OceanusException;
21  import net.sourceforge.joceanus.themis.xanalysis.parser.base.ThemisXAnalysisInstance.ThemisXAnalysisStatementInstance;
22  import net.sourceforge.joceanus.themis.xanalysis.parser.base.ThemisXAnalysisParserDef;
23  
24  /**
25   * Analysis Statement Parser.
26   */
27  public final class ThemisXAnalysisStmtParser {
28      /**
29       * Private Constructor.
30       */
31      private ThemisXAnalysisStmtParser() {
32      }
33  
34      /**
35       * Parse a statement.
36       * @param pParser the parser
37       * @param pStatement the statement
38       * @return the parsed statement
39       * @throws OceanusException on error
40       */
41      public static ThemisXAnalysisStatementInstance parseStatement(final ThemisXAnalysisParserDef pParser,
42                                                                    final Statement pStatement) throws OceanusException {
43          /* Handle null Statement */
44          if (pStatement == null) {
45              return null;
46          }
47  
48          /* Allocate correct Statement */
49          switch (ThemisXAnalysisStatement.determineStatement(pParser, pStatement)) {
50              case ASSERT:       return new ThemisXAnalysisStmtAssert(pParser, pStatement.asAssertStmt());
51              case BLOCK:        return new ThemisXAnalysisStmtBlock(pParser, pStatement.asBlockStmt());
52              case BREAK:        return new ThemisXAnalysisStmtBreak(pParser, pStatement.asBreakStmt());
53              case CONSTRUCTOR:  return new ThemisXAnalysisStmtConstructor(pParser, pStatement.asExplicitConstructorInvocationStmt());
54              case CONTINUE:     return new ThemisXAnalysisStmtContinue(pParser, pStatement.asContinueStmt());
55              case DO:           return new ThemisXAnalysisStmtDo(pParser, pStatement.asDoStmt());
56              case EMPTY:        return new ThemisXAnalysisStmtEmpty(pParser, pStatement.asEmptyStmt());
57              case EXPRESSION:   return new ThemisXAnalysisStmtExpression(pParser, pStatement.asExpressionStmt());
58              case FOR:          return new ThemisXAnalysisStmtFor(pParser, pStatement.asForStmt());
59              case FOREACH:      return new ThemisXAnalysisStmtForEach(pParser, pStatement.asForEachStmt());
60              case IF:           return new ThemisXAnalysisStmtIf(pParser, pStatement.asIfStmt());
61              case LABELED:      return new ThemisXAnalysisStmtLabeled(pParser, pStatement.asLabeledStmt());
62              case LOCALCLASS:   return new ThemisXAnalysisStmtClass(pParser, pStatement.asLocalClassDeclarationStmt());
63              case LOCALRECORD:  return new ThemisXAnalysisStmtRecord(pParser, pStatement.asLocalRecordDeclarationStmt());
64              case RETURN:       return new ThemisXAnalysisStmtReturn(pParser, pStatement.asReturnStmt());
65              case SWITCH:       return new ThemisXAnalysisStmtSwitch(pParser, pStatement.asSwitchStmt());
66              case SYNCHRONIZED: return new ThemisXAnalysisStmtSynch(pParser, pStatement.asSynchronizedStmt());
67              case THROW:        return new ThemisXAnalysisStmtThrow(pParser, pStatement.asThrowStmt());
68              case TRY:          return new ThemisXAnalysisStmtTry(pParser, pStatement.asTryStmt());
69              case WHILE:        return new ThemisXAnalysisStmtWhile(pParser, pStatement.asWhileStmt());
70              case YIELD:        return new ThemisXAnalysisStmtYield(pParser, pStatement.asYieldStmt());
71              default:           throw pParser.buildException("Unsupported Statement Type", pStatement);
72          }
73      }
74  }
