17  package net.sourceforge.joceanus.themis.xanalysis.parser.type;
18  
19  import com.github.javaparser.ast.type.Type;
20  import net.sourceforge.joceanus.oceanus.base.OceanusException;
21  import net.sourceforge.joceanus.themis.xanalysis.parser.base.ThemisXAnalysisInstance.ThemisXAnalysisId;
22  import net.sourceforge.joceanus.themis.xanalysis.parser.base.ThemisXAnalysisParserDef;
23  
24  import java.util.function.Predicate;
25  
26  /**
27   * Analysis TypeType.
28   */
29  public enum ThemisXAnalysisType
30          implements ThemisXAnalysisId {
31      /**
32       * Array.
33       */
34      ARRAY(Type::isArrayType),
35  
36      /**
37       * Class/Interface.
38       */
39      CLASSINTERFACE(Type::isClassOrInterfaceType),
40  
41      /**
42       * Intersection.
43       */
44      INTERSECTION(Type::isIntersectionType),
45  
46      /**
47       * TypeParameter.
48       */
49      PARAMETER(Type::isTypeParameter),
50  
51      /**
52       * Primitive.
53       */
54      PRIMITIVE(Type::isPrimitiveType),
55  
56      /**
57       * Union.
58       */
59      UNION(Type::isUnionType),
60  
61      /**
62       * Unknown.
63       */
64      UNKNOWN(Type::isUnknownType),
65  
66      /**
67       * Var.
68       */
69      VAR(Type::isVarType),
70  
71      /**
72       * Void.
73       */
74      VOID(Type::isVoidType),
75  
76      /**
77       * Wildcard.
78       */
79      WILDCARD(Type::isWildcardType);
80  
81      /**
82       * The test.
83       */
84      private final Predicate<Type> theTester;
85  
86      /**
87       * Constructor.
88       * @param pTester the test method
89       */
90      ThemisXAnalysisType(final Predicate<Type> pTester) {
91          theTester = pTester;
92      }
93  
94      /**
95       * Determine type of type.
96       * @param pParser the parser
97       * @param pType the type
98       * @return the typeType
99       * @throws OceanusException on error
100      */
101     public static ThemisXAnalysisType determineType(final ThemisXAnalysisParserDef pParser,
102                                                     final Type pType) throws OceanusException {
103         /* Loop testing each type */
104         for (ThemisXAnalysisType myType : values()) {
105             if (myType.theTester.test(pType)) {
106                 return myType;
107             }
108         }
109 
110         /* Unrecognised Type */
111         throw pParser.buildException("Unexpected Type", pType);
112     }
113 }
