17  package net.sourceforge.joceanus.themis.lethe.analysis;
18  
19  import java.util.ArrayDeque;
20  import java.util.Deque;
21  
22  import net.sourceforge.joceanus.oceanus.base.OceanusException;
23  import net.sourceforge.joceanus.themis.exc.ThemisDataException;
24  import net.sourceforge.joceanus.themis.lethe.analysis.ThemisAnalysisScanner.ThemisAnalysisSource;
25  
26  /**
27   * Stack of Analysis Elements.
28   */
29  public class ThemisAnalysisStack
30      implements ThemisAnalysisSource {
31      /**
32       * The stack of lines.
33       */
34      private final Deque<ThemisAnalysisElement> theStack;
35  
36      /**
37       * Constructor.
38       * @param pStack the stack
39       */
40      ThemisAnalysisStack(final Deque<ThemisAnalysisElement> pStack) {
41          theStack = pStack;
42      }
43  
44      /**
45       * Constructor.
46       * @param pElement the element
47       */
48      ThemisAnalysisStack(final ThemisAnalysisElement pElement) {
49          theStack = new ArrayDeque<>();
50          theStack.add(pElement);
51      }
52  
53      @Override
54      public boolean hasLines() {
55          return !theStack.isEmpty();
56      }
57  
58      @Override
59      public ThemisAnalysisElement popNextLine() throws OceanusException {
60          /* Check that there is a line to pop */
61          if (theStack.isEmpty()) {
62              throw new ThemisDataException("No more lines");
63          }
64  
65          /* Access the first line and remove from the list */
66          return theStack.removeFirst();
67      }
68  
69      @Override
70      public void pushLine(final ThemisAnalysisElement pLine) {
71          /* Insert the line at the front of the stack */
72          theStack.offerFirst(pLine);
73      }
74  
75      /**
76       * Is the stack empty?
77       * @return true/false
78       */
79      public boolean isEmpty() {
80          return theStack.isEmpty()
81                  || (theStack.size() == 1 && theStack.peekFirst().toString().length() == 0);
82      }
83  
84      /**
85       * Obtain the last line in the stack.
86       * @return the line
87       */
88      ThemisAnalysisElement peekLastLine() {
89          return theStack.peekLast();
90      }
91  
92      /**
93       * Obtain the size of the stack.
94       * @return the size
95       */
96      public int size() {
97          return theStack.size();
98      }
99  
100     /**
101      * Strip parentheses.
102      * @return a new stack with stripped parentheses
103      * @throws OceanusException on error
104      */
105     public ThemisAnalysisStack extractParentheses() throws OceanusException {
106         /* Extract the parenthesised block */
107         final ThemisAnalysisLine myLine = (ThemisAnalysisLine) popNextLine();
108         final ThemisAnalysisScanner myScanner = new ThemisAnalysisScanner(this);
109         final Deque<ThemisAnalysisElement> myParenthesised = myScanner.scanForParenthesis(myLine);
110         final ThemisAnalysisStack myResult = new ThemisAnalysisStack(myParenthesised);
111 
112         /* Strip the parentheses */
113         myResult.stripStartChar(ThemisAnalysisChar.PARENTHESIS_OPEN);
114         myResult.stripEndChar(ThemisAnalysisChar.PARENTHESIS_CLOSE);
115 
116         /* Return the result */
117         return myResult;
118     }
119 
120     /**
121      * Strip parentheses.
122      * @return a new stack with stripped parentheses
123      * @throws OceanusException on error
124      */
125     public ThemisAnalysisStack extractArray() throws OceanusException {
126         /* Extract the parenthesised block */
127         final ThemisAnalysisLine myLine = (ThemisAnalysisLine) popNextLine();
128         final ThemisAnalysisScanner myScanner = new ThemisAnalysisScanner(this);
129         final Deque<ThemisAnalysisElement> myArray = myScanner.scanForArray(myLine);
130         final ThemisAnalysisStack myResult = new ThemisAnalysisStack(myArray);
131 
132         /* Strip the parentheses */
133         myResult.stripStartChar(ThemisAnalysisChar.ARRAY_OPEN);
134         myResult.stripEndChar(ThemisAnalysisChar.ARRAY_CLOSE);
135 
136         /* Return the result */
137         return myResult;
138     }
139 
140     /**
141      * Does the stack start with this character?
142      * @param pChar the character
143      * @return true/false
144      */
145     public boolean startsWithChar(final char pChar) {
146         return !theStack.isEmpty()
147                 && ((ThemisAnalysisLine) theStack.peekFirst()).startsWithChar(pChar);
148     }
149 
150     /**
151      * Strip the starting character.
152      * @param pChar the character
153      */
154     public void stripStartChar(final char pChar) {
155         if (!theStack.isEmpty()) {
156             ((ThemisAnalysisLine) theStack.peekFirst()).stripStartChar(pChar);
157         }
158     }
159 
160     /**
161      * Strip the ending character.
162      * @param pChar the character
163      */
164     public void stripEndChar(final char pChar) {
165         if (!theStack.isEmpty()) {
166             ((ThemisAnalysisLine) theStack.peekLast()).stripEndChar(pChar);
167         }
168     }
169 
170     /**
171      * Rebuild the stack.
172      * @param pSource the source to rebuild from
173      */
174     public void rebuild(final Deque<ThemisAnalysisElement> pSource) {
175         /* Clear the existing stack */
176         theStack.clear();
177 
178         /* Loop through the source */
179         for (ThemisAnalysisElement myElement : pSource) {
180             theStack.offerLast(myElement);
181         }
182     }
183 
184     @Override
185     public String toString() {
186         return ThemisAnalysisBuilder.formatLines(theStack);
187     }
188 }
