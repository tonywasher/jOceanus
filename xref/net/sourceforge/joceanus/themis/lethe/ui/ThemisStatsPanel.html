19  import java.util.Iterator;
20  import java.util.concurrent.atomic.AtomicInteger;
21

23  import net.sourceforge.joceanus.oceanus.event.OceanusEvent;
24  import net.sourceforge.joceanus.tethys.api.base.TethysUIComponent;
25  import net.sourceforge.joceanus.tethys.api.base.TethysUIEvent;
26  import net.sourceforge.joceanus.tethys.api.control.TethysUIHTMLManager;
27  import net.sourceforge.joceanus.tethys.api.control.TethysUISplitTreeManager;
28  import net.sourceforge.joceanus.tethys.api.control.TethysUITreeManager;
29  import net.sourceforge.joceanus.tethys.api.control.TethysUITreeManager.TethysUITreeItem;
30  import net.sourceforge.joceanus.tethys.api.factory.TethysUIFactory;
31  import net.sourceforge.joceanus.themis.lethe.statistics.ThemisStatsBase;
32  import net.sourceforge.joceanus.themis.lethe.statistics.ThemisStatsProject;
33  import net.sourceforge.joceanus.themis.lethe.statistics.ThemisStatsReport;
34  
35  /**
36   * Statistics Panel.
37   */
38  public class ThemisStatsPanel {
39      /**
40       * The Next entryId.
41       */
42      private static final AtomicInteger NEXT_ENTRY_ID = new AtomicInteger(1);
43  
44      /**
45       * The SplitTree Manager.
46       */
47      private final TethysUISplitTreeManager<ThemisStatsEntry> theSplitTree;
48  
49      /**
50       * The Tree Manager.
51       */
52      private final TethysUITreeManager<ThemisStatsEntry> theTree;
53  
54      /**
55       * The HTML Manager.
56       */
57      private final TethysUIHTMLManager theHTML;
58  
59      /**
60       * The Current root.
61       */
62      private TethysUITreeItem<ThemisStatsEntry> theRoot;
63  
64      /**
65       * Constructor.
66       *
67       * @param pFactory the GuiFactory
68       * @throws OceanusException on error
69       */
70      public ThemisStatsPanel(final TethysUIFactory<?> pFactory) throws OceanusException {
71          /* Create the splitTree Manager */
72          theSplitTree = pFactory.controlFactory().newSplitTreeManager();
73          theTree = theSplitTree.getTreeManager();
74          theHTML = theSplitTree.getHTMLManager();
75          theHTML.setCSSContent(ThemisDSMStyleSheet.CSS_DSM);
76          theTree.setVisible(true);
77  
78          /* Listen to the TreeManager */
79          theSplitTree.getEventRegistrar().addEventListener(this::handleSplitTreeAction);
80      }
81  
82      /**
83       * Obtain the component.
84       * @return the component
85       */
86      public TethysUIComponent getComponent() {
87          return theSplitTree;
88      }
89  
90      /**
91       * Get NextId.
92       * @return the next id
93       */
94      static int getNextId() {
95          return NEXT_ENTRY_ID.getAndIncrement();
96      }
97  
98      /**
99       * Initialise tree.
100      * @param pProject the project
101      */
102     void initialiseTree(final ThemisStatsProject pProject) {
103         /* If there is currently a root item */
104         if (theRoot != null) {
105             /* Hide it */
106             theRoot.setVisible(false);
107         }
108 
109         /* Create root item for project */
110         final ThemisStatsEntry myEntry = new ThemisStatsEntry(pProject);
111         theRoot = theTree.addRootItem(myEntry.getUniqueName(), myEntry);
112         theRoot.setVisible(true);
113 
114         /* Create child entries */
115         createChildEntries(theRoot, pProject);
116     }
117 
118     /**
119      * Create child entries tree.
120      * @param pParent the parent
121      * @param pChild the child
122      */
123     void createChildEntries(final TethysUITreeItem<ThemisStatsEntry> pParent,
124                             final ThemisStatsBase pChild) {
125         /* Loop through the root children */
126         final Iterator<ThemisStatsBase> myIterator = pChild.childIterator();
127         while (myIterator.hasNext()) {
128             final ThemisStatsBase myChild = myIterator.next();
129 
130             /* Create a new root entry */
131             final ThemisStatsEntry myEntry = new ThemisStatsEntry(pParent.getItem(), myChild);
132             final TethysUITreeItem<ThemisStatsEntry> myTreeItem = theTree.addChildItem(pParent, myEntry.getUniqueName(), myEntry);
133             myTreeItem.setVisible(true);
134 
135             /* Create child entries */
136             createChildEntries(myTreeItem, myChild);
137         }
138     }
139 
140     /**
141      * Handle the split tree action event.
142      * @param pEvent the event
143      */
144     protected void handleSplitTreeAction(final OceanusEvent<TethysUIEvent> pEvent) {
145         /* If this is a new value */
146         if (pEvent.getEventId() == TethysUIEvent.NEWVALUE) {
147             handleNewTreeItem(pEvent.getDetails(ThemisStatsEntry.class));
148         }
149     }
150 
151     /**
152      * Handle the new tree item.
153      * @param pEntry the new entry
154      */
155     private void handleNewTreeItem(final ThemisStatsEntry pEntry) {
156         /* If we have an entry */
157         if (pEntry != null) {
158             /* Access the StatsHolder */
159             final ThemisStatsBase myHolder = pEntry.getObject();
160 
161             /* Build report */
162             final String myDoc = ThemisStatsReport.reportOnStats(myHolder);
163             theHTML.setHTMLContent(myDoc, "");
164         }
165     }
166 }
