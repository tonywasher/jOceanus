140     /**
141      * Constructor.
142      * @param pFactory the GuiFactory
143      * @throws OceanusException on error
144      */
145     public ThemisDSMPanel(final TethysUIFactory<?> pFactory) throws OceanusException {
146         /* Access GuiFactory */
147         theGuiFactory = pFactory;
148 
149         /* Create the HTML Panes */
150         final TethysUIControlFactory myControls = theGuiFactory.controlFactory();
151         final TethysUIButtonFactory<?> myButtons = theGuiFactory.buttonFactory();
152         theDependencyHTML = myControls.newHTMLManager();
153         theDependencyHTML.setCSSContent(ThemisDSMStyleSheet.CSS_DSM);
154         theMatrixHTML = myControls.newHTMLManager();
155         theMatrixHTML.setCSSContent(ThemisDSMStyleSheet.CSS_DSM);
156         theMatrixHTML.getEventRegistrar().addEventListener(TethysUIEvent.BUILDPAGE, e -> {
157             processReference(e.getDetails(String.class));
158             e.consume();
159         });
160 
161         /* Create the module selection panel */
162         final TethysUIPaneFactory myPanes = theGuiFactory.paneFactory();
163         final TethysUIBoxPaneManager myModuleSelect = myPanes.newHBoxPane();
164         myModuleSelect.addNode(myControls.newLabel("Module:"));
165         theModuleButton = myButtons.newScrollButton(ThemisDSMModule.class);
166         myModuleSelect.addNode(theModuleButton);
167         theModuleButton.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> handleNewModule());
168         theModuleButton.setMenuConfigurator(e -> buildModuleMenu());
169 
170         /* Create the project selection panel */
171         final TethysUIBoxPaneManager myProjectSelect = myPanes.newHBoxPane();
172         myProjectSelect.addNode(myControls.newLabel("Project:"));
173         theProjectButton = myButtons.newButton();
174         theProjectButton.setTextOnly();
175         myProjectSelect.addNode(theProjectButton);
176         theProjectButton.getEventRegistrar().addEventListener(e -> selectProject());
177 
178         /* create the overall matrix status panel */
179         final TethysUIBoxPaneManager myMatrixControl = myPanes.newHBoxPane();
180         myMatrixControl.addSpacer();
181         myMatrixControl.addNode(myProjectSelect);
182         myMatrixControl.addSpacer();
183         myMatrixControl.addNode(myModuleSelect);
184         myMatrixControl.addSpacer();
185 
186         /* create the matrix panel */
187         final TethysUIBorderPaneManager myMatrixPanel = myPanes.newBorderPane();
188         myMatrixPanel.setCentre(theMatrixHTML);
189         myMatrixPanel.setNorth(myMatrixControl);
190 
191         /* Create the from selection panel */
192         final TethysUIBoxPaneManager myFromSelect = myPanes.newHBoxPane();
193         myFromSelect.addNode(myControls.newLabel("From:"));
194         theFromButton = myButtons.newScrollButton(ThemisDSMPackage.class);
195         myFromSelect.addNode(theFromButton);
196         theFromButton.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> handleNewFrom());
197         theFromButton.setMenuConfigurator(e -> buildFromMenu());
198 
199         /* Create the project selection panel */
200         final TethysUIBoxPaneManager myToSelect = myPanes.newHBoxPane();
201         myToSelect.addNode(myControls.newLabel("To:"));
202         theToButton = myButtons.newScrollButton(ThemisDSMPackage.class);
203         myToSelect.addNode(theToButton);
204         theToButton.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> handleNewTo());
205         theToButton.setMenuConfigurator(e -> buildToMenu());
206 
207         /* create the overall matrix status panel */
208         final TethysUIBoxPaneManager myDependencyControl = myPanes.newHBoxPane();
209         myDependencyControl.addSpacer();
210         myDependencyControl.addNode(myFromSelect);
211         myDependencyControl.addSpacer();
212         myDependencyControl.addNode(myToSelect);
213         myDependencyControl.addSpacer();
214 
215         /* create the matrix panel */
216         final TethysUIBorderPaneManager myDependencyPanel = myPanes.newBorderPane();
217         myDependencyPanel.setCentre(theDependencyHTML);
218         myDependencyPanel.setNorth(myDependencyControl);
219 
220         /* Create the TabPane */
221         theTabPane = myPanes.newTabPane();
222         theTabPane.addTabItem("Matrix", myMatrixPanel);
223         theDependencyTab = theTabPane.addTabItem("Dependencies", myDependencyPanel);
224 
225         /* Create the Stats panel */
226         theStatsPanel = new ThemisStatsPanel(theGuiFactory);
227         theTabPane.addTabItem("Stats", theStatsPanel.getComponent());
228 
229         /* Create the Source panel */
230         theSourcePanel = new ThemisSourcePanel(theGuiFactory);
231         theTabPane.addTabItem("Source", theSourcePanel.getComponent());
232 
233         /* Create the log tab */
234         theLogSink = theGuiFactory.getLogSink();
235         theLogTab = theTabPane.addTabItem("Log", theLogSink);
236         theLogSink.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> theLogTab.setVisible(true));
237         theLogSink.getEventRegistrar().addEventListener(TethysUIEvent.WINDOWCLOSED, e -> theLogTab.setVisible(false));
238         theLogTab.setVisible(theLogSink.isActive());
239 
240         /* Initialise status */
241         theProjectButton.setText("None");
242         theDependencyTab.setVisible(false);
243         theModuleButton.setEnabled(false);
244         theFromButton.setEnabled(false);
245         theToButton.setEnabled(false);
246 
247         /* Handle the default location */
248         final File myLocation = getDefaultLocation();
249         if (myLocation != null) {
250             handleNewProject(myLocation);
251         }
252     }

259     /**
260      * Handle select project.
261      */
262     void selectProject() {
263         /* Determine initial directory */
264         File myInit = getDefaultLocation();
265         if (myInit == null) {
266             myInit = new File(System.getProperty("user.home"));
267         }
268 
269         /* Determine the name of the directory to load */
270         final TethysUIDialogFactory myDialogs = theGuiFactory.dialogFactory();
271         final TethysUIDirectorySelector myDialog = myDialogs.newDirectorySelector();
272         myDialog.setTitle("Select Project");
273         myDialog.setInitialDirectory(myInit);
274         final File myFile = myDialog.selectDirectory();
275 
276         /* If we selected a file */
277         if (myFile != null) {
278             /* Handle the new project */
279             handleNewProject(myFile);
280         }
281     }

283     /**
284      * Handle the new project.
285      * @param pProjectDir the new project directory
286      */
287     private void handleNewProject(final File pProjectDir) {
288         /* Parse the project*/
289         theError = null;
290         final ThemisDSMProject myProject  = new ThemisDSMProject(pProjectDir);
291         if (myProject.getError()  != null) {
292             theError = myProject.getError();
293 
294         } else if (myProject.hasModules()) {
295             /* Save details */
296             storeDefaultLocation(pProjectDir);
297 
298             /* Store details */
299             theProject = myProject;
300             theProjectButton.setText(theProject.toString());
301 
302             /* Set the new module */
303             processNewModule(theProject.getDefaultModule());
304 
305             /* Load statistics */
306             handleNewStats(pProjectDir);
307         }
308 
309         /* Display the error */
310         if (theError != null) {
311             theLogSink.writeLogMessage(theError.getMessage());
312             theLogTab.setVisible(true);
313         }
314     }
315 
316     /**
317      * Handle the new project.
318      * @param pProjectDir the new project directory
319      */
320     private void handleNewStats(final File pProjectDir) {
321         /* Analyse source of project */
322         final ThemisAnalysisProject myProj = new ThemisAnalysisProject(pProjectDir);
323         if (myProj.getError() != null) {
324             theError = myProj.getError();
325         } else {
326             /* Parse the base project */
327             final ThemisStatsParser myParser = new ThemisStatsParser();
328             final ThemisStatsProject myProject = myParser.parseProject(myProj);
329             theStatsPanel.initialiseTree(myProject);
330             theSourcePanel.initialiseTree(myProj);
331         }
332     }

341     /**
342      * Process new module.
343      * @param pModule the new Module
344      */
345     private void processNewModule(final ThemisDSMModule pModule) {
346         /* Store the module */
347         theModule = pModule;
348         theModuleButton.setValue(pModule);
349         theModuleButton.setEnabled(true);
350 
351         /* Build matrix report */
352         final String myDoc = ThemisDSMReport.reportOnModule(theModule);
353         theMatrixHTML.setHTMLContent(myDoc, "");
354 
355         /* Process the new from Package */
356         processNewFrom(theModule.getDefaultPackage());
357     }

446     /**
447      * Process the reference.
448      * @param pReference the reference
449      */
450     private void processReference(final String pReference) {
451         /* Split on the "-" */
452         final String[] myTokens = pReference.split(ThemisDSMReport.SEP_REF);
453         if (myTokens.length == 2) {
454             /* Access the relevant packages */
455             final ThemisDSMPackage myFrom = theModule.getIndexedPackage(ThemisDSMReport.getIndexForKey(myTokens[0]));
456             final ThemisDSMPackage myTo = theModule.getIndexedPackage(ThemisDSMReport.getIndexForKey(myTokens[1]));
457 
458             /* Set the new details */
459             processNewFrom(myFrom);
460             processNewTo(myTo);
461 
462             /* Select the tab */
463             theDependencyTab.selectItem();
464         }
465     }

467     /**
468      * Obtain the default location.
469      * @return the default location
470      */
471     private File getDefaultLocation() {
472         final Preferences myPreferences = deriveHandle();
473         final String myLocation = myPreferences.get("DefaultProject", null);
474         return myLocation == null ? null : new File(myLocation);
475     }
476 
477     /**
478      * Store the default location.
479      * @param pLocation the default location
480      */
481     private void storeDefaultLocation(final File pLocation) {
482         /* Protect against exceptions */
483         try {
484             final Preferences myPreferences = deriveHandle();
485             myPreferences.put("DefaultProject", pLocation.getAbsolutePath());
486             myPreferences.flush();
487         } catch (BackingStoreException e) {
488             theError = new ThemisIOException("Failed to save preference", e);
489         }
490     }
491 
492     /**
493      * Derive handle for node.
494      * @return the class name
495      */
496     private Preferences deriveHandle() {
497         /* Obtain the class name */
498         final Class<?> myClass = this.getClass();
499         String myName = myClass.getCanonicalName();
500 
501         /* Obtain the package name */
502         final String myPackage = myClass.getPackage().getName();
503 
504         /* Strip off the package name */
505         myName = myName.substring(myPackage.length() + 1);
506 
507         /* Derive the handle */
508         final Preferences myHandle = Preferences.userNodeForPackage(myClass);
509         return myHandle.node(myName);
510     }
