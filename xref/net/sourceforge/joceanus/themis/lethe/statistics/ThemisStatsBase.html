17  package net.sourceforge.joceanus.themis.lethe.statistics;
18  
19  import java.util.EnumMap;
20  import java.util.Iterator;
21  import java.util.Map;
22  
23  /**
24   * Statistics Base.
25   */
26  public abstract class ThemisStatsBase {
27      /**
28       * The parent.
29       */
30      private ThemisStatsBase theParent;
31  
32      /**
33       * The stats.
34       */
35      private final Map<ThemisStat, Integer> theStats;
36  
37      /**
38       * Constructor.
39       */
40      protected ThemisStatsBase() {
41          theStats = new EnumMap<>(ThemisStat.class);
42      }
43  
44      /**
45       * Obtain the parent.
46       * @return the parent
47       */
48      public ThemisStatsBase getParent() {
49          return theParent;
50      }
51  
52      /**
53       * Set the parent.
54       * @param pParent the parent
55       */
56      public void setParent(final ThemisStatsBase pParent) {
57          theParent = pParent;
58      }
59  
60      /**
61       * Obtain statistic value.
62       * @param pStat the statistic
63       * @return the value
64       */
65      public Integer getStat(final ThemisStat pStat) {
66          return theStats.computeIfAbsent(pStat, s -> 0);
67      }
68  
69      /**
70       * Set statistic value.
71       * @param pStat the statistic
72       * @param pValue the value
73       */
74      private void setStat(final ThemisStat pStat,
75                           final Integer pValue) {
76          theStats.put(pStat, pValue);
77      }
78  
79      /**
80       * Increment statistic.
81       * @param pStat the statistic
82       */
83      void incrementStat(final ThemisStat pStat) {
84          adjustStat(pStat, 1);
85      }
86  
87      /**
88       * Adjust statistic.
89       * @param pStat the statistic
90       * @param pAdjust the adjustment value
91       */
92      void adjustStat(final ThemisStat pStat,
93                      final Integer pAdjust) {
94          /* Adjust the value */
95          final Integer myCurr = getStat(pStat);
96          setStat(pStat, myCurr + pAdjust);
97      }
98  
99      /**
100      * Obtain class iterator.
101      * @return the iterator
102      */
103     public abstract Iterator<ThemisStatsBase> childIterator();
104 
105     /**
106      * Add class to list.
107      * @param pClass the class
108      */
109     void addClass(final ThemisStatsClass pClass) {
110         /* NoOp by default */
111     }
112 
113     /**
114      * Add method to list.
115      * @param pMethod the method
116      */
117     void addMethod(final ThemisStatsMethod pMethod) {
118         /* NoOp by default */
119     }
120 
121     /**
122      * Add all stats into totals.
123      */
124     void addStatsToTotals() {
125         /* Adjust counts */
126         adjustTotals(ThemisStat.LOC, ThemisStat.TLOC);
127         adjustTotals(ThemisStat.LLOC, ThemisStat.TLLOC);
128         adjustTotals(ThemisStat.NOS, ThemisStat.TNOS);
129         adjustTotals(ThemisStat.CLOC, ThemisStat.TCLOC);
130         adjustTotals(ThemisStat.DLOC, ThemisStat.TDLOC);
131         adjustTotals(ThemisStat.NCL, ThemisStat.TNCL);
132         adjustTotals(ThemisStat.NIN, ThemisStat.TNIN);
133         adjustTotals(ThemisStat.NEN, ThemisStat.TNEN);
134         adjustTotals(ThemisStat.NM, ThemisStat.TNM);
135         adjustTotals(ThemisStat.NA, ThemisStat.TNA);
136     }
137 
138     /**
139      * Add method stats into totals.
140      */
141     void addMethodStatsToTotals() {
142         /* Adjust counts */
143         adjustTotals(ThemisStat.NCL, ThemisStat.TNCL);
144         adjustTotals(ThemisStat.NIN, ThemisStat.TNIN);
145         adjustTotals(ThemisStat.NEN, ThemisStat.TNEN);
146         adjustTotals(ThemisStat.NM, ThemisStat.TNM);
147         adjustTotals(ThemisStat.NA, ThemisStat.TNA);
148     }
149 
150     /**
151      * Add method stats into totals.
152      * @param pMethod the method
153      */
154     void addMethodStatsToClass(final ThemisStatsMethod pMethod) {
155         /* Adjust counts */
156         adjustChildStat(pMethod, ThemisStat.LOC);
157         adjustChildStat(pMethod, ThemisStat.LLOC);
158         adjustChildStat(pMethod, ThemisStat.NOS);
159         adjustChildStat(pMethod, ThemisStat.CLOC);
160         adjustChildStat(pMethod, ThemisStat.DLOC);
161     }
162 
163     /**
164      * Adjust totals.
165      * @param pItem the item stat
166      * @param pTotal the total stat
167      */
168     void adjustTotals(final ThemisStat pItem,
169                       final ThemisStat pTotal) {
170         /* Adjust counts */
171         adjustStat(pTotal, getStat(pItem));
172     }
173 
174     /**
175      * Add child totals.
176      * @param pChild the child
177      */
178     void addChildTotals(final ThemisStatsBase pChild) {
179         /* Adjust counts */
180         adjustChildStat(pChild, ThemisStat.TLOC);
181         adjustChildStat(pChild, ThemisStat.TLLOC);
182         adjustChildStat(pChild, ThemisStat.TNCL);
183         adjustChildStat(pChild, ThemisStat.TNIN);
184         adjustChildStat(pChild, ThemisStat.TNEN);
185         adjustChildStat(pChild, ThemisStat.TNM);
186         adjustChildStat(pChild, ThemisStat.TNA);
187         adjustChildStat(pChild, ThemisStat.TNOS);
188         adjustChildStat(pChild, ThemisStat.TCLOC);
189         adjustChildStat(pChild, ThemisStat.TDLOC);
190     }
191 
192     /**
193      * Adjust child stat.
194      * @param pChild the child
195      * @param pStat the stat
196      */
197     void adjustChildStat(final ThemisStatsBase pChild,
198                          final ThemisStat pStat) {
199         adjustChildStat(pChild, pStat, pStat);
200     }
201 
202     /**
203      * Adjust child stat.
204      * @param pChild the child
205      * @param pStat the stat
206      * @param pChildStat the child stat
207      */
208     void adjustChildStat(final ThemisStatsBase pChild,
209                          final ThemisStat pStat,
210                          final ThemisStat pChildStat) {
211         /* Adjust counts */
212         adjustStat(pStat, pChild.getStat(pChildStat));
213     }
214 }
