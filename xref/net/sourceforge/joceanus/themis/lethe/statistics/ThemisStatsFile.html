17  package net.sourceforge.joceanus.themis.lethe.statistics;
18  
19  import java.util.ArrayList;
20  import java.util.Iterator;
21  import java.util.List;
22  
23  import net.sourceforge.joceanus.themis.lethe.analysis.ThemisAnalysisFile;
24  import net.sourceforge.joceanus.themis.lethe.analysis.ThemisAnalysisPackage;
25  
26  /**
27   * File statistics.
28   */
29  public class ThemisStatsFile
30          extends ThemisStatsBase {
31      /**
32       * The file.
33       */
34      private final ThemisAnalysisFile theFile;
35  
36      /**
37       * The child list.
38       */
39      private final List<ThemisStatsBase> theChildren;
40  
41      /**
42       * Constructor.
43       * @param pFile the file
44       */
45      ThemisStatsFile(final ThemisAnalysisFile pFile) {
46          /* Store parameters */
47          theFile = pFile;
48  
49          /* Create lists */
50          theChildren = new ArrayList<>();
51      }
52  
53      /**
54       * Obtain the file.
55       * @return the file
56       */
57      public ThemisAnalysisFile getFile() {
58          return theFile;
59      }
60  
61      @Override
62      public Iterator<ThemisStatsBase> childIterator() {
63          return theChildren.iterator();
64      }
65  
66      @Override
67      public String toString() {
68          return theFile.getName() + ThemisAnalysisPackage.SFX_JAVA;
69      }
70  
71      @Override
72      public void addClass(final ThemisStatsClass pClass) {
73          /* Add class to list */
74          theChildren.add(pClass);
75          pClass.setParent(this);
76  
77          /* Adjust counts */
78          pClass.addStatsToTotals();
79          addChildTotals(pClass);
80      }
81  
82      @Override
83      public void addMethod(final ThemisStatsMethod pMethod) {
84          /* Methods cannot be a direct child of a file */
85      }
86  }
