24  /**
25   * Mac types. Available algorithms.
26   */

28      /**
29       * HMAC.
30       */
31      HMAC,
32  
33      /**
34       * GMAC.
35       */
36      GMAC,
37  
38      /**
39       * CMAC.
40       */
41      CMAC,
42  
43      /**
44       * Poly1305.
45       */
46      POLY1305,
47  
48      /**
49       * Skein.
50       */
51      SKEIN,
52  
53      /**
54       * Blake2.
55       */
56      BLAKE2,
57  
58      /**
59       * Kalyna.
60       */
61      KALYNA,
62  
63      /**
64       * Kupyna.
65       */
66      KUPYNA,
67  
68      /**
69       * VMPC.
70       */
71      VMPC,
72  
73      /**
74       * ZUC.
75       */
76      ZUC,
77  
78      /**
79       * CBCMac.
80       */
81      CBCMAC,

83      /**
84       * CFBMac.
85       */
86      CFBMAC,
87  
88      /**
89       * SipHash.
90       */
91      SIPHASH,

96      KMAC,

99       * GOST.
100      */
101     GOST,

103     /**
104      * Blake3.
105      */
106     BLAKE3;
107 
108     /**
109      * The String name.
110      */
111     private String theName;
112 
113     @Override
114     public String toString() {
115         /* If we have not yet loaded the name */
116         if (theName == null) {
117             /* Load the name */
118             theName = bundleIdForMacType().getValue();
119         }
120 
121         /* return the name */
122         return theName;
123     }
124 
125     /**
126      * Needs re-initialisation after final.
127      * @return true/false
128      */
129     public boolean needsReInitialisation() {
130         return this == GMAC;
131     }
132 
133     /**
134      * Is this MacType valid for largeData?
135      * @return true/false
136      */
137     public boolean supportsLargeData() {
138         return this != ZUC;
139     }
140 
141     /**
142      * Is this MacType valid for keyLength?
143      * @param pKeyLen the keyLength
144      * @return true/false
145      */
146     public boolean validForKeyLength(final GordianLength pKeyLen) {
147         switch (this) {
148             case POLY1305:
149             case GOST:
150             case BLAKE3:
151                 return GordianLength.LEN_256 == pKeyLen;
152             case ZUC:
153                 return GordianLength.LEN_128 == pKeyLen
154                         || GordianLength.LEN_256 == pKeyLen;
155             case KALYNA:
156                 return GordianSymKeyType.KALYNA.validForKeyLength(pKeyLen);
157             case SIPHASH:
158                 return GordianLength.LEN_128 == pKeyLen;
159             default:
160                 return true;
161         }
162     }
163 
164     /**
165      * Obtain the resource bundleId for this macType.
166      * @return the resource bundleId
167      */
168     private GordianBundleId bundleIdForMacType() {
169         /* Create the map and return it */
170         switch (this) {
171             case HMAC: return GordianMacResource.MAC_HMAC;
172             case GMAC: return GordianMacResource.MAC_GMAC;
173             case CMAC: return GordianMacResource.MAC_CMAC;
174             case KMAC: return GordianMacResource.MAC_KMAC;
175             case POLY1305: return GordianMacResource.MAC_POLY;
176             case SKEIN: return GordianMacResource.MAC_SKEIN;
177             case KALYNA: return GordianMacResource.MAC_KALYNA;
178             case KUPYNA: return GordianMacResource.MAC_KUPYNA;
179             case BLAKE2: return GordianMacResource.MAC_BLAKE2;
180             case BLAKE3: return GordianMacResource.MAC_BLAKE3;
181             case VMPC: return GordianMacResource.MAC_VMPC;
182             case ZUC: return GordianCipherResource.STREAMKEY_ZUC;
183             case CBCMAC: return GordianMacResource.MAC_CBC;
184             case CFBMAC: return GordianMacResource.MAC_CFB;
185             case SIPHASH: return GordianMacResource.MAC_SIPHASH;
186             case GOST: return GordianMacResource.MAC_GOST;
187             default:
188                 throw new IllegalArgumentException();
189         }
190     }
191 }
