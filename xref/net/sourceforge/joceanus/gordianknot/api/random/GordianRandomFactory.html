17  package net.sourceforge.joceanus.gordianknot.api.random;
18  
19  import net.sourceforge.joceanus.gordianknot.api.base.GordianException;
20  import net.sourceforge.joceanus.gordianknot.api.base.GordianLength;
21  import net.sourceforge.joceanus.gordianknot.api.cipher.GordianStreamKeySpec;
22  import net.sourceforge.joceanus.gordianknot.api.cipher.GordianSymKeySpec;
23  import net.sourceforge.joceanus.gordianknot.api.digest.GordianDigest;
24  import net.sourceforge.joceanus.gordianknot.api.key.GordianKey;
25  import net.sourceforge.joceanus.gordianknot.api.mac.GordianMac;
26  
27  import java.security.SecureRandom;
28  import java.util.List;
29  import java.util.function.BiPredicate;
30  import java.util.function.Predicate;
31  
32  /**
33   * GordianKnot Random Factory API.
34   */
35  public interface GordianRandomFactory {
36      /**
37       * create SecureRandom.
38       * @param pRandomSpec the randomSpec
39       * @return the new SecureRandom
40       * @throws GordianException on error
41       */
42      SecureRandom createRandom(GordianRandomSpec pRandomSpec) throws GordianException;
43  
44      /**
45       * create CombinedRandom.
46       * @param pCtrSpec the ctrRandomSpec
47       * @param pHashSpec the hashRandomSpec
48       * @return the new SecureRandom
49       * @throws GordianException on error
50       */
51      SecureRandom createRandom(GordianRandomSpec pCtrSpec,
52                                GordianRandomSpec pHashSpec) throws GordianException;
53  
54      /**
55       * Obtain predicate for supported randomSpecs.
56       * @return the predicate
57       */
58      Predicate<GordianRandomSpec> supportedRandomSpecs();
59  
60      /**
61       * Obtain predicate for supported combined randomSpecs.
62       * @return the predicate
63       */
64      BiPredicate<GordianRandomSpec, GordianRandomSpec> supportedCombinedSpecs();
65  
66      /**
67       * generate random GordianDigest.
68       * @param pLargeData only generate a digest that is suitable for processing large amounts of data
69       * @return the new Digest
70       * @throws GordianException on error
71       */
72      GordianDigest generateRandomDigest(boolean pLargeData) throws GordianException;
73  
74      /**
75       * generate random GordianMac.
76       * @param pKeyLen the keyLength
77       * @param pLargeData only generate a Mac that is suitable for parsing large amounts of data
78       * @return the new MAC
79       * @throws GordianException on error
80       */
81      GordianMac generateRandomMac(GordianLength pKeyLen,
82                                   boolean pLargeData) throws GordianException;
83  
84      /**
85       * generate random SymKey.
86       * @param pKeyLen the keyLength
87       * @return the new key
88       * @throws GordianException on error
89       */
90      GordianKey<GordianSymKeySpec> generateRandomSymKey(GordianLength pKeyLen) throws GordianException;
91  
92      /**
93       * generate random GordianStreamKey.
94       * @param pKeyLen the keyLength
95       * @param pLargeData only generate a Mac that is suitable for parsing large amounts of data
96       * @return the new StreamKey
97       * @throws GordianException on error
98       */
99      GordianKey<GordianStreamKeySpec> generateRandomStreamKey(GordianLength pKeyLen,
100                                                              boolean pLargeData) throws GordianException;
101 
102     /**
103      * Obtain a list of supported randomSpecs.
104      * @return the list of supported randomSpecs.
105      */
106     List<GordianRandomSpec> listAllSupportedRandomSpecs();
107 
108     /**
109      * Obtain a list of supported randomSpecs of a given type.
110      * @param pType the random type
111      * @return the list of supported randomSpecs.
112      */
113     List<GordianRandomSpec> listAllSupportedRandomSpecs(GordianRandomType pType);
114 
115     /**
116      * Obtain a list of supported randomSpecs of a given type and keyLength.
117      * <p>Only valid for CTR And X931 types</p>
118      * @param pType the random type
119      * @param pKeyLen the keyLength
120      * @return the list of supported randomSpecs.
121      */
122     List<GordianRandomSpec> listAllSupportedRandomSpecs(GordianRandomType pType,
123                                                         GordianLength pKeyLen);
124 }
