43  import java.util.function.Predicate;
44  
45  /**
46   * Bouncy Signature Factory.
47   */
48  public class BouncySignatureFactory
49          extends GordianCoreSignatureFactory {
50      /**
51       * Constructor.
52       *
53       * @param pFactory the factory
54       */
55      BouncySignatureFactory(final GordianBaseFactory pFactory) {
56          /* Initialise underlying class */
57          super(pFactory);
58      }
59  
60      @Override
61      public Predicate<GordianSignatureSpec> supportedKeyPairSignatures() {
62          return this::validSignatureSpec;
63      }
64  
65      @Override
66      public GordianSignature createSigner(final GordianSignatureSpec pSignatureSpec) throws GordianException {
67          /* Check validity of Signature */
68          checkSignatureSpec(pSignatureSpec);
69  
70          /* Create the signer */
71          return getBCSigner(pSignatureSpec);
72      }
73  
74      /**
75       * Create the BouncyCastle Signer.
76       *
77       * @param pSignatureSpec the signatureSpec
78       * @return the Signer
79       * @throws GordianException on error
80       */
81      private GordianSignature getBCSigner(final GordianSignatureSpec pSignatureSpec) throws GordianException {
82          switch (pSignatureSpec.getKeyPairType()) {
83              case RSA:
84                  return new BouncyRSASignature(getFactory(), pSignatureSpec);
85              case EC:
86                  return new BouncyECSignature(getFactory(), pSignatureSpec);
87              case DSTU4145:
88                  return new BouncyDSTUSignature(getFactory(), pSignatureSpec);
89              case GOST2012:
90                  return new BouncyGOSTSignature(getFactory(), pSignatureSpec);
91              case SM2:
92                  return new BouncySM2Signature(getFactory(), pSignatureSpec);
93              case EDDSA:
94                  return new BouncyEdDSASignature(getFactory(), pSignatureSpec);
95              case DSA:
96                  return new BouncyDSASignature(getFactory(), pSignatureSpec);
97              case SLHDSA:
98                  return new BouncySLHDSASignature(getFactory(), pSignatureSpec);
99              case MLDSA:
100                 return new BouncyMLDSASignature(getFactory(), pSignatureSpec);
101             case FALCON:
102                 return new BouncyFalconSignature(getFactory(), pSignatureSpec);
103             case MAYO:
104                 return new BouncyMayoSignature(getFactory(), pSignatureSpec);
105             case SNOVA:
106                 return new BouncySnovaSignature(getFactory(), pSignatureSpec);
107             case PICNIC:
108                 return new BouncyPicnicSignature(getFactory(), pSignatureSpec);
109             case XMSS:
110                 return new BouncyXMSSSignature(getFactory(), pSignatureSpec);
111             case LMS:
112                 return new BouncyLMSSignature(getFactory(), pSignatureSpec);
113             case COMPOSITE:
114                 return new GordianCompositeSigner(getFactory(), pSignatureSpec);
115             default:
116                 throw new GordianDataException(GordianBaseData.getInvalidText(pSignatureSpec.getKeyPairType()));
117         }
118     }
119 }
