17  package net.sourceforge.joceanus.gordianknot.impl.core.keyset;
18  
19  import net.sourceforge.joceanus.gordianknot.api.base.GordianException;
20  import net.sourceforge.joceanus.gordianknot.api.base.GordianLength;
21  import net.sourceforge.joceanus.gordianknot.api.cipher.GordianCipherParameters;
22  import net.sourceforge.joceanus.gordianknot.impl.core.base.GordianBaseFactory;
23  import net.sourceforge.joceanus.gordianknot.impl.core.cipher.GordianCoreWrapper;
24  
25  /**
26   * GordianKnot base for KeySetWrapper.
27   * <p>
28   * This class extends GordianCoreWrapper to wrap using a keySet rather than a key.
29   */
30  public class GordianKeySetWrapper
31          extends GordianCoreWrapper {
32      /**
33       * The cipher array.
34       */
35      private final GordianSymKeyCipherSet[] theCiphers;
36  
37      /**
38       * Constructor.
39       * @param pFactory the Security Factory
40       * @param pCiphers the CipherSets
41       */
42      GordianKeySetWrapper(final GordianBaseFactory pFactory,
43                           final GordianSymKeyCipherSet[] pCiphers) {
44          super(pFactory, GordianLength.LEN_128.getByteLength());
45          theCiphers = pCiphers;
46      }
47  
48      @Override
49      protected void initCipherForWrapping() throws GordianException {
50          for (GordianSymKeyCipherSet myCipher : theCiphers) {
51              myCipher.getStandardCipher().initForEncrypt(GordianCipherParameters.key(myCipher.getKey()));
52          }
53      }
54  
55      @Override
56      protected void initCipherForUnwrapping() throws GordianException {
57          for (GordianSymKeyCipherSet myCipher : theCiphers) {
58              myCipher.getStandardCipher().initForDecrypt(GordianCipherParameters.key(myCipher.getKey()));
59          }
60      }
61  
62      @Override
63      protected void iterateCipher(final byte[] pInBuffer,
64                                   final int pBufferLen,
65                                   final byte[] pResult) throws GordianException {
66          byte[] myInBuffer = pInBuffer;
67          final byte[][] myTempBuffers = { new byte[pBufferLen], new byte[pBufferLen] };
68          int myNextIndex = 0;
69          byte[] myOutBuffer = myTempBuffers[myNextIndex++];
70          for (GordianSymKeyCipherSet myCipher : theCiphers) {
71              myCipher.getStandardCipher().finish(myInBuffer, 0, pBufferLen, myOutBuffer, 0);
72              myInBuffer = myOutBuffer;
73              myNextIndex %= 2;
74              myOutBuffer = myTempBuffers[myNextIndex++];
75          }
76          System.arraycopy(myInBuffer, 0, pResult, 0, pBufferLen);
77      }
78  }
