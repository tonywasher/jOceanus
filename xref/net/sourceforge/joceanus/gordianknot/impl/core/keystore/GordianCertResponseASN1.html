49  /**
50   * ASN1 Encoding of CertificateResponse.
51   * <pre>
52   * GordianCertResponseASN1 ::= SEQUENCE {
53   *      certReqId    INTEGER
54   *      certRespId   INTEGER
55   *      CHOICE {
56   *          certificate     [1] Certificate,
57   *          encrypted       [2] EnvelopedData
58   *      }
59   *      signerCerts  SEQUENCE SIZE (1..MAX) OF Certificate
60   *      macValue  PKMACValue OPTIONAL
61   * }
62   * </pre>
63   */
64  public final class GordianCertResponseASN1
65          extends GordianASN1Object {
66      /**
67       * The standard certificate tag.
68       */
69      private static final int TAG_STANDARD = 1;
70  
71      /**
72       * The encrypted certificate tag.
73       */
74      private static final int TAG_ENCRYPTED = 2;
75  
76      /**
77       * The requestId.
78       */
79      private final int theReqId;
80  
81      /**
82       * The responseId.
83       */
84      private final int theRespId;
85  
86      /**
87       * The signer certificates.
88       */
89      private final Certificate[] theSignerCerts;
90  
91      /**
92       * The certificate.
93       */
94      private Certificate theCertificate;
95  
96      /**
97       * The encoded certificate.
98       */
99      private EnvelopedData theEncrypted;
100 
101     /**
102      * The macValue.
103      */
104     private PKMACValue theMACValue;
105 
106     /**
107      * Create the ASN1 sequence.
108      * @param pReqId the requestId
109      * @param pRespId the responseId
110      * @param pCertificate the certificate
111      * @param pSignerCerts the signerCertificates
112      */
113     private GordianCertResponseASN1(final int pReqId,
114                                     final int pRespId,
115                                     final Certificate pCertificate,
116                                     final Certificate[] pSignerCerts) {
117         /* Store the Details */
118         theReqId = pReqId;
119         theRespId = pRespId;
120         theCertificate = pCertificate;
121         theSignerCerts = pSignerCerts.clone();
122     }
123 
124     /**
125      * Constructor.
126      * @param pSequence the Sequence
127      * @throws GordianException on error
128      */
129     private GordianCertResponseASN1(final ASN1Sequence pSequence) throws GordianException {
130         /* Protect against exceptions */
131         try {
132             /* Extract the request/responseIds from the sequence */
133             final Enumeration<?> en = pSequence.getObjects();
134             theReqId = ASN1Integer.getInstance(en.nextElement()).getValue().intValue();
135             theRespId = ASN1Integer.getInstance(en.nextElement()).getValue().intValue();
136 
137             /* Extract the certificate from the sequence */
138             final ASN1TaggedObject myTagged = ASN1TaggedObject.getInstance(en.nextElement());
139             switch (myTagged.getTagNo()) {
140                 case TAG_STANDARD:
141                     theCertificate = Certificate.getInstance(myTagged, false);
142                     break;
143                 case TAG_ENCRYPTED:
144                     theEncrypted = EnvelopedData.getInstance(myTagged, false);
145                     break;
146                 default:
147                     throw new GordianDataException("Unexpected tag");
148             }
149 
150             /* Extract the signer Certificates from the sequence */
151             final ASN1Sequence mySignerCerts = ASN1Sequence.getInstance(en.nextElement());
152             final Enumeration<?> enCert = mySignerCerts.getObjects();
153             final int myNumCerts = mySignerCerts.size();
154             theSignerCerts = new Certificate[mySignerCerts.size()];
155             for (int i = 0; i < myNumCerts; i++) {
156                 theSignerCerts[i] = Certificate.getInstance(enCert.nextElement());
157             }
158 
159             /* Extract the Optional MACValue from the sequence */
160             if (en.hasMoreElements()) {
161                 theMACValue = PKMACValue.getInstance(en.nextElement());
162             }
163 
164             /* handle exceptions */
165         } catch (IllegalArgumentException e) {
166             throw new GordianIOException("Unable to parse ASN1 sequence", e);
167         }
168     }
169 
170     /**
171      * Parse the ASN1 object.
172      * @param pObject the object to parse
173      * @return the parsed object
174      * @throws GordianException on error
175      */
176     public static GordianCertResponseASN1 getInstance(final Object pObject) throws GordianException {
177         if (pObject instanceof GordianCertResponseASN1 myASN1) {
178             return myASN1;
179         } else if (pObject != null) {
180             return new GordianCertResponseASN1(ASN1Sequence.getInstance(pObject));
181         }
182         throw new GordianDataException("Null sequence");
183     }

185     /**
186      * Create the certificate response.
187      * @param pReqId the request id
188      * @param pRespId the responseId
189      * @param pChain the certificate chain
190      * @return the response
191      */
192     public static GordianCertResponseASN1 createCertResponse(final int pReqId,
193                                                              final int pRespId,
194                                                              final List<GordianCertificate> pChain) {
195         /* Create the chain */
196         final Certificate[] myChain = new Certificate[pChain.size() - 1];
197 
198         /* Store first element in chain */
199         final Iterator<GordianCertificate> myIterator = pChain.iterator();
200         final Certificate myCert = ((GordianCoreCertificate) myIterator.next()).getCertificate();
201 
202         /* Store subsequent details */
203         for (int i = 1; i < pChain.size(); i++) {
204             myChain[i - 1] = ((GordianCoreCertificate) myIterator.next()).getCertificate();
205         }
206 
207         /* Return the chain */
208         return new GordianCertResponseASN1(pReqId, pRespId, myCert, myChain);
209     }

259     /**
260      * Obtain the certificate.
261      * @param pEncryptor the encryptor
262      * @return the certificate
263      * @throws GordianException on error
264      */
265     public GordianCoreCertificate getCertificate(final GordianCRMEncryptor pEncryptor) throws GordianException {
266         /* If the certificate is still encrypted */
267         if (isEncrypted()) {
268             throw new GordianLogicException("Certificate still encrypted");
269         }
270 
271         /* Convert and return the certificate */
272         return pEncryptor.convertCertificate(theCertificate);
273     }

275     /**
276      * Obtain the certificate chain.
277      * @param pEncryptor the encryptor
278      * @return the chain
279      * @throws GordianException on error
280      */
281     public GordianCoreCertificate[] getCertificateChain(final GordianCRMEncryptor pEncryptor) throws GordianException {
282         /* If the certificate is still encrypted */
283         if (isEncrypted()) {
284             throw new GordianLogicException("Certificate still encrypted");
285         }
286 
287         /* Create the chain */
288         final GordianCoreCertificate[] myChain = new GordianCoreCertificate[theSignerCerts.length + 1];
289 
290         /* Store first element in chain */
291         myChain[0] = pEncryptor.convertCertificate(theCertificate);
292 
293         /* Store subsequent details */
294         for (int i = 0; i < theSignerCerts.length; i++) {
295             myChain[i + 1] = pEncryptor.convertCertificate(theSignerCerts[i]);
296         }
297 
298         /* Return the chain */
299         return myChain;
300     }

302     @Override
303     public ASN1Primitive toASN1Primitive() {
304         final ASN1EncodableVector v = new ASN1EncodableVector();
305         v.add(new ASN1Integer(theReqId));
306         v.add(new ASN1Integer(theRespId));
307         if (theCertificate != null) {
308             v.add(new DERTaggedObject(false, TAG_STANDARD, theCertificate));
309         } else if (theEncrypted != null) {
310             v.add(new DERTaggedObject(false, TAG_ENCRYPTED, theEncrypted));
311         }
312         v.add(new DERSequence(theSignerCerts));
313         if (theMACValue != null) {
314             v.add(theMACValue);
315         }
316 
317         return new DERSequence(v);
318     }

320     /**
321      * Encrypt certificate.
322      * @param pEncryptor the encryptor
323      * @throws GordianException on error
324      */
325     public void encryptCertificate(final GordianCRMEncryptor pEncryptor) throws GordianException {
326         /* Only encrypt if not currently encrypted */
327         if (theEncrypted == null) {
328             /* Prepare for encryption */
329             final GordianCoreCertificate myCert = pEncryptor.convertCertificate(theCertificate);
330             final GordianCRMResult myResult = pEncryptor.prepareForEncryption(myCert);
331 
332             /* Create the encrypted content */
333             final EncryptedContentInfo myInfo = GordianCRMEncryptor.buildEncryptedContentInfo(myResult.getKeySet(), myCert);
334             theEncrypted = new EnvelopedData(null, new BERSet(myResult.getRecipient()), myInfo, (BERSet) null);
335             theCertificate = null;
336         }
337     }
338 
339     /**
340      * Decrypt certificate.
341      * @param pEncryptor the encryptor
342      * @param pKeyPair the keyPair
343      * @throws GordianException on error
344      */
345     public void decryptCertificate(final GordianCRMEncryptor pEncryptor,
346                                    final GordianKeyStorePair pKeyPair) throws GordianException {
347         /* Only decrypt if currently encrypted */
348         if (theCertificate == null) {
349             /* Derive the KeySet */
350             final RecipientInfo myRecipient = RecipientInfo.getInstance(theEncrypted.getRecipientInfos().getObjectAt(0));
351             final KeyTransRecipientInfo myRecInfo = (KeyTransRecipientInfo) myRecipient.getInfo();
352             final GordianKeySet myKeySet = pEncryptor.deriveKeySetFromRecInfo(myRecInfo, pKeyPair.getCertificateChain().get(0), pKeyPair.getKeyPair());
353 
354             /* Decrypt the certificate */
355             final EncryptedContentInfo myInfo = theEncrypted.getEncryptedContentInfo();
356             final byte[] myEncoded = myKeySet.decryptBytes(myInfo.getEncryptedContent().getOctets());
357             theCertificate = Certificate.getInstance(myEncoded);
358             theEncrypted = null;
359         }
