84      @Override
85      public GordianAgreement createAgreement(final byte[] pClientHello) throws GordianException {
86          /* Parse the client hello message */
87          final GordianAgreementMessageASN1 myASN1 = GordianAgreementMessageASN1.getInstance(pClientHello);
88          myASN1.checkMessageType(GordianMessageType.CLIENTHELLO);
89          final AlgorithmIdentifier myAlgId = myASN1.getAgreementId();
90          final GordianAgreementSpec mySpec = getSpecForIdentifier(myAlgId);
91          return createAgreement(mySpec);
92      }

94      /**
95       * Create the BouncyCastle Agreement.
96       *
97       * @param pSpec the agreementSpec
98       * @return the Agreement
99       * @throws GordianException on error
100      */
101     protected GordianAgreement getCompositeAgreement(final GordianAgreementSpec pSpec) throws GordianException {
102         switch (pSpec.getAgreementType()) {
103             case KEM:
104             case ANON:
105                 return new GordianCompositeAnonymousAgreement(getFactory(), pSpec);
106             case BASIC:
107                 return new GordianCompositeBasicAgreement(getFactory(), pSpec);
108             case SIGNED:
109                 return new GordianCompositeSignedAgreement(getFactory(), pSpec);
110             case MQV:
111             case UNIFIED:
112             case SM2:
113                 return new GordianCompositeHandshakeAgreement(getFactory(), pSpec);
114             default:
115                 throw new GordianDataException(GordianBaseData.getInvalidText(pSpec));
116         }
117     }

119     @Override
120     public Predicate<GordianAgreementSpec> supportedAgreements() {
121         return this::validAgreementSpec;

124     /**
125      * Check the agreementSpec.
126      * @param pAgreementSpec the agreementSpec
127      * @throws GordianException on error
128      */
129     protected void checkAgreementSpec(final GordianAgreementSpec pAgreementSpec) throws GordianException {
130         /* Check validity of agreement */
131         if (!validAgreementSpec(pAgreementSpec)) {
132             throw new GordianDataException(GordianBaseData.getInvalidText(pAgreementSpec));
133         }
134     }

144     /**
145      * Check AgreementSpec.
146      *
147      * @param pSpec the agreementSpec
148      * @return true/false
149      */
150     protected boolean validAgreementSpec(final GordianAgreementSpec pSpec) {
151         /* Reject invalid agreementSpec */
152         if (pSpec == null || !pSpec.isValid()) {
153             return false;
154         }
155 
156         /* Check that spec is supported */
157         return pSpec.isSupported();
158     }
159 
160     @Override
161     public boolean validAgreementSpecForKeyPairSpec(final GordianKeyPairSpec pKeyPairSpec,
162                                                     final GordianAgreementSpec pAgreementSpec) {
163         /* Check that the agreementSpec is supported */
164         if (!validAgreementSpec(pAgreementSpec)) {
165             return false;
166         }
167 
168         /* Check agreement matches keySpec */
169         if (!pAgreementSpec.getKeyPairSpec().equals(pKeyPairSpec)) {
170             return false;
171         }
172 
173         /* For Edwards XDH, disallow 512KDF for 25519 and 256KDF for 448 */
174         if (pKeyPairSpec.getKeyPairType() == GordianKeyPairType.XDH) {
175             final GordianEdwardsElliptic myEdwards = pKeyPairSpec.getEdwardsElliptic();
176             switch (pAgreementSpec.getKDFType()) {
177                 case SHA256KDF:
178                 case SHA256CKDF:
179                     return myEdwards.is25519();
180                 case SHA512KDF:
181                 case SHA512CKDF:
182                     return !myEdwards.is25519();
183                 default:
184                     break;
185             }
186         }
187 
188         /* For Composite AgreementSpec */
189         if (pKeyPairSpec.getKeyPairType() == GordianKeyPairType.COMPOSITE) {
190             /* Access the subSpecs  */
191             final List<GordianAgreementSpec> mySubAgrees = GordianCompositeAgreement.getSubAgreements(pAgreementSpec);
192 
193             /* Loop through the subAgreements */
194             for (GordianAgreementSpec mySpec : mySubAgrees) {
195                 if (!validAgreementSpecForKeyPairSpec(mySpec.getKeyPairSpec(), mySpec)) {
196                     return false;
197                 }
198             }
199 
200             /* Check confirmation */
201             if (Boolean.TRUE.equals(pAgreementSpec.withConfirm())
202                 && !pAgreementSpec.getAgreementType().canConfirm()) {
203                 return false;
204             }
205 
206             /* Disallow SM2 with confirm */
207             return pAgreementSpec.getAgreementType() != GordianAgreementType.SM2
208                     || !pAgreementSpec.withConfirm();
209         }
210 
211         /* OK */
212         return true;
213     }

249     @Override
250     public List<GordianAgreementSpec> listAllSupportedAgreements(final GordianKeyPairSpec pKeyPairSpec) {
251         return listPossibleAgreements(pKeyPairSpec)
252                 .stream()
253                 .filter(supportedAgreements())
254                 .filter(s -> validAgreementSpecForKeyPairSpec(pKeyPairSpec, s))
255                 .toList();
256     }

258     /**
259      * Obtain a list of all possible agreements for the keyPairSpec.
260      * @param pKeyPairSpec the keyPairSpec
261      * @return the list
262      */
263     public List<GordianAgreementSpec> listPossibleAgreements(final GordianKeyPairSpec pKeyPairSpec) {
264         /* Create list */
265         final List<GordianAgreementSpec> myAgreements = new ArrayList<>();
266 
267         /* Switch on keyPairType */
268         switch (pKeyPairSpec.getKeyPairType()) {
269             case RSA:
270                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.KEM));
271                 break;
272             case NEWHOPE:
273                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.ANON));
274                 break;
275             case CMCE:
276             case FRODO:
277             case SABER:
278             case MLKEM:
279             case HQC:
280             case BIKE:
281             case NTRU:
282             case NTRUPRIME:
283                 myAgreements.add(GordianAgreementSpecBuilder.kem(pKeyPairSpec, GordianKDFType.NONE));
284                 break;
285             case EC:
286             case SM2:
287             case GOST2012:
288                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.ANON));
289                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.KEM));
290                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.BASIC));
291                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SIGNED));
292                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED));
293                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED, Boolean.TRUE));
294                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.MQV));
295                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.MQV, Boolean.TRUE));
296                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SM2));
297                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SM2, Boolean.TRUE));
298                 break;
299             case DH:
300             case DSTU4145:
301                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.ANON));
302                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.KEM));
303                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.BASIC));
304                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SIGNED));
305                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED));
306                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED, Boolean.TRUE));
307                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.MQV));
308                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.MQV, Boolean.TRUE));
309                 break;
310             case XDH:
311                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.ANON));
312                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.BASIC));
313                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SIGNED));
314                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED));
315                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED, Boolean.TRUE));
316                 break;
317             case COMPOSITE:
318                 /* Loop through the possible keySpecs for the first key */
319                 final Iterator<GordianKeyPairSpec> myIterator = pKeyPairSpec.keySpecIterator();
320                 for (GordianAgreementSpec mySpec : listPossibleAgreements(myIterator.next())) {
321                     final GordianAgreementSpec myTest = new GordianAgreementSpec(pKeyPairSpec, mySpec.getAgreementType(), mySpec.getKDFType(), mySpec.withConfirm());
322                     if (myTest.isValid()) {
323                         myAgreements.add(myTest);
324                     }
325                 }
326                 break;
327             default:
328                 break;
329         }
330 
331         /* Return the list */
332         return myAgreements;
333     }

335     /**
336      * Create list of KDF variants.
337      * @param pKeyPairSpec the keyPairSpec
338      * @param pAgreementType the agreementType
339      * @return the list
340      */
341     public static List<GordianAgreementSpec> listAllKDFs(final GordianKeyPairSpec pKeyPairSpec,
342                                                          final GordianAgreementType pAgreementType) {
343         return listAllKDFs(pKeyPairSpec, pAgreementType, Boolean.FALSE);
344     }

346     @Override
347     public GordianAgreementSpec defaultForKeyPair(final GordianKeyPairSpec pKeySpec) {
348         final Iterator<GordianAgreementSpec> myIterator = listAllSupportedAgreements(pKeySpec).iterator();
349         return myIterator.hasNext() ? myIterator.next() : null;
350     }

352     /**
353      * Create list of KDF variants.
354      * @param pKeyPairSpec the keyPairSpec
355      * @param pAgreementType the agreementType
356      * @param pConfirm with key confirmation
357      * @return the list
358      */
359     public static List<GordianAgreementSpec> listAllKDFs(final GordianKeyPairSpec pKeyPairSpec,
360                                                          final GordianAgreementType pAgreementType,
361                                                          final Boolean pConfirm) {
362         /* Create list */
363         final List<GordianAgreementSpec> myAgreements = new ArrayList<>();
364 
365         /* Loop through the KDFs */
366         for (final GordianKDFType myKDF : GordianKDFType.values()) {
367             myAgreements.add(new GordianAgreementSpec(pKeyPairSpec, pAgreementType, myKDF, pConfirm));
368         }
369 
370         /* Return the list */
371         return myAgreements;
372     }
