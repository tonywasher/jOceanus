134     @Override
135     public void reset() {
136         /* Reset underlying details */
137         super.reset();
138 
139         /* Reset client details */
140         theClient = null;
141         theClientEphemeral = null;
142         theClientConfirmation = null;
143 
144         /* Reset server details */
145         theServer = null;
146         theServerId = null;
147         theServerEphemeral = null;
148         theServerConfirmation = null;

151     @Override
152     protected void processSecret(final byte[] pSecret) throws GordianException {
153         /* If we are using confirmation and are not SM2 */
154         final GordianAgreementSpec mySpec = getAgreementSpec();
155         if (Boolean.TRUE.equals(mySpec.withConfirm())
156                 && mySpec.getAgreementType() != GordianAgreementType.SM2) {
157             /* calculate the confirmation tags */
158             calculateConfirmationTags(pSecret);
159         }
160 
161         /* Pass the call down */
162         super.processSecret(pSecret);
163     }

226     /**
227      * Create the clientHello ASN1.
228      * @param pClient the client keyPair
229      * @return the clientHello message
230      * @throws GordianException on error
231      */
232     public GordianAgreementMessageASN1 createClientHelloASN1(final GordianKeyPair pClient) throws GordianException {
233         /* Check the keyPair */
234         checkKeyPair(pClient);
235 
236         /* Store the keyPair */
237         theClient = pClient;
238 
239         /* Create ephemeral key */
240         final GordianKeyPairFactory myFactory = getFactory().getAsyncFactory().getKeyPairFactory();
241         final GordianKeyPairGenerator myGenerator = myFactory.getKeyPairGenerator(theClient.getKeyPairSpec());
242         theClientEphemeral = myGenerator.generateKeyPair();
243         final X509EncodedKeySpec myKeySpec = myGenerator.getX509Encoding(theClientEphemeral);
244 
245         /* Create the clientHello message */
246         final GordianAgreementMessageASN1 myClientHello = buildClientHelloASN1(myKeySpec);
247 
248         /* Set status */
249         setStatus(GordianAgreementStatus.AWAITING_SERVERHELLO);
250 
251         /* Return the clientHello */
252         return myClientHello;
253     }

255     @Override
256     public byte[] acceptClientHello(final GordianKeyPair pClient,
257                                     final GordianKeyPair pServer,
258                                     final byte[] pClientHello) throws GordianException {
259         /* Must be in clean state */
260         checkStatus(GordianAgreementStatus.CLEAN);
261 
262         /* Access the sequence */
263         final GordianAgreementMessageASN1 myClientHello = GordianAgreementMessageASN1.getInstance(pClientHello);
264         myClientHello.checkMessageType(GordianMessageType.CLIENTHELLO);
265 
266         /* Accept the ASN1 */
267         final GordianAgreementMessageASN1 myServerHello = acceptClientHelloASN1(pClient, pServer, myClientHello);
268         return myServerHello.getEncodedBytes();
269     }

283     /**
284      * Process the incoming clientHello message request.
285      * @param pClient the client keyPair
286      * @param pServer the server keyPair
287      * @param pClientHello the incoming clientHello message
288      * @throws GordianException on error
289      */
290     protected void processClientHelloASN1(final GordianKeyPair pClient,
291                                           final GordianKeyPair pServer,
292                                           final GordianAgreementMessageASN1 pClientHello) throws GordianException {
293         /* Check the keyPair */
294         checkKeyPair(pClient);
295         checkKeyPair(pServer);
296 
297         /* Store the keyPair */
298         theClient = pClient;
299         theServer = pServer;
300 
301         /* Parse the request */
302         parseClientHelloASN1(pClientHello);
303 
304         /* Parse the ephemeral encoding */
305         final X509EncodedKeySpec myKeySpec = pClientHello.getEphemeral();
306 
307         /* Create ephemeral key */
308         final GordianKeyPairFactory myFactory = getFactory().getAsyncFactory().getKeyPairFactory();
309         final GordianKeyPairGenerator myGenerator = myFactory.getKeyPairGenerator(theServer.getKeyPairSpec());
310         theServerEphemeral = myGenerator.generateKeyPair();
311 
312         /* Derive partner ephemeral key */
313         theClientEphemeral = myGenerator.derivePublicOnlyKeyPair(myKeySpec);
314 
315         /* Create the new serverIV */
316         newServerIV();
317     }

319     @Override
320     protected GordianAgreementMessageASN1 buildServerHello() throws GordianException {
321         /* Add server ephemeral and any server confirmation tag to the serverHello */
322         final GordianKeyPairFactory myFactory = getFactory().getAsyncFactory().getKeyPairFactory();
323         final GordianKeyPairGenerator myGenerator = myFactory.getKeyPairGenerator(theServerEphemeral.getKeyPairSpec());
324         final GordianAgreementMessageASN1 myHello = buildServerHello(myGenerator.getX509Encoding(theServerEphemeral), theServerConfirmation);
325         theServerId = myHello.getServerId();
326         return myHello;
327     }

330     public byte[] acceptServerHello(final GordianKeyPair pServer,
331                                     final byte[] pServerHello) throws GordianException {
332         /* Must be in clean state */
333         checkStatus(GordianAgreementStatus.AWAITING_SERVERHELLO);
334 
335         /* Access the sequence */
336         final GordianAgreementMessageASN1 myServerHello = GordianAgreementMessageASN1.getInstance(pServerHello);
337         myServerHello.checkMessageType(GordianMessageType.SERVERHELLO);
338 
339         /* Accept the ASN1 */
340         final GordianAgreementMessageASN1 myConfirm = acceptServerHelloASN1(pServer, myServerHello);
341         return myConfirm == null ? null : myConfirm.getEncodedBytes();
342     }

354     /**
355      * Process the serverHello.
356      * @param pServer the server keyPair
357      * @param pServerHello the serverHello message
358      * @throws GordianException on error
359      */
360     protected void processServerHelloASN1(final GordianKeyPair pServer,
361                                           final GordianAgreementMessageASN1 pServerHello) throws GordianException {
362         /* Check the keyPair */
363         checkKeyPair(pServer);
364 
365         /* Store the server keyPair */
366         theServer = pServer;
367 
368         /* Obtain details from the serverHello */
369         parseServerHelloASN1(pServerHello);
370         theServerConfirmation = pServerHello.getConfirmation();
371         theServerId = pServerHello.getServerId();
372         final X509EncodedKeySpec myKeySpec = pServerHello.getEphemeral();
373 
374         /* Derive partner ephemeral key */
375         final GordianKeyPairFactory myFactory = getFactory().getAsyncFactory().getKeyPairFactory();
376         final GordianKeyPairGenerator myGenerator = myFactory.getKeyPairGenerator(theClient.getKeyPairSpec());
377         theServerEphemeral = myGenerator.derivePublicOnlyKeyPair(myKeySpec);
378     }

380     /**
381      * Build clientConfirm message.
382      * @return the clientConfirm message
383      * @throws GordianException on error
384      */
385     protected GordianAgreementMessageASN1 buildClientConfirmASN1() throws GordianException {
386         /* If there is no client confirmation, return null */
387         if (theClientConfirmation == null) {
388             return null;
389         }
390 
391         /* Create the response */
392         final GordianCoreAgreementFactory myFactory = getAgreementFactory();
393         final AlgorithmIdentifier myAlgId = myFactory.getIdentifierForSpec(getAgreementSpec());
394         return GordianAgreementMessageASN1.newClientConfirm(theServerId)
395                 .setAgreementId(myAlgId)
396                 .setConfirmation(theClientConfirmation);
397     }

401         /* Must be in awaiting clientConfirm state */
402         checkStatus(GordianAgreementStatus.AWAITING_CLIENTCONFIRM);
403 
404         /* Access the sequence */
405         final GordianAgreementMessageASN1 myClientConfirm = GordianAgreementMessageASN1.getInstance(pClientConfirm);
406         myClientConfirm.checkMessageType(GordianMessageType.CLIENTCONFIRM);
407 
408         /* Process the confirmation */
409         acceptClientConfirmASN1(myClientConfirm);
410 
411         /* Set result available status */
412         setStatus(GordianAgreementStatus.RESULT_AVAILABLE);

415     /**
416      * Accept a client confirm message.
417      * @param pClientConfirm the confirm message
418      * @throws GordianException on error
419      */
420     public void acceptClientConfirmASN1(final GordianAgreementMessageASN1 pClientConfirm) throws GordianException {
421         /* Access message parts */
422         final AlgorithmIdentifier myAlgId = pClientConfirm.getAgreementId();
423         final byte[] myConfirm = pClientConfirm.getConfirmation();
424 
425         /* Check serverId */
426         if (!Objects.equals(theServerId, pClientConfirm.getServerId())) {
427             throw new GordianDataException("Mismatch on serverId");
428         }
429 
430         /* Check agreementSpec */
431         final GordianCoreAgreementFactory myFactory = getAgreementFactory();
432         final GordianAgreementSpec mySpec = myFactory.getSpecForIdentifier(myAlgId);
433         if (!Objects.equals(mySpec, getAgreementSpec())) {
434             throw new GordianDataException(ERROR_INVSPEC);
435         }
436 
437         /* Validate the confirmation */
438         if (!Arrays.constantTimeAreEqual(theClientConfirmation, myConfirm)) {
439             throw new GordianDataException("Confirmation failed");
440         }
441     }

443     /**
444      * Create the confirmation tags.
445      * @param pSecret the secret
446      * @throws GordianException on error
447      */
448     private void calculateConfirmationTags(final byte[] pSecret) throws GordianException {
449         /* Derive the key */
450         final byte[] myKey = calculateDerivedSecret(pSecret, GordianDerivationId.TAGS, GordianLength.LEN_512.getByteLength());
451 
452         /* Create the hMac and initialise with the key */
453         final GordianFactory myBaseFactory = getFactory();
454         final GordianMacFactory myMacs = myBaseFactory.getMacFactory();
455         final GordianMacSpec mySpec = GordianMacSpecBuilder.hMac(GordianDerivationId.TAGS.getDigestType());
456         final GordianMac myMac = myMacs.createMac(mySpec);
457         myMac.initKeyBytes(myKey);
458 
459         /* Access the keyPairGenerator and obtain public encodings */
460         final GordianKeyPairFactory myFactory = myBaseFactory.getAsyncFactory().getKeyPairFactory();
461         final GordianKeyPairGenerator myGenerator = myFactory.getKeyPairGenerator(getClientKeyPair().getKeyPairSpec());
462         final byte[] myClient = myGenerator.getX509Encoding(getClientKeyPair()).getEncoded();
463         final byte[] myClientEphemeral = myGenerator.getX509Encoding(getClientEphemeralKeyPair()).getEncoded();
464         final byte[] myServer = myGenerator.getX509Encoding(getServerKeyPair()).getEncoded();
465         final byte[] myServerEphemeral = myGenerator.getX509Encoding(getServerEphemeralKeyPair()).getEncoded();
466 
467         /* Build Server Confirmation tag */
468         myMac.update(myServer);
469         myMac.update(myClient);
470         myMac.update(myServerEphemeral);
471         myMac.update(myClientEphemeral);
472         final byte[] myServerConfirmation = myMac.finish();
473 
474         /* Check that the server value matches any sent value */
475         if (theServerConfirmation != null
476                 && !Arrays.constantTimeAreEqual(myServerConfirmation, theServerConfirmation)) {
477             throw new GordianDataException("Confirmation failed");
478         }
479         theServerConfirmation = myServerConfirmation;
480 
481         /* Build Client Confirmation tag */
482         myMac.update(myClient);
483         myMac.update(myServer);
484         myMac.update(myClientEphemeral);
485         myMac.update(myServerEphemeral);
486         theClientConfirmation = myMac.finish();
487     }
