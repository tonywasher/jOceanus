17  package net.sourceforge.joceanus.gordianknot.impl.core.random;
18  
19  import net.sourceforge.joceanus.gordianknot.api.base.GordianException;
20  import net.sourceforge.joceanus.gordianknot.api.mac.GordianMac;
21  import net.sourceforge.joceanus.gordianknot.impl.core.base.GordianByteArrayInteger;
22  import org.bouncycastle.crypto.prng.EntropySource;
23  import org.bouncycastle.util.Arrays;
24  
25  /**
26   * Implementation of HMacSP800DRBG based on the BouncyCastle Code.
27   * <p>
28   * This implementation is modified so that it accepts any GordianMac.
29   */
30  public final class GordianSP800HMacDRBG
31          implements GordianDRBGenerator {
32      /**
33       * The Update Id.
34       */
35      private static final byte[] UPDATE_ID = { 0 };
36  
37      /**
38       * The ReSeed Id.
39       */
40      private static final byte[] SEED_ID = { 1 };
41  
42      /**
43       * The HMac.
44       */
45      private final GordianMac theHMac;
46  
47      /**
48       * The Entropy Source.
49       */
50      private final EntropySource theEntropy;
51  
52      /**
53       * The ReSeed Counter.
54       */
55      private final GordianByteArrayInteger theReseedCounter;
56  
57      /**
58       * The Key.
59       */
60      private final byte[] theKey;
61  
62      /**
63       * The Hash.
64       */
65      private final byte[] theHash;
66  
67      /**
68       * Construct a SP800-90A Hash DRBG.
69       * @param pHMac Hash MAC to base the DRBG on.
70       * @param pEntropy source of entropy to use for seeding/reSeeding.
71       * @param pSecurityBytes personalisation string to distinguish this DRBG (may be null).
72       * @param pInitVector nonce to further distinguish this DRBG (may be null).
73       */
74      public GordianSP800HMacDRBG(final GordianMac pHMac,
75                                  final EntropySource pEntropy,
76                                  final byte[] pSecurityBytes,
77                                  final byte[] pInitVector) {
78          /* Store hMac and entropy source */
79          theHMac = pHMac;
80          theEntropy = pEntropy;
81  
82          /* Create Seed Material */
83          final byte[] myEntropy = theEntropy.getEntropy();
84          final byte[] mySeed = Arrays.concatenate(myEntropy, pInitVector, pSecurityBytes);
85  
86          /* Initialise buffers */
87          final int myLen = theHMac.getMacSize();
88          theKey = new byte[myLen];
89          theHash = new byte[myLen];
90          Arrays.fill(theHash, (byte) 1);
91  
92          /* Update the state */
93          updateState(mySeed);
94  
95          /* Initialise reSeed counter */
96          theReseedCounter = new GordianByteArrayInteger(Long.BYTES);
97          theReseedCounter.iterate();
98      }
99  
100     /**
101      * Update the state (HMAC_DRBG_Update).
102      * @param pSeed the extra seed material
103      */
104     private void updateState(final byte[] pSeed) {
105         try {
106             updateState(pSeed, UPDATE_ID);
107             if (pSeed != null) {
108                 updateState(pSeed, SEED_ID);
109             }
110         } catch (GordianException e) {
111             throw new IllegalStateException(e);
112         }
113     }
114 
115     /**
116      * Update the state.
117      * @param pSeed optional seed material
118      * @param pCycle the cycle id
119      * @throws GordianException on error
120      */
121     private void updateState(final byte[] pSeed,
122                              final byte[] pCycle) throws GordianException {
123 
124         /* Initialise the hMac */
125         theHMac.initKeyBytes(theKey);
126 
127         /* Update with hash and cycle id */
128         theHMac.update(theHash);
129         theHMac.update(pCycle);
130 
131         /* Add any seed material */
132         if (pSeed != null) {
133             theHMac.update(pSeed);
134         }
135 
136         /* Generate new key */
137         theHMac.finish(theKey, 0);
138 
139         /* Calculate new hash */
140         theHMac.initKeyBytes(theKey);
141         theHMac.update(theHash);
142         theHMac.finish(theHash, 0);
143     }
144 
145     @Override
146     public void reseed(final byte[] pXtraBytes) {
147         /* Create seed material */
148         final byte[] myEntropy = theEntropy.getEntropy();
149         final byte[] mySeed = Arrays.concatenate(myEntropy, pXtraBytes);
150 
151         /* Update the state */
152         updateState(mySeed);
153 
154         /* re-initialise reSeed counter */
155         theReseedCounter.reset();
156         theReseedCounter.iterate();
157     }
158 
159     @Override
160     public int generate(final byte[] pOutput,
161                         final byte[] pXtraBytes,
162                         final boolean isPredictionResistant) {
163         /* Check valid # of bits */
164         final int myLen = pOutput.length;
165         final int myNumBits = myLen << GordianCoreRandomFactory.BIT_SHIFT;
166         if (myNumBits > GordianCoreRandomFactory.MAX_BITS_REQUEST) {
167             throw new IllegalArgumentException("Number of bits per request limited to "
168                     + GordianCoreRandomFactory.MAX_BITS_REQUEST);
169         }
170 
171         /* Check for reSeed required */
172         if (theReseedCounter.compareLimit(GordianCoreRandomFactory.RESEED_MAX)) {
173             return -1;
174         }
175 
176         /* Access XtraBytes */
177         byte[] myXtraBytes = pXtraBytes;
178 
179         /* If we are prediction resistant */
180         if (isPredictionResistant) {
181             /* ReSeed and discard xtraBytes */
182             reseed(myXtraBytes);
183             myXtraBytes = null;
184 
185             /* else if we have extra bytes */
186         } else if (myXtraBytes != null) {
187             /* Update the state */
188             updateState(myXtraBytes);
189         }
190 
191         /* Allocate output buffer */
192         final byte[] myResult = new byte[myLen];
193 
194         /* Protect against exceptions */
195         try {
196             /* Initialise the hMac */
197             theHMac.initKeyBytes(theKey);
198             final int mySize = theHMac.getMacSize();
199 
200             /* while we need to generate more bytes */
201             int myBuilt = 0;
202             while (myBuilt < myLen) {
203                 /* Update the mac */
204                 theHMac.update(theHash);
205                 theHMac.finish(theHash, 0);
206 
207                 /* Determine how many bytes of this hash should be used */
208                 int myNeeded = myLen
209                         - myBuilt;
210                 if (myNeeded > mySize) {
211                     myNeeded = mySize;
212                 }
213 
214                 /* Copy bytes across */
215                 System.arraycopy(theHash, 0, myResult, myBuilt, myNeeded);
216                 myBuilt += myNeeded;
217             }
218         } catch (GordianException e) {
219             throw new IllegalStateException(e);
220         }
221 
222         /* Update the state */
223         updateState(myXtraBytes);
224 
225         /* Iterate the reSeed counter */
226         theReseedCounter.iterate();
227 
228         /* Return the bytes */
229         System.arraycopy(myResult, 0, pOutput, 0, pOutput.length);
230 
231         /* Return the number of bits generated */
232         return myNumBits;
233     }
234 
235     @Override
236     public int getBlockSize() {
237         return theHMac.getMacSize();
238     }
239 
240     @Override
241     public String getAlgorithm() {
242         return GordianCoreRandomFactory.SP800_PREFIX + theHMac.getMacSpec().toString();
243     }
244 }
