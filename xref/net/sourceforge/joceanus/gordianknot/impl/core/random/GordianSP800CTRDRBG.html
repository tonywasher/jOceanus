19  import net.sourceforge.joceanus.gordianknot.api.base.GordianException;
20  import net.sourceforge.joceanus.gordianknot.api.cipher.GordianSymKeySpec;
21  import net.sourceforge.joceanus.gordianknot.impl.core.base.GordianByteArrayInteger;
22  import net.sourceforge.joceanus.gordianknot.impl.core.base.GordianDataConverter;
23  import net.sourceforge.joceanus.gordianknot.impl.core.cipher.GordianCoreCipher;
24  import org.bouncycastle.crypto.prng.EntropySource;
25  import org.bouncycastle.util.Arrays;

27  /**
28   * Implementation of CTRSP800DRBG based on the BouncyCastle Code.
29   * <p>
30   * This implementation is modified so that it accepts any GordianCipher.
31   */
32  public class GordianSP800CTRDRBG
33          implements GordianDRBGenerator {
34      /**
35       * The endData flag.
36       */
37      private static final byte ENDDATA = (byte) 0x80;
38  
39      /**
40       * The Cipher.
41       */
42      private final GordianCoreCipher<GordianSymKeySpec> theCipher;
43  
44      /**
45       * The Entropy Source.
46       */
47      private final EntropySource theEntropy;
48  
49      /**
50       * The ReSeed Counter.
51       */
52      private final GordianByteArrayInteger theReseedCounter;
53  
54      /**
55       * The Seed Length.
56       */
57      private final int theSeedLen;
58  
59      /**
60       * The Key.
61       */
62      private final byte[] theKey;
63  
64      /**
65       * The Variable Source.
66       */
67      private final GordianByteArrayInteger theV;
68  
69      /**
70       * The cipher blockSize (bytes).
71       */
72      private final int theBlockLen;
73  
74      /**
75       * Construct a SP800-90A Hash DRBG.
76       * @param pCipher SIC Cipher to base the DRBG on.
77       * @param pEntropy source of entropy to use for seeding/reSeeding.
78       * @param pSecurityBytes personalisation string to distinguish this DRBG (may be null).
79       * @param pInitVector nonce to further distinguish this DRBG (may be null).
80       * @throws GordianException on error
81       */
82      public GordianSP800CTRDRBG(final GordianCoreCipher<GordianSymKeySpec> pCipher,
83                                 final EntropySource pEntropy,
84                                 final byte[] pSecurityBytes,
85                                 final byte[] pInitVector) throws GordianException {
86          /* Store cipher and entropy source */
87          theCipher = pCipher;
88          theEntropy = pEntropy;
89  
90          /* Initialise buffers */
91          final int myKeyLen = pCipher.getKeyType().getKeyLength().getByteLength();
92          theBlockLen = theCipher.getKeyType().getBlockLength().getByteLength();
93          theKey = new byte[myKeyLen];
94          theV = new GordianByteArrayInteger(theBlockLen);
95          theSeedLen = (myKeyLen + theBlockLen) * Byte.SIZE;
96  
97          /* Create Seed Material */
98          final byte[] myEntropy = theEntropy.getEntropy();
99          final byte[] mySeedInput = Arrays.concatenate(myEntropy, pInitVector, pSecurityBytes);
100         final byte[] mySeed = blockCipherDF(mySeedInput, theSeedLen);
101 
102         /* Update the state */
103         ctrDRBGUpdate(mySeed);
104 
105         /* Initialise reSeed counter */
106         theReseedCounter = new GordianByteArrayInteger(Long.BYTES);
107         theReseedCounter.iterate();
108     }
109 
110     /**
111      * Block Cipher derivation function.
112      * @param pInput the seed material
113      * @param pNumBits the number of bits to return
114      * @return the derived seed
115      * @throws GordianException on error
116      */
117     private byte[] blockCipherDF(final byte[] pInput,
118                                  final int pNumBits) throws GordianException {
119         /* Check valid # of bits */
120         if (pNumBits > GordianCoreRandomFactory.MAX_BITS_REQUEST) {
121             throw new IllegalArgumentException("Number of bits per request limited to "
122                     + GordianCoreRandomFactory.MAX_BITS_REQUEST);
123         }
124 
125         /* Allocate the lengths */
126         final byte[] myL = GordianDataConverter.integerToByteArray(pInput.length);
127         final byte[] myN = GordianDataConverter.integerToByteArray(pNumBits / Byte.SIZE);
128 
129         /* Create the input buffer */
130         final int myKeyLen = theKey.length;
131         final int myBaseLen = (2 * Integer.BYTES) + pInput.length + 1;
132         final int myDataLen = theBlockLen * ((myBaseLen + theBlockLen - 1) / theBlockLen);
133         final byte[] myBlock = new byte[myDataLen];
134         System.arraycopy(myL, 0, myBlock, 0, Integer.BYTES);
135         System.arraycopy(myN, 0, myBlock, Integer.BYTES, Integer.BYTES);
136         System.arraycopy(pInput, 0, myBlock, 2 * Integer.BYTES, pInput.length);
137         myBlock[2 * Integer.BYTES + pInput.length] = ENDDATA;
138 
139         /* Create the key buffer and initialise it */
140         final byte[] myKey = new byte[myKeyLen];
141         for (int i = 0; i < myKeyLen; i++) {
142             myKey[i] = (byte) i;
143         }
144 
145         /* Create the temporary buffers */
146         final byte[] myTemp = new byte[myKeyLen + theBlockLen];
147         final byte[] myOut = new byte[theBlockLen];
148         final byte[] myIV = new byte[theBlockLen];
149 
150         /* while we need to generate more bytes */
151         int myBuilt = 0;
152         final GordianByteArrayInteger myI = new GordianByteArrayInteger();
153         while (myBuilt < myTemp.length) {
154             /* Update the buffer */
155             System.arraycopy(myI.getBuffer(), 0, myIV, 0, Integer.BYTES);
156             blockCC(myOut, myKey, myIV, myBlock);
157 
158             /* Determine how many bytes of this hash should be used */
159             int myNeeded = myTemp.length
160                     - myBuilt;
161             if (myNeeded > theBlockLen) {
162                 myNeeded = theBlockLen;
163             }
164 
165             /* Copy bytes across */
166             System.arraycopy(myOut, 0, myTemp, myBuilt, myNeeded);
167             myBuilt += myNeeded;
168             myI.iterate();
169         }
170 
171         /* Access key and IV from temp */
172         System.arraycopy(myTemp, 0, myKey, 0, myKeyLen);
173         System.arraycopy(myTemp, myKeyLen, myOut, 0, theBlockLen);
174 
175         /* while we need to generate more bytes */
176         myBuilt = 0;
177         while (myBuilt < myTemp.length) {
178             /* Encrypt the bytes */
179             theCipher.initKeyBytes(myKey);
180             theCipher.finish(myOut, 0, myOut.length, myOut, 0);
181 
182             /* Determine how many bytes of this hash should be used */
183             int myNeeded = myTemp.length
184                     - myBuilt;
185             if (myNeeded > theBlockLen) {
186                 myNeeded = theBlockLen;
187             }
188 
189             /* Copy bytes across */
190             System.arraycopy(myOut, 0, myTemp, myBuilt, myNeeded);
191             myBuilt += myNeeded;
192             myI.iterate();
193         }
194 
195         /* Return the derived bytes */
196         return myTemp;
197     }

199     /**
200      * BCC Cipher Chaining.
201      * @param pOutput the output buffer
202      * @param pKey the key
203      * @param pIV the initVector
204      * @param pData the data
205      * @throws GordianException on error
206      */
207     private void blockCC(final byte[] pOutput,
208                          final byte[] pKey,
209                          final byte[] pIV,
210                          final byte[] pData) throws GordianException {
211         /* Build the buffers */
212         final byte[] myChain = new byte[theBlockLen];
213         final byte[] myIn  = new byte[theBlockLen];
214 
215         /* Encrypt the IV */
216         theCipher.initKeyBytes(pKey);
217         theCipher.finish(pIV, 0, theBlockLen, myChain, 0);
218 
219         /* Loop through the data */
220         final int myNumBlocks = pData.length / theBlockLen;
221         for (int i = 0; i < myNumBlocks; i++) {
222             /* Xor in the value */
223             final int offset = i * theBlockLen;
224             for (int j = 0; j < theBlockLen; j++) {
225                 myIn[j] = (byte) (myChain[j] ^ pData[offset + j]);
226             }
227 
228             /* Encrypt it */
229             theCipher.finish(myIn, 0, theBlockLen, myChain, 0);
230         }
231 
232         /* Copy the output */
233         System.arraycopy(myChain, 0, pOutput, 0, theBlockLen);
234     }

236     /**
237      * Update state.
238      * @param pSeed the seed material
239      * @throws GordianException on error
240      */
241     private void ctrDRBGUpdate(final byte[] pSeed) throws GordianException {
242         /* Create the buffers */
243         final byte[] myOut = new byte[theBlockLen];
244         final byte[] myResult = new byte[theSeedLen / Byte.SIZE];
245 
246         /* while we need to generate more bytes */
247         int myBuilt = 0;
248         while (myBuilt < myResult.length) {
249             /* Encrypt the bytes */
250             theCipher.initKeyBytes(theKey);
251             theV.iterate();
252             theCipher.finish(theV.getBuffer(), 0, theBlockLen, myOut, 0);
253 
254             /* Determine how many bytes of this hash should be used */
255             int myNeeded = myResult.length
256                     - myBuilt;
257             if (myNeeded > theBlockLen) {
258                 myNeeded = theBlockLen;
259             }
260 
261             /* Copy bytes across */
262             System.arraycopy(myOut, 0, myResult, myBuilt, myNeeded);
263             myBuilt += myNeeded;
264         }
265 
266         /* Xor in the Seed */
267         for (int i = 0; i < myResult.length; i++) {
268             myResult[i] ^= pSeed[i];
269         }
270 
271         /* Update key and IV from result*/
272         System.arraycopy(myResult, 0, theKey, 0, theKey.length);
273         System.arraycopy(myResult, theKey.length, theV.getBuffer(), 0, theBlockLen);
274     }

276     @Override
277     public void reseed(final byte[] pXtraBytes) {
278         try {
279             /* Hash the new input and add to variable hash */
280             final byte[] myEntropy = theEntropy.getEntropy();
281             final byte[] mySeedInput = Arrays.concatenate(myEntropy, pXtraBytes);
282             final byte[] mySeed = blockCipherDF(mySeedInput, theSeedLen);
283 
284             /* Update the state */
285             ctrDRBGUpdate(mySeed);
286 
287             /* re-initialise reSeed counter */
288             theReseedCounter.reset();
289             theReseedCounter.iterate();
290         } catch (GordianException e) {
291             throw new IllegalStateException(e);
292         }
293     }

295     @Override
296     public int generate(final byte[] pOutput,
297                         final byte[] pXtraBytes,
298                         final boolean isPredictionResistant) {
299         /* Check valid # of bits */
300         final int myNumBits = pOutput.length << GordianCoreRandomFactory.BIT_SHIFT;
301         if (myNumBits > GordianCoreRandomFactory.MAX_BITS_REQUEST) {
302             throw new IllegalArgumentException("Number of bits per request limited to "
303                     + GordianCoreRandomFactory.MAX_BITS_REQUEST);
304         }
305 
306         /* Check for reSeed required */
307         if (theReseedCounter.compareLimit(GordianCoreRandomFactory.RESEED_MAX)) {
308             return -1;
309         }
310 
311         /* If we are prediction resistant */
312         byte[] mySeed = pXtraBytes;
313         if (isPredictionResistant) {
314             /* reSeed */
315             reseed(pXtraBytes);
316             mySeed = null;
317         }
318 
319         try {
320             /* if we have extra bytes */
321             if (mySeed != null) {
322                 /* Derive the new input and process */
323                 mySeed = blockCipherDF(pXtraBytes, theSeedLen);
324                 ctrDRBGUpdate(mySeed);
325 
326                 /* else allocate a buffer of zeroes */
327             } else {
328                 mySeed = new byte[theBlockLen + theKey.length];
329             }
330 
331             /* Create the buffers */
332             final byte[] myOut = new byte[theBlockLen];
333             final byte[] myResult = new byte[pOutput.length];
334 
335             /* while we need to generate more bytes */
336             int myBuilt = 0;
337             while (myBuilt < myResult.length) {
338                 /* Encrypt the bytes */
339                 theCipher.initKeyBytes(theKey);
340                 theV.iterate();
341                 theCipher.finish(theV.getBuffer(), 0, theBlockLen, myOut, 0);
342 
343                 /* Determine how many bytes of this hash should be used */
344                 int myNeeded = myResult.length
345                         - myBuilt;
346                 if (myNeeded > theBlockLen) {
347                     myNeeded = theBlockLen;
348                 }
349 
350                 /* Copy bytes across */
351                 System.arraycopy(myOut, 0, myResult, myBuilt, myNeeded);
352                 myBuilt += myNeeded;
353             }
354 
355             /* Update the state */
356             ctrDRBGUpdate(mySeed);
357 
358             /* Iterate the reSeed counter */
359             theReseedCounter.iterate();
360 
361             /* Return the bytes */
362             System.arraycopy(myResult, 0, pOutput, 0, pOutput.length);
363         } catch (GordianException e) {
364             throw new IllegalStateException(e);
365         }
366 
367         /* Return the number of bits generated */
368         return myNumBits;
369     }
370 
371     @Override
372     public int getBlockSize() {
373         return theBlockLen * Byte.SIZE;
374     }
375 
376     @Override
377     public String getAlgorithm() {
378         return GordianCoreRandomFactory.SP800_PREFIX + theCipher.getCipherSpec().toString();
379     }
