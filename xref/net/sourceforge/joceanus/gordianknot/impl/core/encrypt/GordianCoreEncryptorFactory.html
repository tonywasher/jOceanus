39  /**
40   * GordianKnot base for encryptorFactory.
41   */
42  public abstract class GordianCoreEncryptorFactory
43      implements GordianEncryptorFactory {
44      /**
45       * The factory.
46       */
47      private final GordianBaseFactory theFactory;
48  
49      /**
50       * The algorithm Ids.
51       */
52      private GordianEncryptorAlgId theAlgIds;
53  
54      /**
55       * Constructor.
56       * @param pFactory the factory
57       */
58      protected GordianCoreEncryptorFactory(final GordianBaseFactory pFactory) {
59          theFactory = pFactory;
60      }
61  
62      /**
63       * Obtain the factory.
64       * @return the factory
65       */
66      protected GordianBaseFactory getFactory() {
67          return theFactory;
68      }
69  
70      @Override
71      public Predicate<GordianEncryptorSpec> supportedEncryptors() {
72          return this::validEncryptorSpec;
73      }
74  
75      /**
76       * Check the encryptorSpec.
77       * @param pEncryptorSpec the encryptorSpec
78       * @throws GordianException on error
79       */
80      protected void checkEncryptorSpec(final GordianEncryptorSpec pEncryptorSpec) throws GordianException {
81          /* Check validity of encryptor */
82          if (!validEncryptorSpec(pEncryptorSpec)) {
83              throw new GordianDataException(GordianBaseData.getInvalidText(pEncryptorSpec));
84          }
85      }
86  
87      /**
88       * Check EncryptorSpec.
89       * @param pSpec the encryptorSpec
90       * @return true/false
91       */
92      protected boolean validEncryptorSpec(final GordianEncryptorSpec pSpec) {
93          /* Reject invalid encryptorSpec */
94          if (pSpec == null || !pSpec.isValid()) {
95              return false;
96          }
97  
98          /* For Composite EncryptorSpec */
99          if (pSpec.getKeyPairType() == GordianKeyPairType.COMPOSITE) {
100             /* Loop through the specs */
101             final Iterator<GordianEncryptorSpec> myIterator = pSpec.encryptorSpecIterator();
102             while (myIterator.hasNext()) {
103                 final GordianEncryptorSpec mySpec = myIterator.next();
104                 if (!validEncryptorSpec(mySpec)) {
105                     return false;
106                 }
107             }
108         }
109 
110         /* Check that spec is supported */
111         return pSpec.isSupported();
112     }
113 
114     @Override
115     public boolean validEncryptorSpecForKeyPairSpec(final GordianKeyPairSpec pKeyPairSpec,
116                                                     final GordianEncryptorSpec pEncryptorSpec) {
117         /* Check that the encryptorSpec is supported */
118         if (!validEncryptorSpec(pEncryptorSpec)) {
119             return false;
120         }
121 
122         /* Check encryptor matches keyPair */
123         final GordianKeyPairType myKeyType = pKeyPairSpec.getKeyPairType();
124         final GordianKeyPairType myEncType = pEncryptorSpec.getKeyPairType();
125         switch (myEncType) {
126             case SM2:
127             case EC:
128                 if (!GordianKeyPairType.EC.equals(myKeyType)
129                     && !GordianKeyPairType.GOST2012.equals(myKeyType)
130                     && !GordianKeyPairType.SM2.equals(myKeyType)) {
131                     return false;
132                 }
133                 break;
134             default:
135                 if (!myKeyType.equals(myEncType)) {
136                     return false;
137                 }
138                 break;
139         }
140 
141         /* Disallow EC if the curve does not support encryption */
142         if (GordianKeyPairType.EC.equals(pKeyPairSpec.getKeyPairType())) {
143             return pKeyPairSpec.getElliptic().canEncrypt();
144         }
145 
146         /* If this is a RSA encryption */
147         if (GordianKeyPairType.RSA.equals(pKeyPairSpec.getKeyPairType())) {
148             /* The digest length cannot be too large wrt to the modulus */
149             int myLen = pEncryptorSpec.getDigestSpec().getDigestLength().getByteLength();
150             myLen = (myLen + 1) * Byte.SIZE;
151             return pKeyPairSpec.getRSAModulus().getLength() >= (myLen << 1);
152         }
153 
154         /* For Composite EncryptorSpec */
155         if (pKeyPairSpec.getKeyPairType() == GordianKeyPairType.COMPOSITE) {
156             /* Loop through the keyPairs */
157             final Iterator<GordianKeyPairSpec> pairIterator = pKeyPairSpec.keySpecIterator();
158             final Iterator<GordianEncryptorSpec> encIterator = pEncryptorSpec.encryptorSpecIterator();
159             while (pairIterator.hasNext() && encIterator.hasNext()) {
160                 final GordianKeyPairSpec myPairSpec = pairIterator.next();
161                 final GordianEncryptorSpec myEncSpec = encIterator.next();
162                if (!validEncryptorSpecForKeyPairSpec(myPairSpec, myEncSpec)) {
163                     return false;
164                 }
165             }
166             if (pairIterator.hasNext() || encIterator.hasNext()) {
167                 return false;
168             }
169         }
170 
171         /* OK */
172         return true;
173     }

193     /**
194      * Obtain the encryptor algorithm Ids.
195      * @return the encryptor Algorithm Ids
196      */
197     private GordianEncryptorAlgId getAlgorithmIds() {
198         if (theAlgIds == null) {
199             theAlgIds = new GordianEncryptorAlgId(theFactory);
200         }
201         return theAlgIds;
202     }

204     @Override
205     public List<GordianEncryptorSpec> listAllSupportedEncryptors(final GordianKeyPair pKeyPair) {
206         return listAllSupportedEncryptors(pKeyPair.getKeyPairSpec());
207     }
208 
209     @Override
210     public List<GordianEncryptorSpec> listAllSupportedEncryptors(final GordianKeyPairSpec pKeyPairSpec) {
211         return listPossibleEncryptors(pKeyPairSpec.getKeyPairType())
212                 .stream()
213                 .filter(s -> validEncryptorSpecForKeyPairSpec(pKeyPairSpec, s))
214                 .toList();
215     }
216 
217     @Override
218     public List<GordianEncryptorSpec> listPossibleEncryptors(final GordianKeyPairType pKeyPairType) {
219         /* Create list */
220         final List<GordianEncryptorSpec> myEncryptors = new ArrayList<>();
221 
222         /* Switch on keyPairType */
223         switch (pKeyPairType) {
224             case RSA:
225                 myEncryptors.add(GordianEncryptorSpecBuilder.rsa(GordianDigestSpecBuilder.sha2(GordianLength.LEN_224)));
226                 myEncryptors.add(GordianEncryptorSpecBuilder.rsa(GordianDigestSpecBuilder.sha2(GordianLength.LEN_256)));
227                 myEncryptors.add(GordianEncryptorSpecBuilder.rsa(GordianDigestSpecBuilder.sha2(GordianLength.LEN_384)));
228                 myEncryptors.add(GordianEncryptorSpecBuilder.rsa(GordianDigestSpecBuilder.sha2(GordianLength.LEN_512)));
229                 break;
230             case ELGAMAL:
231                 myEncryptors.add(GordianEncryptorSpecBuilder.elGamal(GordianDigestSpecBuilder.sha2(GordianLength.LEN_224)));
232                 myEncryptors.add(GordianEncryptorSpecBuilder.elGamal(GordianDigestSpecBuilder.sha2(GordianLength.LEN_256)));
233                 myEncryptors.add(GordianEncryptorSpecBuilder.elGamal(GordianDigestSpecBuilder.sha2(GordianLength.LEN_384)));
234                 myEncryptors.add(GordianEncryptorSpecBuilder.elGamal(GordianDigestSpecBuilder.sha2(GordianLength.LEN_512)));
235                 break;
236             case EC:
237             case SM2:
238             case GOST2012:
239                 /* Add EC-ElGamal */
240                 myEncryptors.add(GordianEncryptorSpecBuilder.ec());
241 
242                 /* Loop through the encryptionSpecs */
243                 for (GordianSM2EncryptionSpec mySpec : GordianSM2EncryptionSpec.listPossibleSpecs()) {
244                     myEncryptors.add(GordianEncryptorSpecBuilder.sm2(mySpec));
245                 }
246                 break;
247             default:
248                 break;
249         }
250 
251         /* Return the list */
252         return myEncryptors;
253     }
254 
255     @Override
256     public GordianEncryptorSpec defaultForKeyPair(final GordianKeyPairSpec pKeySpec) {
257         switch (pKeySpec.getKeyPairType()) {
258             case RSA:
259                 return GordianEncryptorSpecBuilder.rsa(GordianDigestSpecBuilder.sha2(GordianLength.LEN_512));
260             case EC:
261             case SM2:
262             case GOST2012:
263                 return GordianEncryptorSpecBuilder.sm2(GordianSM2EncryptionSpec.c1c2c3(GordianDigestSpecBuilder.sm3()));
264             case ELGAMAL:
265                 return GordianEncryptorSpecBuilder.elGamal(GordianDigestSpecBuilder.sha2(GordianLength.LEN_512));
266             case COMPOSITE:
267                 final List<GordianEncryptorSpec> mySpecs = new ArrayList<>();
268                 final Iterator<GordianKeyPairSpec> myIterator = pKeySpec.keySpecIterator();
269                 while (myIterator.hasNext()) {
270                     final GordianKeyPairSpec mySpec = myIterator.next();
271                     mySpecs.add(defaultForKeyPair(mySpec));
272                 }
273                 final GordianEncryptorSpec mySpec = GordianEncryptorSpecBuilder.composite(mySpecs);
274                 return mySpec.isValid() ? mySpec : null;
275             default:
276                 return null;
277         }
278     }
