36  /**
37   * ASN1 Encoding of passwordLockSpec.
38   * <pre>
39   * GordianPasswordLockSpecASN1 ::= SEQUENCE {
40   *      keySetSpec GordianKeySetSpecASN1
41   *      iterations INTEGER
42   * }
43   * </pre>
44   */
45  public class GordianPasswordLockSpecASN1
46          extends GordianASN1Object {
47      /**
48       * The PasswordLockSpec.
49       */
50      private final GordianPasswordLockSpec theLockSpec;
51  
52      /**
53       * Create the ASN1 sequence.
54       * @param pLockSpec the passwordLockSpec
55       */
56      public GordianPasswordLockSpecASN1(final GordianPasswordLockSpec pLockSpec) {
57          /* Store the Spec */
58          theLockSpec = pLockSpec;
59      }
60  
61      /**
62       * Constructor.
63       * @param pSequence the Sequence
64       * @throws GordianException on error
65       */
66      public GordianPasswordLockSpecASN1(final ASN1Sequence pSequence) throws GordianException {
67          /* Protect against exceptions */
68          try {
69              /* Extract the parameters from the sequence */
70              final Enumeration<?> en = pSequence.getObjects();
71              final GordianKeySetSpec mySpec = GordianKeySetSpecASN1.getInstance(en.nextElement()).getSpec();
72              final int myIterations = ASN1Integer.getInstance(en.nextElement()).getValue().intValue();
73  
74              /* Make sure that we have completed the sequence */
75              if (en.hasMoreElements()) {
76                  throw new GordianDataException("Unexpected additional values in ASN1 sequence");
77              }
78  
79              /* Create the lockSpec */
80              theLockSpec = new GordianPasswordLockSpec(myIterations, mySpec);
81  
82              /* handle exceptions */
83          } catch (IllegalArgumentException e) {
84              throw new GordianIOException("Unable to parse ASN1 sequence", e);
85          }
86      }
87  
88      /**
89       * Parse the ASN1 object.
90       * @param pObject the object to parse
91       * @return the parsed object
92       * @throws GordianException on error
93       */
94      public static GordianPasswordLockSpecASN1 getInstance(final Object pObject) throws GordianException {
95          if (pObject instanceof GordianPasswordLockSpecASN1 myASN1) {
96              return myASN1;
97          } else if (pObject != null) {
98              return new GordianPasswordLockSpecASN1(ASN1Sequence.getInstance(pObject));
99          }
100         throw new GordianDataException("Null sequence");
101     }
102 
103     /**
104      * Obtain the lockSpec.
105      * @return the lockSpec
106      */
107     public GordianPasswordLockSpec getLockSpec() {
108         return theLockSpec;
109     }
110 
111     @Override
112     public ASN1Primitive toASN1Primitive() {
113         final ASN1EncodableVector v = new ASN1EncodableVector();
114         v.add(new GordianKeySetSpecASN1(theLockSpec.getKeySetSpec()).toASN1Primitive());
115         v.add(new ASN1Integer(theLockSpec.getKIterations()));
116         return new DERSequence(v);
117     }
118 
119     /**
120      * Obtain the byte length of the encoded sequence.
121      * @return the byte length
122      */
123     static int getEncodedLength() {
124         /* KeyType has type + length + value (all single byte) */
125         int myLength  =  GordianASN1Util.getLengthIntegerField(1);
126         myLength += GordianKeySetSpecASN1.getEncodedLength();
127 
128         /* Calculate the length of the sequence */
129         return GordianASN1Util.getLengthSequence(myLength);
130     }

132     @Override
133     public boolean equals(final Object pThat) {
134         /* Handle trivial cases */
135         if (this == pThat) {
136             return true;
137         }
138         if (pThat == null) {
139             return false;
140         }
141 
142         /* Check that the fields are equal */
143         return pThat instanceof GordianPasswordLockSpecASN1 myThat
144                 && Objects.equals(theLockSpec, myThat.getLockSpec());
145     }

147     @Override
148     public int hashCode() {
149         return Objects.hash(getLockSpec());
150     }
