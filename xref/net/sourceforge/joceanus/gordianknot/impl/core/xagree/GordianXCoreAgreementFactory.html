96      @Override
97      public GordianXAgreement createAgreement(final GordianXAgreementParams pParams) throws GordianException {
98          /* Check validity of Agreement */
99          final GordianAgreementSpec mySpec = pParams.getAgreementSpec();
100         checkAgreementSpec(mySpec);
101 
102         /* Create the agreement */
103         final GordianXCoreAgreementEngine myEngine = createEngine(mySpec);
104         final GordianXCoreAgreement myAgreement = new GordianXCoreAgreement(myEngine);
105 
106         /* Set the details */
107         myAgreement.setClientCertificate(pParams.getClientCertificate());
108         myAgreement.setServerCertificate(pParams.getServerCertificate());
109         myAgreement.setResultType(pParams.getResultType());
110 
111         /* Store the parameters */
112 
113         /* Build the clientHello */
114         myAgreement.buildClientHello();
115 
116         /* Return the agreement */
117         return myAgreement;
118     }

120     @Override
121     public GordianXAgreement parseAgreementMessage(final byte[] pMessage) throws GordianException {
122         /* Parse the message */
123         final GordianXCoreAgreementMessageASN1 myASN1 = GordianXCoreAgreementMessageASN1.getInstance(pMessage);
124 
125         /* Switch on the messageType */
126         switch (myASN1.getMessageType()) {
127             case CLIENTHELLO:
128                 return parseClientHello(myASN1);
129             case SERVERHELLO:
130                 return parseServerHello(myASN1);
131             case CLIENTCONFIRM:
132                 return parseClientConfirm(myASN1);
133             default:
134                 throw new GordianDataException("Unexpected MessageType: " +  myASN1.getMessageType());
135         }
136     }

138     /**
139      * Parse a clientHello message.
140      * @param pClientHello the message
141      * @return the agreement
142      * @throws GordianException error
143      */
144     private GordianXAgreement parseClientHello(final GordianXCoreAgreementMessageASN1 pClientHello) throws GordianException {
145         /* Create a new agreement */
146         final GordianAgreementSpec mySpec = getSpecForIdentifier(pClientHello.getAgreementId());
147         final GordianXCoreAgreementEngine myEngine = createEngine(mySpec);
148         final GordianXCoreAgreement myAgreement = new GordianXCoreAgreement(myEngine);
149 
150         /* If this is a signed agreement */
151         if (GordianAgreementType.SIGNED.equals(mySpec.getAgreementType())) {
152             /* Handle no signer certificate */
153             if (theSignerCertificate == null) {
154                 throw new GordianLogicException("No signer declared for Signed agreement");
155             }
156 
157             /* Declare the signer */
158             myAgreement.setSignerCertificate(theSignSpec, theSignerCertificate);
159         }
160 
161         /* Parse the clientHello */
162         myAgreement.parseClientHello(pClientHello);
163 
164         /* Return the agreement */
165         return myAgreement;
166     }

168     /**
169      * Parse a serverHello message.
170      * @param pServerHello the message
171      * @return the agreement
172      * @throws GordianException error
173      */
174     private GordianXAgreement parseServerHello(final GordianXCoreAgreementMessageASN1 pServerHello) throws GordianException {
175         /* Look up the agreement in the cache */
176         final GordianAgreementSpec mySpec = getSpecForIdentifier(pServerHello.getAgreementId());
177         final GordianXCoreAgreement myAgreement = theCache.lookUpAgreement(pServerHello.getClientId(), mySpec);
178 
179         /* Process the serverHello */
180         myAgreement.processServerHello(pServerHello);
181 
182         /* Return the agreement */
183         return myAgreement;
184     }

204     /**
205      * Create engine.
206      * @param pSpec the agreement Spec
207      * @return the engine
208      * @throws GordianException on error
209      */
210     protected GordianXCoreAgreementEngine createEngine(final GordianAgreementSpec pSpec) throws GordianException {
211         /* If this is a composite agreement */
212         if (pSpec.getKeyPairSpec().getKeyPairType() == GordianKeyPairType.COMPOSITE) {
213             /* Create an engine for each sub-agreement */
214             final List<GordianAgreementSpec> mySpecs = GordianXCoreAgreementComposite.getSubAgreements(pSpec);
215             final List<GordianXCoreAgreementEngine> myEngines = new ArrayList<>();
216             for (GordianAgreementSpec mySpec : mySpecs) {
217                 myEngines.add(createEngine(mySpec));
218             }
219             return new GordianXCoreAgreementComposite(this, pSpec, myEngines);
220         }
221 
222         /* Unsupported spec */
223         throw new GordianDataException(GordianBaseData.getInvalidText(pSpec));
224     }

262     @Override
263     public void setSigner(final GordianCertificate pSigner,
264                           final GordianSignatureSpec pSignSpec) throws GordianException {
265         /* Check that certificate can sign data */
266         if (pSigner == null || !pSigner.getUsage().hasUse(GordianKeyPairUse.SIGNATURE)) {
267             throw new GordianDataException("Certificate must be capable of signing data");
268         }
269 
270         /* Check that certificate can sign data */
271         final GordianSignatureFactory mySignFactory = theFactory.getAsyncFactory().getSignatureFactory();
272         if (!mySignFactory.validSignatureSpecForKeyPair(pSigner.getKeyPair(), pSignSpec)) {
273             throw new GordianDataException(GordianBaseData.getInvalidText(pSignSpec));
274         }
275 
276         /* Store parameters */
277         theSignerCertificate = pSigner;
278         theSignSpec = pSignSpec;
279     }

281     /**
282      * Check AgreementSpec.
283      *
284      * @param pSpec the agreementSpec
285      * @return true/false
286      */
287     protected boolean validAgreementSpec(final GordianAgreementSpec pSpec) {
288         /* Reject invalid agreementSpec */
289         if (pSpec == null || !pSpec.isValid()) {
290             return false;
291         }
292 
293         /* Check that spec is supported */
294         return pSpec.isSupported();
295     }

297     @Override
298     public boolean validAgreementSpecForKeyPairSpec(final GordianKeyPairSpec pKeyPairSpec,
299                                                     final GordianAgreementSpec pAgreementSpec) {
300         /* Check that the agreementSpec is supported */
301         if (!validAgreementSpec(pAgreementSpec)) {
302             return false;
303         }
304 
305         /* Check agreement matches keySpec */
306         if (!pAgreementSpec.getKeyPairSpec().equals(pKeyPairSpec)) {
307             return false;
308         }
309 
310         /* For Edwards XDH, disallow 512KDF for 25519 and 256KDF for 448 */
311         if (pKeyPairSpec.getKeyPairType() == GordianKeyPairType.XDH) {
312             final GordianEdwardsElliptic myEdwards = pKeyPairSpec.getEdwardsElliptic();
313             switch (pAgreementSpec.getKDFType()) {
314                 case SHA256KDF:
315                 case SHA256CKDF:
316                     return myEdwards.is25519();
317                 case SHA512KDF:
318                 case SHA512CKDF:
319                     return !myEdwards.is25519();
320                 default:
321                     break;
322             }
323         }
324 
325         /* For Composite AgreementSpec */
326         if (pKeyPairSpec.getKeyPairType() == GordianKeyPairType.COMPOSITE) {
327             /* Access the subSpecs  */
328             final List<GordianAgreementSpec> mySubAgrees = GordianXCoreAgreementComposite.getSubAgreements(pAgreementSpec);
329 
330             /* Loop through the subAgreements */
331             for (GordianAgreementSpec mySpec : mySubAgrees) {
332                 if (!validAgreementSpecForKeyPairSpec(mySpec.getKeyPairSpec(), mySpec)) {
333                     return false;
334                 }
335             }
336 
337             /* Check confirmation */
338             if (Boolean.TRUE.equals(pAgreementSpec.withConfirm())
339                     && !pAgreementSpec.getAgreementType().canConfirm()) {
340                 return false;
341             }
342 
343             /* Disallow SM2 with confirm */
344             return pAgreementSpec.getAgreementType() != GordianAgreementType.SM2
345                     || !pAgreementSpec.withConfirm();
346         }
347 
348         /* OK */
349         return true;
350     }

388     @Override
389     public List<GordianAgreementSpec> listAllSupportedAgreements(final GordianKeyPairSpec pKeyPairSpec) {
390         return listPossibleAgreements(pKeyPairSpec)
391                 .stream()
392                 .filter(supportedAgreements())
393                 .filter(s -> validAgreementSpecForKeyPairSpec(pKeyPairSpec, s))
394                 .toList();
395     }

397     /**
398      * Obtain a list of all possible agreements for the keyPairSpec.
399      * @param pKeyPairSpec the keyPairSpec
400      * @return the list
401      */
402     private List<GordianAgreementSpec> listPossibleAgreements(final GordianKeyPairSpec pKeyPairSpec) {
403         /* Create list */
404         final List<GordianAgreementSpec> myAgreements = new ArrayList<>();
405 
406         /* Switch on keyPairType */
407         switch (pKeyPairSpec.getKeyPairType()) {
408             case RSA:
409                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.KEM));
410                 break;
411             case NEWHOPE:
412                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.ANON));
413                 break;
414             case CMCE:
415             case FRODO:
416             case SABER:
417             case MLKEM:
418             case HQC:
419             case BIKE:
420             case NTRU:
421             case NTRUPRIME:
422                 myAgreements.add(GordianAgreementSpecBuilder.kem(pKeyPairSpec, GordianKDFType.NONE));
423                 break;
424             case EC:
425             case SM2:
426             case GOST2012:
427                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.ANON));
428                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.KEM));
429                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.BASIC));
430                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SIGNED));
431                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED));
432                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED, Boolean.TRUE));
433                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.MQV));
434                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.MQV, Boolean.TRUE));
435                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SM2));
436                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SM2, Boolean.TRUE));
437                 break;
438             case DH:
439             case DSTU4145:
440                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.ANON));
441                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.KEM));
442                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.BASIC));
443                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SIGNED));
444                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED));
445                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED, Boolean.TRUE));
446                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.MQV));
447                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.MQV, Boolean.TRUE));
448                 break;
449             case XDH:
450                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.ANON));
451                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.BASIC));
452                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.SIGNED));
453                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED));
454                 myAgreements.addAll(listAllKDFs(pKeyPairSpec, GordianAgreementType.UNIFIED, Boolean.TRUE));
455                 break;
456             case COMPOSITE:
457                 /* Loop through the possible keySpecs for the first key */
458                 final Iterator<GordianKeyPairSpec> myIterator = pKeyPairSpec.keySpecIterator();
459                 for (GordianAgreementSpec mySpec : listPossibleAgreements(myIterator.next())) {
460                     final GordianAgreementSpec myTest = new GordianAgreementSpec(pKeyPairSpec, mySpec.getAgreementType(), mySpec.getKDFType(), mySpec.withConfirm());
461                     if (myTest.isValid()) {
462                         myAgreements.add(myTest);
463                     }
464                 }
465                 break;
466             default:
467                 break;
468         }
469 
470         /* Return the list */
471         return myAgreements;
472     }

485     @Override
486     public GordianAgreementSpec defaultForKeyPair(final GordianKeyPairSpec pKeySpec) {
487         final Iterator<GordianAgreementSpec> myIterator = listAllSupportedAgreements(pKeySpec).iterator();
488         return myIterator.hasNext() ? myIterator.next() : null;
489     }

491     /**
492      * Create list of KDF variants.
493      * @param pKeyPairSpec the keyPairSpec
494      * @param pAgreementType the agreementType
495      * @param pConfirm with key confirmation
496      * @return the list
497      */
498     public static List<GordianAgreementSpec> listAllKDFs(final GordianKeyPairSpec pKeyPairSpec,
499                                                          final GordianAgreementType pAgreementType,
500                                                          final Boolean pConfirm) {
501         /* Create list */
502         final List<GordianAgreementSpec> myAgreements = new ArrayList<>();
503 
504         /* Loop through the KDFs */
505         for (final GordianKDFType myKDF : GordianKDFType.values()) {
506             myAgreements.add(new GordianAgreementSpec(pKeyPairSpec, pAgreementType, myKDF, pConfirm));
507         }
508 
509         /* Return the list */
510         return myAgreements;
511     }
