92      /**
93       * Obtain list of subAgreementSpecs.
94       * @param pSpec the composite agreement Spec
95       * @return the list of subAgreementSpecs
96       */
97      static List<GordianAgreementSpec> getSubAgreements(final GordianAgreementSpec pSpec) {
98          /* Create the list */
99          final List<GordianAgreementSpec> mySpecs = new ArrayList<>();
100 
101         /* Switch signed agreements to Basic sub agreements */
102         final GordianAgreementType myType = pSpec.getAgreementType().isSigned()
103                 ? GordianAgreementType.BASIC
104                 : pSpec.getAgreementType();
105         final GordianKDFType myKDF = pSpec.getKDFType();
106 
107         /* Loop through the keyPairs */
108         final GordianKeyPairSpec myKeyPairSpec = pSpec.getKeyPairSpec();
109         final Iterator<GordianKeyPairSpec> myIterator = myKeyPairSpec.keySpecIterator();
110         while (myIterator.hasNext()) {
111             final GordianKeyPairSpec mySpec = myIterator.next();
112             /* Determine the agreementType (note that we have no confirmation) */
113             mySpecs.add(new GordianAgreementSpec(mySpec, myType, myKDF));
114         }
115 
116         /* Return the list */
117         return mySpecs;
118     }

120     @Override
121     public void buildClientHello() throws GordianException {
122         /* Access the client and server keyPairs */
123         final GordianXCoreAgreementParticipant myClient = theState.getClient();
124         final GordianCompositeKeyPair myClientKeyPair = (GordianCompositeKeyPair) myClient.getKeyPair();
125         final Iterator<GordianKeyPair> myClientIterator = myClientKeyPair == null ? null : myClientKeyPair.iterator();
126         final GordianXCoreAgreementParticipant myServer = theState.getServer();
127         final GordianCompositeKeyPair myServerKeyPair = (GordianCompositeKeyPair) myServer.getKeyPair();
128         final Iterator<GordianKeyPair> myServerIterator = myServerKeyPair.iterator();
129 
130         /* Protect against exceptions */
131         try {
132             /* Create vectors for ephemeral keyPairs and encapsulated */
133             final ASN1EncodableVector myEphemeral = new ASN1EncodableVector();
134             final ASN1EncodableVector myEncapsulated = new ASN1EncodableVector();
135 
136             /* Loop through the engines */
137             for (GordianXCoreAgreementEngine myEngine : theEngines) {
138                 /* Access engine details */
139                 final GordianXCoreAgreementState myEngState = myEngine.getBuilder().getState();
140                 final GordianXCoreAgreementParticipant myEngClient = myEngState.getClient();
141                 final GordianXCoreAgreementParticipant myEngServer = myEngState.getServer();
142 
143                 /* Update keyPairs and initVector */
144                 myEngClient.setKeyPair(myClientKeyPair == null ? null : myClientIterator.next());
145                 myEngServer.setKeyPair(myServerIterator.next());
146                 myEngClient.setInitVector(myClient.getInitVector());
147 
148                 /* Build clientHello details in the engine */
149                 myEngine.buildClientHello();
150 
151                 /* Add any Ephemeral to keySpec sequence */
152                 final X509EncodedKeySpec myEngEphemeral = myEngClient.getEphemeralKeySpec();
153                 if (myEngEphemeral != null) {
154                     myEphemeral.add(SubjectPublicKeyInfo.getInstance(myEngEphemeral));
155                 }
156 
157                 /* Add any encapsulated to sequence */
158                 final byte[] myEngEncapsulated = myEngState.getEncapsulated();
159                 if (myEngEncapsulated != null) {
160                     myEncapsulated.add(new DEROctetString(myEngEncapsulated));
161                 }
162             }
163 
164             /* Record combined ephemeral and encapsulated */
165             if (myEphemeral.size() > 0) {
166                 final AlgorithmIdentifier myId = new AlgorithmIdentifier(MiscObjectIdentifiers.id_alg_composite);
167                 final SubjectPublicKeyInfo myInfo = new SubjectPublicKeyInfo(myId, new DERSequence(myEphemeral).getEncoded());
168                 myClient.setEphemeralKeySpec(new X509EncodedKeySpec(myInfo.getEncoded()));
169             }
170             if (myEncapsulated.size() > 0) {
171                 theState.setEncapsulated(new DERSequence(myEncapsulated).getEncoded());
172             }
173 
174             /* catch exceptions */
175         } catch (IOException e) {
176             throw new GordianIOException("Failed to build combined clientHello", e);
177         }
178 
179         /* Sort out the result if anonymous */
180         if (theState.getSpec().getAgreementType().isAnonymous()) {
181             mergeResults();
182         }
183     }

185     /**
186      * Process the clientHello.
187      * @throws GordianException on error
188      */
189     public void processClientHello() throws GordianException {
190         /* Access the client and server keyPairs */
191         final GordianXCoreAgreementParticipant myClient = theState.getClient();
192         final GordianCompositeKeyPair myClientKeyPair = (GordianCompositeKeyPair) myClient.getKeyPair();
193         final Iterator<GordianKeyPair> myClientIterator = myClientKeyPair == null ? null : myClientKeyPair.iterator();
194         final GordianXCoreAgreementParticipant myServer = theState.getServer();
195         final GordianCompositeKeyPair myServerKeyPair = (GordianCompositeKeyPair) myServer.getKeyPair();
196         final Iterator<GordianKeyPair> myServerIterator = myServerKeyPair.iterator();
197         final GordianCompositeKeyPair myEphemeralKeyPair = (GordianCompositeKeyPair) myClient.getEphemeralKeyPair();
198         final Iterator<GordianKeyPair> myEphemeralIterator = myEphemeralKeyPair == null ? null : myEphemeralKeyPair.iterator();
199 
200         /* Protect against exceptions */
201         try {
202             /* Create vector for ephemeral keyPairs and access encapsulated sequence */
203             final ASN1EncodableVector myEphemeral = new ASN1EncodableVector();
204             final byte[] myEncapsulated = theState.getEncapsulated();
205             final Enumeration<?> enEnc = myEncapsulated == null ? null : ASN1Sequence.getInstance(myEncapsulated).getObjects();
206 
207             /* Loop through the engines */
208             for (GordianXCoreAgreementEngine myEngine : theEngines) {
209                 /* Access engine details */
210                 final GordianXCoreAgreementState myEngState = myEngine.getBuilder().getState();
211                 final GordianXCoreAgreementParticipant myEngClient = myEngState.getClient();
212                 final GordianXCoreAgreementParticipant myEngServer = myEngState.getServer();
213 
214                 /* Update keyPairs and initVector */
215                 myEngClient.setKeyPair(myClientKeyPair == null ? null : myClientIterator.next());
216                 myEngServer.setKeyPair(myServerIterator.next());
217                 myEngClient.setInitVector(myClient.getInitVector());
218                 myEngServer.setInitVector(myServer.getInitVector());
219                 myEngClient.setEphemeralKeyPair(myEphemeralKeyPair == null ? null : myEphemeralIterator.next());
220                 myEngState.setEncapsulated(enEnc == null ? null : ASN1OctetString.getInstance(enEnc.nextElement()).getOctets());
221 
222                 /* Process clientHello details in the engine */
223                 myEngine.processClientHello();
224 
225                 /* Add any Ephemeral to keySpec sequence */
226                 final X509EncodedKeySpec myEngEphemeral = myEngClient.getEphemeralKeySpec();
227                 if (myEngEphemeral != null) {
228                     myEphemeral.add(SubjectPublicKeyInfo.getInstance(myEngEphemeral));
229                 }
230             }
231 
232             /* Record combined ephemeral */
233             if (myEphemeral.size() > 0) {
234                 final AlgorithmIdentifier myId = new AlgorithmIdentifier(MiscObjectIdentifiers.id_alg_composite);
235                 final SubjectPublicKeyInfo myInfo = new SubjectPublicKeyInfo(myId, new DERSequence(myEphemeral).getEncoded());
236                 myClient.setEphemeralKeySpec(new X509EncodedKeySpec(myInfo.getEncoded()));
237             }
238 
239             /* catch exceptions */
240         } catch (IOException e) {
241             throw new GordianIOException("Failed to process combined clientHello", e);
242         }
243 
244         /* Sort out the result if no confirm */
245         if (!Boolean.TRUE.equals(theState.getSpec().withConfirm())) {
246             mergeResults();
247         }
248     }

251     public void processServerHello() throws GordianException {
252         /* Access the client and server keyPairs */
253         final GordianXCoreAgreementParticipant myServer = theState.getServer();
254         final GordianCompositeKeyPair myEphemeralKeyPair = (GordianCompositeKeyPair) myServer.getEphemeralKeyPair();
255         final Iterator<GordianKeyPair> myEphemeralIterator = myEphemeralKeyPair == null ? null : myEphemeralKeyPair.iterator();
256 
257         /* Protect against exceptions */
258         try {
259             /* Create vector for ephemeral keyPairs and access encapsulated sequence */
260             final ASN1EncodableVector myEphemeral = new ASN1EncodableVector();
261 
262             /* Loop through the engines */
263             for (GordianXCoreAgreementEngine myEngine : theEngines) {
264                 /* Access engine details */
265                 final GordianXCoreAgreementState myEngState = myEngine.getBuilder().getState();
266                 final GordianXCoreAgreementParticipant myEngServer = myEngState.getServer();
267 
268                 /* Store initVector and ephemeral keyPair */
269                 myEngServer.setInitVector(myServer.getInitVector());
270                 myEngServer.setEphemeralKeyPair(myEphemeralIterator == null ? null : myEphemeralIterator.next());
271 
272                 /* Process serverHello details in the engine */
273                 myEngine.processServerHello();
274 
275                 /* Add any Ephemeral to keySpec sequence */
276                 final X509EncodedKeySpec myEngEphemeral = myEngServer.getEphemeralKeySpec();
277                 if (myEngEphemeral != null) {
278                     myEphemeral.add(SubjectPublicKeyInfo.getInstance(myEngEphemeral));
279                 }
280             }
281 
282             /* Record combined ephemeral */
283             if (myEphemeral.size() > 0) {
284                 final AlgorithmIdentifier myId = new AlgorithmIdentifier(MiscObjectIdentifiers.id_alg_composite);
285                 final SubjectPublicKeyInfo myInfo = new SubjectPublicKeyInfo(myId, new DERSequence(myEphemeral).getEncoded());

304     /**
305      * Merge and store the result.
306      * @throws GordianException on error
307      */
308     private void mergeResults() throws GordianException {
309         /* Protect against exceptions */
310         final GordianHKDFParams myParams = GordianHKDFParams.extractOnly();
311         try {
312             /* Create the HKDF parameters */
313             final GordianDigestSpec myDigestSpec = new GordianDigestSpec(GordianDerivationId.COMPOSITE.getDigestType());
314             Random myRandom = null;
315 
316             /* Loop through the engines */
317             for (GordianXCoreAgreementEngine myEngine : theEngines) {
318                 /* Access the secret result */
319                 final byte[] myPart = (byte[]) myEngine.getBuilder().getState().getResult();
320 
321                 /* Handle random bytes */
322                 if (myRandom == null) {
323                     /* Build the 64-bit seed, create the seeded random and populate bytes */
324                     final long mySeed = GordianDataConverter.byteArrayToLong(myPart);
325                     myRandom = new Random(mySeed);
326                     final byte[] myBytes = new byte[Long.BYTES];
327                     myRandom.nextBytes(myBytes);
328                     myParams.withSalt(myBytes);
329                 }
330 
331                 /* Add part to parameters */
332                 myParams.withIKM(myPart);
333                 Arrays.fill(myPart, (byte) 0);
334             }
335 
336             /* Derive the bytes and store as secret */
337             final GordianHKDFEngine myEngine = new GordianHKDFEngine(theFactory, myDigestSpec);
338             theBuilder.storeSecret(myEngine.deriveBytes(myParams));
339 
340         } finally {
341             myParams.clearParameters();
342         }
343     }
