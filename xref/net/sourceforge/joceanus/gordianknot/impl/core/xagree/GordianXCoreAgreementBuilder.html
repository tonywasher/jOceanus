92      GordianXCoreAgreementBuilder(final GordianXCoreAgreementSupplier pSupplier,
93                                   final GordianAgreementSpec pSpec) throws GordianException {
94          /* Store parameters and access factory and random */
95          theSupplier = pSupplier;
96          theFactory = pSupplier.getFactory();
97          theRandom = theFactory.getRandomSource().getRandom();
98  
99          /* Create the state */
100         theState = new GordianXCoreAgreementState(pSpec);
101         final GordianKeyPairFactory myFactory = theFactory.getAsyncFactory().getKeyPairFactory();
102         theKeyPairGenerator = myFactory.getKeyPairGenerator(pSpec.getKeyPairSpec());
103 
104         /* Create the result calculator */
105         theResultCalc = new GordianXCoreAgreementCalculator(theFactory, theState);
106     }

149     /**
150      * Store secret.
151      * @param pSecret the secret
152      * @throws GordianException on error
153      */
154     public void storeSecret(final byte[] pSecret) throws GordianException {
155         /* Protect against failure */
156         try {
157             /* Just process the secret */
158             processSecret(pSecret);
159 
160             /* Clear buffers */
161         } finally {
162             /* Clear the secret */
163             Arrays.fill(pSecret, (byte) 0);
164         }
165     }
166 
167     /**
168      * Process the secret.
169      * @param pSecret the secret
170      * @throws GordianException on error
171      */
172     private void processSecret(final byte[] pSecret) throws GordianException {
173         /* If we are using confirmation and are not SM2 */
174         final GordianAgreementSpec mySpec = theState.getSpec();
175         if (Boolean.TRUE.equals(mySpec.withConfirm())
176                 && mySpec.getAgreementType() != GordianAgreementType.SM2) {
177             /* calculate the confirmation tags */
178             calculateConfirmationTags(pSecret);
179         }
180 
181         /* Calculate result */
182         theState.setResult(theResultCalc.processSecret(pSecret, theState.getResultType()));
183     }

320     /**
321      * Set the client confirm.
322      * @param pConfirm the clientConfirm
323      * @throws GordianException on error
324      */
325     void setClientConfirm(final byte[] pConfirm) throws GordianException {
326         /* Access any expected value */
327         final GordianXCoreAgreementParticipant myClient = theState.getClient();
328         final byte[] myExpected = myClient.getConfirm();
329 
330         /* If we have an expected value, reject any difference */
331         if (myExpected != null
332                 && !Arrays.constantTimeAreEqual(myExpected, pConfirm)) {
333             throw new GordianDataException("Client Confirmation failed");
334         }
335         myClient.setConfirm(pConfirm);
336     }

356     /**
357      * Build the clientHello.
358      * @return the clientHello message
359      * @throws GordianException on error
360      */
361     GordianXCoreAgreementMessageASN1 newClientHello() throws GordianException {
362         /* Access details */
363         final GordianXCoreAgreementMessageASN1 myMsg = GordianXCoreAgreementMessageASN1.newClientHello();
364         final GordianXCoreAgreementParticipant myClient = theState.getClient();
365         final GordianXCoreAgreementParticipant myServer = theState.getServer();
366 
367         /* Set standard details */
368         myMsg.setAgreementId(theSupplier.getIdentifierForSpec(theState.getSpec()))
369                 .setResultId(theSupplier.getIdentifierForResultType(theState.getResultType()))
370                 .setClientId(myClient.getId())
371                 .setInitVector(myClient.getInitVector())
372                 .setEphemeral(myClient.getEphemeralKeySpec())
373                 .setEncapsulated(theState.getEncapsulated());
374 
375         /* Store certificates */
376         final GordianCoreCertificate myClientCert = (GordianCoreCertificate) myClient.getCertificate();
377         final GordianCoreCertificate myServerCert = (GordianCoreCertificate) myServer.getCertificate();
378         myMsg.setClientCertificate(myClientCert == null ? null : myClientCert.getCertificate())
379                 .setServerCertificate(myServerCert == null ? null : myServerCert.getCertificate());
380 
381         /* Return the message */
382         return myMsg;
383     }

385     /**
386      * Build the serverHello.
387      * @return the serverHello message
388      * @throws GordianException on error
389      */
390     GordianXCoreAgreementMessageASN1 newServerHello() throws GordianException {
391         /* Access details */
392         final GordianXCoreAgreementMessageASN1 myMsg = GordianXCoreAgreementMessageASN1.newServerHello();
393         final GordianXCoreAgreementParticipant myClient = theState.getClient();
394         final GordianXCoreAgreementParticipant myServer = theState.getServer();
395 
396         /* Set standard details */
397         myMsg.setAgreementId(theSupplier.getIdentifierForSpec(theState.getSpec()))
398                 .setClientId(myClient.getId());
399 
400         /* Handle error result */
401         if (theState.getResultType() instanceof String myType) {
402             myMsg.setResultId(theSupplier.getIdentifierForResultType(myType));
403             return myMsg;
404         }
405 
406         /* Set server details */
407         myMsg.setServerId(myServer.getId())
408                 .setInitVector(myServer.getInitVector())
409                 .setEphemeral(myServer.getEphemeralKeySpec())
410                 .setConfirmation(myServer.getConfirm());
411 
412         /* Store signing details */
413         final GordianCoreCertificate mySignerCert = (GordianCoreCertificate) theState.getSignerCertificate();
414         if (mySignerCert != null) {
415             /* Access details */
416             final GordianCoreSignatureFactory mySigns = (GordianCoreSignatureFactory) theFactory.getAsyncFactory().getSignatureFactory();
417             final GordianSignatureSpec mySignSpec = theState.getSignSpec();
418             final GordianKeyPair mySignerPair = mySignerCert.getKeyPair();
419             final AlgorithmIdentifier myAlgId = mySigns.getIdentifierForSpecAndKeyPair(mySignSpec, mySignerPair);
420 
421             /* Build the signature */
422             final GordianSignature mySigner = mySigns.createSigner(mySignSpec);
423             mySigner.initForSigning(GordianSignParams.keyPair(mySignerPair));
424             mySigner.update(myClient.getEphemeralKeySpec().getEncoded());
425             mySigner.update(myClient.getInitVector());
426             mySigner.update(myServer.getEphemeralKeySpec().getEncoded());
427             mySigner.update(myServer.getInitVector());
428             myMsg.setSignerCertificate(mySignerCert.getCertificate())
429                     .setSignature(myAlgId, mySigner.sign());
430         }
431 
432         /* Return the message */
433         return myMsg;
434     }

462     /**
463      * Parse the clientHello.
464      * @param pClientHello the message
465      * @throws GordianException on error
466      */
467     void parseClientHello(final GordianXCoreAgreementMessageASN1 pClientHello) throws GordianException {
468         /* Access details */
469         final GordianXCoreAgreementParticipant myClient = theState.getClient();
470         final GordianXCoreAgreementParticipant myServer = theState.getServer();
471 
472         /* Set standard details */
473         theState.setResultType(theSupplier.getResultTypeForIdentifier(pClientHello.getResultId()))
474                 .setEncapsulated(pClientHello.getEncapsulated());
475 
476         /* Store client details */
477         myClient.setId(pClientHello.getClientId())
478                 .setInitVector(pClientHello.getInitVector());
479         final X509EncodedKeySpec myEphemeral = pClientHello.getEphemeral();
480         if (myEphemeral != null) {
481             myClient.setEphemeralKeySpec(myEphemeral)
482                     .setEphemeralKeyPair(theKeyPairGenerator.derivePublicOnlyKeyPair(myEphemeral));
483         }
484 
485         /* Store certificates */
486         final Certificate myClientCert = pClientHello.getClientCertificate();
487         myClient.setCertificate(myClientCert == null ? null : new GordianCoreCertificate(theFactory, myClientCert));
488         final Certificate myServerCert = pClientHello.getServerCertificate();
489         myServer.setCertificate(myServerCert == null ? null : new GordianCoreCertificate(theFactory, myServerCert));
490     }

492     /**
493      * Parse the serverHello.
494      * @param pServerHello the message
495      * @throws GordianException on error
496      */
497     void parseServerHello(final GordianXCoreAgreementMessageASN1 pServerHello) throws GordianException {
498         /* Access details */
499         final GordianXCoreAgreementParticipant myClient = theState.getClient();
500         final GordianXCoreAgreementParticipant myServer = theState.getServer();
501 
502         /* Handle error result */
503         final AlgorithmIdentifier myResId = pServerHello.getResultId();
504         if (myResId != null) {
505             setError((String) theSupplier.getResultTypeForIdentifier(myResId));
506             return;
507         }
508 
509         /* Store server details */
510         myServer.setId(pServerHello.getServerId())
511                 .setInitVector(pServerHello.getInitVector())
512                 .setConfirm(pServerHello.getConfirmation());
513         final X509EncodedKeySpec myEphemeral = pServerHello.getEphemeral();
514         if (myEphemeral != null) {
515             myServer.setEphemeralKeySpec(myEphemeral)
516                     .setEphemeralKeyPair(theKeyPairGenerator.derivePublicOnlyKeyPair(myEphemeral));
517         }
518 
519         /* Store signing details */
520         final Certificate mySignerCert = pServerHello.getSignerCertificate();
521         if (mySignerCert != null) {
522             /* Access details */
523             final GordianCoreSignatureFactory mySigns = (GordianCoreSignatureFactory) theFactory.getAsyncFactory().getSignatureFactory();
524             final GordianSignatureSpec mySignSpec = mySigns.getSpecForIdentifier(pServerHello.getSignatureId());
525             final GordianCoreCertificate myCert = new GordianCoreCertificate(theFactory, mySignerCert);
526             final GordianKeyPair mySignerPair = myCert.getKeyPair();
527             theState.setSignerCertificate(myCert)
528                     .setSignSpec(mySignSpec);
529             final GordianSignature mySigner = mySigns.createSigner(mySignSpec);
530 
531             /* Build the signature */
532             mySigner.initForVerify(GordianSignParams.keyPair(mySignerPair));
533             mySigner.update(myClient.getEphemeralKeySpec().getEncoded());
534             mySigner.update(myClient.getInitVector());
535             mySigner.update(myServer.getEphemeralKeySpec().getEncoded());
536             mySigner.update(myServer.getInitVector());
537             if (!mySigner.verify(pServerHello.getSignature())) {
538                 setError("Signature Failed");
539             }
540         }
541     }

563     /**
564      * Calculate the confirmation tags.
565      * @param pSecret the secret
566      * @throws GordianException on error
567      */
568     private void calculateConfirmationTags(final byte[] pSecret) throws GordianException {
569         /* Access details */
570         final GordianXCoreAgreementParticipant myClient = theState.getClient();
571         final GordianXCoreAgreementParticipant myServer = theState.getServer();
572 
573         /* Derive the key */
574         final byte[] myKey = theResultCalc.calculateDerivedSecret(pSecret, GordianXDerivationId.TAGS, GordianLength.LEN_512.getByteLength());
575 
576         /* Create the hMac and initialize with the key */
577         final GordianMacFactory myMacs = theFactory.getMacFactory();
578         final GordianMacSpec mySpec = GordianMacSpecBuilder.hMac(GordianXDerivationId.TAGS.getDigestType());
579         final GordianMac myMac = myMacs.createMac(mySpec);
580         myMac.initKeyBytes(myKey);
581 
582         /* Access the public encodings */
583         final byte[] myClientSpec = theKeyPairGenerator.getX509Encoding(myClient.getKeyPair()).getEncoded();
584         final byte[] myClientEphemeral = myClient.getEphemeralKeySpec().getEncoded();
585         final byte[] myServerSpec = theKeyPairGenerator.getX509Encoding(myServer.getKeyPair()).getEncoded();
586         final byte[] myServerEphemeral = myServer.getEphemeralKeySpec().getEncoded();
587 
588         /* Build Server Confirmation tag */
589         myMac.update(myServerSpec);
590         myMac.update(myClientSpec);
591         myMac.update(myServerEphemeral);
592         myMac.update(myClientEphemeral);
593         setServerConfirm(myMac.finish());
594 
595         /* Build Client Confirmation tag */
596         myMac.update(myClientSpec);
597         myMac.update(myServerSpec);
598         myMac.update(myClientEphemeral);
599         myMac.update(myServerEphemeral);
600         setClientConfirm(myMac.finish());
601     }
