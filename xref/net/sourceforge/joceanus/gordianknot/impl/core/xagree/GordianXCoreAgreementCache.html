17  package net.sourceforge.joceanus.gordianknot.impl.core.xagree;
18  
19  import net.sourceforge.joceanus.gordianknot.api.agree.GordianAgreementSpec;
20  import net.sourceforge.joceanus.gordianknot.api.base.GordianException;
21  import net.sourceforge.joceanus.gordianknot.api.xagree.GordianXAgreement;
22  import net.sourceforge.joceanus.gordianknot.impl.core.base.GordianRandomSource;
23  import net.sourceforge.joceanus.gordianknot.impl.core.exc.GordianDataException;
24  
25  import java.util.HashMap;
26  import java.util.Map;
27  import java.util.Objects;
28  import java.util.concurrent.atomic.AtomicLong;
29  
30  /**
31   * Agreement Cache.
32   */
33  public class GordianXCoreAgreementCache {
34      /**
35       * The id control.
36       */
37      private final AtomicLong theNextId;
38  
39      /**
40       * The map of id to agreement.
41       */
42      private final Map<Long, GordianXCoreAgreement> theMap;
43  
44      /**
45       * Constructor.
46       * @param pRandom the random generator.
47       */
48      GordianXCoreAgreementCache(final GordianRandomSource pRandom) {
49          /* Initialize the id bank */
50          final long myInit = ((long) pRandom.getRandom().nextInt()) << Integer.SIZE;
51          theNextId = new AtomicLong(myInit);
52  
53          /* Create the map */
54          theMap = new HashMap<>();
55      }
56  
57      /**
58       * Obtain the nextId.
59       * @return the nextId
60       */
61      Long getNextId() {
62          return theNextId.get();
63      }
64  
65      /**
66       * Lookup id.
67       * @param pId the id to look up
68       * @param pSpec the expected agreementSpec
69       * @return the matching agreement
70       * @throws GordianException on error
71       */
72      GordianXCoreAgreement lookUpAgreement(final Long pId,
73                                            final GordianAgreementSpec pSpec) throws GordianException {
74          /* Look up the Agreement */
75          final GordianXCoreAgreement myAgreement = theMap.get(pId);
76  
77          /* Validate the agreement */
78          if (myAgreement == null) {
79              throw new GordianDataException("No matching cached Agreement found");
80          }
81          if (!Objects.equals(myAgreement.getAgreementSpec(), pSpec)) {
82              throw new GordianDataException("Agreement specification mismatch");
83          }
84  
85          /* Return the agreement */
86          return myAgreement;
87      }
88  
89      /**
90       * Store agreement under id.
91       * @param pId the id
92       * @param pAgreement the agreement
93       */
94      void storeAgreement(final Long pId,
95                          final GordianXAgreement pAgreement) {
96          theMap.put(pId, (GordianXCoreAgreement) pAgreement);
97      }
98  
99      /**
100      * Release agreement.
101      * @param pId the id
102      */
103     void removeAgreement(final Long pId) {
104         theMap.remove(pId);
105     }
106 }
