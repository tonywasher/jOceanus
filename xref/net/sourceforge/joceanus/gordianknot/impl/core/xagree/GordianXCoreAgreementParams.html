177     /**
178      * Check the resultType is valid.
179      * @param pResultType the resultType
180      * @throws GordianException on error
181      */
182     private void checkResultType(final Object pResultType) throws GordianException {
183         /* No need to check FactoryType */
184         if (pResultType instanceof GordianFactoryType) {
185             return;
186         }
187 
188         /* Validate a keySetSpec */
189         if (pResultType instanceof GordianKeySetSpec mySpec) {
190             /* Check Spec */
191             final GordianCoreKeySetFactory myKeySetFactory = (GordianCoreKeySetFactory) theFactory.getKeySetFactory();
192             myKeySetFactory.checkKeySetSpec(mySpec);
193             return;
194         }
195 
196         /* Validate a symCipherSpec */
197         if (pResultType instanceof GordianSymCipherSpec mySpec) {
198             /* Check Spec */
199             final GordianCoreCipherFactory myCipherFactory = (GordianCoreCipherFactory) theFactory.getCipherFactory();
200             myCipherFactory.checkSymCipherSpec(mySpec);
201             return;
202         }
203 
204         /* Validate a streamCipherSpec */
205         if (pResultType instanceof GordianStreamCipherSpec mySpec) {
206             /* Check Spec */
207             final GordianCoreCipherFactory myCipherFactory = (GordianCoreCipherFactory) theFactory.getCipherFactory();
208             myCipherFactory.checkStreamCipherSpec(mySpec);
209             return;
210         }
211 
212         /* Validate a byte array */
213         if (pResultType instanceof Integer myInt) {
214             if (myInt <= 0) {
215                 throw new GordianLogicException("Invalid length for byteArray");
216             }
217             return;
218         }
219 
220         /* Invalid resultType */
221         throw new GordianLogicException("Invalid resultType");
222     }

249     @Override
250     public GordianXAgreementParams setClientCertificate(final GordianCertificate pClient) throws GordianException {
251         /* Not allowed for server parameters */
252         if (!isClient) {
253             throw new GordianDataException("Client Certificate cannot be changed for server");
254         }
255 
256         /* If we have a client certificate */
257         final GordianAgreementSpec mySpec = getAgreementSpec();
258         final GordianAgreementType myType = mySpec.getAgreementType();
259         if (pClient != null) {
260             /* Check that the keySpec matches the agreement and that we have a private key */
261             final GordianKeyPair myKeyPair = pClient.getKeyPair();
262             if (myType.isSigned() || myType.isAnonymous()) {
263                 throw new GordianDataException("Client Certificate not supported for agreement");
264             }
265             if (!Objects.equals(mySpec.getKeyPairSpec(), myKeyPair.getKeyPairSpec())) {
266                 throw new GordianDataException("Client Certificate not valid for agreement");
267             }
268             if (!pClient.getUsage().hasUse(GordianKeyPairUse.AGREEMENT)) {
269                 throw new GordianDataException("Client Certificate must be capable of keyAgreement");
270             }
271             if (myKeyPair.isPublicOnly()) {
272                 throw new GordianDataException("Client Certificate must supply privateKey");
273             }
274         } else if (!myType.isSigned() && !myType.isAnonymous()) {
275             throw new GordianDataException("Null Client Certificate not allowed");
276         }
277 
278         /* Create new updated parameters */
279         final GordianXCoreAgreementParams myParams = new GordianXCoreAgreementParams(this);
280         myParams.theClient = pClient;
281         return myParams;
282     }

284     @Override
285     public GordianXAgreementParams setServerCertificate(final GordianCertificate pServer) throws GordianException {
286         /* If we have a server certificate */
287         final GordianAgreementSpec mySpec = getAgreementSpec();
288         if (pServer != null) {
289             /* Check that the keySpec matches the agreement */
290             final GordianKeyPair myKeyPair = pServer.getKeyPair();
291             if (mySpec.getAgreementType().isSigned()) {
292                 throw new GordianDataException("Server Certificate not supported for agreement");
293             }
294             if (!Objects.equals(mySpec.getKeyPairSpec(), myKeyPair.getKeyPairSpec())) {
295                 throw new GordianDataException("Server Certificate not valid for agreement");
296             }
297             if (!pServer.getUsage().hasUse(GordianKeyPairUse.AGREEMENT)) {
298                 throw new GordianDataException("Server Certificate must be capable of keyAgreement");
299             }
300 
301             /* If we are a server */
302             if (!isClient) {
303                 /* Perform additional checks */
304                 if (myKeyPair.isPublicOnly()) {
305                     throw new GordianDataException("Server Certificate must supply privateKey");
306                 }
307 
308                 /* Check that we match the existing server certificate */
309                 if (!Arrays.equals(theServer.getEncoded(), pServer.getEncoded())) {
310                     throw new GordianDataException("Server Certificate must match requested certificate");
311                 }
312             }
313 
314         } else if (!mySpec.getAgreementType().isSigned()) {
315             throw new GordianDataException("Null Server Certificate not allowed");
316         }
317 
318         /* Create new updated parameters */
319         final GordianXCoreAgreementParams myParams = new GordianXCoreAgreementParams(this);
320         myParams.theServer = pServer;
321         return myParams;
322     }

324     @Override
325     public void setSigner(final GordianCertificate pSigner) throws GordianException {
326         final GordianSignatureFactory mySignFactory = theFactory.getAsyncFactory().getSignatureFactory();
327         final GordianSignatureSpec mySignSpec = pSigner == null ? null : mySignFactory.defaultForKeyPair(pSigner.getKeyPair().getKeyPairSpec());
328         setSigner(pSigner, mySignSpec);
329     }
330 
331     @Override
332     public void setSigner(final GordianCertificate pSigner,
333                           final GordianSignatureSpec pSignSpec) throws GordianException {
334         /* Not allowed for client parameters */
335         if (isClient) {
336             throw new GordianDataException("Signer Certificate cannot be set for client");
337         }
338 
339         /* If we have a signer certificate */
340         final GordianAgreementSpec mySpec = getAgreementSpec();
341         if (pSigner != null) {
342             /* Check that we require a signer */
343             if (!mySpec.getAgreementType().isSigned()) {
344                 throw new GordianDataException("Signer Certificate not allowed");
345             }
346 
347             /* Check that certificate can sign data */
348             if (!pSigner.getUsage().hasUse(GordianKeyPairUse.SIGNATURE)) {
349                 throw new GordianDataException("Certificate must be capable of signing data");
350             }
351 
352             /* Check that signSpec is valid for keyPair */
353             final GordianSignatureFactory mySignFactory = theFactory.getAsyncFactory().getSignatureFactory();
354             if (!mySignFactory.validSignatureSpecForKeyPair(pSigner.getKeyPair(), pSignSpec)) {
355                 throw new GordianDataException(GordianBaseData.getInvalidText(pSignSpec));
356             }
357 
358         } else if (mySpec.getAgreementType().isSigned()) {
359             throw new GordianDataException("Null Signer Certificate not allowed");
360         }
361 
362         /* Store parameters */
363         theSigner = pSigner;
364         theSignSpec = pSignSpec;
365     }
