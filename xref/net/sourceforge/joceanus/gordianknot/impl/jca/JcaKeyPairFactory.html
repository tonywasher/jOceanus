56  /**
57   * Jca KeyPair Factory.
58   */
59  public class JcaKeyPairFactory
60      extends GordianCoreKeyPairFactory {
61      /**
62       * Factory.
63       */
64      private final GordianBaseFactory theFactory;
65  
66      /**
67       * KeyPairGenerator Cache.
68       */
69      private final Map<GordianKeyPairSpec, JcaKeyPairGenerator> theCache;
70  
71      /**
72       * Constructor.
73       *
74       * @param pFactory the factory
75       */
76      JcaKeyPairFactory(final GordianBaseFactory pFactory) {
77          /* Initialize underlying class */
78          super(pFactory);
79          theFactory = pFactory;
80  
81          /* Create the cache */
82          theCache = new HashMap<>();
83      }
84  
85      @Override
86      public GordianKeyPairGenerator getKeyPairGenerator(final GordianKeyPairSpec pKeySpec) throws GordianException {
87          /* Handle composite keyPairGenerator */
88          if (GordianKeyPairType.COMPOSITE.equals(pKeySpec.getKeyPairType())) {
89              return super.getKeyPairGenerator(pKeySpec);
90          }
91  
92          /* Look up in the cache */
93          JcaKeyPairGenerator myGenerator = theCache.get(pKeySpec);
94          if (myGenerator == null) {
95              /* Check the keySpec */
96              checkAsymKeySpec(pKeySpec);
97  
98              /* Create the new generator */
99              myGenerator = getJcaKeyPairGenerator(pKeySpec);
100 
101             /* Add to cache */
102             theCache.put(pKeySpec, myGenerator);
103         }
104         return myGenerator;
105     }
106 
107     /**
108      * Create the Jca KeyPairGenerator.
109      * @param pKeySpec the keySpec
110      * @return the KeyGenerator
111      * @throws GordianException on error
112      */
113     private JcaKeyPairGenerator getJcaKeyPairGenerator(final GordianKeyPairSpec pKeySpec) throws GordianException {
114         switch (pKeySpec.getKeyPairType()) {
115             case RSA:
116                 return new JcaRSAKeyPairGenerator(theFactory, pKeySpec);
117             case ELGAMAL:
118                 return new JcaElGamalKeyPairGenerator(theFactory, pKeySpec);
119             case EC:
120             case SM2:
121             case DSTU4145:
122             case GOST2012:
123                 return new JcaECKeyPairGenerator(theFactory, pKeySpec);
124             case DSA:
125                 return new JcaDSAKeyPairGenerator(theFactory, pKeySpec);
126             case XDH:
127             case EDDSA:
128                 return new JcaEdKeyPairGenerator(theFactory, pKeySpec);
129             case DH:
130                 return new JcaDHKeyPairGenerator(theFactory, pKeySpec);
131             case NEWHOPE:
132                 return new JcaNewHopeKeyPairGenerator(theFactory, pKeySpec);
133             case SLHDSA:
134                 return new JcaSLHDSAKeyPairGenerator(theFactory, pKeySpec);
135             case CMCE:
136                 return new JcaCMCEKeyPairGenerator(theFactory, pKeySpec);
137             case FRODO:
138                 return new JcaFrodoKeyPairGenerator(theFactory, pKeySpec);
139             case SABER:
140                 return new JcaSABERKeyPairGenerator(theFactory, pKeySpec);
141             case MLKEM:
142                 return new JcaMLKEMKeyPairGenerator(theFactory, pKeySpec);
143             case MLDSA:
144                 return new JcaMLDSAKeyPairGenerator(theFactory, pKeySpec);
145             case HQC:
146                 return new JcaHQCKeyPairGenerator(theFactory, pKeySpec);
147             case BIKE:
148                 return new JcaBIKEKeyPairGenerator(theFactory, pKeySpec);
149             case NTRU:
150                 return new JcaNTRUKeyPairGenerator(theFactory, pKeySpec);
151             case NTRUPRIME:
152                 return pKeySpec.getNTRUPrimeKeySpec().getType() == GordianNTRUPrimeType.NTRUL
153                         ? new JcaNTRULPrimeKeyPairGenerator(theFactory, pKeySpec)
154                         : new JcaSNTRUPrimeKeyPairGenerator(theFactory, pKeySpec);
155             case FALCON:
156                 return new JcaFalconKeyPairGenerator(theFactory, pKeySpec);
157             case MAYO:
158                 return new JcaMayoKeyPairGenerator(theFactory, pKeySpec);
159             case SNOVA:
160                 return new JcaSnovaKeyPairGenerator(theFactory, pKeySpec);
161             case PICNIC:
162                 return new JcaPicnicKeyPairGenerator(theFactory, pKeySpec);
163             case XMSS:
164                 return new JcaXMSSKeyPairGenerator(theFactory, pKeySpec);
165             case LMS:
166                 return new JcaLMSKeyPairGenerator(theFactory, pKeySpec);
167             default:
168                 throw new GordianDataException(GordianBaseData.getInvalidText(pKeySpec.getKeyPairType()));
169         }
170     }
