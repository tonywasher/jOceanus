28  import net.sourceforge.joceanus.gordianknot.impl.core.factory.GordianCoreFactory;
29  
30  /**
31   * Jca Factory.
32   */
33  public class JcaFactory
34      extends GordianCoreFactory {
35      /**
36       * Constructor.
37       * @param pGenerator the factory generator
38       * @param pParameters the security parameters
39       * @throws GordianException on error
40       */
41      public JcaFactory(final GordianFactoryGenerator pGenerator,
42                        final GordianParameters pParameters) throws GordianException {
43          /* initialize underlying factory */
44          super(pGenerator, pParameters);
45      }
46  
47      @Override
48      public GordianDigestFactory newDigestFactory(final GordianBaseFactory pFactory) {
49          return new JcaDigestFactory(this);
50      }
51  
52      @Override
53      public GordianCipherFactory newCipherFactory(final GordianBaseFactory pFactory) {
54          return new JcaCipherFactory(this);
55      }
56  
57      @Override
58      public GordianMacFactory newMacFactory(final GordianBaseFactory pFactory) {
59          return new JcaMacFactory(this);
60      }
61  
62      @Override
63      public GordianValidator newValidator() {
64          return new JcaValidator();
65      }
66  
67      @Override
68      public GordianAsyncFactory newAsyncFactory(final GordianBaseFactory pFactory) {
69          return new JcaAsyncFactory(this);
70      }
71  }
