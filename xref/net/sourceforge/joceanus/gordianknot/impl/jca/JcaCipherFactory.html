56   * Jca Cipher Factory.
57   */
58  public class JcaCipherFactory
59      extends GordianCoreCipherFactory {
60      /**
61       * Cipher Algorithm Separator.
62       */
63      private static final Character ALGO_SEP = '/';
64  
65      /**
66       * Kalyna algorithm name.
67       */
68      private static final String KALYNA_ALGORITHM = "DSTU7624";
69  
70      /**
71       * KeyGenerator Cache.
72       */
73      private final Map<GordianKeySpec, JcaKeyGenerator<? extends GordianKeySpec>> theCache;
74  
75      /**
76       * Constructor.
77       *
78       * @param pFactory the factory
79       */
80      JcaCipherFactory(final GordianBaseFactory pFactory) {
81          /* Initialise underlying class */
82          super(pFactory);
83  
84          /* Create the cache */
85          theCache = new HashMap<>();
86      }
87  
88      @Override
89      @SuppressWarnings("unchecked")
90      public <T extends GordianKeySpec> GordianKeyGenerator<T> getKeyGenerator(final T pKeySpec) throws GordianException {
91          /* Look up in the cache */
92          JcaKeyGenerator<T> myGenerator = (JcaKeyGenerator<T>) theCache.get(pKeySpec);
93          if (myGenerator == null) {
94              /* Check the KeySpec */
95              checkKeySpec(pKeySpec);
96  
97              /* Create the new generator */
98              final String myAlgorithm = getKeyAlgorithm(pKeySpec);
99              final KeyGenerator myJavaGenerator = getJavaKeyGenerator(myAlgorithm);
100             myGenerator = new JcaKeyGenerator<>(getFactory(), pKeySpec, myJavaGenerator);
101 
102             /* Add to cache */
103 
104             theCache.put(pKeySpec, myGenerator);
105         }
106         return myGenerator;
107     }

109     @Override
110     public GordianSymCipher createSymKeyCipher(final GordianSymCipherSpec pCipherSpec) throws GordianException {
111         /* Check validity of SymKeySpec */
112         checkSymCipherSpec(pCipherSpec);
113 
114         /* If this is an AAD cipher */
115         if (pCipherSpec.isAAD()) {
116             /* Create the cipher */
117             final Cipher myBCCipher = getJavaCipher(pCipherSpec);
118             return new JcaSymAEADCipher(getFactory(), pCipherSpec, myBCCipher);
119 
120             /* else create the standard cipher */
121         } else {
122             /* Create the cipher */
123             final Cipher myBCCipher = getJavaCipher(pCipherSpec);
124             return new JcaSymCipher(getFactory(), pCipherSpec, myBCCipher);
125         }
126     }
127 
128     @Override
129     public GordianStreamCipher createStreamKeyCipher(final GordianStreamCipherSpec pCipherSpec) throws GordianException {
130         /* Check validity of StreamKeySpec */
131         checkStreamCipherSpec(pCipherSpec);
132 
133         final Cipher myJCACipher = getJavaCipher(pCipherSpec);
134         return pCipherSpec.isAEAD()
135                ? new JcaStreamAEADCipher(getFactory(), pCipherSpec, myJCACipher)
136                : new JcaStreamCipher(getFactory(), pCipherSpec, myJCACipher);
137     }

139     @Override
140     public GordianWrapper createKeyWrapper(final GordianKey<GordianSymKeySpec> pKey) throws GordianException {
141         /* Create the cipher */
142         final JcaKey<GordianSymKeySpec> myKey = JcaKey.accessKey(pKey);
143         final GordianSymCipherSpec mySpec = GordianSymCipherSpecBuilder.ecb(myKey.getKeyType(), GordianPadding.NONE);
144         final JcaSymCipher myJcaCipher = (JcaSymCipher) createSymKeyCipher(mySpec);
145         return createKeyWrapper(myKey, myJcaCipher);
146     }

148     /**
149      * Obtain the algorithm for the keySpec.
150      * @param pKeySpec the keySpec
151      * @param <T> the SpecType
152      * @return the name of the algorithm
153      * @throws GordianException on error
154      */
155     private static <T extends GordianKeySpec> String getKeyAlgorithm(final T pKeySpec) throws GordianException {
156         if (pKeySpec instanceof GordianStreamKeySpec) {
157             return getStreamKeyAlgorithm((GordianStreamKeySpec) pKeySpec);
158         }
159         if (pKeySpec instanceof GordianSymKeySpec) {
160             return getSymKeyAlgorithm((GordianSymKeySpec) pKeySpec);
161         }
162         throw new GordianDataException(GordianBaseData.getInvalidText(pKeySpec));
163     }

165     /**
166      * Create the BouncyCastle KeyGenerator via JCA.
167      * @param pAlgorithm the Algorithm
168      * @return the KeyGenerator
169      * @throws GordianException on error
170      */
171     private static KeyGenerator getJavaKeyGenerator(final String pAlgorithm) throws GordianException {
172         /* Protect against exceptions */
173         try {
174             /* Massage the keyGenerator name */
175             String myAlgorithm = pAlgorithm;
176 
177             /* Note that DSTU7624 has only a single keyGenerator */
178             if (myAlgorithm.startsWith(KALYNA_ALGORITHM)) {
179                 myAlgorithm = KALYNA_ALGORITHM;
180             }
181 
182             /* Return a KeyGenerator for the algorithm */
183             return KeyGenerator.getInstance(myAlgorithm, JcaProvider.BCPROV);
184 
185             /* Catch exceptions */
186         } catch (NoSuchAlgorithmException e) {
187             /* Throw the exception */
188             throw new GordianCryptoException("Failed to create KeyGenerator", e);
189         }
190     }

192     /**
193      * Create the BouncyCastle SymKey Cipher via JCA.
194      * @param pCipherSpec the cipherSpec
195      * @return the Cipher
196      * @throws GordianException on error
197      */
198     private static Cipher getJavaCipher(final GordianSymCipherSpec pCipherSpec) throws GordianException {
199         final String myAlgo = getSymKeyAlgorithm(pCipherSpec.getKeyType())
200                 + ALGO_SEP
201                 + getCipherModeAlgorithm(pCipherSpec)
202                 + ALGO_SEP
203                 + getPaddingAlgorithm(pCipherSpec.getPadding());
204         return getJavaCipher(myAlgo);
205     }
206 
207     /**
208      * Create the BouncyCastle StreamKey Cipher via JCA.
209      * @param pCipherSpec the StreamCipherSpec
210      * @return the Cipher
211      * @throws GordianException on error
212      */
213     private static Cipher getJavaCipher(final GordianStreamCipherSpec pCipherSpec) throws GordianException {
214         final GordianStreamKeySpec myKeySpec = pCipherSpec.getKeyType();
215         String myAlgo = getStreamKeyAlgorithm(myKeySpec);
216         if (pCipherSpec.isAEAD()
217                 && GordianStreamKeyType.CHACHA20 == myKeySpec.getStreamKeyType()) {
218             myAlgo = "CHACHA20-POLY1305";
219         }
220         return getJavaCipher(myAlgo);
221     }
222 
223     /**
224      * Create the StreamKey Cipher via JCA.
225      * @param pAlgorithm the Algorithm
226      * @return the KeyGenerator
227      * @throws GordianException on error
228      */
229     private static Cipher getJavaCipher(final String pAlgorithm) throws GordianException {
230         /* Protect against exceptions */
231         try {
232             /* Return a Cipher for the algorithm */
233             return Cipher.getInstance(pAlgorithm, JcaProvider.BCPROV);
234 
235             /* Catch exceptions */
236         } catch (NoSuchAlgorithmException
237                 | NoSuchPaddingException e) {
238             /* Throw the exception */
239             throw new GordianCryptoException("Failed to create Cipher", e);
240         }
241     }

243     /**
244      * Obtain the SymKey algorithm.
245      * @param pKeySpec the keySpec
246      * @return the Algorithm
247      * @throws GordianException on error
248      */
249     static String getSymKeyAlgorithm(final GordianSymKeySpec pKeySpec) throws GordianException {
250         switch (pKeySpec.getSymKeyType()) {
251             case TWOFISH:
252                 return "TwoFish";
253             case SERPENT:
254                 return "Serpent";
255             case THREEFISH:
256                 return "ThreeFish-" + pKeySpec.getBlockLength().getLength();
257             case GOST:
258                 return "GOST28147";
259             case SHACAL2:
260                 return "Shacal-2";
261             case KUZNYECHIK:
262                 return "GOST3412-2015";
263             case KALYNA:
264                 return KALYNA_ALGORITHM + "-" + pKeySpec.getBlockLength().getLength();
265             case RC5:
266                 return GordianLength.LEN_128.equals(pKeySpec.getBlockLength())
267                        ? "RC5-64"
268                        : "RC5";
269             case AES:
270             case CAMELLIA:
271             case CAST6:
272             case RC6:
273             case ARIA:
274             case NOEKEON:
275             case SM4:
276             case SEED:
277             case SKIPJACK:
278             case TEA:
279             case XTEA:
280             case IDEA:
281             case RC2:
282             case CAST5:
283             case BLOWFISH:
284             case DESEDE:
285                 return pKeySpec.getSymKeyType().name();
286             default:
287                 throw new GordianDataException(GordianBaseData.getInvalidText(pKeySpec));
288         }
289     }

291     /**
292      * Obtain the CipherMode algorithm.
293      * @param pSpec the cipherSpec
294      * @return the Algorithm
295      * @throws GordianException on error
296      */
297     private static String getCipherModeAlgorithm(final GordianSymCipherSpec pSpec) throws GordianException {
298         final GordianCipherMode myMode = pSpec.getCipherMode();
299         switch (pSpec.getCipherMode()) {
300             case ECB:
301             case SIC:
302             case EAX:
303             case CCM:
304             case GCM:
305             case OCB:
306             case GOFB:
307             case GCFB:
308             case CFB8:
309             case OFB8:
310                 return myMode.name();
311             case CBC:
312             case G3413CBC:
313                 return GordianCipherMode.CBC.name();
314             case CFB:
315             case G3413CFB:
316                 return "CFB";
317             case OFB:
318             case G3413OFB:
319                 return GordianCipherMode.OFB.name();
320             case KCTR:
321             case G3413CTR:
322                 return "CTR";
323             case KCCM:
324                 return GordianCipherMode.CCM.name();
325             case KGCM:
326                 return GordianCipherMode.GCM.name();
327             default:
328                 throw new GordianDataException(GordianBaseData.getInvalidText(myMode));
329         }
330     }

332     /**
333      * Obtain the Padding algorithm.
334      * @param pPadding use padding true/false
335      * @return the Algorithm
336      */
337     private static String getPaddingAlgorithm(final GordianPadding pPadding) {
338         switch (pPadding) {
339             case CTS:
340                 return "withCTS";
341             case X923:
342                 return "X923Padding";
343             case PKCS7:
344                 return "PKCS7Padding";
345             case ISO7816D4:
346                 return "ISO7816-4Padding";
347             case TBC:
348                 return "TBCPadding";
349             case NONE:
350             default:
351                 return "NoPadding";
352         }
353     }

355     /**
356      * Obtain the StreamKey algorithm.
357      * @param pKeySpec the keySpec
358      * @return the Algorithm
359      * @throws GordianException on error
360      */
361     private static String getStreamKeyAlgorithm(final GordianStreamKeySpec pKeySpec) throws GordianException {
362         switch (pKeySpec.getStreamKeyType()) {
363             case HC:
364                 return GordianLength.LEN_128 == pKeySpec.getKeyLength()
365                        ? "HC128"
366                        : "HC256";
367             case ZUC:
368                 return GordianLength.LEN_128 == pKeySpec.getKeyLength()
369                        ? "ZUC-128"
370                        : "ZUC-256";
371             case CHACHA20:
372                 return pKeySpec.getSubKeyType() == GordianChaCha20Key.STD
373                        ? "CHACHA"
374                        : "CHACHA7539";
375             case SALSA20:
376                 return pKeySpec.getSubKeyType() == GordianSalsa20Key.STD
377                        ? pKeySpec.getStreamKeyType().name()
378                        : "XSALSA20";
379             case VMPC:
380                 return pKeySpec.getSubKeyType() == GordianVMPCKey.STD
381                        ? pKeySpec.getStreamKeyType().name()
382                        : "VMPC-KSA3";
383             case GRAIN:
384                 return "Grain128";
385             case ISAAC:
386             case RC4:
387                 return pKeySpec.getStreamKeyType().name();
388             default:
389                 throw new GordianDataException(GordianBaseData.getInvalidText(pKeySpec));
390         }
391     }

393     @Override
394     protected boolean validStreamKeySpec(final GordianStreamKeySpec pKeySpec) {
395         /* Check basic validity */
396         if (!super.validStreamKeySpec(pKeySpec)) {
397             return false;
398         }
399 
400         /* Reject XChaCha20 */
401         return pKeySpec.getStreamKeyType() != GordianStreamKeyType.CHACHA20
402                 || pKeySpec.getSubKeyType() != GordianChaCha20Key.XCHACHA;
403     }

405     @Override
406     protected boolean validSymCipherSpec(final GordianSymCipherSpec pCipherSpec) {
407         /* Check standard features */
408         if (!super.validSymCipherSpec(pCipherSpec)) {
409             return false;
410         }
411 
412         /* Disallow GCM-SIV */
413         final GordianCipherMode myMode = pCipherSpec.getCipherMode();
414         if (GordianCipherMode.GCMSIV.equals(myMode)) {
415             return false;
416         }
417 
418         /* Additional Checks */
419         switch (pCipherSpec.getKeyType().getSymKeyType()) {
420             case KALYNA:
421                 /* Disallow OCB, CCM and GCM */
422                 return !GordianCipherMode.OCB.equals(myMode)
423                         && !GordianCipherMode.KCCM.equals(myMode)
424                         && !GordianCipherMode.KGCM.equals(myMode)
425                         && !GordianCipherMode.CCM.equals(myMode)
426                         && !GordianCipherMode.GCM.equals(myMode);
427             case GOST:
428                 /* Disallow OFB and CFB */
429                 return !GordianCipherMode.OFB.equals(myMode)
430                         && !GordianCipherMode.CFB.equals(myMode);
431             case KUZNYECHIK:
432                 /* Disallow OCB, OFB, CFB and CBC */
433                 return !GordianCipherMode.OCB.equals(myMode)
434                         && !GordianCipherMode.OFB.equals(myMode)
435                         && !GordianCipherMode.CFB.equals(myMode)
436                         && !GordianCipherMode.CBC.equals(myMode);
437             default:
438                 return true;
439         }
