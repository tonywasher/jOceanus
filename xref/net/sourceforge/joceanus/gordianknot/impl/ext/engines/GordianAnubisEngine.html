499     /**
500      * Create the Anubis key schedule for a given cipher key. Both encryption and decryption key
501      * schedules are generated.
502      *
503      * @param key The 32N-bit cipher key.
504      */
505     private void nessieKeysetup(final byte[] key) {
506         final int[] kappa = new int[nState];
507         final int[] inter = new int[nState];
508         final int[][] roundKeyEnc = new int[rState + 1][4];
509 
510         /*
511          * map cipher key to initial key state (mu):
512          */
513         for (int i = 0, pos = 0; i < nState; i++, pos += 4) {
514             kappa[i] = (key[pos] << 24)
515                     ^ (key[pos + 1] << 16)
516                     ^ (key[pos + 2] << 8)
517                     ^ (key[pos + 3]);
518         }
519 
520         /*
521          * generate R + 1 round keys:
522          */
523         for (int r = 0; r <= rState; r++) {
524             /*
525              * generate r-th round key K^r:
526              */
527             int k0 = T4[(kappa[nState - 1] >>> 24)];
528             int k1 = T4[(kappa[nState - 1] >>> 16) & 0xff];
529             int k2 = T4[(kappa[nState - 1] >>> 8) & 0xff];
530             int k3 = T4[(kappa[nState - 1]) & 0xff];
531             for (int i = nState - 2; i >= 0; i--) {
532                 k0 = T4[(kappa[i] >>> 24)]
533                         ^ (T5[(k0 >>> 24)] & 0xff000000)
534                         ^ (T5[(k0 >>> 16) & 0xff] & 0x00ff0000)
535                         ^ (T5[(k0 >>> 8) & 0xff] & 0x0000ff00)
536                         ^ (T5[(k0) & 0xff] & 0x000000ff);
537                 k1 = T4[(kappa[i] >>> 16) & 0xff]
538                         ^ (T5[(k1 >>> 24)] & 0xff000000)
539                         ^ (T5[(k1 >>> 16) & 0xff] & 0x00ff0000)
540                         ^ (T5[(k1 >>> 8) & 0xff] & 0x0000ff00)
541                         ^ (T5[(k1) & 0xff] & 0x000000ff);
542                 k2 = T4[(kappa[i] >>> 8) & 0xff]
543                         ^ (T5[(k2 >>> 24)] & 0xff000000)
544                         ^ (T5[(k2 >>> 16) & 0xff] & 0x00ff0000)
545                         ^ (T5[(k2 >>> 8) & 0xff] & 0x0000ff00)
546                         ^ (T5[(k2) & 0xff] & 0x000000ff);
547                 k3 = T4[(kappa[i]) & 0xff]
548                         ^ (T5[(k3 >>> 24)] & 0xff000000)
549                         ^ (T5[(k3 >>> 16) & 0xff] & 0x00ff0000)
550                         ^ (T5[(k3 >>> 8) & 0xff] & 0x0000ff00)
551                         ^ (T5[(k3) & 0xff] & 0x000000ff);
552             }
553             /*
554              * -- this is the code to use with the large U tables: K0 = K1 = K2 = K3 = 0; for (i =
555              * 0; i < N; i++) { K0 ^= U[i][(kappa[i] >> 24) ]; K1 ^= U[i][(kappa[i] >> 16) & 0xff];
556              * K2 ^= U[i][(kappa[i] >> 8) & 0xff]; K3 ^= U[i][(kappa[i] ) & 0xff]; }
557              */
558             roundKeyEnc[r][0] = k0;
559             roundKeyEnc[r][1] = k1;
560             roundKeyEnc[r][2] = k2;
561             roundKeyEnc[r][3] = k3;
562 
563             /*
564              * compute kappa^{r+1} from kappa^r:
565              */
566             if (r == rState) {
567                 break;
568             }
569             for (int i = 0; i < nState; i++) {
570                 int j = i;
571                 inter[i] = T0[(kappa[j--] >>> 24)];
572                 if (j < 0) {
573                     j = nState - 1;
574                 }
575                 inter[i] ^= T1[(kappa[j--] >>> 16) & 0xff];
576                 if (j < 0) {
577                     j = nState - 1;
578                 }
579                 inter[i] ^= T2[(kappa[j--] >>> 8) & 0xff];
580                 if (j < 0) {
581                     j = nState - 1;
582                 }
583                 inter[i] ^= T3[(kappa[j]) & 0xff];
584             }
585             kappa[0] = inter[0] ^ RC[r];
586             for (int i = 1; i < nState; i++) {
587                 kappa[i] = inter[i];
588             }
589         }
590         roundKey = roundKeyEnc;
591 
592         /* If we are decrypting */
593         if (!forEncryption) {
594             /*
595              * generate inverse key schedule: K'^0 = K^R, K'^R = K^0, K'^r = theta(K^{R-r}):
596              */
597             final int[][] roundKeyDec = new int[rState + 1][4];
598             for (int i = 0; i < 4; i++) {
599                 roundKeyDec[0][i] = roundKeyEnc[rState][i];
600                 roundKeyDec[rState][i] = roundKeyEnc[0][i];
601             }
602             for (int r = 1; r < rState; r++) {
603                 for (int i = 0; i < 4; i++) {
604                     final int v = roundKeyEnc[rState - r][i];
605                     roundKeyDec[r][i] = T0[T4[(v >>> 24)] & 0xff]
606                             ^ T1[T4[(v >>> 16) & 0xff] & 0xff]
607                             ^ T2[T4[(v >>> 8) & 0xff] & 0xff]
608                             ^ T3[T4[(v) & 0xff] & 0xff];
609                 }
610             }
611             roundKey = roundKeyDec;
612         }
613     }

615     /**
616      * Either encrypt or decrypt a data block, according to the key schedule.
617      *
618      * @param inBuffer the input buffer.
619      * @param inOff the input offset.
620      * @param outBuffer the output buffer.
621      * @param outOff the output offset
622      */
623     private void crypt(final byte[] inBuffer,
624                        final int inOff,
625                        final byte[] outBuffer,
626                        final int outOff) {
627         final int[] state = new int[4];
628         final int[] inter = new int[4];
629 
630         /*
631          * map plaintext block to cipher state (mu) and add initial round key (sigma[K^0]):
632          */
633         for (int i = 0, pos = inOff; i < 4; i++, pos += 4) {
634             state[i] = (inBuffer[pos] << 24)
635                     ^ ((inBuffer[pos + 1] << 16) & 0xff0000)
636                     ^ ((inBuffer[pos + 2] << 8) & 0xff00)
637                     ^ ((inBuffer[pos + 3]) & 0xff)
638                     ^ roundKey[0][i];
639         }
640 
641         /*
642          * R - 1 full rounds:
643          */
644         for (int r = 1; r < rState; r++) {
645             inter[0] = T0[(state[0] >>> 24)]
646                     ^ T1[(state[1] >>> 24)]
647                     ^ T2[(state[2] >>> 24)]
648                     ^ T3[(state[3] >>> 24)]
649                     ^ roundKey[r][0];
650             inter[1] = T0[(state[0] >>> 16) & 0xff]
651                     ^ T1[(state[1] >>> 16) & 0xff]
652                     ^ T2[(state[2] >>> 16) & 0xff]
653                     ^ T3[(state[3] >>> 16) & 0xff]
654                     ^ roundKey[r][1];
655             inter[2] = T0[(state[0] >>> 8) & 0xff]
656                     ^ T1[(state[1] >>> 8) & 0xff]
657                     ^ T2[(state[2] >>> 8) & 0xff]
658                     ^ T3[(state[3] >>> 8) & 0xff]
659                     ^ roundKey[r][2];
660             inter[3] = T0[(state[0]) & 0xff]
661                     ^ T1[(state[1]) & 0xff]
662                     ^ T2[(state[2]) & 0xff]
663                     ^ T3[(state[3]) & 0xff]
664                     ^ roundKey[r][3];
665             state[0] = inter[0];
666             state[1] = inter[1];
667             state[2] = inter[2];
668             state[3] = inter[3];
669         }
670 
671         /*
672          * last round:
673          */
674         inter[0] = (T0[(state[0] >>> 24)] & 0xff000000)
675                 ^ (T1[(state[1] >>> 24)] & 0x00ff0000)
676                 ^ (T2[(state[2] >>> 24)] & 0x0000ff00)
677                 ^ (T3[(state[3] >>> 24)] & 0x000000ff)
678                 ^ roundKey[rState][0];
679         inter[1] = (T0[(state[0] >>> 16) & 0xff] & 0xff000000)
680                 ^ (T1[(state[1] >>> 16) & 0xff] & 0x00ff0000)
681                 ^ (T2[(state[2] >>> 16) & 0xff] & 0x0000ff00)
682                 ^ (T3[(state[3] >>> 16) & 0xff] & 0x000000ff)
683                 ^ roundKey[rState][1];
684         inter[2] = (T0[(state[0] >>> 8) & 0xff] & 0xff000000)
685                 ^ (T1[(state[1] >>> 8) & 0xff] & 0x00ff0000)
686                 ^ (T2[(state[2] >>> 8) & 0xff] & 0x0000ff00)
687                 ^ (T3[(state[3] >>> 8) & 0xff] & 0x000000ff)
688                 ^ roundKey[rState][2];
689         inter[3] = (T0[(state[0]) & 0xff] & 0xff000000)
690                 ^ (T1[(state[1]) & 0xff] & 0x00ff0000)
691                 ^ (T2[(state[2]) & 0xff] & 0x0000ff00)
692                 ^ (T3[(state[3]) & 0xff] & 0x000000ff)
693                 ^ roundKey[rState][3];
694 
695         /*
696          * map cipher state to ciphertext block (mu^{-1}):
697          */
698         for (int i = 0, pos = outOff; i < 4; i++, pos += 4) {
699             final int w = inter[i];
700             outBuffer[pos] = (byte) (w >> 24);
701             outBuffer[pos + 1] = (byte) (w >> 16);
702             outBuffer[pos + 2] = (byte) (w >> 8);
703             outBuffer[pos + 3] = (byte) (w);
704         }
705     }

717     @Override
718     public void init(final boolean forEncrypt, final CipherParameters pParameters) {
719         /* Reject invalid parameters */
720         if (!(pParameters instanceof KeyParameter)) {
721             throw new IllegalArgumentException("Invalid parameter passed to Anubis init - "
722                     + pParameters.getClass().getName());
723         }
724 
725         /* Determine keySize */
726         final byte[] keyBytes = ((KeyParameter) pParameters).getKey();
727         final int keyBitSize = keyBytes.length * Byte.SIZE;
728         if (keyBitSize != 128 && keyBitSize != 192 && keyBitSize != 256) {
729             throw new IllegalArgumentException("KeyBitSize must be 128, 192 or 256");
730         }
731 
732         /* Record parameters */
733         keyBits = keyBitSize;
734         nState = keyBits >> 5;
735         rState = 8 + nState;
736 
737         /* Set up key */
738         this.forEncryption = forEncrypt;
739         nessieKeysetup(keyBytes);
740     }

742     @Override
743     public int processBlock(final byte[] in, final int inOff, final byte[] out, final int outOff) {
744         if (forEncryption == null) {
745             throw new IllegalStateException("Anubis engine not initialised");
746         }
747         if (inOff > (in.length - BLOCKSIZEB)) {
748             throw new DataLengthException("input buffer too short");
749         }
750         if (outOff > (out.length - BLOCKSIZEB)) {
751             throw new OutputLengthException("output buffer too short");
752         }
753         crypt(in, inOff, out, outOff);
754         return BLOCKSIZEB;
755     }
