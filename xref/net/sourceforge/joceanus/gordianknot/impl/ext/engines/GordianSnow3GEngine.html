154     /**
155      * initialise a Snow3G cipher.
156      * @param forEncryption whether or not we are for encryption.
157      * @param params the parameters required to set up the cipher.
158      * @exception IllegalArgumentException if the params argument is inappropriate.
159      */
160     public void init(final boolean forEncryption,
161                      final CipherParameters params) {
162         /*
163          * encryption and decryption is completely symmetrical, so the 'forEncryption' is
164          * irrelevant. (Like 90% of stream ciphers)
165          */
166 
167         /* Determine parameters */
168         CipherParameters myParams = params;
169         byte[] newKey = null;
170         byte[] newIV = null;
171         if ((myParams instanceof ParametersWithIV)) {
172             final ParametersWithIV ivParams = (ParametersWithIV) myParams;
173             newIV = ivParams.getIV();
174             myParams = ivParams.getParameters();
175         }
176         if (myParams instanceof KeyParameter) {
177             final KeyParameter keyParam = (KeyParameter) myParams;
178             newKey = keyParam.getKey();
179         }
180 
181         /* Initialise engine and mark as initialised */
182         theIndex = 0;
183         theIterations = 0;
184         setKeyAndIV(newKey, newIV);
185 
186         /* Save reset state */
187         theResetState = copy();
188     }

203     @Override
204     public int processBytes(final byte[] in,
205                             final int inOff,
206                             final int len,
207                             final byte[] out,
208                             final int outOff) {
209         /* Check for errors */
210         if (theResetState == null) {
211             throw new IllegalStateException(getAlgorithmName() + " not initialised");
212         }
213         if ((inOff + len) > in.length) {
214             throw new DataLengthException("input buffer too short");
215         }
216         if ((outOff + len) > out.length) {
217             throw new OutputLengthException("output buffer too short");
218         }
219 
220         /* Loop through the input bytes */
221         for (int i = 0; i < len; i++) {
222             out[i + outOff] = returnByte(in[i + inOff]);
223         }
224         return len;
225     }

227     @Override
228     public void reset() {
229         if (theResetState != null) {
230             reset(theResetState);
231         }
232     }

234     @Override
235     public byte returnByte(final byte in) {
236         /* Make the keyStream if required */
237         if (theIndex == 0) {
238             makeKeyStream();
239         }
240 
241         /* Map the next byte and adjust index */
242         final byte out = (byte) (keyStream[theIndex] ^ in);
243         theIndex = (theIndex + 1) % Integer.BYTES;
244 
245         /* Return the mapped character */
246         return out;
247     }

249     /**
250      * Decode a 32-bit value from a buffer (little-endian).
251      *
252      * @param buf the input buffer
253      * @param off the input offset
254      * @return the decoded value
255      */
256     private static int decode32be(final byte[] buf, final int off) {
257         return ((buf[off] & 0xFF) << 24)
258                 | ((buf[off + 1] & 0xFF) << 16)
259                 | ((buf[off + 2] & 0xFF) << 8)
260                 | (buf[off + 3] & 0xFF);
261     }

263     /**
264      * Encode a 32-bit value into a buffer (little-endian).
265      *
266      * @param val the value to encode
267      * @param buf the output buffer
268      * @param off the output offset
269      */
270     private static void encode32be(final int val, final byte[] buf, final int off) {
271         buf[off] = (byte) (val >> 24);
272         buf[off + 1] = (byte) (val >> 16);
273         buf[off + 2] = (byte) (val >> 8);
274         buf[off + 3] = (byte) (val);
275     }

277     /* MULx.
278      * Input v: an 8-bit input.
279      * Input c: an 8-bit input.
280      * Output : an 8-bit output.
281      * See section 3.1.1 for details.
282      */
283     int mulX(final byte v, final int c) {
284         if ((v & 0x80) != 0) {
285             return ((v << 1) ^ c) & 0xFF;
286         } else {
287             return (v << 1) & 0xFF;
288         }
289     }

291     /* MULxPOW.
292      * Input v: an 8-bit input.
293      * Input i: a positive integer.
294      * Input c: an 8-bit input.
295      * Output : an 8-bit output.
296      * See section 3.1.2 for details.
297      */
298     int mulXpow(final byte v, final int i, final int c) {
299         if (i == 0) {
300             return v & 0xFF;
301         } else {
302             return mulX((byte) mulXpow(v, i - 1, c), c);
303         }
304     }

306     /* The function MUL alpha.
307      * Input c: 8-bit input.
308      * Output : 32-bit output.
309      * See section 3.4.2 for details.
310      */
311     int mulAlpha(final byte c) {
312         return ((mulXpow(c, 23, 0xa9) << 24)
313                 | (mulXpow(c, 245, 0xa9) << 16)
314                 | (mulXpow(c, 48, 0xa9) << 8)
315                 | (mulXpow(c, 239, 0xa9)));
316     }

330     /* The 32x32-bit S-Box S1
331      * Input: a 32-bit input.
332      * Output: a 32-bit output of S1 box.
333      * See section 3.3.1.
334      */
335     int s1(final int w) {
336         final byte srw0 = SR[((w >> 24) & 0xff)];
337         final byte srw1 = SR[((w >> 16) & 0xff)];
338         final byte srw2 = SR[((w >> 8) & 0xff)];
339         final byte srw3 = SR[((w) & 0xff)];
340         final int r0 = ((mulX(srw0, 0x1b))
341                 ^ (srw1)
342                 ^ (srw2)
343                 ^ ((mulX(srw3, 0x1b)) ^ srw3)
344         ) & 0xFF;
345         final int r1 = (((mulX(srw0, 0x1b)) ^ srw0)
346                 ^ (mulX(srw1, 0x1b))
347                 ^ (srw2)
348                 ^ (srw3)
349         ) & 0xFF;
350         final int r2 = ((srw0)
351                 ^ ((mulX(srw1, 0x1b)) ^ srw1)
352                 ^ (mulX(srw2, 0x1b))
353                 ^ (srw3)
354         ) & 0xFF;
355         final int r3 = ((srw0)
356                 ^ (srw1)
357                 ^ ((mulX(srw2, 0x1b)) ^ srw2)
358                 ^ (mulX(srw3, 0x1b))
359         ) & 0xFF;
360         return (((r0) << 24) | ((r1) << 16) | ((r2) << 8)
361                 | (r3));
362     }

364     /* The 32x32-bit S-Box S2
365      * Input: a 32-bit input.
366      * Output: a 32-bit output of S2 box.
367      * See section 3.3.2.
368      */
369     int s2(final int w) {
370         final byte sqw0 = SQ[((w >> 24) & 0xff)];
371         final byte sqw1 = SQ[((w >> 16) & 0xff)];
372         final byte sqw2 = SQ[((w >> 8) & 0xff)];
373         final byte sqw3 = SQ[((w) & 0xff)];
374         final int r0 = ((mulX(sqw0, 0x69))
375                 ^ (sqw1)
376                 ^ (sqw2)
377                 ^ ((mulX(sqw3, 0x69)) ^ sqw3)
378         ) & 0xFF;
379         final int r1 = (((mulX(sqw0, 0x69)) ^ sqw0)
380                 ^ (mulX(sqw1, 0x69))
381                 ^ (sqw2)
382                 ^ (sqw3)
383         ) & 0xFF;
384         final int r2 = ((sqw0)
385                 ^ ((mulX(sqw1, 0x69)) ^ sqw1)
386                 ^ (mulX(sqw2, 0x69))
387                 ^ (sqw3)
388         ) & 0xFF;
389         final int r3 = ((sqw0)
390                 ^ (sqw1)
391                 ^ ((mulX(sqw2, 0x69)) ^ sqw2)
392                 ^ (mulX(sqw3, 0x69))
393         ) & 0xFF;
394         return (((r0) << 24) | ((r1) << 16) | ((r2) << 8)
395                 | (r3));

398     /* Clocking LFSR in initialization mode.
399      * LFSR Registers S0 to S15 are updated as the LFSR receives a single clock.
400      * Input f: a 32-bit word comes from output of FSM.
401      * See section 3.4.4.
402      */
403     void clockLFSRInitializationMode(final int f) {
404         final int v = (((lfsrState[0] << 8) & 0xffffff00)
405                 ^ (mulAlpha((byte) ((lfsrState[0] >>> 24) & 0xff)))
406                 ^ (lfsrState[2])
407                 ^ ((lfsrState[11] >>> 8) & 0x00ffffff)
408                 ^ (divAlpha((byte) ((lfsrState[11]) & 0xff)))
409                 ^ (f)
410         );
411         lfsrState[0] = lfsrState[1];
412         lfsrState[1] = lfsrState[2];
413         lfsrState[2] = lfsrState[3];
414         lfsrState[3] = lfsrState[4];
415         lfsrState[4] = lfsrState[5];
416         lfsrState[5] = lfsrState[6];
417         lfsrState[6] = lfsrState[7];
418         lfsrState[7] = lfsrState[8];
419         lfsrState[8] = lfsrState[9];
420         lfsrState[9] = lfsrState[10];
421         lfsrState[10] = lfsrState[11];
422         lfsrState[11] = lfsrState[12];
423         lfsrState[12] = lfsrState[13];
424         lfsrState[13] = lfsrState[14];
425         lfsrState[14] = lfsrState[15];
426         lfsrState[15] = v;
427     }
428 
429     /* Clocking LFSR in keystream mode.
430      * LFSR Registers S0 to S15 are updated as the LFSR receives a single clock.
431      * See section 3.4.5.
432      */
433     void clockLFSRKeyStreamMode() {
434         final int v = (((lfsrState[0] << 8) & 0xffffff00)
435                 ^ (mulAlpha((byte) ((lfsrState[0] >>> 24) & 0xff)))
436                 ^ (lfsrState[2])
437                 ^ ((lfsrState[11] >>> 8) & 0x00ffffff)
438                 ^ (divAlpha((byte) ((lfsrState[11]) & 0xff)))
439         );
440         lfsrState[0] = lfsrState[1];
441         lfsrState[1] = lfsrState[2];
442         lfsrState[2] = lfsrState[3];
443         lfsrState[3] = lfsrState[4];
444         lfsrState[4] = lfsrState[5];
445         lfsrState[5] = lfsrState[6];
446         lfsrState[6] = lfsrState[7];
447         lfsrState[7] = lfsrState[8];
448         lfsrState[8] = lfsrState[9];
449         lfsrState[9] = lfsrState[10];
450         lfsrState[10] = lfsrState[11];
451         lfsrState[11] = lfsrState[12];
452         lfsrState[12] = lfsrState[13];
453         lfsrState[13] = lfsrState[14];
454         lfsrState[14] = lfsrState[15];
455         lfsrState[15] = v;
456     }

458     /* Clocking FSM.
459      * Produces a 32-bit word F.
460      * Updates FSM registers R1, R2, R3.
461      * See Section 3.4.6.
462      */
463     int clockFSM() {
464         final int f = ((lfsrState[15] + fsmState[0]) & 0xffffffff) ^ fsmState[1];
465         final int r = (fsmState[1] + (fsmState[2] ^ lfsrState[5])) & 0xffffffff;
466         fsmState[2] = s2(fsmState[1]);
467         fsmState[1] = s1(fsmState[0]);
468         fsmState[0] = r;
469         return f;
470     }

472     /* Initialization.
473      * Input k[4]: Four 32-bit words making up 128-bit key.
474      * Input IV[4]: Four 32-bit words making 128-bit initialization variable.
475      * Output: All the LFSRs and FSM are initialized for key generation.
476      * See Section 4.1.
477      */
478     void setKeyAndIV(final byte[] key, final byte[] iv) {
479         /* Check lengths */
480         if (key == null || key.length != 16) {
481             throw new IllegalArgumentException("A key of 16 bytes is needed");
482         }
483         if (iv == null || iv.length != 16) {
484             throw new IllegalArgumentException("An IV of 16 bytes is needed");
485         }
486 
487         /* Generate four subkeys */
488         final int k0 = decode32be(key, 12);
489         final int k1 = decode32be(key, 8);
490         final int k2 = decode32be(key, 4);
491         final int k3 = decode32be(key, 0);
492 
493         /* Generate four subvectors */
494         final int i0 = decode32be(iv, 12);
495         final int i1 = decode32be(iv, 8);
496         final int i2 = decode32be(iv, 4);
497         final int i3 = decode32be(iv, 0);
498 
499         lfsrState[15] = k3 ^ i0;
500         lfsrState[14] = k2;
501         lfsrState[13] = k1;
502         lfsrState[12] = k0 ^ i1;
503         lfsrState[11] = k3 ^ 0xffffffff;
504         lfsrState[10] = k2 ^ 0xffffffff ^ i2;
505         lfsrState[9] = k1 ^ 0xffffffff ^ i3;
506         lfsrState[8] = k0 ^ 0xffffffff;
507         lfsrState[7] = k3;
508         lfsrState[6] = k2;
509         lfsrState[5] = k1;
510         lfsrState[4] = k0;
511         lfsrState[3] = k3 ^ 0xffffffff;
512         lfsrState[2] = k2 ^ 0xffffffff;
513         lfsrState[1] = k1 ^ 0xffffffff;
514         lfsrState[0] = k0 ^ 0xffffffff;
515         fsmState[0] = 0x0;
516         fsmState[1] = 0x0;
517         fsmState[2] = 0x0;
518         for (int i = 0; i < 32; i++) {
519             final int f = clockFSM();
520             clockLFSRInitializationMode(f);
521         }
522         /* Clock FSM once. Discard the output. */
523         clockFSM();
524         /* Clock LFSR in keystream mode once. */
525         clockLFSRKeyStreamMode();
526      }
527 
528     /* Generation of Keystream.
529      * input n: number of 32-bit words of keystream.
530      * input z: space for the generated keystream, assumes
531      * memory is allocated already.
532      * output: generated keystream which is filled in z
533      * See section 4.2.
534      */
535     void makeKeyStream() {
536         if (theIterations++ >= getMaxIterations()) {
537             throw new IllegalStateException("Too much data processed by singleKey/IV");
538         }
539 
540         final int f = clockFSM(); // STEP 1 */
541         encode32be(f ^ lfsrState[0], keyStream, 0); // STEP 2 */
542         /* Note that ks[t] corresponds to z_{t+1} in section 4.2 */
543         clockLFSRKeyStreamMode(); // STEP 3 */
544     }

552     public void reset(final Memoable pState) {
553         final GordianSnow3GEngine e = (GordianSnow3GEngine) pState;

556         System.arraycopy(e.keyStream, 0, keyStream, 0, keyStream.length);
557         theIterations = e.theIterations;
558         theIndex = e.theIndex;
559     }
