19  /**
20   * Implementation of the Skein family of parameterised hash functions in 256, 512 and 1024 bit block
21   * sizes, based on the {@link ThreefishEngine Threefish} tweakable block cipher.
22   * <p>
23   * This is the 1.3 version of Skein defined in the Skein hash function submission to the NIST SHA-3
24   * competition in October 2010.
25   * <p>
26   * Skein was designed by Niels Ferguson - Stefan Lucks - Bruce Schneier - Doug Whiting - Mihir
27   * Bellare - Tadayoshi Kohno - Jon Callas - Jesse Walker.
28   * <p>
29   * This implementation is the basis for SkeinDigest and {@link SkeinMac}, implementing the
30   * parameter based configuration system that allows Skein to be adapted to multiple applications. <br>
31   * Initialising the engine with {@link GordianSkeinParameters} allows standard and arbitrary parameters to
32   * be applied during the Skein hash function.
33   * <p>
34   * Implemented:
35   * <ul>
36   * <li>256, 512 and 1024 bit internal states.</li>
37   * <li>Full 96 bit input length.</li>
38   * <li>Parameters defined in the Skein specification, and arbitrary other pre and post message
39   * parameters.</li>
40   * <li>Arbitrary output size in 1 byte intervals.</li>
41   * </ul>
42   * <p>
43   * Not implemented:
44   * <ul>
45   * <li>Sub-byte length input (bit padding).</li>
46   * <li>Tree hashing.</li>
47   * </ul>
48   *
49   * @see GordianSkeinParameters
50   */
51  @SuppressWarnings("checkstyle:MagicNumber")
52  public class GordianSkeinBase
53          implements Memoable {

275     /**
276      * Ubi Tweak class.
277      */
278     private static class UbiTweak {
279         /**
280          * Point at which position might overflow long, so switch to add with carry logic.
281          */
282         private static final long LOW_RANGE = Long.MAX_VALUE - Integer.MAX_VALUE;
283 
284         /**
285          * Bit 127 = final.
286          */
287         private static final long T1_FINAL = 1L << 63;
288 
289         /**
290          * Bit 126 = first.
291          */
292         private static final long T1_FIRST = 1L << 62;
293 
294         /**
295          * UBI uses a 128 bit tweak.
296          */
297         private long[] tweak = new long[2];
298 
299         /**
300          * Whether 64 bit position exceeded.
301          */
302         private boolean extendedPosition;
303 
304         /**
305          * Constructor.
306          */
307         UbiTweak() {
308             reset();
309         }
310 
311         /**
312          * Reset according to source.
313          * @param pTweak the source
314          */
315         public void reset(final UbiTweak pTweak) {
316             this.tweak = Arrays.clone(pTweak.tweak, this.tweak);
317             this.extendedPosition = pTweak.extendedPosition;
318         }
319 
320         /**
321          * Reset.
322          */
323         public void reset() {
324             tweak[0] = 0;
325             tweak[1] = 0;
326             extendedPosition = false;
327             setFirst(true);
328         }
329 
330         public void setType(final int type) {
331             // Bits 120..125 = type
332             tweak[1] = (tweak[1] & 0xFFFFFFC000000000L) | ((type & 0x3FL) << 56);
333         }
334 
335         public int getType() {
336             return (int) ((tweak[1] >>> 56) & 0x3FL);
337         }
338 
339         public void setTreeLocation(final int level, final int hiOffset, final long loOffset) {
340             // Bits 112..119 = level
341             tweak[1] = (tweak[1] & 0xFF00FFC000000000L) | ((level & 0xFFL) << 48);
342 
343             /* Calculate high and lo parts of offset */
344             tweak[0] = loOffset;
345             tweak[1] |= hiOffset;
346 
347             /* Determine whether we have an extended position */
348             extendedPosition = loOffset < 0 || loOffset > LOW_RANGE || hiOffset != 0;
349         }
350 
351         public int getLevel() {
352             return (int) ((tweak[1] >>> 48) & 0xFFL);
353         }
354 
355         public void setFirst(final boolean first) {
356             if (first) {
357                 tweak[1] |= T1_FIRST;
358             } else {
359                 tweak[1] &= ~T1_FIRST;
360             }
361         }
362 
363         public boolean isFirst() {
364             return ((tweak[1] & T1_FIRST) != 0);
365         }
366 
367         public void setFinal(final boolean last) {
368             if (last) {
369                 tweak[1] |= T1_FINAL;
370             } else {
371                 tweak[1] &= ~T1_FINAL;
372             }
373         }
374 
375         public boolean isFinal() {
376             return ((tweak[1] & T1_FINAL) != 0);
377         }
378 
379         /**
380          * Advances the position in the tweak by the specified value.
381          * @param advance the advance count
382          */
383         public void advancePosition(final int advance) {
384             // Bits 0..95 = position
385             if (extendedPosition) {
386                 final long[] parts = new long[3];
387                 parts[0] = tweak[0] & 0xFFFFFFFFL;
388                 parts[1] = (tweak[0] >>> 32) & 0xFFFFFFFFL;
389                 parts[2] = tweak[1] & 0xFFFFFFFFL;
390 
391                 long carry = advance;
392                 for (int i = 0; i < parts.length; i++) {
393                     carry += parts[i];
394                     parts[i] = carry;
395                     carry >>>= 32;
396                 }
397                 tweak[0] = ((parts[1] & 0xFFFFFFFFL) << 32) | (parts[0] & 0xFFFFFFFFL);
398                 tweak[1] = (tweak[1] & 0xFFFFFFFF00000000L) | (parts[2] & 0xFFFFFFFFL);
399             } else {
400                 long position = tweak[0];
401                 position += advance;
402                 tweak[0] = position;
403                 if (position > LOW_RANGE) {
404                     extendedPosition = true;
405                 }
406             }
407         }
408 
409         public long[] getWords() {
410             return tweak;
411         }
412 
413         public String toString() {
414             return getType() + " first: " + isFirst() + ", final: " + isFinal();
415         }

419     /**
420      * The Unique Block Iteration chaining mode.
421      */
422      private class UBI {
423         /**
424          * The tweak.
425          */
426         private final UbiTweak tweak = new UbiTweak();
427 
428         /**
429          * Buffer for the current block of message data.
430          */
431         private byte[] currentBlock;
432 
433         /**
434          * Offset into the current message block.
435          */
436         private int currentOffset;
437 
438         /**
439          * Buffer for message words for feedback into encrypted block.
440          */
441         private long[] message;
442 
443         UBI(final int blockSize) {
444             currentBlock = new byte[blockSize];
445             message = new long[currentBlock.length / 8];
446         }
447 
448         public void reset(final UBI pUbi) {
449             currentBlock = Arrays.clone(pUbi.currentBlock, currentBlock);
450             currentOffset = pUbi.currentOffset;
451             message = Arrays.clone(pUbi.message, this.message);
452             tweak.reset(pUbi.tweak);
453         }
454 
455         public void reset(final int type) {
456             tweak.reset();
457             tweak.setType(type);
458             currentOffset = 0;
459         }
460 
461         public void setTreeLocation(final int level, final long offSet, final int pShift) {
462             int shift = getBlockSize() << 3;
463             shift += pShift; //level == 0 ? leafLen : fanOut;
464             final int hiOffset = (int) offSet >>> 64 - shift;
465             final long loOffset = offSet << shift;
466             tweak.setTreeLocation(level, hiOffset, loOffset);
467         }
468 
469         public void update(final byte[] value, final int offset, final int len, final long[] output) {
470             /*
471              * Buffer complete blocks for the underlying Threefish cipher, only flushing when there
472              * are subsequent bytes (last block must be processed in doFinal() with final=true set).
473              */
474             int copied = 0;
475             while (len > copied) {
476                 if (currentOffset == currentBlock.length) {
477                     processBlock(output);
478                     tweak.setFirst(false);
479                     currentOffset = 0;
480                 }
481 
482                 final int toCopy = Math.min((len - copied), currentBlock.length - currentOffset);
483                 System.arraycopy(value, offset + copied, currentBlock, currentOffset, toCopy);
484                 copied += toCopy;
485                 currentOffset += toCopy;
486                 tweak.advancePosition(toCopy);
487             }
488         }
489 
490         private void processBlock(final long[] output) {
491             threefish.init(true, chain, tweak.getWords());
492             for (int i = 0; i < message.length; i++) {
493                 message[i] = Pack.littleEndianToLong(currentBlock, i * 8);
494             }
495 
496             threefish.processBlock(message, output);
497 
498             for (int i = 0; i < output.length; i++) {
499                 output[i] ^= message[i];
500             }
501         }
502 
503         public void doFinal(final long[] output) {
504             // Pad remainder of current block with zeroes
505             for (int i = currentOffset; i < currentBlock.length; i++) {
506                 currentBlock[i] = 0;
507             }
508 
509             tweak.setFinal(true);
510             processBlock(output);
511         }
512     }

564     /**

568      *                       {@link #SKEIN_1024}.

572     public GordianSkeinBase(final int blockSizeBits, final int outputSizeBits) {
573         if (outputSizeBits % 8 != 0) {
574             throw new IllegalArgumentException("Output size must be a multiple of 8 bits. :" + outputSizeBits);
575         }
576         this.outputSizeBytes = outputSizeBits / Byte.SIZE;

578         this.threefish = new ThreefishEngine(blockSizeBits);
579         this.ubi = new UBI(threefish.getBlockSize());

600     private static Parameter[] clone(final Parameter[] data, final Parameter[] existing)    {
601         if (data == null) {
602             return null;
603         }
604         Parameter[] myExisting = existing;
605         if ((myExisting == null) || (myExisting.length != data.length)) {
606             myExisting = new Parameter[data.length];
607         }
608         System.arraycopy(data, 0, myExisting, 0, myExisting.length);
609         return existing;
610     }

650     /**
651      * Initialises the Skein engine with the provided parameters. See {@link GordianSkeinParameters} for
652      * details on the parameterisation of the Skein hash function.
653      *
654      * @param params the parameters to apply to this engine, or <code>null</code> to use no parameters.
655      */
656     @SuppressWarnings("unchecked")
657     public void init(final GordianSkeinParameters params) {
658         this.chain = null;
659         this.key = null;
660         this.preMessageParameters = null;
661         this.postMessageParameters = null;
662 
663         if (params != null) {
664             final byte[] theKey = params.getKey();
665             if (theKey != null && theKey.length < 16) {
666                 throw new IllegalArgumentException("Skein key must be at least 128 bits.");
667             }
668             initParams(params.getParameters());
669         }
670         createInitialState();
671 
672         // Initialise message block
673         ubiInit(PARAM_TYPE_MESSAGE);
674     }
675 
676     private void initParams(final Hashtable<Integer, byte[]> parameters) {
677         final Enumeration<Integer> keys = parameters.keys();
678         final Vector<Parameter> pre = new Vector<>();
679         final Vector<Parameter> post = new Vector<>();
680 
681         while (keys.hasMoreElements()) {
682             final Integer type = keys.nextElement();
683             final byte[] value = parameters.get(type);
684 
685             if (type == PARAM_TYPE_KEY) {
686                 this.key = value;
687             } else if (type < PARAM_TYPE_MESSAGE) {
688                 pre.addElement(new Parameter(type, value));
689             } else {
690                 post.addElement(new Parameter(type, value));
691             }
692         }
693         preMessageParameters = new Parameter[pre.size()];
694         pre.copyInto(preMessageParameters);
695         sort(preMessageParameters);
696 
697         postMessageParameters = new Parameter[post.size()];
698         post.copyInto(postMessageParameters);
699         sort(postMessageParameters);
700     }
701 
702     private static void sort(final Parameter[] params) {
703         if (params == null) {
704             return;
705         }
706 
707         // Insertion sort, for Java 1.1 compatibility
708         for (int i = 1; i < params.length; i++) {
709             final Parameter param = params[i];
710             int hole = i;
711             while (hole > 0 && param.getType() < params[hole - 1].getType()) {
712                 params[hole] = params[hole - 1];
713                 hole = hole - 1;
714             }
715             params[hole] = param;
716         }
717     }

719     /**
720      * Calculate the initial (pre message block) chaining state.
721      */
722     private void createInitialState() {
723         final boolean xtendedConfig = theConfig != null;
724         final long[] precalc = xtendedConfig ? null : INITIAL_STATES.get(variantIdentifier(getBlockSize(), getOutputSize()));
725         if ((key == null) && (precalc != null)) {
726             // Precalculated UBI(CFG)
727             chain = Arrays.clone(precalc);
728         } else {
729             // Blank initial state
730             chain = new long[getBlockSize() / 8];
731 
732             // Process key block
733             if (key != null) {
734                 ubiComplete(SkeinParameters.PARAM_TYPE_KEY, key);
735             }
736 
737             // Process configuration block
738             final Configuration myConfig = xtendedConfig ? theConfig : new Configuration(outputSizeBytes * 8L);
739             ubiComplete(PARAM_TYPE_CONFIG, myConfig.getBytes());
740         }
741 
742         // Process additional pre-message parameters
743         if (preMessageParameters != null) {
744             for (int i = 0; i < preMessageParameters.length; i++) {
745                 final Parameter param = preMessageParameters[i];
746                 ubiComplete(param.getType(), param.getValue());
747             }
748         }
749         initialState = Arrays.clone(chain);
750     }

754      * accept message input.
755      */
756     public void reset() {

759         ubiInit(PARAM_TYPE_MESSAGE);

807     void calculateNode(final byte[] out, final int outOff) {
808         checkInitialised();
809         if (out.length < (outOff + outputSizeBytes)) {
810             throw new OutputLengthException("Output buffer is too short to hold output");
811         }
812 
813         // Finalise message block
814         ubiFinal();
815 
816         /* Output the state */
817         for (int i = 0; i < chain.length; i++) {
818             Pack.longToLittleEndian(chain[i], out, outOff + (i * 8));
819         }
820     }

822     /**
823      * Finalise the digest.
824      * @param out the output buffer
825      * @param outOff the output offset
826      * @return the number of bytes returned
827      */
828     public int doFinal(final byte[] out, final int outOff) {
829         checkInitialised();
830         if (out.length < (outOff + outputSizeBytes)) {
831             throw new OutputLengthException("Output buffer is too short to hold output");
832         }
833 
834         // Initiate output
835         initiateOutput();
836 
837         // Perform the output transform
838         final int blockSize = getBlockSize();
839         final int blocksRequired = ((outputSizeBytes + blockSize - 1) / blockSize);
840         for (int i = 0; i < blocksRequired; i++) {
841             final int toWrite = Math.min(blockSize, outputSizeBytes - (i * blockSize));
842             output(i, out, outOff + (i * blockSize), toWrite);
843         }
844 
845         reset();
846 
847         return outputSizeBytes;
848     }

852         if (postMessageParameters != null) {

856             }
857         }

861         // Finalise message block
862         ubiFinal();

868     void restoreForOutput(final byte[] pState) {
869         /* Restore the state */
870         for (int i = 0; i < chain.length; i++) {
871             chain[i] = Pack.littleEndianToLong(pState, i * 8);
872         }

874         // Initiate output
875         initiateOutput();

878     void output(final long outputSequence, final byte[] out, final int outOff, final int outputBytes) {
879         final byte[] currentBytes = new byte[8];
880         Pack.longToLittleEndian(outputSequence, currentBytes, 0);
881 
882         // Output is a sequence of UBI invocations all of which use and preserve the pre-output
883         // state
884         final long[] outputWords = new long[chain.length];
885         ubiInit(PARAM_TYPE_OUTPUT);
886         this.ubi.update(currentBytes, 0, currentBytes.length, outputWords);
887         ubi.doFinal(outputWords);
888 
889         final int wordsRequired = ((outputBytes + 8 - 1) / 8);
890         for (int i = 0; i < wordsRequired; i++) {
891             final int toWrite = Math.min(8, outputBytes - (i * 8));
892             if (toWrite == 8) {
893                 Pack.longToLittleEndian(outputWords[i], out, outOff + (i * 8));
894             } else  {
895                 Pack.longToLittleEndian(outputWords[i], currentBytes, 0);
896                 System.arraycopy(currentBytes, 0, out, outOff + (i * 8), toWrite);
897             }
898         }
899     }
