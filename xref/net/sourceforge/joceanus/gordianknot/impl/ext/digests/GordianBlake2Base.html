208         if (pKey == null || pKey.length == 0) {

302     /**
303      * Set the treeConfig.
304      * @param pFanOut the fanOut.
305      * @param pMaxDepth the maxDepth.
306      * @param pLeafLen the leafLength.
307      */
308     void setTreeConfig(final int pFanOut,
309                        final int pMaxDepth,
310                        final int pLeafLen) {
311         /* Check that fanOut value makes sense */
312         if (pFanOut < 0 || pFanOut > MAXBYTE) {
313             throw new IllegalArgumentException("FanOut out of range");
314         }
315         final boolean seqMode = pFanOut == 1;
316 
317         /* Check that maxDepth value makes sense */
318         final boolean xofMode = pMaxDepth == 0;
319         if (pMaxDepth < 0 || pMaxDepth > MAXBYTE) {
320             throw new IllegalArgumentException("MaxDepth out of range");
321         }
322         if (seqMode != (pMaxDepth == 1)) {
323             throw new IllegalArgumentException("Inconsistent treeConfig for sequentialMode");
324         }
325 
326         /* Check that leaf value makes sense */
327         if (pLeafLen < 0) {
328             throw new IllegalArgumentException("LeafLength out of range");
329         }
330         if (seqMode != (pLeafLen == 0)) {
331             throw new IllegalArgumentException("Inconsistent treeConfig for LeafLen and fanOut");
332         }
333         if (xofMode && pFanOut != 0) {
334             throw new IllegalArgumentException("Inconsistent treeConfig for xofMode");
335         }
336 
337         /* Record the values */
338         theFanOut = (short) pFanOut;
339         theMaxDepth = (short) pMaxDepth;
340         theLeafLen = pLeafLen;
341     }

443     @Override
444     public void update(final byte b) {
445         /* If the buffer is full */
446         final int blockLen = theBuffer.length;
447         final int remainingLength = blockLen - thePos;
448         if (remainingLength == 0) {
449             /* Process the buffer */
450             adjustCounter(blockLen);
451             compressF(theBuffer, 0);
452 
453             /* Reset the buffer */
454             Arrays.fill(theBuffer, (byte) 0);
455             thePos = 0;
456         }
457 
458         /* Store the byte */
459         theBuffer[thePos] = b;
460         thePos++;
461     }
462 
463     @Override
464     public void update(final byte[] pMessage,
465                        final int pOffset,
466                        final int pLen) {
467         /* Ignore null operation */
468         if (pMessage == null || pLen == 0) {
469             return;
470         }
471 
472         /* Process any bytes currently in the buffer */
473         final int blockLen = theBuffer.length;
474         int remainingLen = 0; // left bytes of buffer
475         if (thePos != 0) {
476             /* Calculate space remaining in the buffer */
477             remainingLen = blockLen - thePos;
478 
479             /* If there is sufficient space in the buffer */
480             if (remainingLen >= pLen) {
481                 /* Copy date into byffer and return */
482                 System.arraycopy(pMessage, pOffset, theBuffer, thePos, pLen);
483                 thePos += pLen;
484                 return;
485             }
486 
487             /* Fill the buffer */
488             System.arraycopy(pMessage, pOffset, theBuffer, thePos, remainingLen);
489 
490             /* Adjust bytes count */
491             adjustCounter(blockLen);
492 
493             /* Process the buffer */
494             compressF(theBuffer, 0);
495 
496             /* Reset the buffer */
497             thePos = 0;
498             Arrays.fill(theBuffer, (byte) 0);
499         }
500 
501         /* process all blocks except the last one */
502         int messagePos;
503         final int blockWiseLastPos = pOffset + pLen - blockLen;
504         for (messagePos = pOffset + remainingLen; messagePos < blockWiseLastPos; messagePos += blockLen) {
505             /* Adjust bytes count */
506             adjustCounter(blockLen);
507 
508             /* Process the buffer */
509             compressF(pMessage, messagePos);
510         }
511 
512         /* Fill the buffer with the remaining bytes of the message */
513         final int len = pLen - messagePos;
514         System.arraycopy(pMessage, messagePos, theBuffer, 0, pOffset + len);
515         thePos += pOffset + len;
516     }

518     @Override
519     public int doFinal(final byte[] pOut,
520                        final int pOutOffset) {
521         /* Adjust flags and counter */
522         isLastBlock = true;
523         completeCounter(thePos);
524 
525         /* Process the buffer */
526         compressF(theBuffer, 0);
527         Arrays.fill(theBuffer, (byte) 0);
528 
529         /* Output the digest */
530         outputDigest(pOut, pOutOffset);
531 
532         /* Reset the state */
533         reset();
534 
535         /* Return the digest length */
536         return theDigestLen;
537     }

539     @Override
540     public void reset() {
541         /* Reset flags */
542         isLastBlock = false;
543         isLastNode = false;
544 
545         /* Reset the data Buffer */
546         thePos = 0;
547         Arrays.fill(theBuffer, (byte) 0);
548 
549         /* Activate */
550         activateH();
551     }

618     /**
619      * Obtain the Sigma for the round.
620      * @param pRound the round
621      * @return the Sigma
622      */
623     private static byte[] getSigmaForRound(final int pRound) {
624         return SIGMA[pRound % SIGMA.length];
625     }

627     /**
628      * Compress a message.
629      * @param pMessage the message buffer
630      * @param pMsgPos the position within the message buffer
631      */
632     private void compressF(final byte[] pMessage,
633                            final int pMsgPos) {
634         /* Initialise the buffers */
635         initV();
636         initM(pMessage, pMsgPos);
637 
638         /* Loop through the rounds */
639         for (int round = 0; round < theRounds; round++) {
640             /* Obtain the relevant SIGMA */
641             final byte[] sigma = getSigmaForRound(round);
642 
643             /* Apply to columns of V */
644             mixG(sigma[0], sigma[1], 0, 4, 8, 12);
645             mixG(sigma[2], sigma[3], 1, 5, 9, 13);
646             mixG(sigma[4], sigma[5], 2, 6, 10, 14);
647             mixG(sigma[6], sigma[7], 3, 7, 11, 15);
648 
649             /* Apply to diagonals of V */
650             mixG(sigma[8], sigma[9], 0, 5, 10, 15);
651             mixG(sigma[10], sigma[11], 1, 6, 11, 12);
652             mixG(sigma[12], sigma[13], 2, 7, 8, 13);
653             mixG(sigma[14], sigma[15], 3, 4, 9, 14);
654         }
655 
656         /* Adjust H */
657         adjustH();
658     }
