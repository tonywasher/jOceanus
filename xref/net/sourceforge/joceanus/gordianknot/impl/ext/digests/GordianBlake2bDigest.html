17  package net.sourceforge.joceanus.gordianknot.impl.ext.digests;
18  
19  import org.bouncycastle.util.Memoable;
20  import org.bouncycastle.util.Pack;
21  
22  /**
23   * Blake2b digest.
24   */
25  @SuppressWarnings("checkstyle:MagicNumber")
26  public class GordianBlake2bDigest
27          extends GordianBlake2Base {
28      /**
29       * Number of Rounds.
30       */
31      private static final int ROUNDS = 12;
32  
33      /**
34       * Block length.
35       */
36      private static final int BLOCK_LENGTH_BYTES = NUMWORDS * Long.BYTES << 1;
37  
38      /**
39       * Blake2b Initialization Vector.
40       */
41      private static final long[] IV = {
42              // Produced from the square root of primes 2, 3, 5, 7, 11, 13, 17, 19.
43              // The same as SHA-512 IV.
44  
45                      0x6a09e667f3bcc908L, 0xbb67ae8584caa73bL, 0x3c6ef372fe94f82bL,
46                      0xa54ff53a5f1d36f1L, 0x510e527fade682d1L, 0x9b05688c2b3e6c1fL,
47                      0x1f83d9abfb41bd6bL, 0x5be0cd19137e2179L
48              };
49  
50      /**
51       * The state.
52       */
53      private final long[] theH = new long[NUMWORDS];
54  
55      /**
56       * The workBuffer.
57       */
58      private final long[] theV = new long[NUMWORDS << 1];
59  
60      /**
61       * The messageBuffer.
62       */
63      private final long[] theM = new long[NUMWORDS << 1];
64  
65      /**
66       * Low Counter.
67       */
68      private long t0;
69  
70      /**
71       * High Counter.
72       */
73      private long t1;
74  
75      /**
76       * Constructor.
77       */
78      public GordianBlake2bDigest() {
79          /* Default to 512 bits */
80          this(512);
81      }
82  
83      /**
84       * Constructor.
85       * @param pLength the digest length in bits.
86       */
87      public GordianBlake2bDigest(final int pLength) {
88          /* Initialise underlying class */
89          super(ROUNDS, BLOCK_LENGTH_BYTES);
90  
91          /* Check digest length */
92          if ((pLength % Byte.SIZE) != 0 || pLength < 0 || pLength > 512) {
93              throw new IllegalArgumentException("Incorrect digest length");
94          }
95          setDigestLength(pLength / Byte.SIZE);
96          activateH();
97      }
98  
99      /**
100      * Constructor.
101      * @param pSource the source digest.
102      */
103     private GordianBlake2bDigest(final GordianBlake2bDigest pSource) {
104         /* Initialise underlying class */
105         super(pSource);
106 
107         /* Initialise from source */
108         reset((Memoable) pSource);
109     }
110 
111     @Override
112     public String getAlgorithmName() {
113         return "Blake2b-" + getDigestSize() * Byte.SIZE;
114     }
115 
116     @Override
117     public int getByteLength() {
118         return BLOCK_LENGTH_BYTES;
119     }
120 
121     @Override
122     public void reset() {
123         /* Reset counter */
124         t0 = 0L;
125         t1 = 0L;
126 
127         /* reset underlying class */
128         super.reset();
129     }
130 
131     @Override
132     public void reset(final Memoable pSource) {
133         /* Access source */
134         final GordianBlake2bDigest mySource = (GordianBlake2bDigest) pSource;
135 
136         /* reset underlying class */
137         super.reset(mySource);
138 
139         /* Reset counter */
140         t0 = mySource.t0;
141         t1 = mySource.t1;
142 
143         /* Copy state */
144         System.arraycopy(mySource.theH, 0, theH, 0, theH.length);
145     }
146 
147     @Override
148     public GordianBlake2bDigest copy() {
149         return new GordianBlake2bDigest(this);
150     }
151 
152     @Override
153     void adjustCounter(final int pCount) {
154         t0 += pCount;
155         if (t0 == 0) {
156             t1++;
157         }
158     }
159 
160     @Override
161     void completeCounter(final int pCount) {
162         t0 += pCount;
163         if (pCount > 0 && t0 == 0) {
164             t1++;
165         }
166     }
167 
168     @Override
169     void outputDigest(final byte[] pOut,
170                       final int pOutOffset) {
171         /* Loop to provide the output */
172         final int myDigestLen = getDigestSize();
173         for (int i = 0, j = 0; i < NUMWORDS && j < myDigestLen; i++, j += Long.BYTES) {
174             /* Convert the next word to bytes */
175             final byte[] bytes = Pack.longToLittleEndian(theH[i]);
176 
177             if (j + Long.BYTES < myDigestLen) {
178                 System.arraycopy(bytes, 0, pOut, pOutOffset + j, Long.BYTES);
179             } else {
180                 System.arraycopy(bytes, 0, pOut, pOutOffset + j, myDigestLen - j);
181             }
182         }
183     }
184 
185     @Override
186     void activateH() {
187         /* Initialise from IV */
188         System.arraycopy(IV, 0, theH, 0, IV.length);
189 
190         /* Initialise first word */
191         theH[0] ^= getDigestSize() | (getKeyLen() << Byte.SIZE);
192         theH[0] ^= (getFanOut() | (getMaxDepth() << Byte.SIZE)) << Short.SIZE;
193         theH[0] ^= ((long) getLeafLen()) << Integer.SIZE;
194 
195         /* Initialise second word */
196         theH[1] ^= getNodeOffset() | (((long) getXofLen()) << Integer.SIZE);
197 
198         /* Initialise third word */
199         theH[2] ^= (getNodeDepth() | (getInnerLen() << Byte.SIZE));
200 
201         /* Build salt section */
202         final byte[] mySalt = getSalt();
203         if (mySalt != null) {
204             theH[4] ^= Pack.littleEndianToLong(mySalt, 0);
205             theH[5] ^= Pack.littleEndianToLong(mySalt, Long.BYTES);
206         }
207 
208         /* Build personalisation section */
209         final byte[] myPersonal = getPersonal();
210         if (myPersonal != null) {
211             theH[6] ^= Pack.littleEndianToLong(myPersonal, 0);
212             theH[7] ^= Pack.littleEndianToLong(myPersonal, Long.BYTES);
213         }
214 
215         /* Initialise any keyBlock */
216         initKeyBlock();
217     }
218 
219     @Override
220     void initV() {
221         /* Copy in H and IV */
222         System.arraycopy(theH, 0, theV, 0, NUMWORDS);
223         System.arraycopy(IV, 0, theV, NUMWORDS, NUMWORDS);
224 
225         /* Fold in counters */
226         theV[12] ^= t0;
227         theV[13] ^= t1;
228 
229         /* Fold in finalisation flags */
230         if (isLastBlock()) {
231             theV[14] ^= -1L;
232             if (isLastNode()) {
233                 theV[15] ^= -1L;
234             }
235         }
236     }
237 
238     @Override
239     void initM(final byte[] pMessage,
240                final int pMsgPos) {
241         /* Copy message bytes into word array */
242         for (int i = 0; i < NUMWORDS << 1; i++) {
243             theM[i] = Pack.littleEndianToLong(pMessage, pMsgPos + i * Long.BYTES);
244         }
245     }
246 
247     @Override
248     void adjustH() {
249         /* Combine V into H */
250         for (int i = 0; i < NUMWORDS; i++) {
251             theH[i] ^= theV[i] ^ theV[i + NUMWORDS];
252         }
253     }
254 
255     @Override
256     void mixG(final int msgIdx1,
257               final int msgIdx2,
258               final int posA,
259               final int posB,
260               final int posC,
261               final int posD) {
262         /* Perform the Round */
263         theV[posA] += theV[posB] + theM[msgIdx1];
264         theV[posD] = rotr64(theV[posD] ^ theV[posA], 32);
265         theV[posC] += theV[posD];
266         theV[posB] = rotr64(theV[posB] ^ theV[posC], 24);
267         theV[posA] += theV[posB] + theM[msgIdx2];
268         theV[posD] = rotr64(theV[posD] ^ theV[posA], 16);
269         theV[posC] += theV[posD];
270         theV[posB] = rotr64(theV[posB] ^ theV[posC], 63);
271     }
272 
273     /**
274      * Rotate a long right.
275      * @param x the value to rotate
276      * @param rot the number of bits to rotate
277      * @return the result
278      */
279     private static long rotr64(final long x,
280                                final int rot) {
281         return x >>> rot | (x << (Long.SIZE - rot));
282     }
283 }
