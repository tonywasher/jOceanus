24  /**
25   * Groestl Digest.
26   * <p>
27   * The embedded GroestlFastDigest class is ported from the C implementation in
28   * groestl_opt.h/tables.h in the "NIST submission package" at http://www.groestl.info with tweaks to
29   * interface to the BouncyCastle libraries
30   */
31  @SuppressWarnings({"checkstyle:MagicNumber", "checkstyle:JavadocVariable"})
32  public class GordianGroestlDigest
33          implements ExtendedDigest, Memoable {
34      /**
35       * The underlying digest.
36       */
37      private final GordianGroestlFastDigest theDigest;
38  
39      /**
40       * The digest length.
41       */
42      private final int theDigestLen;
43  
44      /**
45       * Constructor.
46       * @param pHashBitLen the hash bit length
47       */
48      public GordianGroestlDigest(final int pHashBitLen) {
49          theDigest = new GordianGroestlFastDigest(pHashBitLen);
50          theDigestLen = pHashBitLen / Byte.SIZE;
51      }
52  
53      /**
54       * Constructor.
55       * @param pDigest the digest to copy
56       */
57      public GordianGroestlDigest(final GordianGroestlDigest pDigest) {
58          theDigestLen = pDigest.theDigestLen;
59          theDigest = new GordianGroestlFastDigest(theDigestLen * Byte.SIZE);
60          theDigest.copyIn(pDigest.theDigest);
61      }
62  
63      @Override
64      public int doFinal(final byte[] pHash, final int pOffset) {
65          theDigest.finalise(pHash, pOffset);
66          return getDigestSize();
67      }
68  
69      @Override
70      public String getAlgorithmName() {
71          return "Groestl";
72      }
73  
74      @Override
75      public int getDigestSize() {
76          return theDigestLen;
77      }
78  
79      @Override
80      public void reset() {
81          theDigest.reset();
82      }
83  
84      @Override
85      public void update(final byte arg0) {
86          final byte[] myByte = new byte[] { arg0 };
87          update(myByte, 0, 1);
88      }
89  
90      @Override
91      public void update(final byte[] pData, final int pOffset, final int pLength) {
92          theDigest.update(pData, pOffset, ((long) pLength) * Byte.SIZE);
93      }
94  
95      @Override
96      public int getByteLength() {
97          return theDigest.getBufferSize();
98      }
99  
100     @Override
101     public GordianGroestlDigest copy() {
102         return new GordianGroestlDigest(this);
103     }

105     @Override
106     public void reset(final Memoable pState) {
107         final GordianGroestlDigest d = (GordianGroestlDigest) pState;
108         theDigest.copyIn(d.theDigest);
109     }

528             long myValue = pValue;

535             int pos = pOffset + pLength - i;
536             while (i-- > 0) {
537                 pBuffer[pos++] = (byte) (myValue & 0xff);
538                 myValue >>= 8;
539             }

668         /* the compression function (short variants) */
669         private void f512(final long[] h, final byte[] m, final int pOffset) {
670             for (int i = 0; i < COLS512; i++) {
671                 final long inVal = leBytesToLong(m, pOffset, i);
672                 tmpZ[i] = inVal;
673                 tmpInP[i] = h[i] ^ inVal;
674             }
675 
676             /* compute Q(m) */
677             rnd512Q(tmpZ, tmpY, 0x0000000000000000L);
678             rnd512Q(tmpY, tmpZ, 0x0100000000000000L);
679             rnd512Q(tmpZ, tmpY, 0x0200000000000000L);
680             rnd512Q(tmpY, tmpZ, 0x0300000000000000L);
681             rnd512Q(tmpZ, tmpY, 0x0400000000000000L);
682             rnd512Q(tmpY, tmpZ, 0x0500000000000000L);
683             rnd512Q(tmpZ, tmpY, 0x0600000000000000L);
684             rnd512Q(tmpY, tmpZ, 0x0700000000000000L);
685             rnd512Q(tmpZ, tmpY, 0x0800000000000000L);
686             rnd512Q(tmpY, tmpOutQ, 0x0900000000000000L);
687 
688             /* compute P(h+m) */
689             rnd512P(tmpInP, tmpZ, 0x0000000000000000L);
690             rnd512P(tmpZ, tmpY, 0x0000000000000001L);
691             rnd512P(tmpY, tmpZ, 0x0000000000000002L);
692             rnd512P(tmpZ, tmpY, 0x0000000000000003L);
693             rnd512P(tmpY, tmpZ, 0x0000000000000004L);
694             rnd512P(tmpZ, tmpY, 0x0000000000000005L);
695             rnd512P(tmpY, tmpZ, 0x0000000000000006L);
696             rnd512P(tmpZ, tmpY, 0x0000000000000007L);
697             rnd512P(tmpY, tmpZ, 0x0000000000000008L);
698             rnd512P(tmpZ, tmpY, 0x0000000000000009L);
699 
700             /* h' == h + Q(m) + P(h+m) */
701             for (int i = 0; i < COLS512; i++) {
702                 h[i] ^= tmpOutQ[i] ^ tmpY[i];
703             }
704         }

706         /* the compression function (long variants) */
707         private void f1024(final long[] h, final byte[] m, final int pOffset) {
708             for (int i = 0; i < COLS1024; i++) {
709                 final long inVal = leBytesToLong(m, pOffset, i);
710                 tmpZ[i] = inVal;
711                 tmpInP[i] = h[i] ^ inVal;
712             }
713 
714             /* compute Q(m) */
715             rnd1024Q(tmpZ, tmpY, 0);
716             for (int i = 1; i < ROUNDS1024 - 1; i += 2) {
717                 rnd1024Q(tmpY, tmpZ, u64Big(i));
718                 rnd1024Q(tmpZ, tmpY, u64Big(i + 1));
719             }
720             rnd1024Q(tmpY, tmpOutQ, u64Big((long) ROUNDS1024 - 1));
721 
722             /* compute P(h+m) */
723             rnd1024P(tmpInP, tmpZ, 0);
724             for (int i = 1; i < ROUNDS1024 - 1; i += 2) {
725                 rnd1024P(tmpZ, tmpY, u64Big(((long) i) << 56));
726                 rnd1024P(tmpY, tmpZ, u64Big(((long) (i + 1)) << 56));
727             }
728             rnd1024P(tmpZ, tmpY, u64Big(((long) (ROUNDS1024 - 1)) << 56));

731             for (int i = 0; i < COLS1024; i++) {

736         /* digest up to msglen bytes of input (full blocks only) */
737         private void transform(final byte[] input, final int pOffset, final int msglen) {
738             /*
739              * determine variant, SHORT or LONG, and select underlying compression function based on
740              * the variant
741              */
742             int myOffset = pOffset;
743             int myMsgLen = msglen;
744             if (isShort) {
745                 /* increment block counter */
746                 blockCounter += myMsgLen / SIZE512;
747                 while (myMsgLen >= SIZE512) {
748                     f512(chaining, input, myOffset);
749                     myMsgLen -= SIZE512;
750                     myOffset += SIZE512;
751                 }
752             } else {
753                 /* increment block counter */
754                 blockCounter += myMsgLen / SIZE1024;
755                 while (myMsgLen >= SIZE1024) {
756                     f1024(chaining, input, myOffset);
757                     myMsgLen -= SIZE1024;
758                     myOffset += SIZE1024;
759                 }
760             }
761         }

768                     tmpInP[j] = chaining[j];

778                 rnd512P(tmpY, tmpZ, 0x0000000000000008L);

842             }

853         /* update state with databitlen bits of input */
854         void update(final byte[] input, final int pOffset, final long databitlen) {
855             int index = 0;
856             final int msglen = (int) (databitlen / 8);
857             final int rem = (int) (databitlen % 8);
858 
859             /*
860              * non-integral number of message bytes can only be supplied in the last call to this
861              * function
862              */
863             if (bitsInLastByte != 0) {
864                 throw new IllegalStateException("FAIL");
865             }
866 
867             /* Ensure that we are initialised */
868             ensureInitialised();
869 
870             /*
871              * if the buffer contains data that has not yet been digested, first add data to buffer
872              * until full
873              */
874             if (bufPtr != 0) {
875                 while (bufPtr < size && index < msglen) {
876                     buffer[bufPtr++] = input[pOffset + index++];
877                 }
878                 if (bufPtr < size) {
879                     /* buffer still not full, return */
880                     if (rem != 0) {
881                         bitsInLastByte = rem;
882                         buffer[bufPtr++] = input[pOffset + index];
883                     }
884                     return;
885                 }
886 
887                 /* digest buffer */
888                 bufPtr = 0;
889                 transform(buffer, 0, size);
890             }
891 
892             /* digest bulk of message */
893             transform(input, pOffset + index, msglen - index);
894             index += ((msglen - index) / size) * size;
895 
896             /* store remaining data in buffer */
897             while (index < msglen) {
898                 buffer[bufPtr++] = input[pOffset + index++];
899             }
900 
901             /*
902              * if non-integral number of bytes have been supplied, store remaining bits in last
903              * byte, together with information about number of bits
904              */
905             if (rem != 0) {
906                 bitsInLastByte = rem;
907                 buffer[bufPtr++] = input[pOffset + index];
908             }
909         }

925         /*
926          * finalise: process remaining data (including padding), perform output transformation, and
927          * write hash result to 'output'
928          */
929         void finalise(final byte[] hashval, final int pOffset) {
930             final int hashbytelen = hashbitlen / 8;
931 
932             /* Ensure that we are initialised */
933             ensureInitialised();
934 
935             /* pad with '1'-bit and first few '0'-bits */
936             if (bitsInLastByte != 0) {
937                 buffer[bufPtr - 1] &= ((1 << bitsInLastByte) - 1) << (8 - bitsInLastByte);
938                 buffer[bufPtr - 1] ^= 0x1 << (7 - bitsInLastByte);
939                 bitsInLastByte = 0;
940             } else {
941                 buffer[bufPtr++] = (byte) 0x80;
942             }
943 
944             /* pad with '0'-bits */
945             if (bufPtr > size - LENGTHFIELDLEN) {
946                 /* padding requires two blocks */
947                 while (bufPtr < size) {
948                     buffer[bufPtr++] = 0;
949                 }
950                 /* digest first padding block */
951                 transform(buffer, 0, size);
952                 bufPtr = 0;
953             }
954             while (bufPtr < size - LENGTHFIELDLEN) {
955                 buffer[bufPtr++] = 0;
956             }
957 
958             /* length padding */
959             blockCounter++;
960             bufPtr = size;
961             while (bufPtr > size - LENGTHFIELDLEN) {
962                 buffer[--bufPtr] = (byte) blockCounter;
963                 blockCounter >>>= 8;
964             }
965 
966             /* digest final padding block */
967             transform(buffer, 0, size);
968 
969             /* perform output transformation */
970             outputTransformation();
971 
972             /* store hash result in output */
973             buildHashFromState(hashval, pOffset, hashbytelen);
974 
975             /* Reset the digest */
976             reset();
977         }
