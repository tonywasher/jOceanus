118     /**

122      * @param pControls the controls factory
123      */
124     TethysUIFXHTMLManager(final TethysUICoreFactory<?> pFactory,
125                           final TethysUIFXControlFactory pControls) {
126         /* Initialise underlying class */
127         super(pFactory);
128 
129         /* Store the factory */
130         theControls = pControls;
131 
132         /* Allocate the Pane */
133         thePane = new BorderPane();
134         theNode = new TethysUIFXNode(thePane);
135 
136         /* Attach a listener to the Factory */
137         if (theControls.getStage() == null) {
138             pFactory.getEventRegistrar().addEventListener(e -> allocateWebView());
139         } else {
140             allocateWebView();
141         }
142 
143         /* Create the hyperLink listener */
144         theListener = this::handleClick;
145     }

177     @Override
178     protected void loadHTMLContent(final String pHTMLString) {
179         /* Load the content */
180         if (theWebView != null) {
181             Platform.runLater(() -> theWebEngine.loadContent(pHTMLString));

187         /* reLoad the content */

191     /**
192      * Allocate webView.
193      */
194     private void allocateWebView() {
195         /* If we do not have a webView */
196         if (theWebView == null) {
197             /* Create WebView and access the engine */
198             theWebView = new WebView();
199             theWebEngine = theWebView.getEngine();
200             theWebEngine.setJavaScriptEnabled(true);
201 
202             /* Attach to the pane */
203             thePane.setCenter(theWebView);
204 
205             /* Create the load listener */
206             theWebEngine.getLoadWorker().stateProperty().addListener((v, o, n) -> handleStateChange(n));
207 
208             /* load CSS contents */
209             loadCSSContents();
210         }
211     }

213     @Override
214     public void scrollToReference(final String pReference) {
215         /* If the webPage is fully loaded */
216         if (theWebEngine.getLoadWorker().getState().equals(Worker.State.SUCCEEDED)) {
217             /* Obtain the key objects */
218             final JSObject myWin = (JSObject) theWebEngine.executeScript("window");
219             final JSObject myDoc = (JSObject) theWebEngine.executeScript("document");
220             final JSObject myEl = (JSObject) myDoc.call("getElementById", pReference);
221 
222             /* If we found the reference */
223             if (myEl != null) {
224                 /* Access element rectangle */
225                 final JSObject myBox = (JSObject) myEl.call("getBoundingClientRect");
226 
227                 /* Calculate vertical scroll */
228                 final Integer myHeight = (Integer) myWin.getMember("innerHeight");
229                 final Double myTop = getJSDouble(myBox.getMember("top"));
230                 final Double myBottom = getJSDouble(myBox.getMember("bottom"));
231                 Double myAdjustY = myTop;
232                 if (myBottom - myTop > myHeight) {
233                     myAdjustY = myBottom - myHeight;
234                 }
235 
236                 /* Calculate horizontal scroll */
237                 final Integer myWidth = (Integer) myWin.getMember("innerWidth");
238                 final Double myLeft = getJSDouble(myBox.getMember("left"));
239                 final Double myRight = getJSDouble(myBox.getMember("right"));
240                 Double myAdjustX = myLeft;
241                 if (myRight - myLeft > myWidth) {
242                     myAdjustX = myRight - myWidth;
243                 }
244 
245                 /* Perform the scroll */
246                 final Integer myXOffset = getJSInteger(myWin.getMember("pageXOffset"));
247                 final Integer myYOffset = getJSInteger(myWin.getMember("pageYOffset"));
248                 myWin.call("scrollTo", myXOffset + myAdjustX, myYOffset + myAdjustY);
249             } else {
250                 LOGGER.error("Failed to locate reference <%s>", pReference);
251             }
252 
253             /* else still loading */
254         } else {
255             /* Register pending scroll */
256             thePendingScroll = pReference;
257         }
258     }

260     /**
261      * Obtain JSObject Double.
262      *
263      * @param pObject the returned number (either Double or Integer)
264      * @return the Double
265      */
266     private static Double getJSDouble(final Object pObject) {
267         if (pObject instanceof Double d) {
268             return d;
269         }
270         if (pObject instanceof Integer i) {
271             return i.doubleValue();
272         }
273         throw new IllegalArgumentException();
274     }
275 
276     /**
277      * Obtain JSObject Integer.
278      *
279      * @param pObject the returned number (either Double or Integer)
280      * @return the Integer
281      */
282     private static Integer getJSInteger(final Object pObject) {
283         if (pObject instanceof Integer i) {
284             return i;
285         }
286         if (pObject instanceof Double d) {
287             return d.intValue();
288         }
289         throw new IllegalArgumentException();
290     }

292     /**
293      * Handle click.
294      *
295      * @param pEvent the event
296      */
297     private void handleClick(final Event pEvent) {
298         if (EVENT_TYPE_CLICK.equals(pEvent.getType())) {
299             processReference(((Element) pEvent.getTarget()).getAttribute(ATTR_REF));
300         }
301     }
302 
303     /**
304      * Handle load state change.
305      *
306      * @param pNewState the new state
307      */
308     private void handleStateChange(final State pNewState) {
309         if (pNewState == Worker.State.SUCCEEDED) {
310             attachListenerToDoc();
311         }
312     }

314     @Override
315     public void printIt() {
316         /* Prepare to print the webPage */
317         final PrinterJob job = PrinterJob.createPrinterJob();
318         if (job != null
319                 && job.showPrintDialog(theControls.getStage())) {
320             /* Access printer and determine orientation */
321             final Printer myPrinter = job.getPrinter();
322 
323             /* Create page layout of correct type */
324             final PageLayout myLayout = myPrinter.createPageLayout(Paper.A4, PageOrientation.PORTRAIT, Printer.MarginType.DEFAULT);
325             job.getJobSettings().setPageLayout(myLayout);
326 
327             /* Print the WebPage */
328             theWebEngine.print(job);
329             job.endJob();
330         }
331     }

333     /**
334      * Attach hyperLinkListener.
335      */
336     private void attachListenerToDoc() {
337         /* LookUp all anchor elements in the HTML */
338         final Document myDoc = theWebEngine.getDocument();
339 
340         /* If we have CSS to add */
341         final String myCSS = getProcessedCSS();
342         if (myCSS != null) {
343             /* Create the style element */
344             final Element myElement = myDoc.createElement(ELEMENT_STYLE);
345             final Text myContent = myDoc.createTextNode(myCSS);
346             myElement.appendChild(myContent);
347 
348             /* Obtain the head and add a style element */
349             final NodeList headList = myDoc.getElementsByTagName(ELEMENT_HEAD);
350             if (headList.getLength() > 0) {
351                 headList.item(0).appendChild(myElement);
352             }
353         }
354 
355         /* Attach the listener to listen for clicks */
356         final NodeList nodeList = myDoc.getElementsByTagName(ELEMENT_ANCHOR);
357         for (int i = 0; i < nodeList.getLength(); i++) {
358             /* Only listen to anchors that contain an HRef */
359             final org.w3c.dom.Node myNode = nodeList.item(i);
360             final NamedNodeMap myAttrMap = myNode.getAttributes();
361             if (myAttrMap.getNamedItem(ATTR_REF) != null) {
362                 ((EventTarget) myNode).addEventListener(EVENT_TYPE_CLICK, theListener, false);
363             }
364         }
365 
366         /* If we have a pending scroll */
367         if (thePendingScroll != null) {
368             /* We can perform the scroll now */
369             scrollToReference(thePendingScroll);
370             thePendingScroll = null;
371         }
372     }
