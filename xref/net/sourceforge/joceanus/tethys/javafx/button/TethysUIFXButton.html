17  package net.sourceforge.joceanus.tethys.javafx.button;
18  
19  import javafx.geometry.Insets;
20  import javafx.geometry.Pos;
21  import javafx.scene.control.Button;
22  import javafx.scene.control.ContentDisplay;
23  import javafx.scene.control.Tooltip;
24  
25  import net.sourceforge.joceanus.tethys.api.base.TethysUIArrowIconId;
26  import net.sourceforge.joceanus.tethys.api.base.TethysUIIcon;
27  import net.sourceforge.joceanus.tethys.api.base.TethysUIIconId;
28  import net.sourceforge.joceanus.tethys.core.button.TethysUICoreButton;
29  import net.sourceforge.joceanus.tethys.core.factory.TethysUICoreFactory;
30  import net.sourceforge.joceanus.tethys.javafx.base.TethysUIFXArrowIcon;
31  import net.sourceforge.joceanus.tethys.javafx.base.TethysUIFXIcon;
32  import net.sourceforge.joceanus.tethys.javafx.base.TethysUIFXNode;
33  import net.sourceforge.joceanus.tethys.javafx.base.TethysUIFXUtils;
34  
35  /**
36   * Tethys FX Button.
37   */
38  public class TethysUIFXButton
39          extends TethysUICoreButton {
40      /**
41       * The "Null" Margins for a button.
42       */
43      private static final int NULL_MARGIN = 5;
44      /**
45       * The node.
46       */
47      private final TethysUIFXNode theNode;
48  
49      /**
50       * The button.
51       */
52      private final Button theButton;
53  
54      /**
55       * Constructor.
56       *
57       * @param pFactory the GUI Factory
58       */
59      TethysUIFXButton(final TethysUICoreFactory<?> pFactory) {
60          super(pFactory);
61          theButton = new Button();
62          theNode = new TethysUIFXNode(theButton);
63          theButton.setOnAction(e -> handlePressed());
64      }
65  
66      @Override
67      public TethysUIFXNode getNode() {
68          return theNode;
69      }
70  
71      @Override
72      public void setEnabled(final boolean pEnabled) {
73          theButton.setDisable(!pEnabled);
74      }
75  
76      @Override
77      public void setVisible(final boolean pVisible) {
78          theNode.setManaged(pVisible);
79          theNode.setVisible(pVisible);
80      }
81  
82      @Override
83      public void setText(final String pText) {
84          theButton.setText(pText);
85      }
86  
87      @Override
88      public void setIcon(final TethysUIArrowIconId pIcon) {
89          setIcon(TethysUIFXArrowIcon.getIconForId(pIcon));
90      }
91  
92      @Override
93      public void setIcon(final TethysUIIconId pId) {
94          setIcon(TethysUIFXUtils.getIconAtSize(pId, getIconSize()));
95      }
96  
97      @Override
98      public void setIcon(final TethysUIIcon pIcon) {
99          theButton.setGraphic(TethysUIFXIcon.getIcon(pIcon));
100     }
101 
102     @Override
103     public void setToolTip(final String pTip) {
104         final Tooltip myToolTip = pTip == null
105                 ? null
106                 : new Tooltip(pTip);
107         theButton.setTooltip(myToolTip);
108     }
109 
110     @Override
111     public void setNullMargins() {
112         theButton.setPadding(new Insets(NULL_MARGIN, NULL_MARGIN, NULL_MARGIN, NULL_MARGIN));
113     }
114 
115     @Override
116     public void setIconOnly() {
117         theButton.setContentDisplay(ContentDisplay.GRAPHIC_ONLY);
118         theButton.setAlignment(Pos.CENTER);
119         theButton.setMaxWidth(Double.MAX_VALUE);
120     }
121 
122     @Override
123     public void setTextAndIcon() {
124         theButton.setContentDisplay(ContentDisplay.RIGHT);
125         theButton.setAlignment(Pos.CENTER);
126         theButton.setMaxWidth(Double.MAX_VALUE);
127         theButton.setMaxHeight(Double.MAX_VALUE);
128     }
129 
130     @Override
131     public void setIconAndText() {
132         theButton.setContentDisplay(ContentDisplay.LEFT);
133         theButton.setAlignment(Pos.CENTER);
134         theButton.setMaxWidth(Double.MAX_VALUE);
135         theButton.setMaxHeight(Double.MAX_VALUE);
136     }
137 
138     @Override
139     public void setTextOnly() {
140         theButton.setContentDisplay(ContentDisplay.TEXT_ONLY);
141         theButton.setAlignment(Pos.CENTER);
142         theButton.setMaxWidth(Double.MAX_VALUE);
143     }
144 
145     @Override
146     public void setPreferredWidth(final Integer pWidth) {
147         theButton.setPrefWidth(pWidth);
148     }
149 
150     @Override
151     public void setPreferredHeight(final Integer pHeight) {
152         theButton.setPrefHeight(pHeight);
153     }
154 
155     @Override
156     public void setBorderPadding(final Integer pPadding) {
157         super.setBorderPadding(pPadding);
158         theNode.createWrapperPane(getBorderTitle(), getBorderPadding());
159     }
160 
161     @Override
162     public void setBorderTitle(final String pTitle) {
163         super.setBorderTitle(pTitle);
164         theNode.createWrapperPane(getBorderTitle(), getBorderPadding());
165     }
166 }
