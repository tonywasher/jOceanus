17  package net.sourceforge.joceanus.tethys.javafx.base;
18  
19  import java.util.EnumMap;
20  import java.util.Map;
21  import javafx.geometry.Point2D;
22  import javafx.scene.paint.Color;
23  import javafx.scene.shape.Polygon;
24  
25  import net.sourceforge.joceanus.tethys.api.base.TethysUIArrowIconId;
26  
27  /**
28   * Arrow Icons.
29   */
30  public enum TethysUIFXArrowIcon {
31      /**
32       * Up Arrow.
33       */
34      UP(new Point2D(1, 9), new Point2D(5, 1), new Point2D(9, 9)),
35  
36      /**
37       * Down Arrow.
38       */
39      DOWN(new Point2D(1, 1), new Point2D(5, 9), new Point2D(9, 1)),
40  
41      /**
42       * Left Arrow.
43       */
44      LEFT(new Point2D(1, 5), new Point2D(9, 1), new Point2D(9, 9)),
45  
46      /**
47       * Right Arrow.
48       */
49      RIGHT(new Point2D(1, 1), new Point2D(1, 9), new Point2D(9, 5)),
50  
51      /**
52       * DoubleUp Arrow.
53       */
54      DOUBLEUP(new Point2D(1, 5), new Point2D(5, 1), new Point2D(9, 5), new Point2D(5, 5), new Point2D(9, 9), new Point2D(1, 9), new Point2D(5, 5)),
55  
56      /**
57       * DoubleDown Arrow.
58       */
59      DOUBLEDOWN(new Point2D(1, 1), new Point2D(1, 9), new Point2D(5, 5), new Point2D(9, 5), new Point2D(5, 9), new Point2D(1, 5), new Point2D(5, 5)),
60  
61      /**
62       * DoubleLeft Arrow.
63       */
64      DOUBLELEFT(new Point2D(1, 5), new Point2D(5, 1), new Point2D(5, 5), new Point2D(9, 1), new Point2D(9, 9), new Point2D(5, 5), new Point2D(5, 9)),
65  
66      /**
67       * DoubleRight Arrow.
68       */
69      DOUBLERIGHT(new Point2D(1, 1), new Point2D(1, 9), new Point2D(5, 5), new Point2D(5, 9), new Point2D(9, 5), new Point2D(5, 1), new Point2D(5, 5));
70  
71      /**
72       * The Icon Map.
73       */
74      private static final Map<TethysUIArrowIconId, TethysUIFXArrowIcon> ICON_MAP = buildIconMap();
75  
76      /**
77       * locations of points.
78       */
79      private final Double[] thePoints;
80  
81      /**
82       * Constructor.
83       * @param pPoints the icon points
84       */
85      TethysUIFXArrowIcon(final Point2D... pPoints) {
86          /* Allocate arrays */
87          final int myNumPoints = pPoints.length;
88          thePoints = new Double[myNumPoints << 1];
89  
90          /* Loop through the points */
91          for (int i = 0, j = 0; i < myNumPoints; i++, j += 2) {
92              /* Store locations */
93              thePoints[j] = pPoints[i].getX();
94              thePoints[j + 1] = pPoints[i].getY();
95          }
96      }
97  
98      /**
99       * Obtain a polygon for the arrow.
100      * @return a new polygon
101      */
102     public Polygon getArrow() {
103         /* Allocate new polygon */
104         final Polygon myArrow = new Polygon();
105 
106         /* Initialise graphics */
107         myArrow.setStroke(Color.GRAY);
108         myArrow.setFill(Color.BLACK);
109 
110         /* Add the points */
111         myArrow.getPoints().addAll(thePoints);
112 
113         /* Return the arrow */
114         return myArrow;
115     }
116 
117     /**
118      * Build icon map.
119      * @return the map
120      */
121     private static Map<TethysUIArrowIconId, TethysUIFXArrowIcon> buildIconMap() {
122         /* Create the map and return it */
123         final Map<TethysUIArrowIconId, TethysUIFXArrowIcon> myMap = new EnumMap<>(TethysUIArrowIconId.class);
124         myMap.put(TethysUIArrowIconId.UP, UP);
125         myMap.put(TethysUIArrowIconId.DOWN, DOWN);
126         myMap.put(TethysUIArrowIconId.LEFT, LEFT);
127         myMap.put(TethysUIArrowIconId.RIGHT, RIGHT);
128         myMap.put(TethysUIArrowIconId.DOUBLEUP, DOUBLEUP);
129         myMap.put(TethysUIArrowIconId.DOUBLEDOWN, DOUBLEDOWN);
130         myMap.put(TethysUIArrowIconId.DOUBLELEFT, DOUBLELEFT);
131         myMap.put(TethysUIArrowIconId.DOUBLERIGHT, DOUBLERIGHT);
132         return myMap;
133     }
134 
135     /**
136      * Obtain icon for id.
137      * @param pId the id
138      * @return the icon
139      */
140     public static TethysUIFXIcon getIconForId(final TethysUIArrowIconId pId) {
141         final TethysUIFXArrowIcon myArrow = ICON_MAP.get(pId);
142         return myArrow == null
143                 ? null
144                 : new TethysUIFXIcon(myArrow.getArrow());
145     }
146 }
