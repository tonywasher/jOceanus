223         makeTable();
224

228         /* Create the locale list */
229         makeLocaleList();

231         /* Create the format list */
232         makeFormatList();
233 
234         /* Create the additional labels */
235         final TethysUIControlFactory myControls = theFactory.controlFactory();
236         final TethysUILabel myFormat = myControls.newLabel("Format:");
237         final TethysUILabel myLocale = myControls.newLabel("Locale:");
238         final TethysUILabel myStart = myControls.newLabel("Start:");
239         final TethysUILabel myEnd = myControls.newLabel("End:");
240         final TethysUILabel mySelRange = myControls.newLabel("SelectedRange:");

242         /* Create a Range sub-panel */
243         final TethysUIPaneFactory myPanes = theFactory.paneFactory();
244         final TethysUIGridPaneManager myRange = myPanes.newGridPane();
245         myRange.setHGap(GRID_GAP);
246         myRange.setVGap(GRID_GAP);
247         myRange.setBorderPadding(INSET_DEPTH);

263         myStyle.setHGap(GRID_GAP);
264         myStyle.setVGap(GRID_GAP);

314     /**
315      * Create the Table.
316      */
317     private void makeTable() {
318         /* Create the table */
319         theTable = theFactory.tableFactory().newTable();
320 
321         /* Create the list */
322         final List<DateItem> myList = new ArrayList<>();
323         myList.add(new DateItem(DATE_FIRST, "First Entry"));
324         myList.add(new DateItem(DATE_SECOND, "Second Entry"));
325         myList.add(new DateItem(DATE_THIRD, "Third Entry"));
326         myList.add(new DateItem(DATE_FOURTH, "Fourth Entry"));
327         myList.add(new DateItem(DATE_FIFTH, "Fifth Entry"));
328 
329         /* Declare the list to the table */
330         theTable.setItems(myList);
331 
332         /* Create the date column */
333         final TethysUITableDateColumn<String, DateItem> myDateColumn = theTable.declareDateColumn(DateItem.PROP_DATE);
334         myDateColumn.setCellValueFactory(DateItem::getDate);
335         myDateColumn.setOnCommit(DateItem::setDate);
336         myDateColumn.setColumnWidth(COL_1_WIDTH);
337         myDateColumn.setDateConfigurator((r, c) -> {
338             c.setEarliestDate(theStartDate.getSelectedDate());
339             c.setLatestDate(theEndDate.getSelectedDate());
340             c.setShowNarrowDays(theNarrowSelect.isSelected());
341         });
342 
343         /* Create the comments column */
344         final TethysUITableStringColumn<String, DateItem> myCommentsColumn = theTable.declareStringColumn(DateItem.PROP_COMMENTS);
345         myCommentsColumn.setCellValueFactory(DateItem::getComments);
346         myCommentsColumn.setOnCommit(DateItem::setComments);
347         myCommentsColumn.setColumnWidth(COL_2_WIDTH);
348     }

350     /**
351      * Create the list of available locales.
352      */
353     private void makeLocaleList() {
354         /* Create the Combo box and populate it */
355         theLocaleList = theFactory.buttonFactory().newScrollButton(ShortLocale.class);
356         final TethysUIScrollMenu<ShortLocale> myMenu = theLocaleList.getMenu();
357         for (final ShortLocale myLocale : ShortLocale.values()) {
358             /* Add the Locale to the list */
359             myMenu.addItem(myLocale);
360         }
361 
362         /* Set the default item */
363         theLocaleList.setValue(ShortLocale.UK);
364 
365         /* Action selections */
366         theLocaleList.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> {
367             /* Store the new locale */
368             final ShortLocale myLocale = e.getDetails(ShortLocale.class);
369             theLocale = myLocale.getLocale();
370             applyLocale();
371             theTable.repaintColumn(DateItem.PROP_DATE);
372             theNarrowSelect.setSelected(myLocale.showNarrowDays());
373         });
374     }

420     /**
421      * Create the range buttons.
422      */
423     private void makeRangeButtons() {
424         /* Create the buttons */
425         final TethysUIButtonFactory<?> myButtons = theFactory.buttonFactory();
426         theStartDate = myButtons.newDateButton();
427         theEndDate = myButtons.newDateButton();
428 
429         /* Create the range selection */
430         theRangeSelect = myButtons.newDateRangeSelector(true);
431         theRangeSelect.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> handleNewRange());
432 
433         /* Set the values */
434         theStartDate.setSelectedDate(DATE_START);
435         theEndDate.setSelectedDate(DATE_END);
436 
437         /* Apply the range */
438         applyRange();
439 
440         /* Add Listener to buttons */
441         theStartDate.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> applyRange());
442         theEndDate.getEventRegistrar().addEventListener(TethysUIEvent.NEWVALUE, e -> applyRange());
443 
444         /* Create the range report label */
445         theSelectedRange = theFactory.controlFactory().newLabel(theRangeSelect.getRange().toString());
446     }

497     /**
498      * Apply Locale to underlying objects.
499      */
500     private void applyLocale() {
501         /* Set locale for formatter */
502         theFormatter.setLocale(theLocale);
503         theRangeSelect.setLocale(theLocale);
504 
505         /* Set range locale */
506         theSelectedRange.setText(theDateFormatter.formatDateRange(theRangeSelect.getRange()));
507 
508         /* Note that we should redraw the table */
509         theTable.repaintColumn(DateItem.PROP_DATE);
510     }

512     /**
513      * Apply format to underlying objects.
514      */
515     private void applyFormat() {
516         /* Set format for formatter */
517         theFormatter.setFormat(theFormat);
518 
519         /* Set range format */
520         theSelectedRange.setText(theDateFormatter.formatDateRange(theRangeSelect.getRange()));
521 
522         /* Note that we should redraw the table */
523         theTable.repaintColumn(DateItem.PROP_COMMENTS);
524     }

526     /**
527      * Apply Range to underlying objects.
528      */
529     private void applyRange() {
530         /* Access the Start/End Dates */
531         final OceanusDate myStart = theStartDate.getSelectedDate();
532         final OceanusDate myEnd = theEndDate.getSelectedDate();
533 
534         /* Set the select-able range for the start/end buttons */
535         theStartDate.setLatestDate(myEnd);
536         theEndDate.setEarliestDate(myStart);
537 
538         /* set the range select range */
539         theRangeSelect.setOverallRange(new OceanusDateRange(myStart, myEnd));
540     }

542     /**
543      * Handle the new range.
544      */
545     private void handleNewRange() {
546         final OceanusDateRange myRange = theRangeSelect.getRange();
547         if (theSelectedRange != null) {
548             theSelectedRange.setText(theDateFormatter.formatDateRange(myRange));
549         }
550     }

621     /**

624     private enum ShortLocale {
625         /**
626          * China (shorten day names to one character, and shrink from the right to make sure they
627          * are different).
628          */
629         CHINA(Locale.CHINA, true),
630 
631         /**
632          * Germany.
633          */
634         GERMANY(Locale.GERMANY),
635 
636         /**
637          * France.
638          */
639         FRANCE(Locale.FRANCE),
640 
641         /**
642          * Italy.
643          */
644         ITALY(Locale.ITALY),
645 
646         /**
647          * Japan (shorten day names to one character).
648          */
649         JAPAN(Locale.JAPAN, true),
650 
651         /**
652          * Korea (shorten day names to one character).
653          */
654         KOREA(Locale.KOREA, true),
655 
656         /**
657          * US.
658          */
659         US(Locale.US),
660 
661         /**
662          * UK (shorten day names to two characters).
663          */
664         UK(Locale.UK);
665 
666         /**
667          * Locale property.
668          */
669         private final Locale theLocale;
670 
671         /**
672          * Show narrow days.
673          */
674         private final boolean doShowNarrowDays;
675 
676         /**
677          * Constructor.
678          * @param pLocale the locale
679          */
680         ShortLocale(final Locale pLocale) {
681             /* Store the Locale */
682             this(pLocale, false);
683         }
684 
685         /**
686          * Constructor.
687          * @param pLocale the locale
688          * @param pShowNarrowDays true/false
689          */
690         ShortLocale(final Locale pLocale,
691                     final boolean pShowNarrowDays) {
692             /* Store the Locale */
693             theLocale = pLocale;
694             doShowNarrowDays = pShowNarrowDays;
695         }
696 
697         /**
698          * Obtain locale value.
699          * @return the locale
700          */
701         public Locale getLocale() {
702             return theLocale;
703         }
704 
705         /**
706          * Show narrow days.
707          * @return true/false
708          */
709         public boolean showNarrowDays() {
710             return doShowNarrowDays;
711         }
