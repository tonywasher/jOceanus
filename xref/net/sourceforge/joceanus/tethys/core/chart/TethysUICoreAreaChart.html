34  /**
35   * Area Chart.
36   * <p>
37   * The EventProvider fires the following events.
38   * <ul>
39   *    <li>TethysUIEvent.PRESSED is fired when a series is selected
40   * </ul>
41   */
42  public abstract class TethysUICoreAreaChart
43          extends TethysUICoreComponent
44          implements TethysUIAreaChart {
45      /**
46       * The angle for labels.
47       */
48      protected static final int LABEL_ANGLE = -45;
49  
50      /**
51       * The formatter.
52       */
53      private final OceanusDecimalFormatter theFormatter;
54  
55      /**
56       * The id.
57       */
58      private final Integer theId;
59  
60      /**
61       * The Event Manager.
62       */
63      private final OceanusEventManager<TethysUIEvent> theEventManager;
64  
65      /**
66       * The sectionMap.
67       */
68      private final Map<String, TethysUIAreaChartSeries> theSeriesMap;
69  
70      /**
71       * Constructor.
72       * @param pFactory the Gui Factory
73       */
74      protected TethysUICoreAreaChart(final TethysUICoreFactory<?> pFactory) {
75          /* Build standard fields */
76          theId = pFactory.getNextId();
77          theEventManager = new OceanusEventManager<>();
78          theFormatter = pFactory.getDataFormatter().getDecimalFormatter();
79  
80          /* Create the section map */
81          theSeriesMap = new HashMap<>();
82      }
83  
84      @Override
85      public Integer getId() {
86          return theId;
87      }
88  
89      @Override
90      public OceanusEventRegistrar<TethysUIEvent> getEventRegistrar() {
91          return theEventManager.getEventRegistrar();
92      }
93  
94      /**
95       * Obtain the formatter.
96       * @return the formatter
97       */
98      protected OceanusDecimalFormatter getFormatter() {
99          return theFormatter;
100     }
101 
102     @Override
103     public void updateAreaChart(final TethysUIAreaChartData pData) {
104         /* clear the data */
105         resetData();
106 
107         /* Iterate through the sections */
108         final Iterator<TethysUIAreaChartSeries> myIterator = pData.seriesIterator();
109         while (myIterator.hasNext()) {
110             final TethysUIAreaChartSeries myBase = myIterator.next();
111             final String myKey = myBase.getName();
112             theSeriesMap.put(myKey, myBase);
113 
114             /* Iterate through the sections */
115             final Iterator<TethysUIAreaChartDataPoint> myPointIterator = myBase.pointIterator();
116             while (myPointIterator.hasNext()) {
117                 final TethysUIAreaChartDataPoint myPoint = myPointIterator.next();
118 
119                 /* Add the section */
120                 createPoint(myKey, myPoint);
121             }
122         }
123     }
124 
125     /**
126      * Reset chart data.
127      */
128     protected void resetData() {
129         /* Clear existing data  */
130         theSeriesMap.clear();
131     }
132 
133     /**
134      * Add chart point.
135      * @param pName the name of the series
136      * @param pPoint the point to add
137      */
138     protected abstract void createPoint(String pName,
139                                         TethysUIAreaChartDataPoint pPoint);
140 
141     /**
142      * Obtain tooltip for series.
143      * @param pName the series name
144      * @param pValue the value
145      * @return the tooltip
146      */
147     protected String getToolTip(final String pName,
148                                 final OceanusMoney pValue) {
149         return pName  + " = " + theFormatter.formatMoney(pValue);
150     }
151 
152     /**
153      * handle selection.
154      * @param pName the series name
155      */
156     protected void selectSeries(final String pName) {
157         final TethysUIAreaChartSeries mySeries = theSeriesMap.get(pName);
158         theEventManager.fireEvent(TethysUIEvent.PRESSED, mySeries);
159     }

161     /**
162      * AreaChart Data.
163      */
164     public static final class TethysUICoreAreaChartData
165             implements TethysUIAreaChartData {
166         /**
167          * The XAxis default label.
168          */
169         static final String XAXIS_LABEL = "Date";
170 
171         /**
172          * The XAxis default label.
173          */
174         static final String YAXIS_LABEL = "Value";
175 
176         /**
177          * The SeriesMap.
178          */
179         private final Map<String, TethysUIAreaChartSeries> theSeriesMap;
180 
181         /**
182          * The Chart Title.
183          */
184         private final String theTitle;
185 
186         /**
187          * The Chart XAxisLabel.
188          */
189         private String theXAxisLabel;
190 
191         /**
192          * The Chart YAxisLabel.
193          */
194         private String theYAxisLabel;
195 
196         /**
197          * Constructor.
198          * @param pTitle the title
199          */
200         TethysUICoreAreaChartData(final String pTitle) {
201             this(pTitle, XAXIS_LABEL, YAXIS_LABEL);
202         }
203 
204         /**
205          * Constructor.
206          * @param pTitle the title
207          * @param pXAxisLabel the XAxis label
208          * @param pYAxisLabel the YAxis label
209          */
210         private TethysUICoreAreaChartData(final String pTitle,
211                                           final String pXAxisLabel,
212                                           final String pYAxisLabel) {
213             /* Store parameters */
214             theTitle = pTitle;
215             theXAxisLabel = pXAxisLabel;
216             theYAxisLabel = pYAxisLabel;
217 
218             /* Create map */
219             theSeriesMap = new LinkedHashMap<>();
220         }
221 
222         @Override
223         public TethysUIAreaChartData setXAxisLabel(final String pLabel) {
224             theXAxisLabel = pLabel;
225             return this;
226         }
227 
228         @Override
229         public TethysUIAreaChartData setYAxisLabel(final String pLabel) {
230             theYAxisLabel = pLabel;
231             return this;
232         }
233 
234         @Override
235         public String getTitle() {
236             return theTitle;
237         }
238 
239         @Override
240         public String getXAxisLabel() {
241             return theXAxisLabel;
242         }
243 
244         @Override
245         public String getYAxisLabel() {
246             return theYAxisLabel;
247         }
248 
249         @Override
250         public Iterator<TethysUIAreaChartSeries> seriesIterator() {
251             return theSeriesMap.values().iterator();
252         }
253 
254         @Override
255         public TethysUIAreaChartSeries createSeries(final String pName,
256                                                     final Object pSource) {
257             final TethysUIAreaChartSeries mySeries = new TethysUICoreAreaChartSeries(pName, pSource);
258             theSeriesMap.put(pName, mySeries);
259             return mySeries;
260         }
261     }

263     /**
264      * The Series definition.
265      */
266     public static final class TethysUICoreAreaChartSeries
267             implements TethysUIAreaChartSeries {
268         /**
269          * The name of the series.
270          */
271         private final String theName;
272 
273         /**
274          * The source of the series.
275          */
276         private final Object theSource;
277 
278         /**
279          * The pointMap of the series.
280          */
281         private final Map<OceanusDate, TethysUIAreaChartDataPoint> thePointMap;
282 
283         /**
284          * Constructor.
285          * @param pName the name
286          * @param pSource the source
287          */
288         TethysUICoreAreaChartSeries(final String pName,
289                                     final Object pSource) {
290             theName = pName;
291             theSource = pSource;
292             thePointMap = new HashMap<>();
293         }
294 
295         @Override
296         public void addPoint(final OceanusDate pDate,
297                              final OceanusMoney pValue) {
298             thePointMap.put(pDate, new TethysUICoreAreaChartDataPoint(this, pDate, pValue));
299         }
300 
301         @Override
302         public String getName() {
303             return theName;
304         }
305 
306         @Override
307         public Iterator<TethysUIAreaChartDataPoint> pointIterator() {
308             return thePointMap.values().iterator();
309         }
310 
311         @Override
312         public Object getSource() {
313             return theSource;
314         }
315     }
