17  package net.sourceforge.joceanus.tethys.swing.menu;
18  
19  import net.sourceforge.joceanus.tethys.core.factory.TethysUICoreFactory;
20  import net.sourceforge.joceanus.tethys.core.menu.TethysUICoreToolBarManager;
21  import net.sourceforge.joceanus.tethys.swing.base.TethysUISwingIcon;
22  import net.sourceforge.joceanus.tethys.swing.base.TethysUISwingNode;
23  import net.sourceforge.joceanus.tethys.swing.base.TethysUISwingUtils;
24  
25  import javax.swing.JButton;
26  import javax.swing.JToolBar;
27  import javax.swing.SwingConstants;
28  
29  /**
30   * JavaFX ToolBar Manager.
31   */
32  public class TethysUISwingToolBarManager
33          extends TethysUICoreToolBarManager {
34      /**
35       * The Node.
36       */
37      private final TethysUISwingNode theNode;
38  
39      /**
40       * The ToolBar.
41       */
42      private final JToolBar theToolBar;
43  
44      /**
45       * Constructor.
46       * @param pFactory the factory
47       */
48      TethysUISwingToolBarManager(final TethysUICoreFactory<?> pFactory) {
49          super(pFactory);
50          theToolBar = new JToolBar();
51          theToolBar.setFloatable(false);
52          theNode = new TethysUISwingNode(theToolBar);
53      }
54  
55      @Override
56      public TethysUISwingNode getNode() {
57          return theNode;
58      }
59  
60      @Override
61      public void setEnabled(final boolean pEnabled) {
62          theToolBar.setEnabled(pEnabled);
63      }
64  
65      @Override
66      public void setVisible(final boolean pVisible) {
67          theToolBar.setVisible(pVisible);
68      }
69  
70      @Override
71      public void setPreferredWidth(final Integer pWidth) {
72          theNode.setPreferredWidth(pWidth);
73      }
74  
75      @Override
76      public void setPreferredHeight(final Integer pHeight) {
77          theNode.setPreferredHeight(pHeight);
78      }
79  
80      @Override
81      public TethysUISwingIconElement newIcon(final TethysUIToolBarId pId) {
82          final TethysUISwingIconElement myIcon = new TethysUISwingIconElement(this, pId);
83          theToolBar.add(myIcon.getButton());
84          return myIcon;
85      }
86  
87      @Override
88      public void newSeparator() {
89          theToolBar.addSeparator();
90      }
91  
92      /**
93       * SwingIcon.
94       */
95      public static class TethysUISwingIconElement
96              extends TethysUICoreToolElement {
97          /**
98           * The Button.
99           */
100         private final JButton theButton;
101 
102         /**
103          * Constructor.
104          * @param pManager the manager
105          * @param pId the Id
106          */
107         TethysUISwingIconElement(final TethysUISwingToolBarManager pManager,
108                                  final TethysUIToolBarId pId) {
109             super(pManager, pId);
110             theButton = new JButton();
111             theButton.addActionListener(e -> handlePressed());
112             setIcon(pId);
113         }
114 
115         /**
116          * Obtain the button.
117          * @return the button
118          */
119         protected JButton getButton() {
120             return theButton;
121         }
122 
123         @Override
124         protected TethysUISwingToolBarManager getManager() {
125             return (TethysUISwingToolBarManager) super.getManager();
126         }
127 
128         @Override
129         protected void setVisible(final boolean pVisible) {
130             theButton.setVisible(pVisible);
131         }
132 
133         @Override
134         protected void enableItem(final boolean pEnabled) {
135             theButton.setEnabled(pEnabled);
136         }
137 
138         @Override
139         public void setText(final String pText) {
140             theButton.setText(pText);
141         }
142 
143         @Override
144         public void setToolTip(final String pTip) {
145             theButton.setToolTipText(pTip);
146         }
147 
148         /**
149          * Set the icon.
150          * @param pIcon the icon
151          */
152         private void setIcon(final TethysUIToolBarId pIcon) {
153             theButton.setIcon(TethysUISwingIcon.getIcon(TethysUISwingUtils.getIconAtSize(pIcon, getManager().getIconSize())));
154         }
155 
156         @Override
157         public void setIconOnly() {
158             theButton.setHorizontalAlignment(SwingConstants.CENTER);
159         }
160 
161         @Override
162         public void setTextAndIcon() {
163             theButton.setHorizontalAlignment(SwingConstants.CENTER);
164             theButton.setVerticalAlignment(SwingConstants.CENTER);
165             theButton.setHorizontalTextPosition(SwingConstants.CENTER);
166             theButton.setVerticalTextPosition(SwingConstants.BOTTOM);
167         }
168 
169         @Override
170         public void setTextOnly() {
171             theButton.setHorizontalAlignment(SwingConstants.CENTER);
172             theButton.setVerticalAlignment(SwingConstants.CENTER);
173             theButton.setHorizontalTextPosition(SwingConstants.CENTER);
174         }
175     }
176 }
