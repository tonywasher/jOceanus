49  /**
50   * javafx Factory.
51   */
52  public class TethysUISwingFactory
53          extends TethysUICoreFactory<Color>
54          implements TethysUISwingFieldAdjustSupplier {
55      /**
56       * The field adjust.
57       */
58      private final TethysUISwingDataFieldAdjust theFieldAdjust;
59  
60      /**
61       * The pane factory.
62       */
63      private final TethysUISwingPaneFactory thePaneFactory;
64  
65      /**
66       * The button factory.
67       */
68      private final TethysUISwingButtonFactory theButtonFactory;
69  
70      /**
71       * The chart factory.
72       */
73      private final TethysUISwingChartFactory theChartFactory;
74  
75      /**
76       * The control factory.
77       */
78      private final TethysUISwingControlFactory theControlFactory;
79  
80      /**
81       * The dialog factory.
82       */
83      private final TethysUISwingDialogFactory theDialogFactory;
84  
85      /**
86       * The field factory.
87       */
88      private final TethysUISwingFieldFactory theFieldFactory;
89  
90      /**
91       * The menu factory.
92       */
93      private final TethysUISwingMenuFactory theMenuFactory;
94  
95      /**
96       * The table factory.
97       */
98      private final TethysUISwingTableFactory theTableFactory;
99  
100     /**
101      * The thread factory.
102      */
103     private final TethysUISwingThreadFactory theThreadFactory;
104 
105     /**
106      * Constructor.
107      * @param pProgram the program definitions
108      */
109     public TethysUISwingFactory(final TethysUIProgram pProgram) {
110         super(pProgram);
111         theFieldAdjust = new TethysUISwingDataFieldAdjust(this);
112         thePaneFactory = new TethysUISwingPaneFactory(this);
113         theButtonFactory = new TethysUISwingButtonFactory(this);
114         theChartFactory = new TethysUISwingChartFactory(this);
115         theControlFactory = new TethysUISwingControlFactory(this, theFieldAdjust);
116         theDialogFactory = new TethysUISwingDialogFactory(this);
117         theFieldFactory = new TethysUISwingFieldFactory(this);
118         theMenuFactory = new TethysUISwingMenuFactory(this);
119         theTableFactory = new TethysUISwingTableFactory(this);
120         theThreadFactory = new TethysUISwingThreadFactory(this, pProgram.useSliderStatus());
121         establishLogSink();
122     }
123 
124     /**
125      * Set the frame.
126      * @param pFrame the frame
127      */
128     public void setFrame(final JFrame pFrame) {
129         theDialogFactory.setFrame(pFrame);
130     }
131 
132     @Override
133     public TethysUIIcon resolveIcon(final TethysUIIconId pIconId,
134                                     final int pWidth) {
135         return pIconId == null
136                 ? null
137                 : TethysUISwingUtils.getIconAtSize(pIconId, pWidth);
138     }
139 
140     @Override
141     public TethysUIType getGUIType() {
142         return TethysUIType.SWING;
143     }
144 
145     @Override
146     public TethysUIPaneFactory paneFactory() {
147         return thePaneFactory;
148     }
149 
150     @Override
151     public TethysUIButtonFactory<Color> buttonFactory() {
152         return theButtonFactory;
153     }
154 
155     @Override
156     public TethysUIChartFactory chartFactory() {
157         return theChartFactory;
158     }
159 
160     @Override
161     public TethysUIControlFactory controlFactory() {
162         return theControlFactory;
163     }
164 
165     @Override
166     public TethysUIDialogFactory dialogFactory() {
167         return theDialogFactory;
168     }
169 
170     @Override
171     public TethysUIFieldFactory fieldFactory() {
172         return theFieldFactory;
173     }
174 
175     @Override
176     public TethysUIMenuFactory menuFactory() {
177         return theMenuFactory;
178     }
179 
180     @Override
181     public TethysUIThreadFactory threadFactory() {
182         return theThreadFactory;
183     }
184 
185     @Override
186     public TethysUITableFactory tableFactory() {
187         return theTableFactory;
188     }
189 
190     @Override
191     public TethysUISwingDataFieldAdjust getFieldAdjuster() {
192         return theFieldAdjust;
193     }
194 }
