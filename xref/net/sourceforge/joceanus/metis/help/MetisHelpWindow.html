36  /**
37   * Help Manager class, responsible for displaying the help.
38   */
39  public class MetisHelpWindow
40          implements OceanusEventProvider<TethysUIEvent> {
41      /**
42       * The Height of the window.
43       */
44      protected static final int WINDOW_WIDTH = 900;
45  
46      /**
47       * The Height of the window.
48       */
49      protected static final int WINDOW_HEIGHT = 600;
50  
51      /**
52       * The Factory.
53       */
54      private final TethysUIFactory<?> theFactory;
55  
56      /**
57       * The Event Manager.
58       */
59      private final OceanusEventManager<TethysUIEvent> theEventManager;
60  
61      /**
62       * The split tree.
63       */
64      private final TethysUISplitTreeManager<MetisHelpEntry> theSplitTree;
65  
66      /**
67       * The tree manager.
68       */
69      private final TethysUITreeManager<MetisHelpEntry> theTree;
70  
71      /**
72       * The help dialog.
73       */
74      private TethysUIChildDialog theDialog;
75  
76      /**
77       * The HTML manager.
78       */
79      private final TethysUIHTMLManager theHtml;
80  
81      /**
82       * Constructor.
83       * @param pFactory the GUI factory
84       */
85      public MetisHelpWindow(final TethysUIFactory<?> pFactory) {
86          /* Store parameters */
87          theFactory = pFactory;
88  
89          /* Create the event manager */
90          theEventManager = new OceanusEventManager<>();
91  
92          /* Create the splitTree manager and obtain details */
93          theSplitTree = pFactory.controlFactory().newSplitTreeManager();
94          theTree = theSplitTree.getTreeManager();
95          theHtml = theSplitTree.getHTMLManager();
96  
97          /* Listen to the TreeManager */
98          theSplitTree.getEventRegistrar().addEventListener(this::handleSplitTreeAction);
99      }
100 
101     @Override
102     public OceanusEventRegistrar<TethysUIEvent> getEventRegistrar() {
103         return theEventManager.getEventRegistrar();
104     }
105 
106     /**
107      * Obtain the SplitTree Manager.
108      * @return the tree manager
109      */
110     public TethysUISplitTreeManager<MetisHelpEntry> getSplitTreeManager() {
111         return theSplitTree;
112     }
113 
114     /**
115      * Obtain the Tree Manager.
116      * @return the tree manager
117      */
118     public TethysUITreeManager<MetisHelpEntry> getTreeManager() {
119         return theTree;
120     }
121 
122     /**
123      * Obtain the HTML Manager.
124      * @return the HTML manager
125      */
126     public TethysUIHTMLManager getHTMLManager() {
127         return theHtml;
128     }
129 
130     /**
131      * Fire event.
132      * @param pEventId the eventId
133      * @param pValue the relevant value
134      */
135     protected void fireEvent(final TethysUIEvent pEventId,
136                              final Object pValue) {
137         theEventManager.fireEvent(pEventId, pValue);
138     }
139 
140     /**
141      * show the dialog.
142      */
143     public void showDialog() {
144         /* If the dialog does not exist */
145         if (theDialog == null) {
146             /* Create a new dialog */
147             theDialog = theFactory.dialogFactory().newChildDialog();
148             theDialog.setTitle(MetisHelpResource.TITLE.getValue());
149 
150             /* Create the help panel */
151             final TethysUIBorderPaneManager myPanel = theFactory.paneFactory().newBorderPane();
152             myPanel.setCentre(theSplitTree);
153             myPanel.setPreferredWidth(WINDOW_WIDTH);
154             myPanel.setPreferredHeight(WINDOW_HEIGHT);
155             theDialog.setContent(myPanel);
156 
157             /* Set listener */
158             theDialog.getEventRegistrar().addEventListener(TethysUIEvent.WINDOWCLOSED, e -> {
159                 theTree.setVisible(false);
160                 fireEvent(TethysUIEvent.WINDOWCLOSED, null);
161             });
162         }
163 
164         /* If the dialog is not showing */
165         if (!theDialog.isShowing()) {
166             /* Make sure that the dialog is showing */
167             theTree.setVisible(true);
168             theDialog.showDialog();
169         }
170     }
171 
172     /**
173      * Hide the dialog.
174      */
175     public void hideDialog() {
176         /* If the dialog exists */
177         if (theDialog != null
178             && theDialog.isShowing()) {
179             /* Make sure that the dialog is hidden */
180             theDialog.hideDialog();
181         }
182     }
183 
184     /**
185      * CloseWindow on parent termination.
186      */
187     public void closeWindow() {
188         hideDialog();
189         if (theDialog != null) {
190             theDialog.closeDialog();
191         }
192     }
193 
194     /**
195      * Set the help module.
196      * @param pModule the helpModule
197      * @throws OceanusException on error
198      */
199     public void setModule(final MetisHelpModule pModule) throws OceanusException {
200         /* Access the Help entries and list */
201         final List<MetisHelpEntry> myEntries = pModule.getHelpEntries();
202 
203         /* Declare CSS */
204         final TethysUIStyleSheetId myCSS = pModule.getCSS();
205         if (myCSS != null) {
206             theHtml.setCSSContent(myCSS);
207         }
208 
209         /* Create the tree */
210         createTree(pModule.getTitle(), myEntries);
211     }
212 
213     /**
214      * Handle the split tree action event.
215      * @param pEvent the event
216      */
217     protected void handleSplitTreeAction(final OceanusEvent<TethysUIEvent> pEvent) {
218         switch (pEvent.getEventId()) {
219             case NEWVALUE:
220                 handleNewTreeItem(pEvent.getDetails(MetisHelpEntry.class));
221                 break;
222             case BUILDPAGE:
223             default:
224                 break;
225         }
226     }
227 
228     /**
229      * Handle the new tree item.
230      * @param pEntry the new entry
231      */
232     private void handleNewTreeItem(final MetisHelpEntry pEntry) {
233         if (pEntry != null) {
234             final String myHtml = pEntry.getHtml();
235             if (myHtml != null) {
236                 theHtml.setHTMLContent(myHtml, pEntry.getName());
237             }
238         }
239     }
240 
241     /**
242      * Construct a top level Tree Node from a set of help entries.
243      * @param pTitle the title for the tree
244      * @param pEntries the help entries
245      * @return the Tree node
246      */
247     private TethysUITreeItem<MetisHelpEntry> createTree(final String pTitle,
248                                                         final List<MetisHelpEntry> pEntries) {
249         /* Obtain the root node */
250         final TethysUITreeItem<MetisHelpEntry> myRoot = theTree.getRoot();
251         theTree.setRootName(pTitle);
252         theTree.setRootVisible();
253 
254         /* Clear existing children */
255         myRoot.removeChildren();
256 
257         /* Add the entries into the node */
258         addHelpEntries(myRoot, pEntries);
259 
260         /* Return the root */
261         return myRoot;
262     }
263 
264     /**
265      * Add array of Help entries.
266      * @param pParent the parent to add to
267      * @param pEntries the entries to add
268      */
269     private void addHelpEntries(final TethysUITreeItem<MetisHelpEntry> pParent,
270                                 final List<MetisHelpEntry> pEntries) {
271         /* Loop through the entries */
272         for (MetisHelpEntry myEntry : pEntries) {
273             /* Create the entry */
274             final TethysUITreeItem<MetisHelpEntry> myItem = theTree.addChildItem(pParent, myEntry.getName(), myEntry);
275 
276             /* If we have children */
277             final List<MetisHelpEntry> myChildren = myEntry.getChildren();
278             if (myChildren != null) {
279                 /* Add the children into the tree */
280                 addHelpEntries(myItem, myChildren);
281             }
282         }
283     }
