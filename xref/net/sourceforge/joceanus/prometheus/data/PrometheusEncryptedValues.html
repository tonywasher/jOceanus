17  package net.sourceforge.joceanus.prometheus.data;
18  
19  import net.sourceforge.joceanus.metis.data.MetisDataItem.MetisDataFieldId;
20  import net.sourceforge.joceanus.metis.field.MetisFieldItem.MetisFieldDef;
21  import net.sourceforge.joceanus.metis.field.MetisFieldItem.MetisFieldSetDef;
22  import net.sourceforge.joceanus.metis.field.MetisFieldItem.MetisFieldVersionedDef;
23  import net.sourceforge.joceanus.metis.field.MetisFieldVersionValues;
24  import net.sourceforge.joceanus.oceanus.base.OceanusException;
25  
26  import java.util.Iterator;
27  
28  /**
29   * Prometheus Set of versioned Values.
30   */
31  public class PrometheusEncryptedValues
32          extends MetisFieldVersionValues {
33      /**
34       * Constructor.
35       * @param pItem the associated item
36       */
37      public PrometheusEncryptedValues(final PrometheusEncryptedDataItem pItem) {
38          super(pItem);
39      }
40  
41      @Override
42      public PrometheusEncryptedValues cloneIt() {
43          /* Create the valueSet and initialise to existing values */
44          final PrometheusEncryptedValues mySet = new PrometheusEncryptedValues(getItem());
45          mySet.copyFrom(this);
46          return mySet;
47      }
48  
49      @Override
50      protected PrometheusEncryptedDataItem getItem() {
51          return (PrometheusEncryptedDataItem) super.getItem();
52      }
53  
54      @Override
55      public Object getValue(final MetisFieldDef pField) {
56          /* Access the underlying object */
57          final Object myValue = super.getValue(pField);
58  
59          /* If this is an encrypted value */
60          return myValue instanceof PrometheusEncryptedPair myPair
61                  ? myPair.getValue()
62                  : myValue;
63      }
64  
65      /**
66       * Obtain the encrypted bytes.
67       * @param pFieldId the fieldId
68       * @return the encrypted bytes
69       */
70      public byte[] getEncryptedBytes(final MetisDataFieldId pFieldId) {
71          /* Access the underlying object */
72          final MetisFieldDef myField = getItem().getDataFieldSet().getField(pFieldId);
73          return getEncryptedBytes(myField);
74      }
75  
76      /**
77       * Obtain the encrypted bytes.
78       * @param pField the field
79       * @return the encrypted bytes
80       */
81      public byte[] getEncryptedBytes(final MetisFieldDef pField) {
82          /* Access the underlying object */
83          final Object myValue = super.getValue(pField);
84  
85          /* If this is an encrypted value */
86          return myValue instanceof PrometheusEncryptedPair myPair
87                  ? myPair.getBytes()
88                  : null;
89      }
90  
91      /**
92       * Obtain the encrypted pair.
93       * @param pFieldId the fieldId
94       * @return the encrypted pair
95       */
96      public PrometheusEncryptedPair getEncryptedPair(final MetisDataFieldId pFieldId) {
97          /* Access the underlying object */
98          final MetisFieldDef myField = getItem().getDataFieldSet().getField(pFieldId);
99          return getEncryptedPair(myField);
100     }
101 
102     /**
103      * Obtain the encrypted pair.
104      * @param pField the field
105      * @return the encrypted pair
106      */
107     public PrometheusEncryptedPair getEncryptedPair(final MetisFieldDef pField) {
108         /* Access the underlying object */
109         final Object myValue = super.getValue(pField);
110 
111         /* If this is an encrypted value */
112         return myValue instanceof PrometheusEncryptedPair myPair
113                 ? myPair
114                 : null;
115     }
116 
117     @Override
118     public void setValue(final MetisFieldDef pField,
119                          final Object pValue) throws OceanusException {
120         /* Reject if not in valueSet */
121         if (!(pField instanceof MetisFieldVersionedDef)) {
122             throw new IllegalArgumentException(ERROR_NOTVERSIONED);
123         }
124 
125         /* check the value type */
126         checkValueType(pField, pValue);
127 
128         /* If this is an encrypted field */
129         Object myValue = pValue;
130         if (pField instanceof PrometheusEncryptedField
131                 //&& theEncryptor != null
132                 && myValue != null) {
133             /* Encrypt the value */
134             final PrometheusEncryptedDataItem myItem = getItem();
135             final PrometheusEncryptor myEncryptor = myItem.getEncryptor();
136             myValue = myEncryptor.encryptValue(pValue, pField);
137         }
138 
139         /* Store the value */
140         setUncheckedValue(pField, myValue);
141     }
142 
143     /**
144      * Update security for the values.
145      * @throws OceanusException on error
146      */
147     public void updateSecurity() throws OceanusException {
148         /* Loop through the fields */
149         final PrometheusEncryptedDataItem myItem = getItem();
150         final PrometheusEncryptor myEncryptor = myItem.getEncryptor();
151         final MetisFieldSetDef myFieldSet = getFields();
152         final Iterator<MetisFieldDef> myIterator = myFieldSet.fieldIterator();
153         while (myIterator.hasNext()) {
154             final MetisFieldDef myField = myIterator.next();
155 
156             /* Ignore non-encrypted fields */
157             if (!(myField instanceof PrometheusEncryptedField)) {
158                 continue;
159             }
160 
161             /* Access the value */
162             final Object myValue = getValue(myField);
163 
164             /* Encrypt the value */
165             setUncheckedValue(myField, myEncryptor.encryptValue(myValue, myField));
166         }
167     }
168 
169     /**
170      * Adopt security for the values.
171      * @param pBaseValues the base values
172      * @throws OceanusException on error
173      */
174     public void adoptSecurity(final PrometheusEncryptedValues pBaseValues) throws OceanusException {
175         /* Loop through the fields */
176         final PrometheusEncryptedDataItem myItem = getItem();
177         final PrometheusEncryptor myEncryptor = myItem.getEncryptor();
178         final MetisFieldSetDef myFieldSet = getFields();
179         final Iterator<MetisFieldDef> myIterator = myFieldSet.fieldIterator();
180         while (myIterator.hasNext()) {
181             final MetisFieldDef myField = myIterator.next();
182 
183             /* Ignore non-encrypted fields */
184             if (!(myField instanceof PrometheusEncryptedField)) {
185                 continue;
186             }
187             final PrometheusEncryptedPair myValue = getEncryptedPair(myField);
188             if (myValue == null) {
189                 continue;
190             }
191 
192             /* Access the base object */
193             Object myBaseObj = pBaseValues == null
194                     ? null
195                     : pBaseValues.getValue(myField);
196             if (!(myBaseObj instanceof PrometheusEncryptedPair)) {
197                 myBaseObj = null;
198             }
199 
200             /* Adopt encryption */
201             myEncryptor.adoptEncryption(myValue, (PrometheusEncryptedPair) myBaseObj);
202         }
203     }
204 }
