195     /**
196      * Construct a new item based on an old item.
197      * @param pList the list that this item is associated with
198      * @param pBase the old item
199      */
200     protected PrometheusDataItem(final PrometheusDataList<?> pList,
201                                  final PrometheusDataItem pBase) {
202         /* Initialise using standard constructor */
203         this(pList, pBase.getIndexedId());
204 
205         /* Initialise the valueSet */
206         getValues().copyFrom(pBase.getValues());
207 
208         /* Access the varying styles and the source state */
209         final PrometheusListStyle myStyle = pList.getStyle();
210         final PrometheusListStyle myBaseStyle = pBase.getList().getStyle();
211         final MetisDataState myState = pBase.getState();
212 
213         /* Switch on the styles */
214         switch (myStyle) {
215             /* We are building an update list (from Core) */
216             case UPDATE:
217                 switch (myState) {
218                     /* NEW/DELNEW need to be at version 1 */
219                     case DELNEW:
220                     case NEW:
221                         getValues().setVersion(1);
222                         break;
223 
224                     case DELETED:
225                         getValuesHistory().pushHistory(1);
226                         break;
227 
228                     /*
229                      * Changed items need to have new values at version 1 and originals at version 0
230                      */
231                     case CHANGED:
232                         setHistory(pBase);
233                         break;
234 
235                     /* No change for other states */
236                     default:
237                         break;
238                 }
239 
240                 /* Record the base item */
241                 theBase = pBase;
242                 break;
243 
244             /* We are building an edit item (from Core/Edit) */
245             case EDIT:
246                 /* Switch on the base style */
247                 switch (myBaseStyle) {
248                     /* New item from core we need to link back and copy flags */
249                     case CORE:
250                         theBase = pBase;
251                         copyFlags(pBase);
252                         break;
253                     /* Duplication in edit */
254                     case EDIT:
255                         /* set as a new item */
256                         getValues().setVersion(pList.getVersion() + 1);
257 
258                         /* Reset the Id */
259                         setIndexedId(0);
260                         pList.setNewId(this);
261                         break;
262                     default:
263                         break;
264                 }
265                 break;
266 
267             /* We are building a CORE item */
268             case CORE:
269                 /* set as a new item */
270                 getValues().setVersion(pList.getVersion() + 1);
271 
272                 /* If we are adding from Edit */
273                 if (myBaseStyle == PrometheusListStyle.EDIT) {
274                     /* Reset the Id */
275                     setIndexedId(0);
276                     pList.setNewId(this);
277                 }
278                 break;
279 
280             /* Creation of copy element not allowed */
281             case COPY:
282                 throw new IllegalArgumentException("Illegal creation of COPY element");
283 
284                 /* Nothing special for other styles */
285             case CLONE:
286             case DIFFER:
287             default:
288                 break;
289         }
290     }

496     @Override
497     public void rewindToVersion(final int pVersion) {
498         /* If the item was newly created */
499         if (getOriginalValues().getVersion() > pVersion) {
500             /* Remove from list */
501             unLink();
502             deRegister();
503 
504             /* Return */
505             return;
506         }
507 
508         /* Loop while version is too high */
509         while (getValues().getVersion() > pVersion) {
510             /* Pop history */
511             getValuesHistory().popTheHistory();
512         }
513     }

570     /**
571      * Resolve a data link into a list.
572      * @param pFieldId the fieldId to resolve
573      * @param pList the list to resolve against
574      * @throws OceanusException on error
575      */
576     protected void resolveDataLink(final MetisDataFieldId pFieldId,
577                                    final PrometheusDataList<?> pList) throws OceanusException {
578         /* Access the values */
579         final MetisFieldVersionValues myValues = getValues();
580 
581         /* Access value for field */
582         Object myValue = myValues.getValue(pFieldId);
583 
584         /* Convert dataItem reference to Id */
585         if (myValue instanceof PrometheusDataItem myItem) {
586             myValue = myItem.getIndexedId();
587         }
588 
589         /* Lookup Id reference */
590         if (myValue instanceof Integer i) {
591             final PrometheusDataItem myItem = pList.findItemById(i);
592             if (myItem == null) {
593                 addError(ERROR_UNKNOWN, pFieldId);
594                 throw new PrometheusDataException(this, ERROR_RESOLUTION);
595             }
596             myValues.setValue(pFieldId, myItem);
597 
598             /* Lookup Name reference */
599         } else if (myValue instanceof String s) {
600             final PrometheusDataItem myItem = pList.findItemByName(s);
601             if (myItem == null) {
602                 addError(ERROR_UNKNOWN, pFieldId);
603                 throw new PrometheusDataException(this, ERROR_RESOLUTION);
604             }
605             myValues.setValue(pFieldId, myItem);
606         }
607     }

618     @Override
619     public boolean equals(final Object pThat) {
620         /* Handle the trivial cases */
621         if (this == pThat) {
622             return true;
623         }
624         if (pThat == null) {
625             return false;
626         }
627 
628         /* Make sure that the object is the same class */
629         if (pThat.getClass() != getClass()) {
630             return false;
631         }
632 
633         /* Access the object as a DataItem */
634         final PrometheusDataItem myItem = (PrometheusDataItem) pThat;
635 
636         /* Check the id */
637         if (compareId(myItem) != 0) {
638             return false;
639         }
640 
641         /* Loop through the fields */
642         final Iterator<MetisFieldDef> myIterator = getDataFieldSet().fieldIterator();
643         while (myIterator.hasNext()) {
644             /* Access Field */
645             final MetisFieldDef myField = myIterator.next();
646 
647             /* Skip if not used in equality */
648             if (!(myField instanceof MetisFieldVersionedDef myVersioned)
649                 || !myVersioned.isEquality()) {
650                 continue;
651             }
652 
653             /* Access the values */
654             final Object myValue = myField.getFieldValue(this);
655             final Object myNew = myField.getFieldValue(myItem);
656 
657             /* Check the field */
658             if (!MetisDataDifference.isEqual(myValue, myNew)) {
659                 return false;
660             }
661         }
662 
663         /* Return identical */
664         return true;
665     }

673     @Override
674     public int compareTo(final Object pThat) {
675         /* Handle the trivial cases */
676         if (this.equals(pThat)) {
677             return 0;
678         }
679         if (pThat == null) {
680             return -1;
681         }
682 
683         /* Non-DataItems are last */
684         if (!(pThat instanceof PrometheusDataItem)) {
685             return -1;
686         }
687 
688         /* Check data type */
689         final PrometheusDataItem myThat = (PrometheusDataItem) pThat;
690         int iDiff = getItemType().getItemKey() - myThat.getItemType().getItemKey();
691         if (iDiff != 0) {
692             return iDiff;
693         }
694 
695         /* Check values and finally id */
696         iDiff = compareValues(myThat);
697         return iDiff != 0 ? iDiff : compareId(myThat);
698     }

754     /**
755      * Does the string contain only valid characters (no control chars)?
756      * @param pString the string
757      * @param pDisallowed the set of additional disallowed characters
758      * @return true/false
759      */
760     public static boolean validString(final String pString,
761                                       final String pDisallowed) {
762         /* Loop through the string */
763         for (int i = 0; i < pString.length(); i++) {
764             final int myChar = pString.codePointAt(i);
765             /* Check for ISO control */
766             if (Character.isISOControl(myChar)) {
767                 return false;
768             }
769 
770             /* Check for disallowed value */
771             if (pDisallowed != null
772                     && pDisallowed.indexOf(myChar) != -1) {
773                 return false;
774             }
775         }
776         return true;
777     }

795     /**
796      * Obtain the field state.
797      * @param pField the field
798      * @return the state
799      */
800     public MetisFieldState getFieldState(final MetisDataFieldId pField) {
801         /* Determine DELETED state */
802         if (isDeleted()) {
803             return MetisFieldState.DELETED;
804 
805             /* Determine Error state */
806         } else if (hasErrors() && hasErrors(pField)) {
807             return MetisFieldState.ERROR;
808 
809             /* Determine Changed state */
810         } else if (fieldChanged(pField).isDifferent()) {
811             return MetisFieldState.CHANGED;
812 
813             /* Determine standard states */
814         } else {
815             switch (getState()) {
816                 case NEW:
817                     return MetisFieldState.NEW;
818                 case RECOVERED:
819                     return MetisFieldState.RESTORED;
820                 default:
821                     return MetisFieldState.NORMAL;
822             }
823         }
824     }

826     /**
827      * Obtain the item State.
828      * @return the state
829      */
830     public MetisFieldState getItemState() {
831         /* Determine DELETED state */
