17  package net.sourceforge.joceanus.prometheus.service.sheet.odf;
18  
19  import java.util.ListIterator;
20  
21  import net.sourceforge.joceanus.prometheus.service.sheet.PrometheusSheetCellPosition;
22  import net.sourceforge.joceanus.prometheus.service.sheet.PrometheusSheetCellRange;
23  import net.sourceforge.joceanus.prometheus.service.sheet.PrometheusSheetRow;
24  import net.sourceforge.joceanus.prometheus.service.sheet.PrometheusSheetSheet;
25  import net.sourceforge.joceanus.oceanus.base.OceanusException;
26  
27  /**
28   * Sheet class.
29   */
30  public class PrometheusOdfSheet
31          extends PrometheusSheetSheet {
32      /**
33       * The underlying sheet.
34       */
35      private final PrometheusOdfSheetCore theSheet;
36  
37      /**
38       * The index.
39       */
40      private final int theIndex;
41  
42      /**
43       * Constructor.
44       * @param pBook the workBook
45       * @param pSheet the underlying sheet
46       * @param pIndex the index
47       * @param pReadOnly is the sheet readOnly?
48       */
49      PrometheusOdfSheet(final PrometheusOdfWorkBook pBook,
50                         final PrometheusOdfSheetCore pSheet,
51                         final int pIndex,
52                         final boolean pReadOnly) {
53          super(pBook, pSheet.getName(), pReadOnly);
54          theSheet = pSheet;
55          theIndex = pIndex;
56      }
57  
58      @Override
59      public int getRowCount() {
60          return theSheet.getRowCount();
61      }
62  
63      @Override
64      public PrometheusOdfRow getReadOnlyRowByIndex(final int pRowIndex) {
65          return theSheet.getReadOnlyRowByIndex(this, pRowIndex);
66      }
67  
68      @Override
69      protected ListIterator<PrometheusSheetRow> iteratorForRange(final int pFirstIndex,
70                                                                  final int pLastIndex) {
71          return theSheet.iteratorForRange(this, pFirstIndex, pLastIndex);
72      }
73  
74      @Override
75      public PrometheusOdfRow getMutableRowByIndex(final int pRowIndex) {
76          return isReadOnly()
77                 ? null
78                 : theSheet.getMutableRowByIndex(this, pRowIndex);
79      }
80  
81      @Override
82      public PrometheusOdfColumn getReadOnlyColumnByIndex(final int pColIndex) {
83          return theSheet.getReadOnlyColumnByIndex(this, pColIndex);
84      }
85  
86      @Override
87      public PrometheusOdfColumn getMutableColumnByIndex(final int pColIndex) {
88          return isReadOnly()
89                 ? null
90                 : theSheet.getMutableColumnByIndex(this, pColIndex);
91      }
92  
93      @Override
94      public int getSheetIndex() {
95          return theIndex;
96      }
97  
98      @Override
99      public boolean isHidden() {
100         return theSheet.isHidden();
101     }
102 
103     @Override
104     public void setHidden(final boolean isHidden) {
105         if (!isReadOnly()) {
106             theSheet.setHidden(isHidden);
107         }
108     }
109 
110     @Override
111     public void declareRange(final String pName,
112                              final PrometheusSheetCellPosition pFirstCell,
113                              final PrometheusSheetCellPosition pLastCell) throws OceanusException {
114         if (!isReadOnly()) {
115             /* Build the range */
116             final PrometheusSheetCellRange myRange = new PrometheusSheetCellRange(getName(), pFirstCell, pLastCell);
117 
118             /* Declare it */
119             theSheet.declareRange(pName, myRange);
120         }
121     }
122 
123     @Override
124     public void applyDataValidation(final PrometheusSheetCellPosition pFirstCell,
125                                     final PrometheusSheetCellPosition pLastCell,
126                                     final String pName) throws OceanusException {
127         if (!isReadOnly()) {
128             /* Declare the validation */
129             theSheet.applyDataValidation(pFirstCell, pLastCell, pName);
130         }
131     }
132 
133     @Override
134     public void applyDataFilter(final PrometheusSheetCellPosition pBaseCell,
135                                 final int pNumRows) throws OceanusException {
136         if (!isReadOnly()) {
137             /* Build the range */
138             final PrometheusSheetCellPosition myEnd = new PrometheusSheetCellPosition(pBaseCell.getColumnIndex(), pNumRows - 1);
139             final PrometheusSheetCellRange myRange = new PrometheusSheetCellRange(getName(), pBaseCell, myEnd);
140 
141             /* Declare it */
142             theSheet.applyDataFilter(myRange);
143         }
144     }
145 
146     @Override
147     public void createFreezePane(final PrometheusSheetCellPosition pFreezeCell) {
148         /* NoOp */
149     }
150 }
