125     /**
126      * Build sheet map.
127      */
128     private void buildSheetMap() {
129         /* Access the list of tables */
130         final List<Element> mySheets = theParser.getAllNamedChildren(theSpreadSheet, PrometheusOdfTableItem.TABLE);
131 
132         /* Loop through the list */
133         for (Element mySheet : mySheets) {
134             /* Store the sheet in the map */
135             theSheets.put(theParser.getAttribute(mySheet, PrometheusOdfTableItem.NAME), mySheet);
136         }
137     }
138 
139     /**
140      * Obtain existing sheet.
141      * @param pName the name of the sheet
142      * @return the sheet
143      * @throws OceanusException on error
144      */
145     PrometheusOdfSheet getSheet(final String pName) throws OceanusException {
146         /* Obtain the existing sheet */
147         Object myCore = theSheets.get(pName);
148         if (myCore instanceof Element myElement) {
149             myCore = new PrometheusOdfSheetCore(this, theNumTables++, myElement);
150         }
151         return myCore == null
152                ? null
153                : ((PrometheusOdfSheetCore) myCore).getReadOnlySheet();
154     }

156     /**
157      * Create a new sheet.
158      * @param pName the name of the sheet.
159      * @param pNumRows the initial number of rows
160      * @param pNumCols the initial number of columns
161      * @return the new sheet
162      * @throws OceanusException on error
163      */
164     PrometheusOdfSheet newSheet(final String pName,
165                                 final int pNumRows,
166                                 final int pNumCols) throws OceanusException {
167         /* Check for existing sheet */
168         if (theSheets.get(pName) != null) {
169             throw new PrometheusSheetException("Sheet "
170                     + pName
171                     + "already exists in workbook");
172         }
173 
174         /* Create the new Sheet */
175         final Element myTable = theParser.newElement(PrometheusOdfTableItem.TABLE);
176         theParser.setAttribute(myTable, PrometheusOdfTableItem.NAME, pName);
177 
178         /* Access the expressions */
179         final Element myExpressions = theParser.getFirstNamedChild(theSpreadSheet, PrometheusOdfTableItem.EXPRESSIONS);
180         PrometheusOdfParser.addAsPriorSibling(myTable, myExpressions);
181 
182         /* Create the Core sheet and add to the map */
183         final PrometheusOdfSheetCore myCore = new PrometheusOdfSheetCore(this, theNumTables++, pNumRows, pNumCols, myTable);
184         theSheets.put(pName, myCore);
185 
186         /* Return the sheet */
187         return myCore.getMutableSheet();
188     }
189 
190     /**
191      * Build range map.
192      */
193     private void buildRangeMap() {
194         /* Access the list of ranges */
195         final Element myExpressions = theParser.getFirstNamedChild(theSpreadSheet, PrometheusOdfTableItem.EXPRESSIONS);
196         final List<Element> myRanges = theParser.getAllNamedChildren(myExpressions, PrometheusOdfTableItem.RANGE);
197 
198         /* Loop through the named ranges */
199         for (Element myRange : myRanges) {
200             /* Add range to map */
201             theRanges.put(theParser.getAttribute(myRange, PrometheusOdfTableItem.NAME), myRange);
202         }
203     }
204 
205     /**
206      * Obtain the range view.
207      * @param pName the name of the view
208      * @return the range
209      * @throws OceanusException on error
210      */
211     PrometheusSheetView getRangeView(final String pName) throws OceanusException {
212         /* Locate the named range in the map */
213         final Element myNamedRange = theRanges.get(pName);
214         if (myNamedRange == null) {
215             return null;
216         }
217 
218         /* Obtain the address */
219         final String myRange = theParser.getAttribute(myNamedRange, PrometheusOdfTableItem.CELLRANGEADDRESS);
220         final PrometheusSheetCellRange myCellRange = new PrometheusSheetCellRange(myRange);
221         final PrometheusSheetCellPosition myFirstCell = myCellRange.getFirstCell().getPosition();
222         final PrometheusSheetCellPosition myLastCell = myCellRange.getLastCell().getPosition();
223 
224         /* Obtain the sheet and reject if missing */
225         final PrometheusSheetSheet mySheet = getSheet(myCellRange.getFirstCell().getSheetName());
226         if (mySheet == null) {
227             throw new PrometheusSheetException("Sheet for "
228                     + pName
229                     + " not found in workbook");
230         }
231 
232         /* Return the view */
233         return new PrometheusSheetView(mySheet, myFirstCell, myLastCell);
234     }

236     /**
237      * Declare the named range.
238      * @param pName the name of the range
239      * @param pRange the range to declare
240      * @throws OceanusException on error
241      */
242     void declareRange(final String pName,
243                       final PrometheusSheetCellRange pRange) throws OceanusException {
244         /* Check for existing range */
245         if (theRanges.get(pName) != null) {
246             throw new PrometheusSheetException("Name "
247                     + pName
248                     + "already exists in workbook");
249         }
250 
251         /* Access the expressions */
252         final Element myExpressions = theParser.getFirstNamedChild(theSpreadSheet, PrometheusOdfTableItem.EXPRESSIONS);
253 
254         /* Protect against exceptions */
255         try {
256             /* Add the new range */
257             final Element myRange = theParser.newElement(PrometheusOdfTableItem.RANGE);
258             theParser.setAttribute(myRange, PrometheusOdfTableItem.NAME, pName);
259             theParser.setAttribute(myRange, PrometheusOdfTableItem.CELLRANGEADDRESS, pRange.toString());
260             theParser.setAttribute(myRange, PrometheusOdfTableItem.BASECELLADDRESS, pRange.getFirstCell().toString());
261             myExpressions.appendChild(myRange);
262             theRanges.put(pName, myRange);
263         } catch (Exception e) {
264             throw new PrometheusSheetException("Failed to declare range", e);
265         }
266     }

268     /**
269      * Apply Data Validation.
270      * @param pSheet the workSheet containing the cells
271      * @param pFirstCell the the first cell in the range
272      * @param pLastCell the last cell in the range
273      * @param pValidRange the name of the validation range
274      */
275     void applyDataValidation(final PrometheusOdfSheetCore pSheet,
276                              final PrometheusSheetCellPosition pFirstCell,
277                              final PrometheusSheetCellPosition pLastCell,
278                              final String pValidRange) {
279         /* Access constraint */
280         Element myConstraint = theConstraints.get(pValidRange);
281         if (myConstraint == null) {
282             /* Create the constraint */
283             myConstraint = createDataConstraint(pValidRange);
284 
285             /* Store the constraint */
286             theConstraints.put(pValidRange, myConstraint);
287         }
288 
289         /* Apply the constraint */
290         pSheet.applyValidation(theParser.getAttribute(myConstraint, PrometheusOdfTableItem.NAME), pFirstCell, pLastCell);
291     }

293     /**
294      * Apply Data Validation.
295      * @param pSheet the workSheet containing the cells
296      * @param pFirstCell the the first cell in the range
297      * @param pLastCell the last cell in the range
298      * @param pValueList the value list
299      */
300     void applyDataValidation(final PrometheusOdfSheetCore pSheet,
301                              final PrometheusSheetCellPosition pFirstCell,
302                              final PrometheusSheetCellPosition pLastCell,
303                              final String[] pValueList) {
304         /* Access constraint */
305         Element myConstraint = theConstraints.get(pValueList);
306         if (myConstraint == null) {
307             /* Build the constraint list */
308             final StringBuilder myBuilder = new StringBuilder();
309 
310             /* Loop through the values */
311             for (String myValue : pValueList) {
312                 /* If this is not the first element */
313                 if (myBuilder.length() > 0) {
314                     /* Add a comma */
315                     myBuilder.append(',');
316                 }
317                 /* Add the escaped value */
318                 myBuilder.append(PrometheusSheetCellAddress.escapeApostrophes(myValue));
319             }
320 
321             /* Create the constraint */
322             myConstraint = createDataConstraint(myBuilder.toString());
323 
324             /* Store the constraint */
325             theConstraints.put(pValueList, myConstraint);
326         }
327 
328         /* Apply the constraint */
329         pSheet.applyValidation(theParser.getAttribute(myConstraint, PrometheusOdfTableItem.NAME), pFirstCell, pLastCell);
330     }

332     /**
333      * Create Data Constraint.
334      * @param pConstraint the constraint list
335      * @return the constraint
336      */
337     private Element createDataConstraint(final String pConstraint) {
338         /* Build the name */
339         final String myName = "val"
340                 + ++theNumConstraints;
341 
342         /* Access the constraints */
343         Element myConstraints = theParser.getFirstNamedChild(theSpreadSheet, PrometheusOdfTableItem.VALIDATIONS);
344         if (myConstraints == null) {
345             myConstraints = theParser.newElement(PrometheusOdfTableItem.VALIDATIONS);
346             theSpreadSheet.insertBefore(myConstraints, theSpreadSheet.getFirstChild());
347         }
348 
349         /* Create the new constraint */
350         final Element myConstraint = theParser.newElement(PrometheusOdfTableItem.VALIDATION);
351         myConstraints.appendChild(myConstraint);
352         theParser.setAttribute(myConstraint, PrometheusOdfTableItem.NAME, myName);
353 
354         /* Create the rule */
355         final String myRule = "of:cell-content-is-in-list("
356                 + pConstraint
357                 + ")";
358         theParser.setAttribute(myConstraint, PrometheusOdfTableItem.CONDITION, myRule);
359         theParser.setAttribute(myConstraint, PrometheusOdfTableItem.ALLOWEMPTYCELL, Boolean.TRUE);
360         final Element myError = theParser.newElement(PrometheusOdfTableItem.ERRORMSG);
361         myConstraint.appendChild(myError);
362         theParser.setAttribute(myError, PrometheusOdfTableItem.DISPLAY, Boolean.TRUE);
363 
364         /* Store the constraint */
365         return myConstraint;
366     }

368     /**
369      * Apply Data Filter.
370      * @param pRange the range
371      */
372     void applyDataFilter(final PrometheusSheetCellRange pRange) {
373         /* Access the dbRanges */
374         Element myDBRanges = theParser.getFirstNamedChild(theSpreadSheet, PrometheusOdfTableItem.DATARANGES);
375         if (myDBRanges == null) {
376             myDBRanges = theParser.newElement(PrometheusOdfTableItem.DATARANGES);
377             theSpreadSheet.appendChild(myDBRanges);
378         }
379 
380         /* Build the name */
381         final String myName = "dbRange"
382                 + ++theNumDBRanges;
383 
384         /* Create the new filter */
385         final Element myFilter = theParser.newElement(PrometheusOdfTableItem.DATARANGE);
386         theParser.setAttribute(myFilter, PrometheusOdfTableItem.NAME, myName);
387         theParser.setAttribute(myFilter, PrometheusOdfTableItem.TARGETRANGE, pRange.toString());
388         theParser.setAttribute(myFilter, PrometheusOdfTableItem.DISPLAYFILTER, Boolean.TRUE);
389         theParser.setAttribute(myFilter, PrometheusOdfTableItem.ORIENTATION, PrometheusOdfValue.COLUMN);
390     }
