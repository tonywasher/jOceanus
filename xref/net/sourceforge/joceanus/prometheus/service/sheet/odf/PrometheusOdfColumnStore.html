18  
19  import java.util.Arrays;
20  import java.util.Objects;
21  
22  import org.w3c.dom.Element;
23  
24  import net.sourceforge.joceanus.prometheus.service.sheet.PrometheusSheetCellStyleType;

26  /**
27   * Hold columns as a list of values.
28   */
29  class PrometheusOdfColumnStore {
30      /**
31       * Column Expansion.
32       */
33      private static final int COL_EXPAND = 20;
34  
35      /**
36       * Underlying sheet.
37       */
38      private final PrometheusOdfSheetCore theSheet;
39  
40      /**
41       * The Parser.
42       */
43      private final PrometheusOdfParser theParser;
44  
45      /**
46       * Array of hiddens.
47       */
48      private Boolean[] theHiddens;
49  
50      /**
51       * Array of styles.
52       */
53      private PrometheusSheetCellStyleType[] theStyles;
54  
55      /**
56       * Number of Columns.
57       */
58      private int theNumCols;
59  
60      /**
61       * ReadOnly Constructor.
62       * @param pSheet the owning sheet.
63       */
64      PrometheusOdfColumnStore(final PrometheusOdfSheetCore pSheet) {
65          /* Store details */
66          theSheet = pSheet;
67          theParser = theSheet.getParser();
68          theHiddens = new Boolean[COL_EXPAND];
69          theStyles = new PrometheusSheetCellStyleType[COL_EXPAND];
70      }
71  
72      /**
73       * Mutable Constructor.
74       * @param pSheet the owning sheet.
75       * @param pNumCols the initial # of columns
76       */
77      PrometheusOdfColumnStore(final PrometheusOdfSheetCore pSheet,
78                               final int pNumCols) {
79          /* Store details */
80          theSheet = pSheet;
81          theParser = theSheet.getParser();
82          theNumCols = pNumCols;
83          theHiddens = new Boolean[theNumCols];
84          theStyles = new PrometheusSheetCellStyleType[theNumCols];
85      }
86  
87      /**
88       * Obtain OasisSheet.
89       * @return the row.
90       */
91      PrometheusOdfSheetCore getSheet() {
92          return theSheet;
93      }
94  
95      /**
96       * Obtain Column count.
97       * @return the column count.
98       */
99      int getColumnCount() {
100         return theNumCols;
101     }
102 
103     /**
104      * Process a column Element.
105      * @param pColumn the column to process
106      */
107     void processColumn(final Element pColumn) {
108         /* Determine the number of repeated columns */
109         final String myRepeatStr = theParser.getAttribute(pColumn, PrometheusOdfTableItem.COLUMNREPEAT);
110         int myRepeat = myRepeatStr == null
111                        ? 1
112                        : Integer.parseInt(myRepeatStr);
113 
114         /* Parse the value */
115         final Boolean myValue = parseColumnElement(pColumn);
116 
117         /* Add the additional columns */
118         addAdditionalCols(myRepeat);
119 
120         /* If we have a value */
121         if (myValue != null) {
122             /* Loop through the cells in reverse order */
123             for (int iIndex = theNumCols - 1;
124                  myRepeat > 0; iIndex--, myRepeat--) {
125                 /* Set the value */
126                 theHiddens[iIndex] = myValue;
127             }
128         }
129      }
130 
131     /**
132      * Add additional columns to table.
133      * @param pXtraCols the number of columns to add.
134      */
135     private void addAdditionalCols(final int pXtraCols) {
136         /* Adjust the # of columns */
137         theNumCols += pXtraCols;
138 
139         /* Determine the expansion length */
140         final int myLen = (((theNumCols + 1) / COL_EXPAND) + 1)  * COL_EXPAND;
141         theHiddens = Arrays.copyOf(theHiddens, myLen);
142         theStyles = Arrays.copyOf(theStyles, myLen);
143         theSheet.addAdditionalCols(pXtraCols);
144     }
145 
146     /**
147      * Obtain a readOnly column by its index.
148      * @param pSheet the owning sheet
149      * @param pColIndex the index of the column.
150      * @return the column if it exists, else null
151      */
152     PrometheusOdfColumn getReadOnlyColumnByIndex(final PrometheusOdfSheet pSheet,
153                                                  final int pColIndex) {
154         /* Handle index out of range */
155         if (pColIndex < 0 || pColIndex >= theNumCols) {
156             return null;
157         }
158 
159         /* Just return the column */
160         return new PrometheusOdfColumn(this, pSheet, pColIndex, true);
161     }
162 
163     /**
164      * Obtain a mutable column by its index, creating column if it does not exist.
165      * @param pSheet the owning sheet
166      * @param pColIndex the index of the column.
167      * @return the column
168      */
169     PrometheusOdfColumn getMutableColumnByIndex(final PrometheusOdfSheet pSheet,
170                                                 final int pColIndex) {
171         /* Handle negative column index */
172         if (pColIndex < 0) {
173             return null;
174         }
175 
176         /* If we need to extend the table */
177         if (pColIndex >= theNumCols) {
178             /* Determine the number of extra columns required */
179             final int myXtraCols = pColIndex
180                     - theNumCols
181                     + 1;
182 
183             /* Add additional columns */
184             addAdditionalCols(myXtraCols);
185         }
186 
187         /* Return the required column */
188         return new PrometheusOdfColumn(this, pSheet, pColIndex, false);
189     }
190 
191     /**
192      * Get hidden flag at index.
193      * @param pIndex the index
194      * @return the value
195      */
196     boolean getHiddenAtIndex(final int pIndex) {
197         return theHiddens[pIndex] != null;
198     }
199 
200     /**
201      * Set hidden flag at index.
202      * @param pIndex the index
203      * @param pHidden true/false
204      */
205     void setHiddenAtIndex(final int pIndex,
206                           final boolean pHidden) {
207        theHiddens[pIndex] = pHidden;
208     }
209 
210     /**
211      * Set style at index.
212      * @param pIndex the index
213      * @param pStyle the style
214      */
215     void setStyleAtIndex(final int pIndex,
216                          final PrometheusSheetCellStyleType pStyle) {
217         theStyles[pIndex] = pStyle;
218     }
219 
220     /**
221      * Populate Sheet children.
222      * @param pTable the sheet element
223      */
224     void populateSheetChildren(final Element pTable) {
225         /* Loop through the columns */
226         int myRepeat;
227         for (int iIndex = 0; iIndex < theNumCols; iIndex += myRepeat) {
228             /* Create a new column element */
229             final Element myColumn = theParser.newElement(PrometheusOdfTableItem.COLUMN);
230             pTable.appendChild(myColumn);
231 
232             /* Populate it */
233             populateColumn(myColumn, iIndex);
234 
235             /* Determine the repeat count */
236             myRepeat = getRepeatCountForIndex(iIndex);
237             if (myRepeat > 1) {
238                 /* Set attribute and adjust index */
239                 theParser.setAttribute(myColumn, PrometheusOdfTableItem.COLUMNREPEAT, myRepeat);
240             }
241         }
242     }
243 
244     /**
245      * Populate Column value.
246      * @param pElement the element
247      * @param pIndex the column index
248      */
249     private void populateColumn(final Element pElement,
250                                 final int pIndex) {
251         /* Set hidden if required */
252         if (theHiddens[pIndex] != null) {
253             theParser.setAttribute(pElement, PrometheusOdfTableItem.VISIBILITY, PrometheusOdfValue.COLLAPSE);
254         }
255         final PrometheusSheetCellStyleType myStyle = theStyles[pIndex];
256         if (myStyle != null) {
257             theSheet.setColumnStyle(pElement, myStyle);
258             theSheet.setDefaultCellStyle(pElement, myStyle);
259         }
260     }
261 
262     /**
263      * Obtain repeat count for index.
264      * @param pIndex the index to start at
265      * @return the repeat count
266      */
267     private int getRepeatCountForIndex(final int pIndex) {
268         /* Access the current values */
269         final Boolean myHidden = theHiddens[pIndex];
270         final PrometheusSheetCellStyleType myStyle = theStyles[pIndex];
271 
272         /* Loop through the remaining indices */
273         for (int i = pIndex + 1; i < theNumCols; i++) {
274             /* Access the next object */
275             final Boolean nextHidden = theHiddens[i];
276             final PrometheusSheetCellStyleType nextStyle = theStyles[i];
277 
278             /* Test for equality */
279             if (!Objects.equals(myHidden, nextHidden)
280                 || !Objects.equals(myStyle, nextStyle)) {
281                 return i - pIndex;
282             }
283         }
284 
285         /* All remaining values are equal */
286         return theNumCols - pIndex;
287     }
288 
289     /**
290      * parse Column element.
291      * @param pElement the element
292      * @return the Cell
293       */
294     private Boolean parseColumnElement(final Element pElement) {
295         /* Determine whether the column is hidden */
296         final String myHidden = theParser.getAttribute(pElement, PrometheusOdfTableItem.VISIBILITY);
297         return PrometheusOdfValue.COLLAPSE.getValue().equals(myHidden)
298                 ? Boolean.TRUE
299                 : null;
300     }
