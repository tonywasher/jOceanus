120     /**
121      * Constructor.
122      * @param pInput the input stream
123      * @throws OceanusException on error
124      */
125     PrometheusExcelHSSFWorkBook(final InputStream pInput) throws OceanusException {
126         try {
127             /* Load the book and set null map */
128             theBook = new HSSFWorkbook(pInput);
129             theStyleMap = null;
130             theRangeConstraintMap = null;
131             theValueConstraintMap = null;
132             theStyleEngine = null;
133             theNumberFont = null;
134             theValueFont = null;
135             theHeaderFont = null;
136 
137             /* Allocate the formatter */
138             theDataFormatter = createFormatter();
139 
140             /* Create evaluator and formatter */
141             theEvaluator = new HSSFFormulaEvaluator(theBook);
142             theExcelFormatter = new DataFormatter();
143 
144             /* Note readOnly */
145             isReadOnly = true;
146 
147         } catch (IOException e) {
148             throw new PrometheusSheetException("Failed to load workbook", e);
149         }
150     }

152     /**
153      * Constructor.
154      */
155     PrometheusExcelHSSFWorkBook() {
156         /* Create new book and map */
157         theBook = new HSSFWorkbook();
158         theStyleMap = new HashMap<>();
159         theRangeConstraintMap = new HashMap<>();
160         theValueConstraintMap = new HashMap<>();
161 
162         /* Allocate the formatter */
163         theDataFormatter = createFormatter();
164 
165         /* Create evaluator and formatter */
166         theEvaluator = new HSSFFormulaEvaluator(theBook);
167         theExcelFormatter = new DataFormatter();
168 
169         /* Ensure that we can create data formats */
170         theStyleEngine = theBook.createDataFormat();
171 
172         /* Create the Standard fonts */
173         theValueFont = theBook.createFont();
174         theValueFont.setFontName(PrometheusSheetFormats.FONT_VALUE);
175         theValueFont.setFontHeightInPoints((short) PrometheusSheetFormats.FONT_HEIGHT);
176         theNumberFont = theBook.createFont();
177         theNumberFont.setFontName(PrometheusSheetFormats.FONT_NUMERIC);
178         theNumberFont.setFontHeightInPoints((short) PrometheusSheetFormats.FONT_HEIGHT);
179         theHeaderFont = theBook.createFont();
180         theHeaderFont.setFontName(PrometheusSheetFormats.FONT_VALUE);
181         theHeaderFont.setBold(true);
182         theHeaderFont.setFontHeightInPoints((short) PrometheusSheetFormats.FONT_HEIGHT);
183 
184         /* Note writable */
185         isReadOnly = false;
186     }

268     @Override
269     public PrometheusSheetView getRangeView(final String pName) {
270         /* Find the range of cells */
271         final Name myName = theBook.getName(pName);
272         if (myName == null) {
273             return null;
274         }
275 
276         /* Parse the name reference */
277         final AreaReference myArea = new AreaReference(myName.getRefersToFormula(), SpreadsheetVersion.EXCEL2007);
278 
279         /* Determine extent of Range */
280         CellReference myRef = myArea.getLastCell();
281         final PrometheusSheetCellPosition myLastCell = new PrometheusSheetCellPosition(myRef.getCol(), myRef.getRow());
282         myRef = myArea.getFirstCell();
283         final PrometheusSheetCellPosition myFirstCell = new PrometheusSheetCellPosition(myRef.getCol(), myRef.getRow());
284 
285         /* Obtain the sheet */
286         final PrometheusSheetSheet mySheet = getSheet(myRef.getSheetName());
287 
288         /* Return the view */
289         return new PrometheusSheetView(mySheet, myFirstCell, myLastCell);
290     }

292     /**
293      * Declare the named range.
294      * @param pName the name of the range
295      * @param pRange the range to declare
296      * @throws OceanusException on error
297      */
298     void declareRange(final String pName,
299                       final AreaReference pRange) throws OceanusException {
300         /* Check for existing range */
301         Name myName = theBook.getName(pName);
302         if (myName != null) {
303             throw new PrometheusSheetException("Name "
304                                           + pName
305                                           + " already exists in workbook");
306         }
307 
308         /* Build the basic name */
309         myName = theBook.createName();
310         myName.setNameName(pName);
311 
312         /* Set into Name */
313         final String myRef = pRange.formatAsString();
314         myName.setRefersToFormula(myRef);
315     }

317     /**
318      * Apply Data Validation.
319      * @param pSheet the workSheet containing the cells
320      * @param pCells the Cells to apply validation to
321      * @param pValidRange the name of the validation range
322      */
323     void applyDataValidation(final HSSFSheet pSheet,
324                              final CellRangeAddressList pCells,
325                              final String pValidRange) {
326         /* Access the constraint */
327         final HSSFDataValidationHelper myHelper = new HSSFDataValidationHelper(pSheet);
328         final DataValidationConstraint myConstraint = theRangeConstraintMap.computeIfAbsent(pValidRange,
329                 myHelper::createFormulaListConstraint);
330 
331         /* Link the two and use drop down arrow */
332         final DataValidation myValidation = myHelper.createValidation(myConstraint, pCells);
333         myValidation.setSuppressDropDownArrow(false);
334 
335         /* Apply to the sheet */
336         pSheet.addValidationData(myValidation);
337     }

345     protected void applyDataValidation(final HSSFSheet pSheet,

348         /* Access the constraint */
349         final HSSFDataValidationHelper myHelper = new HSSFDataValidationHelper(pSheet);
350         final DataValidationConstraint myConstraint = theValueConstraintMap.computeIfAbsent(pValueList,
351                 myHelper::createExplicitListConstraint);
352 
353         /* Link the two and use drop down arrow */
354         final DataValidation myValidation = myHelper.createValidation(myConstraint, pCells);
355         myValidation.setSuppressDropDownArrow(false);
356 
357         /* Apply to the sheet */
358         pSheet.addValidationData(myValidation);
359     }

372     /**
373      * Obtain alignment for a cell.
374      * @param pType the cell type
375      * @return the alignment
376      */
377     private static HorizontalAlignment getStyleAlignment(final PrometheusSheetCellStyleType pType) {
378         switch (pType) {
379             case HEADER:
380             case BOOLEAN:
381                 return HorizontalAlignment.CENTER;
382             case DATE:
383             case STRING:
384                 return HorizontalAlignment.LEFT;
385             default:
386                 return HorizontalAlignment.RIGHT;
387         }

407     /**
408      * Obtain the required CellStyle.
409      * @param pType the CellStyleType
410      * @return the required CellStyle
411      */
412     HSSFCellStyle getCellStyle(final PrometheusSheetCellStyleType pType) {
413         /* Determine the correct format */
414         final String myStyleName = PrometheusSheetFormats.getFormatName(pType);
415 
416         /* Look for existing format */
417         HSSFCellStyle myStyle = theStyleMap.get(myStyleName);
418         if (myStyle != null) {
419             return myStyle;
420         }
421 
422         /* Create the New Cell Style */
423         myStyle = theBook.createCellStyle();
424         myStyle.setFont(getStyleFont(pType));
425         myStyle.setAlignment(getStyleAlignment(pType));
426 
427         /* If we have a data format */
428         if (PrometheusSheetFormats.hasDataFormat(pType)) {
429             /* Determine the format */
430             final String myFormat = PrometheusSheetFormats.getDataFormatString(pType);
431             myStyle.setDataFormat(theStyleEngine.getFormat(myFormat));
432         }
433 
434         /* Add to the map and return new style */
435         theStyleMap.put(myStyleName, myStyle);
436         return myStyle;
437     }

439     /**
440      * Obtain the required CellStyle.
441      * @param pValue the Cell Value
442      * @return the required CellStyle
443      */
444     HSSFCellStyle getCellStyle(final Object pValue) {
445         /* Determine the correct format */
446         final String myStyleName = PrometheusSheetFormats.getFormatName(pValue);
447 
448         /* Look for existing format */
449         HSSFCellStyle myStyle = theStyleMap.get(myStyleName);
450         if (myStyle != null) {
451             return myStyle;
452         }
453 
454         /* Determine the CellStyleType */
455         final PrometheusSheetCellStyleType myType = PrometheusSheetFormats.getCellStyleType(pValue);
456 
457         /* Create the New Cell Style */
458         myStyle = theBook.createCellStyle();
459         myStyle.setFont(getStyleFont(myType));
460         myStyle.setAlignment(getStyleAlignment(myType));
461 
462         /* If we have a data format */
463         if (myType != PrometheusSheetCellStyleType.BOOLEAN
464             && PrometheusSheetFormats.hasDataFormat(myType)) {
465             /* Determine the format */
466             final String myFormat = PrometheusSheetFormats.getDataFormatString(pValue);
467             myStyle.setDataFormat(theStyleEngine.getFormat(myFormat));
468         }
469 
470         /* Add to the map and return new style */
471         theStyleMap.put(myStyleName, myStyle);
472         return myStyle;
473     }

475     /**
476      * Obtain the required alternate CellStyle.
477      * @param pValue the Cell Value
478      * @return the required CellStyle
479      */
480     HSSFCellStyle getAlternateCellStyle(final Object pValue) {
481         /* Determine the correct format */
482         final String myStyleName = PrometheusSheetFormats.getAlternateFormatName(pValue);
483 
484         /* Look for existing format */
485         HSSFCellStyle myStyle = theStyleMap.get(myStyleName);
486         if (myStyle != null) {
487             return myStyle;
488         }
489 
490         /* Create the New Cell Style */
491         myStyle = theBook.createCellStyle();
492 
493         /* Determine the CellStyleType */
494         PrometheusSheetCellStyleType myType = PrometheusSheetFormats.getCellStyleType(pValue);
495 
496         /* Handle the header style */
497         if (myType == PrometheusSheetCellStyleType.STRING) {
498             myType = PrometheusSheetCellStyleType.HEADER;
499             myStyle.setLocked(true);
500         }
501 
502         /* Set font and style */
503         myStyle.setFont(getStyleFont(myType));
504         myStyle.setAlignment(getStyleAlignment(myType));
505 
506         /* If we have a data format */
507         if (myType != PrometheusSheetCellStyleType.BOOLEAN
508             && PrometheusSheetFormats.hasDataFormat(myType)) {
509             /* Determine the format */
510             final String myFormat = PrometheusSheetFormats.getAlternateFormatString(pValue);
511             myStyle.setDataFormat(theStyleEngine.getFormat(myFormat));
512         }
513 
514         /* Add to the map and return new style */
515         theStyleMap.put(myStyleName, myStyle);
516         return myStyle;
517     }
