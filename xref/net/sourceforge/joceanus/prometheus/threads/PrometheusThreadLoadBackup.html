37  /**
38   * Thread to load changes from an encrypted backup. Once the backup is loaded, the current database
39   * is loaded and the backup is re-based onto the database so that a correct list of additions,
40   * changes and deletions is built. These changes remain in memory and should be committed to the
41   * database later.
42   * @author Tony Washer
43   */
44  public class PrometheusThreadLoadBackup
45          implements TethysUIThread<PrometheusDataSet> {
46      /**
47       * Select Backup Task.
48       */
49      private static final String TASK_SELECTFILE = PrometheusThreadResource.TASK_SELECT_BACKUP.getValue();
50  
51      /**
52       * Data control.
53       */
54      private final PrometheusDataControl theControl;
55  
56      /**
57       * Constructor (Event Thread).
58       * @param pControl data control
59       */
60      public PrometheusThreadLoadBackup(final PrometheusDataControl pControl) {
61          theControl = pControl;
62      }
63  
64      @Override
65      public String getTaskName() {
66          return PrometheusThreadId.RESTOREBACKUP.toString();
67      }
68  
69      @Override
70      public PrometheusDataSet performTask(final TethysUIThreadManager pManager) throws OceanusException {
71          /* Access the thread manager */
72          final PrometheusToolkit myPromToolkit = (PrometheusToolkit) pManager.getThreadData();
73          final MetisToolkit myToolkit = myPromToolkit.getToolkit();
74          final PrometheusSecurityPasswordManager myPasswordMgr = myPromToolkit.getPasswordManager();
75  
76          /* Initialise the status window */
77          pManager.initTask(getTaskName());
78  
79          /* Access the Sheet preferences */
80          final MetisPreferenceManager myMgr = theControl.getPreferenceManager();
81          final PrometheusBackupPreferences myProperties = myMgr.getPreferenceSet(PrometheusBackupPreferences.class);
82  
83          /* Determine the archive name */
84          final File myBackupDir = new File(myProperties.getStringValue(PrometheusBackupPreferenceKey.BACKUPDIR));
85  
86          /* Determine the name of the file to load */
87          final TethysUIFileSelector myDialog = myToolkit.getGuiFactory().dialogFactory().newFileSelector();
88          myDialog.setTitle(TASK_SELECTFILE);
89          myDialog.setInitialDirectory(myBackupDir);
90          myDialog.setExtension(GordianUtilities.SECUREZIPFILE_EXT);
91          final File myFile = myDialog.selectFile();
92  
93          /* If we did not select a file */
94          if (myFile == null) {
95              /* Throw cancelled exception */
96              pManager.throwCancelException();
97          }
98  
99          /* Load workbook */
100         final PrometheusSpreadSheet mySheet = theControl.getSpreadSheet();
101         final PrometheusDataSet myData = theControl.getNewData();
102         mySheet.loadBackup(pManager, myPasswordMgr, myData, myFile);
103 
104         /* Create interface */
105         final PrometheusDataStore myDatabase = theControl.getDatabase();
106 
107         /* Load underlying database */
108         final PrometheusDataSet myStore = theControl.getNewData();
109         myDatabase.loadDatabase(pManager, myStore);
110 
111         /* Check security on the database */
112         myStore.checkSecurity(pManager);
113         if (myData.hasUpdates()) {
114             /* Store any updates */
115             myDatabase.updateDatabase(pManager, myData);
116         }
117 
118         /* Re-base the loaded backup onto the database image */
119         myData.reBase(pManager, myStore);
120 
121         /* State that we have completed */
122         pManager.setCompletion();
123 
124         /* Return the Data */
125         return myData;
126     }
127 
128     @Override
129     public void processResult(final PrometheusDataSet pResult) {
130         theControl.setData(pResult);
131     }
132 }
