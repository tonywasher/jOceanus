29  import net.sourceforge.joceanus.prometheus.security.PrometheusSecurityPasswordManager;

35  import java.io.File;
36  
37  /**
38   * LoaderThread extension to create an XML backup.
39   */
40  public class PrometheusThreadCreateXMLFile
41          implements TethysUIThread<Void> {
42      /**
43       * Buffer length.
44       */
45      private static final int BUFFER_LEN = 100;
46  
47      /**
48       * Number 10.
49       */
50      private static final int TEN = 10;
51  
52      /**
53       * File indicator.
54       */
55      private static final String SUFFIX_FILE = "XML";
56  
57      /**
58       * Data Control.
59       */
60      private final PrometheusDataControl theControl;
61  
62      /**
63       * Is this a Secure backup?
64       */
65      private final boolean isSecure;
66  
67      /**
68       * Constructor (Event Thread).
69       * @param pControl data control
70       * @param pSecure is this a secure backup
71       */
72      public PrometheusThreadCreateXMLFile(final PrometheusDataControl pControl,
73                                           final boolean pSecure) {
74          isSecure = pSecure;
75          theControl = pControl;
76      }
77  
78      @Override
79      public String getTaskName() {
80          return isSecure
81                  ? PrometheusThreadId.CREATEXML.toString()
82                  : PrometheusThreadId.CREATEXTRACT.toString();
83      }
84  
85      @Override
86      public Void performTask(final TethysUIThreadManager pManager) throws OceanusException {
87          /* Access the thread manager */
88          final PrometheusToolkit myPromToolkit = (PrometheusToolkit) pManager.getThreadData();
89          final PrometheusSecurityPasswordManager myPasswordMgr = myPromToolkit.getPasswordManager();
90          boolean doDelete = false;
91          File myFile = null;
92  
93          /* Catch Exceptions */
94          try {
95              /* Initialise the status window */
96              pManager.initTask(getTaskName());
97  
98              /* Access the Backup preferences */
99              final MetisPreferenceManager myMgr = theControl.getPreferenceManager();
100             final PrometheusBackupPreferences myProperties = myMgr.getPreferenceSet(PrometheusBackupPreferences.class);
101 
102             /* Determine the archive name */
103             final String myBackupDir = myProperties.getStringValue(PrometheusBackupPreferenceKey.BACKUPDIR);
104             final String myPrefix = myProperties.getStringValue(PrometheusBackupPreferenceKey.BACKUPPFIX);
105             final boolean doTimeStamp = myProperties.getBooleanValue(PrometheusBackupPreferenceKey.BACKUPTIME);
106 
107             /* Create the name of the file */
108             final StringBuilder myName = new StringBuilder(BUFFER_LEN);
109             myName.append(myBackupDir);
110             myName.append(File.separator);
111             myName.append(myPrefix);
112             myName.append(SUFFIX_FILE);
113 
114             /* If we are doing time-stamps */
115             if (doTimeStamp) {
116                 /* Obtain the current date/time */
117                 final OceanusDate myNow = new OceanusDate();
118 
119                 myName.append(myNow.getYear());
120                 if (myNow.getMonth() < TEN) {
121                     myName.append('0');
122                 }
123                 myName.append(myNow.getMonth());
124                 if (myNow.getDay() < TEN) {
125                     myName.append('0');
126                 }
127                 myName.append(myNow.getDay());
128             }
129 
130             /* Set the standard backup name */
131             myFile = new File(myName.toString() + (isSecure
132                     ? GordianUtilities.SECUREZIPFILE_EXT
133                     : GordianUtilities.ZIPFILE_EXT));
134 
135             /* Access the data */
136             final PrometheusDataSet myOldData = theControl.getData();
137 
138             /* Create a new formatter */
139             final PrometheusDataValuesFormatter myFormatter = new PrometheusDataValuesFormatter(pManager, myPasswordMgr);
140 
141             /* Create backup */
142             if (isSecure) {
143                 myFormatter.createBackup(myOldData, myFile);
144             } else {
145                 myFormatter.createExtract(myOldData, myFile);
146             }
147 
148             /* If this is a secure backup */
149             if (isSecure) {
150                 /* File created, so delete on error */
151                 doDelete = true;
152 
153                 /* Check for cancellation */
154 
155                 /* Initialise the status window */
156                 pManager.initTask("Reading Backup");
157 
158                 /* Load workbook */
159                 final PrometheusDataSet myNewData = theControl.getNewData();
160                 myFormatter.loadZipFile(myNewData, myFile);
161 
162                 /* Initialise the security, from the original data */
163                 myNewData.initialiseSecurity(pManager, myOldData);
164 
165                 /* Initialise the status window */
166                 pManager.initTask("Verifying Backup");
167 
168                 /* Create a difference set between the two data copies */
169                 final PrometheusDataSet myDiff = myNewData.getDifferenceSet(pManager, myOldData);
170 
171                 /* If the difference set is non-empty */
172                 if (!myDiff.isEmpty()) {
173                     /* Throw an exception */
174                     throw new PrometheusDataException(myDiff, "Backup is inconsistent");
175                 }
176 
177                 /* OK so switch off flag */
178                 doDelete = false;
179             }
180 
181         } finally {
182             /* Try to delete the file if required */
183             if (doDelete) {
184                 MetisToolkit.cleanUpFile(myFile);
185             }
186         }
187 
188         /* Return nothing */
189         return null;
190     }
191 }
