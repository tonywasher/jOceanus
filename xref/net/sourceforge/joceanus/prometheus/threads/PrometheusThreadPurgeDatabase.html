17  package net.sourceforge.joceanus.prometheus.threads;
18  
19  import net.sourceforge.joceanus.prometheus.data.PrometheusDataSet;
20  import net.sourceforge.joceanus.prometheus.database.PrometheusDataStore;
21  import net.sourceforge.joceanus.prometheus.views.PrometheusDataControl;
22  import net.sourceforge.joceanus.oceanus.base.OceanusException;
23  import net.sourceforge.joceanus.tethys.api.thread.TethysUIThread;
24  import net.sourceforge.joceanus.tethys.api.thread.TethysUIThreadManager;
25  
26  /**
27   * Thread to purge tables in a database that represent a data set. Existing loaded data will be
28   * marked as new so that it will be written to the database via the store command.
29   */
30  public class PrometheusThreadPurgeDatabase
31          implements TethysUIThread<Void> {
32      /**
33       * Data Control.
34       */
35      private final PrometheusDataControl theControl;
36  
37      /**
38       * Constructor (Event Thread).
39       * @param pControl data control
40       */
41      public PrometheusThreadPurgeDatabase(final PrometheusDataControl pControl) {
42          theControl = pControl;
43      }
44  
45      @Override
46      public String getTaskName() {
47          return PrometheusThreadId.PURGEDB.toString();
48      }
49  
50      @Override
51      public Void performTask(final TethysUIThreadManager pManager) throws OceanusException {
52          /* Initialise the status window */
53          pManager.initTask(getTaskName());
54  
55          /* Create interface */
56          final PrometheusDataStore myDatabase = theControl.getDatabase();
57  
58          /* Protect against failures */
59          try {
60              /* Purge database */
61              myDatabase.purgeTables(pManager);
62  
63              /* Re-base this set on a null set */
64              final PrometheusDataSet myNull = theControl.getNewData();
65              final PrometheusDataSet myData = theControl.getData();
66              myData.reBase(pManager, myNull);
67  
68              /* Derive the new set of updates */
69              theControl.deriveUpdates();
70  
71              /* State that we have completed */
72              pManager.setCompletion();
73  
74              /* Return null */
75              return null;
76  
77              /* Make sure that the database is closed */
78          } finally {
79              /* Close the database */
80              myDatabase.close();
81          }
82      }
83  }
