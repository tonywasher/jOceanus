17  package net.sourceforge.joceanus.prometheus.database;
18  
19  import net.sourceforge.joceanus.metis.data.MetisDataItem.MetisDataFieldId;
20  import net.sourceforge.joceanus.prometheus.data.PrometheusControlKeySet;
21  import net.sourceforge.joceanus.prometheus.data.PrometheusDataKeySet;
22  import net.sourceforge.joceanus.prometheus.data.PrometheusDataResource;
23  import net.sourceforge.joceanus.prometheus.data.PrometheusDataSet;
24  import net.sourceforge.joceanus.prometheus.data.PrometheusDataSet.PrometheusCryptographyDataType;
25  import net.sourceforge.joceanus.prometheus.data.PrometheusDataValues;
26  import net.sourceforge.joceanus.oceanus.base.OceanusException;
27  
28  /**
29   * Database table class for ControlKey.
30   */
31  public class PrometheusTableControlKeySet
32          extends PrometheusTableDataItem<PrometheusControlKeySet> {
33      /**
34       * The name of the DataKeySet table.
35       */
36      protected static final String TABLE_NAME = PrometheusControlKeySet.LIST_NAME;
37  
38      /**
39       * Constructor.
40       * @param pDatabase the database control
41       */
42      protected PrometheusTableControlKeySet(final PrometheusDataStore pDatabase) {
43          super(pDatabase, TABLE_NAME);
44          final PrometheusTableDefinition myTableDef = getTableDef();
45  
46          /* Define the columns */
47          myTableDef.addReferenceColumn(PrometheusCryptographyDataType.CONTROLKEY, PrometheusTableControlKeys.TABLE_NAME);
48          myTableDef.addBinaryColumn(PrometheusDataResource.KEYSET_KEYSETDEF, PrometheusDataKeySet.WRAPLEN);
49      }
50  
51      @Override
52      protected void declareData(final PrometheusDataSet pData) {
53          setList(pData.getControlKeySets());
54      }
55  
56      @Override
57      protected PrometheusDataValues loadValues() throws OceanusException {
58          /* Access the table definition */
59          final PrometheusTableDefinition myTableDef = getTableDef();
60  
61          /* Build data values */
62          final PrometheusDataValues myValues = getRowValues(PrometheusDataKeySet.OBJECT_NAME);
63          myValues.addValue(PrometheusCryptographyDataType.CONTROLKEY, myTableDef.getIntegerValue(PrometheusCryptographyDataType.CONTROLKEY));
64          myValues.addValue(PrometheusDataResource.KEYSET_KEYSETDEF, myTableDef.getBinaryValue(PrometheusDataResource.KEYSET_KEYSETDEF));
65  
66          /* Return the values */
67          return myValues;
68      }
69  
70      @Override
71      protected void setFieldValue(final PrometheusControlKeySet pItem,
72                                   final MetisDataFieldId iField) throws OceanusException {
73          /* Switch on field id */
74          final PrometheusTableDefinition myTableDef = getTableDef();
75          if (PrometheusCryptographyDataType.CONTROLKEY.equals(iField)) {
76              myTableDef.setIntegerValue(iField, pItem.getControlKeyId());
77          } else if (PrometheusDataResource.KEYSET_KEYSETDEF.equals(iField)) {
78              myTableDef.setBinaryValue(iField, pItem.getSecuredKeySetDef());
79          } else {
80              super.setFieldValue(pItem, iField);
81          }
82      }
83  }
