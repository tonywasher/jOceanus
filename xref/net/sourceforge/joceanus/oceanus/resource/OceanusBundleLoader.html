17  package net.sourceforge.joceanus.oceanus.resource;
18  
19  import java.lang.ref.WeakReference;
20  import java.util.Map;
21  import java.util.ResourceBundle;
22  import java.util.function.Function;
23  
24  /**
25   * Bundle Loader.
26   */
27  public final class OceanusBundleLoader {
28      /**
29       * String Builder.
30       */
31      private final StringBuilder theBuilder = new StringBuilder();
32  
33      /**
34       * BundleName.
35       */
36      private final String theBundleName;
37  
38      /**
39       * BundleLoader.
40       */
41      private final Function<String, ResourceBundle> theLoader;
42  
43      /**
44       * ResourceBundle.
45       */
46      private WeakReference<ResourceBundle> theBundle;
47  
48      /**
49       * Standard constructor.
50       * @param pBundleName the Bundle Name
51       * @param pLoader the loader
52       */
53      private OceanusBundleLoader(final String pBundleName,
54                                  final Function<String, ResourceBundle> pLoader) {
55          theBundleName = pBundleName;
56          theLoader = pLoader;
57      }
58  
59      /**
60       * Obtain standard loader.
61       * @param pBundleName the Bundle Name
62       * @param pLoader the bundle loader
63       * @return the loader
64       */
65      public static OceanusBundleLoader getLoader(final String pBundleName,
66                                                  final Function<String, ResourceBundle> pLoader) {
67          return new OceanusBundleLoader(pBundleName, pLoader);
68      }
69  
70      /**
71       * Build name of resource for Key.
72       * @param <K> the resource key data type
73       * @param pKey the resource Key
74       * @return the resourceName
75       */
76      public <K extends Enum<K> & OceanusBundleId> String getValue(final K pKey) {
77          /* Build the required name */
78          theBuilder.setLength(0);
79          theBuilder.append(pKey.getNameSpace());
80          theBuilder.append('.');
81          theBuilder.append(pKey.getKeyName());
82  
83          /* Access the cached bundle */
84          ResourceBundle myBundle = theBundle == null
85                                                      ? null
86                                                      : theBundle.get();
87  
88          /* If the bundle is not cached */
89          if (myBundle == null) {
90              /* Access and cache the bundle */
91              myBundle = theLoader.apply(theBundleName);
92              theBundle = new WeakReference<>(myBundle);
93          }
94  
95          /* Access the required resource */
96          return myBundle.getString(theBuilder.toString());
97      }
98  
99      /**
100      * Obtain key for enum.
101      * @param <E> the enum type
102      * @param pMap the map
103      * @param pValue the enum value
104      * @return the resource key
105      */
106     public static <E extends Enum<E>> OceanusBundleId getKeyForEnum(final Map<E, OceanusBundleId> pMap,
107                                                                     final E pValue) {
108         final OceanusBundleId myId = pMap.get(pValue);
109         if (myId == null) {
110             throw new IllegalArgumentException(OceanusResourceLoader.getErrorNoResource(pValue));
111         }
112         return myId;
113     }
114 }
