1   /**
2    * Provides classes to represent decimal numbers with fixed numbers of decimal digits
3    * {@link #theScale} as Long integers. The decimal value is multiplied by 10 to the power of the
4    * number of decimals for the number ({@link #theFactor}). The integral part of the number can be
5    * expressed as (Value / Factor) and the fractional part as (Value % Factor). Arithmetic is then
6    * performed as whole number arithmetic on these values, with due care taken on multiplication and
7    * division to express the result to the correct number of decimals without losing any part of the
8    * answer to overflow.
9    */
10  package net.sourceforge.joceanus.oceanus.decimal;

15  /**
16   * Decimal class performing integer arithmetic on large decimals.
17   */
18  public class OceanusNewDecimal {

24      /**

77      /**
78       * Constructor.
79       * @param pSource the source BigDecimal
80       */
81      public OceanusNewDecimal(final BigDecimal pSource) {
82          /* Store sign and scale */
83          theSign = pSource.signum();
84          theScale = pSource.scale();
85          checkValidScale(theScale);
86          theFactor = getFactor(theScale);
87  
88          /* Extract the integral and fractional parts */
89          theIntegral = pSource.longValue();
90          theIntegral *= theSign;
91          long myFractional = pSource.movePointRight(theScale).longValue() * theSign;
92          myFractional %= theFactor;
93          theFractional = (int) myFractional;
94      }

111         theIntegral = pIntegral;
112         theFractional = pFractional;
113         theSign = pSign;
114         theScale = pScale;

118     /**
119      * Check that the scale is valid.
120      * @param pScale the scale
121      */
122     private static void checkValidScale(final int pScale) {
123         if (pScale < 0 || pScale > MAX_DECIMALS) {
124             throw new IllegalArgumentException("Invalid scale - " + pScale);
125         }
126     }

160     /**
161      * Add a decimal to value.
162      * @param pDecimal the decimal to add to this value
163      */
164     public void add(final OceanusNewDecimal pDecimal) {
165         /* Access the integral/fractional part of the second decimal at the same scale */
166         long myIntegral = pDecimal.theIntegral;
167         long myFractional = adjustDecimals(pDecimal.theFractional, theScale - pDecimal.theScale);
168         if (myFractional >= theFactor) {
169             myFractional %= theFactor;
170             myIntegral++;
171         }
172         add(myIntegral, (int) myFractional);
173     }

175     /**
176      * Subtract a decimal from value.
177      * @param pDecimal the decimal to subtract from this value
178      */
179     public void subtract(final OceanusNewDecimal pDecimal) {
180         /* Access the integral/fractional part of the second decimal at the same scale */
181         long myIntegral = pDecimal.theIntegral;
182         long myFractional = adjustDecimals(pDecimal.theFractional, theScale - pDecimal.theScale);
183         if (myFractional >= theFactor) {
184             myFractional %= theFactor;
185             myIntegral++;
186         }
187         add(-myIntegral, (int) -myFractional);
188     }

190     /**
191      * Add a decimal to this value.
192      * @param pIntegral the integral part of the decimal.
193      * @param pFractional the fractional part of the decimal
194      */
195     private void add(final long pIntegral,
196                      final int pFractional) {
197          /* Add the fractional and non-fractional parts of the sum */
198         theFractional = pFractional + fractionalValue();
199         theIntegral = pIntegral + integralValue();
200 
201         /* If we have a positive integral # */
202         if (theIntegral > 0) {
203             /* Handle fractional too small */
204             if (theFractional < 0) {
205                 theFractional += theFactor;
206                 theIntegral--;
207 
208                 /* Handle fractional too large */
209             } else if (theFractional >= theFactor) {
210                 theFractional -= theFactor;
211                 theIntegral++;
212             }
213 
214             /* Set sign */
215             theSign = 1;
216 
217             /* If we have a negative integral # */
218         } else if (theIntegral < 0) {
219             /* Handle fractional too large */
220             if (theFractional > 0) {
221                 theFractional -= theFactor;
222                 theIntegral++;
223 
224                 /* Handle fractional too small */
225             } else if (theFractional <= -theFactor) {
226                 theFractional += theFactor;
227                 theIntegral--;
228             }
229 
230             /* Set sign */
231             theSign = -1;
232             theFractional = -theFractional;
233             theIntegral = -theIntegral;
234 
235             /* else we have a zero integral */
236         } else {
237             /* Handle fractional too large */
238             if (theFractional >= theFactor) {
239                 theFractional -= theFactor;
240                 theIntegral = 1;
241                 theSign = 1;
242 
243                 /* Handle Fractional too small */
244             } else if (theFractional <= -theFactor) {
245                 theFractional += theFactor;
246                 theSign = -1;
247                 theIntegral = 1;
248                 theFractional = -theFractional;
249 
250                 /* Handle positive fractional */
251             } else if (theFractional > 0) {
252                 theSign = 1;
253 
254                 /* Handle negative fractional */
255             } else if (theFractional < 0) {
256                 theSign = -1;
257                 theFractional = -theFractional;
258 
259                 /* Handle zero fractional */
260             } else {
261                 theSign = 0;
262             }
263         }
264     }

266     /**
267      * Multiply by another decimal
268      * <p>
269      * This function splits the values into three separate integers and then performs long arithmetic to
270      * prevent loss of precision. The value is represented as (x,y,z,s) where the decimal may be written as
271      * x*2<sup>32</sup> + y + z*10<sup>-s</sup> and x,y,z,s are all integers.
272      * <p>
273      * The product of (x<sub>1</sub>, y<sub>1</sub>, z<sub>1</sub>, s) by
274      * (x<sub>2</sub>, y<sub>2</sub>, z<sub>2</sub>, t)
275      * is therefore x<sub>1</sub>*x<sub>2</sub>*2<sup>64</sup> (discardable)
276      * + (x<sub>1</sub>*y<sub>2</sub> + x<sub>2</sub>*y<sub>1</sub>)*2<sup>32</sup>
277      * + x<sub>2</sub>*y<sub>2</sub>
278      * + x<sub>1</sub>*z<sub>2</sub>*2<sup>32</sup>*10<sup>-t</sup>
279      * + x<sub>2</sub>*z<sub>1</sub>*2<sup>32</sup>*10<sup>-s</sup>
280      * + y<sub>1</sub>*z<sub>2</sub>*10<sup>-t</sup>
281      * + y<sub>2</sub>*z<sub>1</sub>*10<sup>-s</sup>
282      * + z<sub>1</sub>*z<sub>2</sub>*10<sup>-s-t</sup>
283      *
284      * @param pMultiplicand the decimal to multiply by
285      */
286     public void multiply(final OceanusNewDecimal pMultiplicand) {
287         /* Access the parts of this value */
288         final long myX1 = theIntegral >>> Integer.SIZE;
289         final long myY1 = theIntegral & INTEGER_MASK;
290         final long myZ1 = theFractional;
291         final int myS = theScale;
292         final int mySFactor = theFactor;
293 
294         /* Access the parts of the multiplicand */
295         final long myX2 = pMultiplicand.theIntegral >>> Integer.SIZE;
296         final long myY2 = pMultiplicand.theIntegral & INTEGER_MASK;
297         final long myZ2 = pMultiplicand.theFractional;
298         final int myT = pMultiplicand.theScale;
299         final int myTFactor = pMultiplicand.theFactor;
300         final long mySTFactor = mySFactor * (long) myTFactor;
301 
302         /* Calculate integral products */
303         long myIntegral = ((myX1 * myY2) + (myY1 * myX2)) << Integer.SIZE;
304         myIntegral += myY2 * myX2;
305 
306         /* Calculate product of X2 and Z1 and multiply by 2^32 */
307         long myProduct = myX2 * myZ1;
308         long myIntPart = myProduct / mySFactor;
309         long myFracPart = myProduct % mySFactor;
310         myIntegral += myIntPart << Integer.SIZE;
311         myFracPart *= INTEGER_BOOST;
312         myIntegral += myFracPart / mySFactor;
313         long myFractional = adjustDecimals(myFracPart % mySFactor, myT);
314 
315         /* Calculate products of Y2 and Z1 */
316         myProduct = myY2 * myZ1;
317         myIntegral += myProduct / mySFactor;
318         myFractional += adjustDecimals(myProduct % mySFactor, myT);
319 
320         /* Calculate product of X1 and Z2 and multiply by 2^32 */
321         myProduct = myX1 * myZ2;
322         myIntPart = myProduct / myTFactor;
323         myFracPart = myProduct % myTFactor;
324         myIntegral += myIntPart << Integer.SIZE;
325         myFracPart *= INTEGER_BOOST;
326         myIntegral += myFracPart / myTFactor;
327         myFractional += adjustDecimals(myFracPart % myTFactor, myS);
328 
329         /* Calculate products of Y1 and Z2 */
330         myProduct = myY1 * myZ2;
331         myIntegral += myProduct / myTFactor;
332         myFractional += adjustDecimals(myProduct % myTFactor, myS);
333 
334         /* Calculate products of Z1 and Z2 */
335         myFractional += myZ1 * myZ2;
336 
337         /* Handle wrap of fractional */
338         myIntegral += myFracPart / mySTFactor;
339         myFractional %= mySTFactor;
340 
341         /* Adjust decimals */
342         myFractional = adjustDecimals(myFractional, -myT);
343         if (myFractional >= mySFactor) {
344             myFractional %= mySFactor;
345             myIntegral++;
346         }
347 
348         /* Store the result */
349         theIntegral = myIntegral;
350         theFractional = (int) myFractional;
351         theSign *= pMultiplicand.theSign;
352     }

354     /**
355      * Divide by another decimal.
356      * <p>
357      * This function uses BigDecimal to perform the calculation
358      * @param pDivisor the decimal to divide by
359      */
360     public void divide(final OceanusNewDecimal pDivisor) {
361         /* Calculate the result */
362         final BigDecimal myNumerator = toBigDecimal();
363         final BigDecimal myDenominator = pDivisor.toBigDecimal();
364         final BigDecimal myResult = myNumerator.divide(myDenominator, theScale, RoundingMode.HALF_UP);
365 
366         /* Extract the integral and fractional parts */
367         theSign = myResult.signum();
368         theIntegral = myResult.longValue();
369         theIntegral *= theSign;
370         long myFractional = myResult.movePointRight(theScale).longValue() * theSign;
371         myFractional %= theFactor;
372         theFractional = (int) myFractional;
373     }

375     /**
376      * Convert to BigDecimal.
377      * @return the BigDecimal equivalent
378      */
379     public BigDecimal toBigDecimal() {
380         final BigDecimal myIntegral = new BigDecimal(theIntegral);
381         final BigDecimal myFractional = new BigDecimal(theFractional).movePointLeft(theScale);
382         final BigDecimal myResult = myIntegral.add(myFractional);
383         return theSign == -1 ? myResult.negate() : myResult;
384     }

386     @Override
387     public String toString() {
388         /* Format the string */
389         final StringBuilder myString = new StringBuilder(100);
390         if (theSign == -1) {
391             myString.append('-');
392         }
393         myString.append(theIntegral);
394         if (theScale > 0) {
395             final int myLen = myString.length();
396             myString.append(theFractional + theFactor);
397             myString.setCharAt(myLen, '.');
398         }
399 
400         /* Return the string */
401         return myString.toString();
402     }

404     /**
405      * Build powers of ten.
406      * @param pMax maximum power of ten
407      * @return array of powers of ten
408      */
409     private static int[] getPowersOfTen(final int pMax) {
410         /* Allocate the array */
411         final int[] myArray = new int[pMax + 1];
412 
413         /* Initialise array */
414         int myValue = 1;
415         myArray[0] = myValue;
416 
417         /* Loop through array */
418         for (int i = 1; i < pMax + 1; i++) {
419             /* Adjust value and record it */
420             myValue *= RADIX_TEN;
421             myArray[i] = myValue;
422         }
423 
424         /* Return the array */
425         return myArray;
426     }

437     /**
438      * Adjust a value to a different number of decimals.
439      * <p>
440      * If the adjustment is to reduce the number of decimals, the most significant digit of the
441      * discarded digits is examined to determine whether to round up. If the number of decimals is
442      * to be increased, zeros are simply added to the end.
443      * @param pValue the value to adjust
444      * @param iAdjust the adjustment (positive if # of decimals are to increase, negative if they
445      * are to decrease)
446      * @return the adjusted value
447      */
448     private static long adjustDecimals(final long pValue,
449                                        final int iAdjust) {
450         /* Take a copy of the value */
451         long myValue = pValue;
452 
453         /* If we need to reduce decimals */
454         if (iAdjust < 0) {
455             /* If we have more than one decimal to remove */
456             if (iAdjust + 1 < 0) {
457                 /* Calculate division factor (minus one) */
458                 final long myFactor = getFactor(-(iAdjust + 1));
459 
460                 /* Reduce to 10 times required value */
461                 myValue /= myFactor;
462             }
463 
464             /* Access last digit */
465             long myDigit = myValue
466                     % RADIX_TEN;
467 
468             /* Handle negative values */
469             int myAdjust = 1;
470             if (myDigit < 0) {
471                 myAdjust = -1;
472                 myDigit = -myDigit;
473             }
474 
475             /* Reduce final decimal and round up if required */
476             myValue /= RADIX_TEN;
477             if (myDigit >= (RADIX_TEN >> 1)) {
478                 myValue += myAdjust;
479             }
480 
481             /* else if we need to expand fractional product */
482         } else if (iAdjust > 0) {
483             myValue *= getFactor(iAdjust);
484         }
485 
486         /* Return the adjusted value */
487         return myValue;
488     }
